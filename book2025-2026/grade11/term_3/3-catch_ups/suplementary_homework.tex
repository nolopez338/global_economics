		\SubsectionBox{Critical values used}
		\vspace{-1cm}
		\[
		z_{0.95}=1.645
		\qquad
		z_{0.975}=1.960
		\qquad
		z_{0.99}=2.326
		\]
		
		For right-tailed tests, reject when \(z>z_{\text{critical}}\). For left-tailed tests, reject when \(z<-z_{\text{critical}}\). For two-tailed tests, reject when \(|z|>z_{\text{critical}}\).

		\SubsectionBox{Formula tools}
		\vspace{-0.65cm}
		\[
		\begin{array}{c@{\qquad}c}
			\text{Mean-count test} & \text{Proportion test} \\[0.35cm]
			z=\dfrac{\bar x-\mu_0}{\frac{\sigma}{\sqrt{n}}}
			&
			z=\dfrac{\hat p-p_0}{\sqrt{\frac{p_0(1-p_0)}{n}}}
		\end{array}
		\]
		
		Minimum sample-size tools:
		\[
		n=\left(\frac{z\sigma}{\bar x-\mu_0}\right)^2
		\qquad 
		n=\frac{z^2p_0(1-p_0)}{(\hat p-p_0)^2}
		\]

		\subsection*{C1}
		
		Organize the statistical information for both groups and both settings.
		
		\begin{enumerate}
			\item Describe the real-world context, the two groups, and the counted variable.
			\item Define the binary threshold used to transform each count into \(0\) or \(1\).
			\item Show a few sample rows using the required data-table format.
			\item Summarize the required statistical information for both groups:
			\[
			n_A,\quad \bar x_A,\quad \sigma_A,\quad x_A,\quad \hat p_A
			\]
			\[
			n_B,\quad \bar x_B,\quad \sigma_B,\quad x_B,\quad \hat p_B
			\]
			\item Define the population parameters for the mean-count setting and the proportion setting.
		\end{enumerate}
		
		\subsection*{C2}
		
		Write the null and alternative hypotheses for all setup cases.
		
		\begin{enumerate}
			\item Mean-count setting using Group A as the benchmark and Group B as the test group.
			\item Proportion setting using Group A as the benchmark and Group B as the test group.
			\item Mean-count setting using Group B as the benchmark and Group A as the test group.
			\item Proportion setting using Group B as the benchmark and Group A as the test group.
		\end{enumerate}
		
		Use \(\mu\) for the mean-count setting and \(p\) for the proportion setting.
		
		\subsection*{C3}
		
		Classify each setup case as right-tailed, left-tailed, or two-tailed.
		
		\begin{enumerate}
			\item For each setup case from C2, state whether the test is right-tailed, left-tailed, or two-tailed.
			\item Justify the classification using the sign in the alternative hypothesis.
			\item Draw an approximate rejection-region graph for each setup case. Each graph must match the selected significance level and the correct tail direction.
		\end{enumerate}
		
		\subsection*{Benchmark direction for C4--C7}
		
		After completing C1--C3, choose only one benchmark direction.
		
		\[
		\text{Selected benchmark direction: } \underline{\hspace{3cm}}
		\]
		
		All work from C4 to C7 must use only this selected direction. Do not complete C4--C7 for the reverse direction.
		
		\subsection*{C4}
		
		Apply the hypothesis test in the proportion setting for your selected benchmark direction.
		
		\begin{enumerate}
			\item State the benchmark proportion \(p_0\), the sample proportion \(\hat p\), and the sample size \(n\).
			\item State the significance level and the correct critical value.
			\item Compute the test statistic:
			\[
			z=\dfrac{\hat p-p_0}{\sqrt{\frac{p_0(1-p_0)}{n}}}
			\]
			\item Compare the test statistic with the appropriate critical value and state the decision about \(H_0\).
			\item Interpret the decision in the context given.
		\end{enumerate}
		
		\subsection*{C5}
		
		Apply the hypothesis test in the mean-count setting for your selected benchmark direction.
		
		\begin{enumerate}
			\item State the benchmark mean \(\mu_0\), the sample mean \(\bar x\), the standard deviation \(\sigma\), and the sample size \(n\).
			\item State the significance level and the correct critical value.
			\item Compute the test statistic:
			\[
			z=\dfrac{\bar x-\mu_0}{\frac{\sigma}{\sqrt{n}}}
			\]
			\item Compare the test statistic with the appropriate critical value and state the decision about \(H_0\).
			\item Interpret the decision in the context given.
		\end{enumerate}
		
		\subsection*{C7}
		
		Determine the minimum sample size needed to reject \(H_0\) for the selected benchmark direction.
		
		\begin{enumerate}
			\item Find the minimum sample size for the proportion setting:
			\[
			n=\frac{z^2p_0(1-p_0)}{(\hat p-p_0)^2}
			\]
			\item Find the minimum sample size for the mean-count setting:
			\[
			n=\left(\frac{z\sigma}{\bar x-\mu_0}\right)^2
			\]
			\item Round each result up to the next whole number.
			\item Verify that each minimum sample size satisfies the rejection condition.
		\end{enumerate}
		
		\subsection*{C6}
		
		Revisit the initial-sample-size decision and the minimum-sample-size decision for both settings.
		
		\begin{enumerate}
			\item For the proportion setting, compare the initial decision and the decision using the minimum sample size found in C7.
			\item For the mean-count setting, compare the initial decision and the decision using the minimum sample size found in C7.
			\item For each case, identify the possible error type: Type I error or Type II error.
			\item Explain the practical consequence of each possible error in the context given.
		\end{enumerate}
