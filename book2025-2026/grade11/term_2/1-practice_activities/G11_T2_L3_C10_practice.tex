	\ExamSection{C10: Conclusions about a mean parameter in finance and economy}
	\begin{ExamProblems}
		\item
		\subsection*{Problem description}
		\textbf{Context.}
		A commercial bank tracks daily net fee revenue (thousand USD) from a new SME payments package for 16 days:
		\[
		12,\ 13,\ 14,\ 13,\ 12,\ 14,\ 13,\ 13,\ 14,\ 12,\ 13,\ 14,\ 13,\ 12,\ 14,\ 13
		\]

		\textbf{Hypotheses.}
		\[
		H_0:\mu=12.90
		\qquad
		H_a:\mu\neq 12.90
		\]

		\textbf{Sample information.}
		\[
		n=16,\quad \bar{x}=13.0625,\quad s=0.7719,\quad \alpha=0.05
		\]

		\textbf{Given test result.}
		\[
		t=0.842
		\qquad
		p\text{-value}=0.4130
		\]

		\textbf{Questions/Tasks.}
		\begin{enumerate}[label=\alph*)]
			\item Based on the p-value and the significance level, should $H_0$ be rejected or should we fail to reject $H_0$?
			\item Write the conclusion in the context of the bank's mean daily net fee revenue.
			\item Identify which type of error could occur after making this decision.
			\item Explain what that error would mean in this real-world context.
			\item Explain the implications of the decision for pricing, monitoring, or management decisions for the SME payments package.
		\end{enumerate}

		\subsection*{C10}
		\textbf{Decision rule.}
		\[
		\text{Reject }H_0\text{ if }p\text{-value}<\alpha.
		\]
		\[
		\text{Fail to reject }H_0\text{ if }p\text{-value}\geq\alpha.
		\]
		Since $0.4130\geq 0.05$, we fail to reject $H_0$.

		\textbf{Conclusion in context.}
		At the 5\% significance level, there is not enough statistical evidence to conclude that the bank's true mean daily net fee revenue differs from 12.90 thousand USD.

		\textbf{Relevant error type.}
		Because the decision is to fail to reject $H_0$, the possible error is a Type II error: failing to reject $H_0$ even though the alternative hypothesis is actually true.

		\textbf{Real-world implication.}
		If a Type II error occurs, management could keep assuming the package performs at 12.90 thousand USD when the true mean is actually different, delaying needed pricing or strategy adjustments.

		\subsection*{Problem description}
		\textbf{Context.}
		An export firm groups daily freight insurance cost (USD per shipment) into classes for 30 days:
		\[
		\begin{array}{c|c}
		\text{Class interval} & \text{Frequency}\\ \hline
		\text{[80,90)} & 3\\
		\text{[90,100)} & 7\\
		\text{[100,110)} & 9\\
		\text{[110,120)} & 6\\
		\text{[120,130)} & 5
		\end{array}
		\]

		\textbf{Hypotheses.}
		\[
		H_0:\mu=112
		\qquad
		H_a:\mu\neq 112
		\]

		\textbf{Sample information.}
		Using class midpoints,
		\[
		n=30,\quad \bar{x}\approx 106.00,\quad s\approx 12.42,\quad \alpha=0.05
		\]

		\textbf{Given test result.}
		\[
		t=-2.647
		\qquad
		p\text{-value}=0.0130
		\]

		\textbf{Questions/Tasks.}
		\begin{enumerate}[label=\alph*)]
			\item Based on the p-value and the significance level, should $H_0$ be rejected or should we fail to reject $H_0$?
			\item Write the conclusion in the context of the firm's mean daily insurance cost.
			\item Identify which type of error could occur after making this decision.
			\item Explain what that error would mean in this real-world context.
			\item Explain the implications of the decision for budgeting and freight-cost planning.
		\end{enumerate}

		\subsection*{C10}
		\textbf{Decision rule.}
		\[
		\text{Reject }H_0\text{ if }p\text{-value}<\alpha.
		\]
		\[
		\text{Fail to reject }H_0\text{ if }p\text{-value}\geq\alpha.
		\]
		Since $0.0130<0.05$, we reject $H_0$.

		\textbf{Conclusion in context.}
		At the 5\% significance level, there is enough statistical evidence that the firm's true mean daily freight insurance cost is different from 112 USD per shipment.

		\textbf{Relevant error type.}
		Because the decision is to reject $H_0$, the possible error is a Type I error: rejecting $H_0$ even though $H_0$ is actually true.

		\textbf{Real-world implication.}
		If a Type I error occurs, managers might change budgets and contracts away from the 112 USD benchmark even though that benchmark is actually correct, creating unnecessary operational disruption.

		\subsection*{Problem description}
		\textbf{Context.}
		A fintech support unit studies the mean ticket-resolution time (hours). It has two data sources in the same month:
		\begin{itemize}
			\item Individual observations (8 tickets):
			\[
			3,\ 3,\ 4,\ 3,\ 3,\ 2,\ 4,\ 4
			\]
			\item Additional tickets summarized in grouped form:
			\[
			\begin{array}{c|c}
			\text{Class interval} & \text{Frequency}\\ \hline
			\text{[2,3)} & 3\\
			\text{[3,4)} & 5\\
			\text{[4,5)} & 4
			\end{array}
			\]
		\end{itemize}

		\textbf{Hypotheses.}
		\[
		H_0:\mu=3.10
		\qquad
		H_a:\mu\neq 3.10
		\]

		\textbf{Sample information.}
		Using midpoint approximations for the grouped part,
		\[
		n=20,\quad \bar{x}=3.45,\quad s\approx 0.7592,\quad \alpha=0.10
		\]

		\textbf{Given test result.}
		\[
		t=2.061
		\qquad
		p\text{-value}=0.0532
		\]

		\textbf{Questions/Tasks.}
		\begin{enumerate}[label=\alph*)]
			\item Based on the p-value and the significance level, should $H_0$ be rejected or should we fail to reject $H_0$?
			\item Write the conclusion in the context of mean ticket-resolution time.
			\item Identify which type of error could occur after making this decision.
			\item Explain what that error would mean in this real-world context.
			\item Explain the implications of the decision for staffing, service targets, and customer support policy.
		\end{enumerate}

		\subsection*{C10}
		\textbf{Decision rule.}
		\[
		\text{Reject }H_0\text{ if }p\text{-value}<\alpha.
		\]
		\[
		\text{Fail to reject }H_0\text{ if }p\text{-value}\geq\alpha.
		\]
		Since $0.0532<0.10$, we reject $H_0$.

		\textbf{Conclusion in context.}
		At the 10\% significance level, there is enough statistical evidence that the true mean ticket-resolution time differs from 3.10 hours.

		\textbf{Relevant error type.}
		Because the decision is to reject $H_0$, the possible error is a Type I error: rejecting $H_0$ even though $H_0$ is actually true.

		\textbf{Real-world implication.}
		If a Type I error occurs, the unit might launch costly process changes or staffing increases based on a perceived shift in mean resolution time that does not actually exist.
