\section{(5) Example Uniform Investment Plan (C1)}
\label{sec:example-uniform-investment-plan-c1}
A variable $V$ follows a uniform model on $[8,16]$. Use the graph and density model to answer the tasks.

\textbf{Questions/Tasks.}
\begin{enumerate}[label=\textbf{\alph*)}]
    \item \TaskHyperref{subsec:example-uniform-investment-plan-c1-a}{\textbf{(2)} Write the density function and support.}
    \item \TaskHyperref{subsec:example-uniform-investment-plan-c1-b}{\textbf{(3)} Compute $P(10\le V\le 14)$ and justify with area.}
\end{enumerate}

\subsection*{a) \textbf{(2)} Write the density function and support.}
\phantomsection
\label{subsec:example-uniform-investment-plan-c1-a}
\[
f_V(v)=\begin{cases}
\dfrac{1}{16-8}=\dfrac18, & 8\le v\le 16,\\
0, & \text{otherwise.}
\end{cases}
\]

\[
8\le v\le 16
\qquad \longrightarrow \qquad
f_V(v)=\frac18
\qquad \longrightarrow \qquad
\int_8^{16} f_V(v)\,dv=1
\]

\secInicio[sec:example-uniform-investment-plan-c1]

\subsection*{b) \textbf{(3)} Compute $P(10\le V\le 14)$ and justify with area.}
\phantomsection
\label{subsec:example-uniform-investment-plan-c1-b}
\begin{center}
    \setlength{\tabcolsep}{3pt}
    \renewcommand{\arraystretch}{1.08}
    \begin{tabular}{@{}p{0.38\textwidth}p{0.58\textwidth}@{}}
        \centering
        \InvestmentThresholdGraph{8}{16}{1/8}{10}{14}{0.18} &
        \[
        \begin{aligned}
        P(10\le V\le 14)
        &=\int_{10}^{14} \frac18\,dv\\
        &=\frac18(14-10)=\frac48=0.5
        \end{aligned}
        \]
        The shaded area represents half of the support interval.
    \end{tabular}
\end{center}

\section{(6) Example Linear Density Model (C2)}
\label{sec:example-linear-density-model-c2}
Suppose $X$ has density $f(x)=kx$ for $0\le x\le 2$.

\textbf{Questions/Tasks.}
\begin{enumerate}[label=\textbf{\alph*)}]
    \item \TaskHyperref{subsec:example-linear-density-model-c2-a}{\textbf{(3)} Find the value of $k$.}
    \item \TaskHyperref{subsec:example-linear-density-model-c2-b}{\textbf{(3)} Compute $P(0.5\le X\le 1.5)$.}
\end{enumerate}

\subsection*{a) \textbf{(3)} Find the value of $k$.}
\phantomsection
\label{subsec:example-linear-density-model-c2-a}
\[
\begin{aligned}
1&=\int_0^2 kx\,dx
= k\left[\frac{x^2}{2}\right]_0^2
=2k
\end{aligned}
\quad\Rightarrow\quad k=\frac12.
\]

\COneGraph{0}{2}{0.5*x}{0.5}{1.5}{0,0.5,1,1.5,2}{1.2}

\secInicio[sec:example-linear-density-model-c2]

\subsection*{b) \textbf{(3)} Compute $P(0.5\le X\le 1.5)$.}
\phantomsection
\label{subsec:example-linear-density-model-c2-b}
\[
P(0.5\le X\le 1.5)=\int_{0.5}^{1.5}\frac12x\,dx
=\left[\frac{x^2}{4}\right]_{0.5}^{1.5}=\frac{1.5^2-0.5^2}{4}=0.5
\]

\secInicio[sec:example-linear-density-model-c2]
