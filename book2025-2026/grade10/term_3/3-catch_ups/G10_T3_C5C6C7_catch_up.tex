\SubsectionBox{1. Standard Normal Table Problems (C5)}\vspace{-0.25cm}

Let \(Z\sim N(0,1)\). Let \(\Phi(z)=P(Z<z)\).

This section trains students to compute standard-normal probabilities using the table.

\textbf{Questions/Tasks.}
\begin{enumerate}[label=\textbf{\alph*)}]
    \item \textbf{(6) Direct and complement-rule probabilities.}
    \begin{enumerate}[label=\textbf{\arabic*)}]
        \item Find \(P(Z<1.24)\).
        \item Find \(P(Z>-0.67)\).
        \item Find \(P(Z>1.18)\).
        \item Find \(P(Z<-0.52)\).
        \item Find \(P(Z<0.36)\).
        \item Find \(P(Z>2.05)\).
        \item Find \(P(Z<-1.41)\).
        \item Find \(P(Z>-1.09)\).
        \item Find \(P(Z>0.73)\).
        \item Find \(P(Z<1.96)\).
        \item Find \(P(Z<-0.88)\).
        \item Find \(P(Z>-2.17)\).
    \end{enumerate}

    \item \textbf{(4) Interval probabilities.}
    \begin{enumerate}[label=\textbf{\arabic*)}]
        \item Find \(P(0.42<Z<1.37)\).
        \item Find \(P(-0.95<Z<1.12)\).
        \item Find \(P(-1.75<Z<-0.40)\).
        \item Find \(P(-2.10<Z<0.65)\).
    \end{enumerate}

    \item \textbf{(4) Tail-union probabilities.}
    \begin{enumerate}[label=\textbf{\arabic*)}]
        \item Find \(P(Z<-1.25\text{ or }Z>1.80)\).
        \item Find \(P(Z<0.45\text{ or }Z>1.70)\).
        \item Find \(P(Z<-2.00\text{ or }Z>-0.60)\).
        \item Find \(P(Z<-0.30\text{ or }Z>2.25)\).
    \end{enumerate}
\end{enumerate}

\SubsectionBox{2. Contextual Normal Distribution Problems (C6)}\vspace{-0.25cm}

Use
\[
Z=\frac{X-\mu}{\sigma}
\]
to convert each contextual value into a standard normal value.

\textbf{Questions/Tasks.}
\begin{enumerate}[label=\textbf{\arabic*)}]
    \item \textbf{(1) Exam scores.} If exam scores follow \(X\sim N(75,10^2)\), find \(P(X<87.4)\) and \(P(X>68.3)\).
    \item \textbf{(1) Delivery times.} If delivery times follow \(X\sim N(40,6^2)\), find \(P(X>47.08)\) and \(P(X<36.88)\).
    \item \textbf{(1) Product weights.} If product weights follow \(X\sim N(500,20^2)\), find \(P(X<507.2)\) and \(P(X>541)\).
    \item \textbf{(1) Commute times.} If commute times follow \(X\sim N(35,8^2)\), find \(P(X<23.72)\) and \(P(X>26.28)\).
    \item \textbf{(1) Customer spending.} If customer spending follows \(X\sim N(120,15^2)\), find \(P(X>130.95)\) and \(P(X<149.4)\).
    \item \textbf{(1) Machine lifetimes.} If machine lifetimes follow \(X\sim N(1000,100^2)\), find \(P(X<912)\) and \(P(X>783)\).
    \item \textbf{(2) Weekly app usage.} If weekly app usage follows \(X\sim N(20,4^2)\), find \(P(21.68<X<25.48)\) and \(P(16.2<X<24.48)\).
    \item \textbf{(2) Crop yields.} If crop yields follow \(X\sim N(60,5^2)\), find \(P(51.25<X<58)\) and \(P(49.5<X<63.25)\).
    \item \textbf{(2) Battery duration.} If battery duration follows \(X\sim N(12,2^2)\), find \(P(X<9.5\text{ or }X>15.6)\) and \(P(X<12.9\text{ or }X>15.4)\).
    \item \textbf{(2) Scholarship scores.} If scholarship scores follow \(X\sim N(300,30^2)\), find \(P(X<240\text{ or }X>282)\) and \(P(X<291\text{ or }X>367.5)\).
\end{enumerate}

\vspace{0.25cm}
\SubsectionBox{3. Inverse Normal Problems (C7)}\vspace{-0.25cm}

Use
\[
Z=\frac{X-\mu}{\sigma}
\qquad \text{and} \qquad
x=\mu+\sigma z
\]
to convert between standard normal values and contextual values.

\textbf{Questions/Tasks.}
\begin{enumerate}[label=\textbf{\arabic*)}]
    \item \textbf{(1) Commute times.} Commute times follow \(X\sim N(38,7^2)\).
    \begin{enumerate}[label=\textbf{\alph*)}]
        \item Find \(x\) such that \(P(X<x)=0.80\).
        \item Find \(x\) such that \(P(X>x)=0.10\).
    \end{enumerate}
    \item \textbf{(1) Exam scores.} Exam scores follow \(X\sim N(72,9^2)\).
    \begin{enumerate}[label=\textbf{\alph*)}]
        \item Find \(x\) such that \(P(X>x)=0.80\).
        \item Find \(x\) such that \(P(X<x)=0.25\).
    \end{enumerate}
    \item \textbf{(1) Delivery times.} Delivery times follow \(X\sim N(56,5^2)\).
    \begin{enumerate}[label=\textbf{\alph*)}]
        \item Find \(x\) such that \(P(X<x)=0.60\).
        \item Find \(x\) such that \(P(X>x)=0.30\).
    \end{enumerate}
    \item \textbf{(1) Product weights.} Product weights follow \(X\sim N(500,20^2)\).
    \begin{enumerate}[label=\textbf{\alph*)}]
        \item Find \(x\) such that \(P(X<x)=0.10\).
        \item Find \(x\) such that \(P(X>x)=0.75\).
    \end{enumerate}
    \item \textbf{(1) Customer spending.} Customer spending follows \(X\sim N(120,15^2)\).
    \begin{enumerate}[label=\textbf{\alph*)}]
        \item Find \(x\) such that \(P(X>x)=0.05\).
        \item Find \(x\) such that \(P(X<x)=0.70\).
    \end{enumerate}
    \item \textbf{(1) App usage.} App usage follows \(X\sim N(20,4^2)\).
    \begin{enumerate}[label=\textbf{\alph*)}]
        \item Find \(x\) such that \(P(X<x)=0.40\).
        \item Find \(x\) such that \(P(X>x)=0.90\).
    \end{enumerate}
    \item \textbf{(2) Battery life.} Battery life follows \(X\sim N(20,2.5^2)\).
    \begin{enumerate}[label=\textbf{\alph*)}]
        \item Find \([a,b]\) such that \(P(a<X<b)=0.90\), with a central interval.
        \item Find \([a,b]\) such that \(P(X<a)=0.60\) and \(P(X<b)=0.90\).
    \end{enumerate}
    \item \textbf{(2) Machine lifetimes.} Machine lifetimes follow \(X\sim N(1000,100^2)\).
    \begin{enumerate}[label=\textbf{\alph*)}]
        \item Find \([a,b]\) such that \(P(X<a)=0.10\) and \(P(X<b)=0.40\).
        \item Find \([a,b]\) such that \(P(X<a)=0.25\) and \(P(X<b)=0.75\).
    \end{enumerate}
    \item \textbf{(2) Metal rod diameters.} Metal rod diameters follow \(X\sim N(10.00,0.08^2)\).
    \begin{enumerate}[label=\textbf{\alph*)}]
        \item Find \(a\) and \(b\) such that \(P(X<a\text{ or }X>b)=0.10\), with symmetric tails.
        \item Find \(a\) and \(b\) such that \(P(X<a\text{ or }X>b)=0.70\), with \(P(X<a)=0.60\) and \(P(X>b)=0.10\).
    \end{enumerate}
    \item \textbf{(2) Crop yields.} Crop yields follow \(X\sim N(60,5^2)\).
    \begin{enumerate}[label=\textbf{\alph*)}]
        \item Find \(a\) and \(b\) such that \(P(X<a\text{ or }X>b)=0.70\), with \(P(X<a)=0.10\) and \(P(X>b)=0.60\).
        \item Find \(a\) and \(b\) such that \(P(X<a\text{ or }X>b)=0.45\), with \(P(X<a)=0.20\) and \(P(X>b)=0.25\).
    \end{enumerate}
\end{enumerate}
