\section*{Evaluated criterion}
\textbf{C4.} Interprets the concept of expected value, variance, and standard deviation of a probability function.

\textbf{Uniform Distribution}

For a uniform distribution $U[a,b]$,
\[
\mu=\frac{a+b}{2}
\qquad \qquad
\sigma=\frac{b-a}{\sqrt{12}}
\]

\section{(1) Uniform: match graphs with their means}
Four uniform densities are shown. The distributions have different spreads, but the task is only to match each graph with its mean.

Match each graph with one mean option. Some options will not be used.
\begin{itemize}
\item Mean options: $\{-2,\ 0,\ 2,\ 5,\ 8,\ 11\}$
\end{itemize}
\UniformFourGraph
{13cm}{6.5cm}{-5}{16}{0.55}
{-4,0}
{-1,5}
{3,7}
{7,15}

\subsection*{Solution}
\textbf{Final match.}
\begin{itemize}
\item Graph A $\to \mu=-2$
\item Graph B $\to \mu=2$
\item Graph C $\to \mu=5$
\item Graph D $\to \mu=11$
\end{itemize}

\textbf{Triangular Distribution}

For a triangular distribution with left endpoint $a$, mode $c$, and right endpoint $b$,
\[
\mu=\frac{a+b+c}{3}
\qquad \qquad
\sigma=\sqrt{\frac{a^2+b^2+c^2-ab-ac-bc}{18}}
\]

\section{(1) Triangular: same spread, different mean}
Four triangular densities are shown. All four distributions have the same standard deviation, so only the expected value changes.
Match each graph with one mean option. Some options will not be used.
\begin{itemize}
\item Mean options: $\left\{-\frac{10}{3},\ 0,\ \frac{4}{3},\ \frac{11}{3},\ 6,\ \frac{25}{3},\ 10\right\}$
\end{itemize}

{-5,-4,-1}

{2,3,6}
{6,9,10}

\subsection*{Solution}

\begin{itemize}

\item Graph B $\to \mu=\frac{4}{3}$

\item Graph D $\to \mu=\frac{25}{3}$

\textbf{Normal Distribution}

For a normal distribution $N(\mu,\sigma^2)$,
\[
E[X]=\mu
\qquad \qquad
\mathrm{SD}(X)=\sigma
\]

\section{(1) Normal I: match two graphs with their mean and standard deviation}
Two normal densities are shown. Both the expected value and the standard deviation change from one graph to the other.
Match each graph with one mean option and one standard deviation option. Some options will not be used.
\begin{itemize}
\item Mean options: $\{2.0,\ 2.8,\ 4.1,\ 4.6,\ 5.2\}$
\item Standard deviation options: $\{1.0,\ 1.3\}$
\end{itemize}
\NormalTwoGraph
{12cm}{6.5cm}{0}{7}{0.65}
{2.8,1.0}
{4.1,1.3}

\subsection*{Solution}
\textbf{Final match.}
\begin{itemize}
\item Graph A $\to \mu=2.8$, $\sigma=1.0$
\item Graph B $\to \mu=4.1$, $\sigma=1.3$
\end{itemize}
