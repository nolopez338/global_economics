	\section*{Evaluated criteria}
	\begin{itemize}
		\item C5. Uses the normal standard distribution and its tables to calculate probabilities in contextual situations.
	\end{itemize}
	
	
	\noindent\textbf{Meaning of the notation.}
	The expression \(P(Z<x)\) means the probability that the standard normal random variable \(Z\) is less than the value \(x\). In the graph of the standard normal distribution, this corresponds to the area under the curve to the left of \(x\).
	
	For example $P(Z<0.85)$ means the area under the standard normal curve to the left of \(x=0.85\). Since the table gives values of \(\Phi(z)=P(Z<x)\), this value can be read directly from the table as
	\[
	P(Z<0.85)=\Phi(0.85)=0.8023.
	\]
	Here are five examples of left-tail probabilities read directly from the table.
	\begin{center}
		\begin{tabular}{c c c}
			\toprule
			\(x\) & \(P(Z<x)=\Phi(x)\) & Value \\
			\midrule
			\(0.25\) & \(P(Z<0.25)=\Phi(0.25)\) & \(0.5987\) \\
			\(0.70\) & \(P(Z<0.70)=\Phi(0.70)\) & \(0.7580\) \\
			\(0.85\) & \(P(Z<0.85)=\Phi(0.85)\) & \(0.8023\) \\
			\(1.20\) & \(P(Z<1.20)=\Phi(1.20)\) & \(0.8849\) \\
			\(1.65\) & \(P(Z<1.65)=\Phi(1.65)\) & \(0.9505\) \\
			\bottomrule
		\end{tabular}
	\end{center}
	
	If the value is negative, such as $	P(Z<-1.20)$, the expression means the area to the left of \(z=-1.20\). Since the table only shows positive \(z\)-values, symmetry and the complement rule can be used:
	\[
	P(Z<-1.20)=P(Z>1.20)=1-P(Z<1.20)=1-\Phi(1.20).
	\]
	Since \(\Phi(1.20)=0.8849\),
	\[
	P(Z<-1.20)=1-0.8849=0.1151.
	\]
	
	The same idea can be summarized as follows.
	
	\begin{center}
		\begin{tabular}{c c c c c}
			\toprule
			\(X\) & \(-X\) & \(P(Z<-X)\) & \(1-P(Z<-X)\) & \(P(Z<X)\) \\
			\midrule
			\(-0.25\) & \(0.25\) & \(0.5987\) & \(0.4013\) & \(0.4013\) \\
			\(-0.70\) & \(0.70\) & \(0.7580\) & \(0.2420\) & \(0.2420\) \\
			\(-0.85\) & \(0.85\) & \(0.8023\) & \(0.1977\) & \(0.1977\) \\
			\(-1.20\) & \(1.20\) & \(0.8849\) & \(0.1151\) & \(0.1151\) \\
			\(-1.65\) & \(1.65\) & \(0.9505\) & \(0.0495\) & \(0.0495\) \\
			\bottomrule
		\end{tabular}
	\end{center}

	\section{Problem 1: Left Tail with Positive and Negative Bounds}
	\textbf{Problem.} Let \(Z\sim N(0,1)\).
	
	\textbf{Questions/Tasks.}
	\begin{enumerate}[label=\textbf{\alph*)}]
		\item Find \(P(Z<0.85)\) using the table.
		\item Find \(P(Z<-1.20)\) using the table.
	\end{enumerate}
	
	\subsection*{Solution}
	
	\textbf{a)}
	From the table, \(\Phi(0.85)=0.8023\), so
	\[
	P(Z<0.85)=0.8023.
	\]
	
	\StandardNormalLeftTailGraph{0.85}{z=0.85}{P(Z<0.85)=0.8023}
	
	\textbf{b)}
	The goal is to find the left-tail probability
	\[
	P(Z<-1.20).
	\]
	
	\StandardNormalLeftTailGraph{-1.20}{z=-1.20}{P(Z<-1.20)}
	
	\[	\text{By Symmetry: } P(Z<-1.20)=P(Z>1.20).	\]
	
	\StandardNormalSymmetryGraph{1.20}{P(Z<-1.20)}{P(Z>1.20)}
	
	Now use the table value
	\[
	P(Z<1.20)=\Phi(1.20)=0.8849.
	\]
	Using the complement rule
	\[
	P(Z>1.20)=1-\Phi(1.20)=1-0.8849=0.1151.
	\]
	
	\StandardNormalComplementMirrorGraph
	{1.20}
	{P(Z<1.20)=0.8849}
	{P(Z>1.20)=0.1151}
	
	Therefore,
	\[
	P(Z<-1.20)=P(Z>1.20)=0.1151.
	\]
	
	\textbf{Interpretation.} About 80.2\% of values are below 0.85, while 11.5\% are below \(-1.20\).

\section{Problem 4: Central Band with Positive Bounds}
\textbf{Problem.} Let \(Z\sim N(0,1)\).

\textbf{Questions/Tasks.} Compute \(P(0.35<Z<1.40)\).

\subsection*{C5}
The target central-band probability is
\[
P(0.35<Z<1.40).
\]
\StandardNormalCentralBandGraph{0.35}{1.40}{z=0.35}{z=1.40}{P(0.35<Z<1.40)}
Using \(P(a<Z<b)=\Phi(b)-\Phi(a)\), first read the cumulative areas:
\[
P(Z<1.40)=\Phi(1.40)=0.9192
\]
\[
P(Z<0.35)=\Phi(0.35)=0.6368.
\]
The central band is the difference of these cumulative areas:
\[
P(0.35<Z<1.40)=\Phi(1.40)-\Phi(0.35)=0.9192-0.6368=0.2824.
\]
\textbf{Interpretation.} About 28.2\% of values are between 0.35 and 1.40.

\StandardNormalTwoTailGraph{0.55}{1.65}{z=0.55}{z=1.65}{P(Z<0.55)}{P(Z>1.65)}

\StandardNormalComplementMirrorGraph
{1.65}

{P(Z>1.65)=0.0495}

P(Z>1.65)=1-\Phi(1.65)=1-0.9505=0.0495.
