% \NormalMeanSDGraph: 6 parameters
% #1 = variable symbol
% #2 = mean mu
% #3 = standard deviation sigma
% #4 = xmin
% #5 = xmax
% #6 = ymax
\newcommand{\NormalMeanSDGraph}[6]{%
	\begin{center}
		\begin{tikzpicture}
			\pgfmathsetmacro{\Mu}{#2}
			\pgfmathsetmacro{\Sigma}{#3}
			\pgfmathsetmacro{\LeftSD}{\Mu-\Sigma}

			\pgfmathsetmacro{\RightTwoSD}{\Mu+2*\Sigma}
			\pgfmathsetmacro{\PeakHeight}{1/(\Sigma*sqrt(2*pi))}
			\pgfmathsetmacro{\OneSDHeight}{1/(\Sigma*sqrt(2*pi))*exp(-0.5)}
			\pgfmathsetmacro{\TwoSDHeight}{1/(\Sigma*sqrt(2*pi))*exp(-2)}
			
			\begin{axis}[
				width=10cm,
				height=5.5cm,
				axis lines=left,
				xlabel={$#1$},
				xlabel style={at={(axis description cs:1.02,-0.06)}, anchor=north west},
				ylabel={$f(#1)$},
				xmin=#4,
				xmax=#5,
				ymin=0,
				ymax=#6,
				grid=both,
				xtick={\LeftTwoSD,\LeftSD,\Mu,\RightSD,\RightTwoSD},
				xticklabels={
					$\pgfmathprintnumber[precision=2]{\LeftTwoSD}$,
					,
					$E[#1]=\pgfmathprintnumber[precision=2]{\Mu}$,
					,
					$\pgfmathprintnumber[precision=2]{\RightTwoSD}$
				},
				ytick=\empty
				]
				
				\addplot[
				draw=none,
				fill=red,
				fill opacity=0.5,
				domain=\LeftSD:\RightSD,
				samples=160
				]
				{1/(\Sigma*sqrt(2*pi))*exp(-((x-\Mu)^2)/(2*\Sigma^2))}
				\closedcycle;
				
				\addplot[
				ultra thick,
				blue,
				domain=#4:#5,
				samples=240
				]
				{1/(\Sigma*sqrt(2*pi))*exp(-((x-\Mu)^2)/(2*\Sigma^2))};
				
				\addplot[ultra thick, orange]
				coordinates {(\Mu,0) (\Mu,\PeakHeight)};
				
				\addplot[red, densely dotted, thick]
				coordinates {(\LeftSD,0) (\LeftSD,\OneSDHeight)};
				
				\addplot[red, densely dotted, thick]
				coordinates {(\RightSD,0) (\RightSD,\OneSDHeight)};
				
				\addplot[densely dotted, thick]
				coordinates {(\LeftTwoSD,0) (\LeftTwoSD,\TwoSDHeight)};
				
				\addplot[densely dotted, thick]
				coordinates {(\RightTwoSD,0) (\RightTwoSD,\TwoSDHeight)};
				
				\node[font=\small, above] at (axis cs:\LeftSD,\OneSDHeight) {$E[#1]-\sigma$};
				\node[font=\small, above] at (axis cs:\RightSD,\OneSDHeight) {$E[#1]+\sigma$};
				
			\end{axis}
		\end{tikzpicture}

\section{Uniform Distribution}

\textbf{Problem.}
A student's daily streaming time \(X\) (minutes) is modeled by
\[
X\sim \text{Uniform}(2,14).
\]

\textbf{Questions/Tasks.}
\begin{enumerate}[label=\textbf{\alph*)}]
	\item Find \(E[X]\).
	\item Find \(\mathrm{Var}(X)\) and the standard deviation.
	\item Write the PDF of \(X\).
	\item Draw the graph of the PDF, indicate \(E[X]\), and draw an interval of one standard deviation around \(E[X]\).
	\item Calculate the probability of the highlighted interval.
	\item Compare the probabilities of intervals of the same length around \(4\) and around \(12\). What is the difference between these probabilities?
\end{enumerate}

\subsection*{C4}
For \(X\sim\text{Uniform}(a,b)\):
\[
E[X]=\frac{a+b}{2},\qquad \mathrm{Var}(X)=\frac{(b-a)^2}{12}.
\]

\begin{enumerate}[label=\textbf{\alph*)}]
	\item With \(a=2\) and \(b=14\),
	\[
	E[X]=\frac{2+14}{2}=8.
	\]
	
	\item Using the same endpoints,
	\[
	\mathrm{Var}(X)=\frac{(14-2)^2}{12}=\frac{144}{12}=12,
	\]
	so
	\[
	\sigma=\sqrt{12}=2\sqrt{3}\approx 3.46.
	\]
	
	\item Since \(X\sim\text{Uniform}(2,14)\), the PDF is constant on the support \([2,14]\). Its height is
	\[
	\frac{1}{14-2}=\frac{1}{12}.
	\]
	Therefore,
	\[
	f(x)=
	\begin{cases}
		\frac{1}{12}, & 2\le x\le 14,\\
		0, & \text{otherwise}.
	\end{cases}
	\]
	
	\item From parts a and b, the mean is \(E[X]=8\) and the standard deviation is \(\sigma=2\sqrt{3}\). Therefore, one standard deviation around the mean is
	\[
	\left[E[X]-\sigma,\;E[X]+\sigma\right]
	=
	\left[8-2\sqrt{3},\;8+2\sqrt{3}\right]
	\approx [4.54,\;11.46].
	\]
	
	\UniformMeanSDGraph{X}{0}{16}{0.13}{2}{4.54}{8}{11.46}{14}
	
	\item The highlighted interval is the interval found in part d:
	\[
	\left[8-2\sqrt{3},\;8+2\sqrt{3}\right].
	\]
	So the probability of the highlighted interval is
	\[
	P\left(8-2\sqrt{3}\le X\le 8+2\sqrt{3}\right).
	\]
	
	Using the PDF from part c, the height of the rectangle is \(\frac{1}{12}\). The base of the rectangle is the length of the highlighted interval:
	\[
	\left(8+2\sqrt{3}\right)-\left(8-2\sqrt{3}\right)=4\sqrt{3}.
	\]
	
	Therefore,
	\[
	P\left(8-2\sqrt{3}\le X\le 8+2\sqrt{3}\right)
	=
	(\text{base})(\text{height})
	=
	(4\sqrt{3})\left(\frac{1}{12}\right).
	\]
	Thus,
	\[
	P\left(8-2\sqrt{3}\le X\le 8+2\sqrt{3}\right)
	=
	\frac{4\sqrt{3}}{12}
	=
	\frac{\sqrt{3}}{3}
	\approx 0.577.
	\]
	
	
	\item Since this is a continuous model, the probability of any exact single value is \(0\). So we compare intervals of the same length around \(4\) and around \(12\).
	
	Using the PDF from part c, any interval inside \([2,14]\) with length \(L\) has probability
	\[
	P(\text{interval})=L\left(\frac{1}{12}\right)=\frac{L}{12}.
	\]
	
	The following graph shows the \(8\) intervals. Intervals of length \(1\) are the darkest, and intervals of length \(4\) are the lightest.
	
	\UniformIntervalComparisonGraph{X}{0}{16}{0.13}{2}{14}{4}{12}

			\hline
			\(1\) & \([3.5,4.5]\) & \(\frac{1}{12}\) & \([11.5,12.5]\) & \(\frac{1}{12}\) & \(\frac{0}{12}\) \\
			\(2\) & \([3,5]\) & \(\frac{2}{12}\) & \([11,13]\) & \(\frac{2}{12}\) & \(\frac{0}{12}\) \\
			\(3\) & \([2.5,5.5]\) & \(\frac{3}{12}\) & \([10.5,13.5]\) & \(\frac{3}{12}\) & \(\frac{0}{12}\) \\

	In each row, the two intervals have the same length and lie completely inside the support \([2,14]\). Since the distribution is uniform, their probabilities are equal.
	
	The difference between the probabilities around \(4\) and around \(12\) is always
	\[
	0.
	\]
	
\end{enumerate}

\section{Triangular Distribution 1}

\textbf{Problem.}
A bus wait time \(W\) (minutes) is modeled by a triangular distribution with minimum \(a=2\), mode \(c=2\), and maximum \(b=14\):
\[
W\sim \text{Triangular}(2,2,14).
\]

\textbf{Questions/Tasks.}
\begin{enumerate}[label=\textbf{\alph*)}]
	\item Find \(E[W]\).
	\item Find \(\mathrm{Var}(W)\) and the standard deviation.
	\item Write the PDF of \(W\).
	\item Draw the graph of the PDF, indicate \(E[W]\), and draw an interval of one standard deviation around \(E[W]\).
	\item Calculate the probability of the highlighted interval.
	\item Compare the probabilities of intervals of the same length around \(4\) and around \(12\). What is the difference between these probabilities?
\end{enumerate}

\subsection*{C4}
For \(\text{Triangular}(a,c,b)\):
\[
E[W]=\frac{a+b+c}{3},\qquad
\mathrm{Var}(W)=\frac{a^2+b^2+c^2-ab-ac-bc}{18}.
\]

\begin{enumerate}[label=\textbf{\alph*)}]
	\item With \(a=2\), \(c=2\), and \(b=14\),
	\[
	E[W]=\frac{2+14+2}{3}=\frac{18}{3}=6.
	\]
	
	\item Using the same parameters,
	\[
	\mathrm{Var}(W)=\frac{2^2+14^2+2^2-(2)(14)-(2)(2)-(14)(2)}{18}
	=
	\frac{4+196+4-28-4-28}{18}
	=
	\frac{144}{18}=8.
	\]
	Therefore,
	\[
	\sigma=\sqrt{8}=2\sqrt{2}\approx 2.83.
	\]
	
		\item Since the mode is at the left endpoint, the PDF is a decreasing straight line from \(w=2\) to \(w=14\). The total area under the PDF must be \(1\). Therefore, the maximum height is found from the area of the full triangle:
	\[
	A_{\triangle} = \frac{1}{2}(\text{base})(\text{height})=1
	\qquad , \qquad
	\text{base}=14-2=12.
	\]
	So
	\[
	A_{\triangle}=\frac{1}{2}(12)(\text{height})=1
	\qquad \longrightarrow \qquad

\section{Normal Distribution}

\textbf{Problem.}
Quiz scores \(S\) are modeled by
\[
S\sim N(6,4^2).
\]

\textbf{Questions/Tasks.}
\begin{enumerate}[label=\textbf{\alph*)}]
	\item State \(E[S]\).
	\item State \(\mathrm{Var}(S)\) and the standard deviation.
	\item Write the PDF of \(S\).
	\item Give the intervals \([\mu-\sigma,\mu+\sigma]\) and \([\mu-2\sigma,\mu+2\sigma]\), draw the graph of the PDF, indicate \(E[S]\), and draw the interval of one standard deviation around \(E[S]\).
	\item Find the probability of the highlighted interval \([\mu-\sigma,\mu+\sigma]\).
	\item Compare the probabilities of intervals of the same length around \(6\) and around \(-2\). What is the difference between these probabilities?
\end{enumerate}

\subsection*{C4}
For \(N(\mu,\sigma^2)\),
\[
E[S]=\mu,
\qquad
\mathrm{Var}(S)=\sigma^2.
\]

\begin{enumerate}[label=\textbf{\alph*)}]
	\item Since
	\[
	S\sim N(6,4^2),
	\]
	the mean is
	\[
	E[S]=\mu=6.
	\]
	
	\item Using the same parameters,
	\[
	\mathrm{Var}(S)=\sigma^2=4^2=16.
	\]
	Therefore,
	\[
	\sigma=4.
	\]
	
	\item For a normal distribution, the PDF is
	\[
	f(s)=\frac{1}{\sigma\sqrt{2\pi}}e^{-\frac{(s-\mu)^2}{2\sigma^2}}.
	\]
	
	Using \(\mu=6\) and \(\sigma=4\), the PDF is
	\[
	f(s)=\frac{1}{4\sqrt{2\pi}}e^{-\frac{(s-6)^2}{2(4)^2}}.
	\]
	Equivalently,
	\[
	f(s)=\frac{1}{4\sqrt{2\pi}}e^{-\frac{(s-6)^2}{32}},
	\qquad -\infty<s<\infty.
	\]
	
	\item From parts a and b, the mean is \(E[S]=6\) and the standard deviation is \(\sigma=4\). Therefore,
	\[
	[\mu-\sigma,\mu+\sigma]
	=
	[6-4,\;6+4]
	=
	[2,10].
	\]
	Also,
	\[
	[\mu-2\sigma,\mu+2\sigma]
	=
	[6-2(4),\;6+2(4)]
	=
	[-2,14].
	\]
	
	\NormalMeanSDGraph{S}{6}{4}{-6}{18}{0.11}
	
	\item The highlighted interval is the interval found in part d:
	\[
	[2,10].
	\]
	The probability of this interval is represented by the area under the normal curve above the highlighted interval:
	\[
	P(2\le S\le 10).
	\]
	
	To use the standard normal distribution, standardize with
	\[
	Z=\frac{S-\mu}{\sigma}
	=
	\frac{S-6}{4}.
	\]
	Then
	\[
	P(2\le S\le 10)
	=
	P\left(\frac{2-6}{4}\le Z\le \frac{10-6}{4}\right).
	\]
	So
	\[
	P(2\le S\le 10)
	=
	P(-1\le Z\le 1).
	\]
	
	Using the standard normal distribution,
	\[
	P(-1\le Z\le 1)\approx 0.6827.
	\]
	Therefore,
	\[
	P(2\le S\le 10)\approx 0.6827.
	\]
	
		\item Since this is a continuous model, the probability of any exact single value is \(0\). So we compare intervals of the same length around \(4\) and around \(12\).
	
	Using the standardization formula from part e,
	\[
	Z=\frac{S-6}{4},
	\]
	each interval probability can be found by converting the endpoints to \(z\)-scores and using the standard normal distribution.
	
	The following graph shows the \(8\) intervals. Intervals of length \(1\) are the darkest, and intervals of length \(4\) are the lightest.
	
	\NormalIntervalComparisonGraph{S}{6}{4}{-6}{18}{0.11}{4}{12}
	
	\begin{center}

			\hline
			Interval length & Around \(4\) & Probability & Around \(12\) & Probability & Difference \\
			\hline
			\(1\) & \([3.5,4.5]\) & \(0.0878\) & \([11.5,12.5]\) & \(0.0325\) & \(0.0553\) \\
			\(2\) & \([3,5]\) & \(0.1747\) & \([11,13]\) & \(0.0656\) & \(0.1091\) \\

		\end{tabular}
	\end{center}
	
	The intervals around \(4\) have larger probability than the intervals around \(12\). This happens because \(4\) is closer to the mean \(6\), where the normal density is highest. The value \(12\) is farther from the mean, so intervals around \(12\) have less area under the curve.
	
\end{enumerate}

\section{Uniform with Parameters}

\textbf{Problem.}
A delivery time \(U\) (minutes) for a local route is modeled by
\[
U\sim \text{Uniform}(h-2w,\,h+w),\qquad w>0.
\]

\textbf{Questions/Tasks.}
\begin{enumerate}[label=\textbf{\alph*)}]
  \item Derive \(E[U]\), \(\mathrm{Var}(U)\), and \(\sigma_U\) in terms of \(h\) and \(w\).
  \item Identify which parameter controls the center and which controls the interval width.
  \item Compare Case A: \((h,w)=(30,4)\) and Case B: \((h,w)=(35,4)\). Does the distribution shift, widen, or both?
  \item Compare Case C: \((h,w)=(30,4)\) and Case D: \((h,w)=(30,7)\). How do center, width, and spread change?
  \item If both \(h\) and \(w\) increase by 2 at the same time, explain whether the model shifts, widens, or both.
\end{enumerate}

\subsection*{C4}
For \(U\sim\text{Uniform}(a,b)\), \(E[U]=\frac{a+b}{2}\), \(\mathrm{Var}(U)=\frac{(b-a)^2}{12}\).
Here \(a=h-2w\), \(b=h+w\), so
\[
E[U]=\frac{(h-2w)+(h+w)}{2}=h-\frac{w}{2},
\]
\[
\mathrm{Var}(U)=\frac{\big((h+w)-(h-2w)\big)^2}{12}=\frac{(3w)^2}{12}=\frac{3w^2}{4},
\]
\[
\sigma_U=\sqrt{\frac{3w^2}{4}}=\frac{\sqrt{3}}{2}\,w \quad (w>0).
\]
The center \(h-\frac{w}{2}\) depends on both \(h\) and \(w\), while the width \(b-a=3w\) depends only on \(w\).
Case A \(\to\) B keeps \(w\) fixed, so width and spread stay the same, and the model shifts right by 5.
Case C \(\to\) D keeps \(h\) fixed and increases \(w\), so width grows from \(12\) to \(21\), spread increases, and the center moves from \(28\) to \(26.5\) (left).
If \(h\) and \(w\) both increase by 2, then width increases by \(3(2)=6\) (widens) and
\[
E[U]'-(E[U])=\left(h+2-\frac{w+2}{2}\right)-\left(h-\frac{w}{2}\right)=1,
\]
so it also shifts right by 1.

\section{Triangular with Parameters}

\textbf{Problem.}
A machine-cycle time \(T\) (seconds) is modeled by
\[
T\sim \text{Triangular}(k,\,k+r,\,k+3r),\qquad r>0.
\]

\textbf{Questions/Tasks.}
\begin{enumerate}[label=\textbf{\alph*)}]
  \item Derive \(E[T]\), \(\mathrm{Var}(T)\), and \(\sigma_T\) in terms of \(k\) and \(r\).
  \item State the validity conditions on \(k\) and \(r\), and explain why they are sufficient.
  \item How do \(k\) and \(r\) separately affect center and spread?
  \item Compare shape with the symmetric triangular model having the same endpoints \(k\) and \(k+3r\). Is this model symmetric, left-skewed, or right-skewed?
  \item For fixed \(k\), compare \(r=2\) and \(r=5\): discuss changes in mean, variance, and skewness direction.
\end{enumerate}

\subsection*{C4}
For \(\text{Triangular}(a,c,b)\):
\[
E[T]=\frac{a+b+c}{3},\qquad
\mathrm{Var}(T)=\frac{a^2+b^2+c^2-ab-ac-bc}{18}.
\]
With \(a=k\), \(c=k+r\), \(b=k+3r\):
\[
E[T]=\frac{k+(k+r)+(k+3r)}{3}=k+\frac{4r}{3}.
\]
For variance,
\[
\mathrm{Var}(T)=\frac{k^2+(k+3r)^2+(k+r)^2-k(k+3r)-k(k+r)-(k+3r)(k+r)}{18}
=\frac{7r^2}{18}.
\]
Thus
\[
\sigma_T=\sqrt{\frac{7r^2}{18}}=r\sqrt{\frac{7}{18}}.
\]
Validity requires \(a\le c\le b\), i.e.
\[
k\le k+r\le k+3r,
\]
which holds whenever \(r\ge 0\); with \(r>0\), spread is nonzero, so the model is valid and non-degenerate for any real \(k\).
Parameter \(k\) shifts location only. Parameter \(r\) increases both mean offset (\(\frac{4r}{3}\)) and spread (\(\propto r^2\)).
For endpoints \(k\) and \(k+3r\), symmetry would need mode at midpoint \(k+\frac{3r}{2}\), but here mode is \(k+r\), left of midpoint; therefore the model is right-skewed.
Comparing \(r=2\) and \(r=5\) (fixed \(k\)): mean increases by \(\frac{4}{3}(5-2)=4\), variance scales from \(\frac{28}{18}\) to \(\frac{175}{18}\), and skewness direction stays right-skewed.

\[

\]

\begin{enumerate}[label=\textbf{\alph*)}]
  \item Derive \(E[N]\), \(\mathrm{Var}(N)\), and \(\sigma_N\) in terms of \(p\) and \(q\).
  \item Explain the role of \(p\) and \(q\): which controls location and which controls spread?
  \item Compare Case A: \((p,q)=(0,1)\) and Case B: \((p,q)=(-2,1)\). What changes in center, concentration around the mean, and tail likelihood?
  \item Compare Case C: \((p,q)=(0,1)\) and Case D: \((p,q)=(0,4)\). What changes in center, spread, concentration near the mean, and likelihood of values far from the mean?
  \item Between models \(M_1:(p,q)=(1,0)\) and \(M_2:(p,q)=(-1,3)\), which model is more likely to produce values at least 15 GB above its own mean? Justify conceptually.
\end{enumerate}

\subsection*{C4}
For \(N(\mu,\sigma^2)\), \(E[N]=\mu\), \(\mathrm{Var}(N)=\sigma^2\), \(\sigma>0\).
So
\[
E[N]=50+3p,\qquad
\mathrm{Var}(N)=(2+q)^2,\qquad
\sigma_N=2+q \quad (q>-2).
\]
Parameter \(p\) changes only location (horizontal shift). Parameter \(q\) changes spread: larger \(q\) gives larger \(\sigma\), wider curve, lower peak, and heavier tails.
Case A \(\to\) B keeps \(q=1\), so \(\sigma=3\) is unchanged: concentration and tail behavior relative to each mean stay the same, but mean shifts from \(50\) to \(44\) (left by 6).
Case C \(\to\) D keeps \(p=0\), so mean stays 50, but \(\sigma\) increases from \(3\) to \(6\): values are less concentrated near 50 and far-from-mean values become more likely.
For \(M_1\), \(\sigma=2\); for \(M_2\), \(\sigma=5\). A value 15 GB above its own mean is \(15/\sigma\) standard deviations above mean: \(7.5\sigma\) for \(M_1\) versus \(3\sigma\) for \(M_2\). The \(3\sigma\) event is much less extreme, so \(M_2\) is more likely.
