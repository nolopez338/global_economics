	% --- SECTION 1: DIRECT PAYOFF MODELS ---
	\section{Direct Payoff Modeling} \label{sec:direct}
	
	\subsection{Local Café Investment (Problem 1)}
	A café is deciding whether to launch artisan desserts. Revenue is generated from daily sales, and costs cover ingredients and labor.
	\begin{itemize}
		\item \textbf{Launch}: Revenue 60 (High traffic) / 20 (Low); Cost 18 (High) / 12 (Low).
		\item \textbf{Menu}: Revenue 40 (High) / 30 (Low); Cost 12 (High) / 10 (Low).
	\end{itemize}
	\textbf{a) (C1) Interpretation}: Alternatives: Launch vs. Menu. States: High (0.60), Low (0.40). Events: Actual foot traffic.
	\textbf{b) (C2) Payoff Table}: Compute $P = R - C$.
	\begin{center} \begin{tabular}{lll} \toprule State & Launch & Menu \\ \midrule High & $60-18=42$ & $40-12=28$ \\ Low & $20-12=8$ & $30-10=20$ \\ \bottomrule \end{tabular} \end{center}
	
	\subsection{Green Energy Production Mix (Problem 2)}
	An energy firm selects a generation mix. Revenue comes from MWh sales; costs are operational.
	\begin{itemize}
		\item \textbf{Wind}: Revenue 120 (Windy), 60 (Sunny), 40 (Cloudy); Costs 30, 15, 10.
		\item \textbf{Solar}: Revenue 50, 130, 35; Costs 15, 35, 10.
		\item \textbf{Hybrid}: Revenue 90, 85, 80; Costs 20, 20, 20.
	\end{itemize}
	\textbf{b) (C2) Payoff Table}: Wind Windy: $120-30=90$; Solar Sunny: $130-35=95$.
	
	% --- SECTION 2: COMPLEX DIRECT AND ALT-VARIABLE MODELS ---
	\section{Advanced Direct and Alternative-Variable Models}
	
	\subsection{Global Shipping Network Design (Problem 3)}
	A firm chooses between Centralized vs. Regional hubs across four trade states.
	\begin{itemize}
		\item \textbf{Centralized}: Revenue 300, 200, 100, 50; Costs 80, 60, 60, 80.
		\item \textbf{Regional}: Revenue 250, 210, 150, 100; Costs 70, 50, 40, 40.
	\end{itemize}
	\textbf{b) (C2) Payoff Table}: Centralized Boom: $300-80=220$; Severe Disruptions: $50-80=-30$.
	
	\subsection{Community Bakery Lunch Boxes (Problem 4)}
	Payoff is based on boxes sold times profit margin.
	\begin{itemize}
		\item \textbf{Standard}: 900 (High demand), 550 (Low). Revenue \$0.10/box, Cost \$0.06/box.
		\item \textbf{Premium}: 650 (High), 250 (Low). Revenue \$0.15/box, Cost \$0.08/box.
	\end{itemize}
	\textbf{b) (C2) Payoff Table}: Standard High: $900(0.10-0.06)=36$.
