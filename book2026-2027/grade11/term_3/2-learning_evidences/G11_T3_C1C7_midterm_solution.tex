\begin{enumerate}

	\begin{itemize}

	\end{itemize}
	
	\item \textbf{C2.} Write hypotheses for both benchmark directions in both settings.
	\begin{itemize}
		\item Mean-count setting with Group A as benchmark.
		\item Mean-count setting with Group B as benchmark.
		\item Proportion setting with Group A as benchmark.
		\item Proportion setting with Group B as benchmark.
		\item To pass C2, each \(H_0\) and \(H_a\) must use the correct parameter, \(\mu\) for means and \(p\) for proportions, such that the initial tests will clearly fail to reject the null hypothesis.
	\end{itemize}
	
	\item \textbf{C3.} Identify and justify the tail direction for both benchmark directions in both settings.
	\begin{itemize}
		\item To pass C3, each setup case must be classified as right-tailed, left-tailed, or two-tailed.
		\item Each classification must be justified using the sign in the alternative hypothesis.
		\item To pass C3, each setup case must include an approximate hand-drawn normal graph that represents the proposed significance level. The graph does not need to be exact, but the shaded rejection region must be close enough to the stated significance level and must match the correct tail direction.
	\end{itemize}
	
	\item \textbf{C4.} Apply the hypothesis test in the \textbf{proportion setting} for \textbf{one selected benchmark direction}.
	\begin{itemize}
		\item Identify \(p_0\), \(\hat p\), and \(n\).
		\item Calculate the test statistic with the initial sample size and conclude that the test fails to reject \(H_0\).
		\item Calculate the test statistic with the minimum sample size found in C7 and conclude that the test rejects \(H_0\).
		\item To pass C4, both decisions must be justified with the correct critical value.
	\end{itemize}
	\[
	z=\frac{\hat p-p_0}{\sqrt{\frac{p_0(1-p_0)}{n}}}
	\]
	
	\item \textbf{C5.} Apply the hypothesis test in the \textbf{mean-count setting} for \textbf{one selected benchmark direction}.
	\begin{itemize}
		\item Identify \(\mu_0\), \(\bar x\), \(\sigma\), and \(n\).
		\item Calculate the test statistic with the initial sample size and conclude that the test fails to reject \(H_0\).
		\item Calculate the test statistic with the minimum sample size found in C7 and conclude that the test rejects \(H_0\).
		\item To pass C5, both decisions must be justified with the correct critical value.
	\end{itemize}
	\[
	z=\frac{\bar x-\mu_0}{\frac{\sigma}{\sqrt{n}}}
	\]
	
	\item \textbf{C7.} Find the minimum sample size needed to reject \(H_0\) for \textbf{one selected benchmark direction}.
	\begin{itemize}
		\item Find the minimum sample size for the proportion setting.
		\item Find the minimum sample size for the mean-count setting.
		\item Use the correct inequality for the selected test direction.
		\item Round up to the next whole number.
		\item To pass C7, the minimum sample size must make the test reject \(H_0\).
	\end{itemize}
	\[
	n=\left(\frac{z\sigma}{\bar x-\mu_0}\right)^2
	\qquad 
	n=\frac{z^2p_0(1-p_0)}{(\hat p-p_0)^2}
	\]
	
	\item \textbf{C6.} Identify the possible error type in context for the four decision cases from \textbf{one selected benchmark direction}.
	\begin{itemize}
		\item Proportion setting with initial sample size.
		\item Mean-count setting with initial sample size.
		\item Proportion setting with minimum sample size.
		\item Mean-count setting with minimum sample size.
		\item To pass C6, each case must state the decision, identify the possible error as Type I or Type II, and explain the error in the context of the scenario.
	\end{itemize}
\end{enumerate}

	\subsubsection*{C4}
	
	\[
	\hat{p}_{25}=0.2333
	\qquad
	\hat{p}_{26}=0.40
	\qquad
	n_{26}=30
	\qquad

	\[

	\]

	\[
	z_\mu=\frac{0.97}{\frac{3.24}{\sqrt{31}}}\approx 1.67>1.645
	\]

	Now both tests reject \(H_0\).

	For the original decision (fail to reject), the possible error is Type II error.

	For the recalculated decision (reject), the possible error is Type I error.

	\textbf{Proportion test.}
	
	\[
	H_0\colon p=p_0
	\qquad
	H_a\colon p<p_0
	\]
	
	\[
	p=p_{25}
	\qquad
	p_0=\hat{p}_{26}=\frac{12}{30}=0.40
	\]
	
	\[
	H_0\colon p_{25}=0.40
	\qquad
	H_a\colon p_{25}<0.40
	\]
	
	\textbf{Mean test.}
	
	\[
	H_0\colon \mu=\mu_0
	\qquad
	H_a\colon \mu<\mu_0
	\]
	
	\[
	\mu=\mu_{25}
	\qquad
	\mu_0=\bar{x}_{26}=8.60
	\]
	
	\[
	H_0\colon \mu_{25}=8.60
	\qquad
	H_a\colon \mu_{25}<8.60
	\]
	
	\subsubsection*{C3}
	
	Both tests are left-tailed.
	
	\[
	H_a\colon p_{25}<0.40
	\qquad
	H_a\colon \mu_{25}<8.60
	\]
	
	\ZNormalLeft{-1.645}{Left tail, \(\alpha=0.05\)}
	
	\subsubsection*{C4}
	
	\[
	\hat{p}_{26}=0.40
	\qquad
	\hat{p}_{25}=0.2333
	\qquad
	n_{25}=30
	\]
	
	\[
	z=
	\frac{\hat{p}_{25}-\hat{p}_{26}}{\sqrt{\frac{\hat{p}_{26}(1-\hat{p}_{26})}{n_{25}}}}
	=
	\frac{0.2333-0.40}{\sqrt{\frac{0.40(0.60)}{30}}}
	\]
	
	\[
	z=
	\frac{-0.1667}{0.0894}
	\approx -1.86
	\]
	
	\[
	z_{\text{critical}}=-1.645
	\]
	
	\[
	-1.86<-1.645
	\]
	
	\RejectDecision{There is enough evidence to conclude that \(p_{25}<0.40\).}
	
	
	\subsubsection*{C5}
	
	\[
	\bar{x}_{26}=8.60
	\qquad
	\bar{x}_{25}=7.63
	\qquad
	\sigma_{26}=3.24
	\qquad
	\sigma_{25}=3.40
	\qquad
	n_{25}=30
	\]
	
	The test sample is Prom 2025, so the statistic uses \(\sigma_{25}\).
	
	\[
	z=
	\frac{\bar{x}_{25}-\bar{x}_{26}}{\frac{\sigma_{25}}{\sqrt{n_{25}}}}
	=
	\frac{7.63-8.60}{\frac{3.40}{\sqrt{30}}}
	\]
	
	\[
	z=
	\frac{-0.97}{0.6207}
	\approx
	-1.56
	\]
	
	\[
	z_{\text{critical}}=-1.645
	\]
	
	\[
	-1.56>-1.645
	\]
	
	\FailDecision{There is not enough evidence to conclude that \(\mu_{25}<8.60\).}
	
	\subsubsection*{C7}

	\textbf{Proportion test.}

	\[
	\frac{0.1667}{\sqrt{1.50\,\frac{0.40(0.60)}{n_{25}}}}>1.645
	\Rightarrow
	n_{25}>
	\left(
	\frac{1.645\sqrt{1.50\cdot 0.40(0.60)}}{0.1667}
	\right)^2
	\approx 44.54
	\]

	\[
	n_{25,\min}=45
	\]

	\textbf{Mean test.}

	\[
	\frac{0.97}{\frac{3.40}{\sqrt{n_{25}}}}>1.645
	\Rightarrow
	n_{25}>
	\left(\frac{1.645(3.40)}{0.97}\right)^2
	\approx 33.23
	\]

	\[
	n_{25,\min}=34
	\]

	\subsubsection*{C6}

	With the original sample size \(n_{25}=30\), both tests failed to reject \(H_0\).

	Using the minimum sample sizes from C7:

	\[
	z_p=\frac{-0.1667}{\sqrt{1.50\,\frac{0.40(0.60)}{45}}}\approx -1.66<-1.645
	\]

	\[
	z_\mu=\frac{-0.97}{\frac{3.40}{\sqrt{34}}}\approx -1.66<-1.645
	\]

	Now both tests reject \(H_0\).

	For the original decision (fail to reject), the possible error is Type II error.

	For the recalculated decision (reject), the possible error is Type I error.
