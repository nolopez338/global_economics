	\ExamSection{C8: Comparison of Bayes, Maximin, and Minimax Regret on the same dataset}
	\begin{ExamProblems}
		\item
		\subsection*{Problem description}
		A manufacturing cooperative must choose one production plan for the next month: Plan $A_1$, Plan $A_2$, or Plan $A_3$.
		There are six market-cost states with probabilities:
		\[
		S_1:0.18,\ S_2:0.17,\ S_3:0.16,\ S_4:0.15,\ S_5:0.19,\ S_6:0.15,
		\]
		which sum to $1.00$.
		Payoffs are in thousand USD. Apply Bayes (EV), Maximin, and Minimax Regret.

		\subsection*{C8}
		Step 1. Build the payoff table.
		\begin{center}
			\textit{Payoff table} \\
			\begin{tabular}{l c c c c}
				\toprule
				State of nature & Probability & $A_1$ & $A_2$ & $A_3$ \\
				\midrule
				$S_1$ & 0.18 & 120 & 110 & 140 \\
				$S_2$ & 0.17 & 95 & 100 & 90 \\
				$S_3$ & 0.16 & 80 & 85 & 70 \\
				$S_4$ & 0.15 & 60 & 70 & 65 \\
				$S_5$ & 0.19 & 40 & 55 & 35 \\
				$S_6$ & 0.15 & 20 & 30 & 25 \\
				\bottomrule
			\end{tabular}
		\end{center}

		Step 2. Bayes criterion (Expected Value) using $EV(A_i)=\sum p_j\cdot x_{ij}$.
		\[
		\begin{aligned}
		EV(A_1) &= 0.18(120)+0.17(95)+0.16(80)+0.15(60)+0.19(40)+0.15(20)\\
		&=21.60+16.15+12.80+9.00+7.60+3.00=70.15,\\
		EV(A_2) &= 0.18(110)+0.17(100)+0.16(85)+0.15(70)+0.19(55)+0.15(30)\\
		&=19.80+17.00+13.60+10.50+10.45+4.50=75.85,\\
		EV(A_3) &= 0.18(140)+0.17(90)+0.16(70)+0.15(65)+0.19(35)+0.15(25)\\
		&=25.20+15.30+11.20+9.75+6.65+3.75=71.85.
		\end{aligned}
		\]
		EV ranking (best $\to$ worst):
		\[
		A_2\,(75.85) > A_3\,(71.85) > A_1\,(70.15).
		\]

		Step 3. Maximin criterion.
		\[
		\begin{aligned}
		\min(A_1)&=\min\{120,95,80,60,40,20\}=20,\\
		\min(A_2)&=\min\{110,100,85,70,55,30\}=30,\\
		\min(A_3)&=\min\{140,90,70,65,35,25\}=25.
		\end{aligned}
		\]
		Maximin ranking (best $\to$ worst):
		\[
		A_2\,(\min=30) > A_3\,(\min=25) > A_1\,(\min=20).
		\]

		Step 4. Minimax Regret criterion.
		Find the best payoff in each state:
		\[
		\begin{aligned}
		\max(S_1)&=140,\ \max(S_2)=100,\ \max(S_3)=85,\\
		\max(S_4)&=70,\ \max(S_5)=55,\ \max(S_6)=30.
		\end{aligned}
		\]
		Build regret table using $r_{ij}=\max(S_j)-x_{ij}$.
		\begin{center}
			\begin{tabular}{l c c c c c c c}
				\toprule
				Alternative & $S_1$ & $S_2$ & $S_3$ & $S_4$ & $S_5$ & $S_6$ & Max regret \\
				\midrule
				$A_1$ & $140-120=20$ & $100-95=5$ & $85-80=5$ & $70-60=10$ & $55-40=15$ & $30-20=10$ & 20 \\
				$A_2$ & $140-110=30$ & $100-100=0$ & $85-85=0$ & $70-70=0$ & $55-55=0$ & $30-30=0$ & 30 \\
				$A_3$ & $140-140=0$ & $100-90=10$ & $85-70=15$ & $70-65=5$ & $55-35=20$ & $30-25=5$ & 20 \\
				\bottomrule
			\end{tabular}
		\end{center}
		Minimax Regret ranking (best $\to$ worst, lower is better):
		\[
		A_1\,(20)=A_3\,(20) > A_2\,(30).
		\]
		There is a tie between $A_1$ and $A_3$ under Minimax Regret.

		Step 5. Conclusion.
		Bayes and Maximin both select $A_2$.
		Minimax Regret gives a tie between $A_1$ and $A_3$.
		The ordered lists show why strategies may disagree even on the same payoff table.

	\begin{ExamProblems}
		\item
		\subsection*{Problem description}
		A city logistics office must choose one policy among $B_1,B_2,B_3,B_4,B_5,B_6$.
		Payoffs (thousand USD) and probabilities are:
		\begin{center}
			\begin{tabular}{l c c c c c c c}
				\toprule
				State & Probability & $B_1$ & $B_2$ & $B_3$ & $B_4$ & $B_5$ & $B_6$ \\
				\midrule
				$T_1$ & 0.50 & 95 & 90 & 110 & 85 & 100 & 92 \\
				$T_2$ & 0.30 & 70 & 75 & 60 & 80 & 65 & 78 \\
				$T_3$ & 0.20 & 30 & 40 & 10 & 55 & 35 & 20 \\
				\bottomrule
			\end{tabular}
		\end{center}
		Ordered criterion results (already computed):
		\[
		\text{EV ranking: }B_4(77.50)>B_5(76.50)>B_2(75.50)>B_3(75.00)>B_1(74.50)>B_6(73.40).
		\]
		\[
		\text{Maximin ranking: }B_4(55)>B_2(40)>B_5(35)>B_1(30)>B_6(20)>B_3(10).
		\]
		\[
		\text{Minimax Regret ranking: }B_2(20)=B_5(20)>B_1(25)=B_4(25)>B_6(35)>B_3(45).
		\]
		Filtering rule:
		(1) keep the best 4 EV options;
		(2) from those, keep the best 2 Maximin options;
		(3) from those, choose the smallest Minimax Regret.

		\subsection*{C9}
		Step 1. Reconfirm ordered lists.
		\begin{itemize}
			\item EV top 4: $\{B_4,B_5,B_2,B_3\}$.
			\item On this subset, Maximin values are $B_4(55),\ B_2(40),\ B_5(35),\ B_3(10)$, so top 2 are $\{B_4,B_2\}$.
			\item On $\{B_4,B_2\}$, Max regrets are $B_4(25),\ B_2(20)$.
		\end{itemize}

		Step 2. Decision tree for staged filtering.
		\begin{center}
			\begin{tabular}{l}
				\toprule
				Decision tree (filtering nodes and leaves) \\
				\midrule
				Root decision node $D_0$: All alternatives $\{B_1,B_2,B_3,B_4,B_5,B_6\}$ \\
				$\rightarrow$ Node $D_1$ (EV filter): keep $\{B_4,B_5,B_2,B_3\}$ \\
				$\rightarrow$ Node $D_2$ (Maximin filter): keep $\{B_4,B_2\}$ \\
				$\rightarrow$ Node $D_3$ (Minimax Regret filter): choose smallest max regret \\
				Leaves: $B_2$ (20), $B_4$ (25) \\
				Final leaf selected: $B_2$ \\
				\bottomrule
			\end{tabular}
		\end{center}

		Step 3. Final decision.
		The decision tree selects $B_2$.

		\item
		\subsection*{Problem description}
		A food-processing company must choose one packaging strategy for next quarter:
		$P$ (premium line), $Q$ (standard line), or $R$ (hybrid line).
		Demand can be high ($0.60$) or low ($0.40$).
		Operating cost can be low ($0.50$), medium ($0.30$), or high ($0.20$).
		Revenues (thousand USD):
		\begin{itemize}
			\item High demand: $P=500,\ Q=540,\ R=520$.
			\item Low demand: $P=320,\ Q=300,\ R=310$.
		\end{itemize}
		Costs (thousand USD):
		\begin{itemize}
			\item $P$: low $180$, medium $220$, high $260$.
			\item $Q$: low $210$, medium $250$, high $295$.
			\item $R$: low $195$, medium $235$, high $275$.
		\end{itemize}
		Build the payoff table, apply EV, Maximin, and Minimax Regret, produce ordered rankings, then use a staged decision tree to choose a final strategy.

		\subsection*{C10}
		Step 1. Build the payoff table from narrative data.
		Joint state probabilities:
		\[
		\begin{aligned}
		&\Pr(H,L)=0.60\cdot 0.50=0.30,\quad \Pr(H,M)=0.60\cdot 0.30=0.18,\quad \Pr(H,Hc)=0.60\cdot 0.20=0.12,\\
		&\Pr(L,L)=0.40\cdot 0.50=0.20,\quad \Pr(L,M)=0.40\cdot 0.30=0.12,\quad \Pr(L,Hc)=0.40\cdot 0.20=0.08.
		\end{aligned}
		\]
		Payoff = revenue $-$ cost.
		\begin{center}
			\begin{tabular}{l c c c c}
				\toprule
				State of nature & Probability & $P$ & $Q$ & $R$ \\

				Low demand, low cost & 0.20 & $320-180=140$ & $300-210=90$ & $310-195=115$ \\
				Low demand, medium cost & 0.12 & $320-220=100$ & $300-250=50$ & $310-235=75$ \\
				Low demand, high cost & 0.08 & $320-260=60$ & $300-295=5$ & $310-275=35$ \\
				\bottomrule
			\end{tabular}
		\end{center}

		Step 2. Bayes criterion (EV).
		\[
		\begin{aligned}
		EV(P)&=0.30(320)+0.18(280)+0.12(240)+0.20(140)+0.12(100)+0.08(60)\\

		\end{aligned}
		\]
		EV ranking:
		\[
		P\,(220.0)>R\,(213.0)>Q\,(205.0).
		\]

		Step 3. Maximin criterion.
		\[
		\min(P)=60,\quad \min(Q)=5,\quad \min(R)=35.
		\]
		Maximin ranking:
		\[
		P\,(60)>R\,(35)>Q\,(5).
		\]

		Step 4. Minimax Regret criterion.
		State best payoffs:
		\[
		\max=\{330,290,245,140,100,60\}.
		\]
		Regret table:
		\begin{center}
			\begin{tabular}{l c c c c c c c}
				\toprule
				Alternative & $HL$ & $HM$ & $HHc$ & $LL$ & $LM$ & $LHc$ & Max regret \\
				\midrule
				$P$ & $330-320=10$ & $290-280=10$ & $245-240=5$ & $140-140=0$ & $100-100=0$ & $60-60=0$ & 10 \\
				$Q$ & $330-330=0$ & $290-290=0$ & $245-245=0$ & $140-90=50$ & $100-50=50$ & $60-5=55$ & 55 \\
				$R$ & $330-325=5$ & $290-285=5$ & $245-245=0$ & $140-115=25$ & $100-75=25$ & $60-35=25$ & 25 \\
				\bottomrule
			\end{tabular}
		\end{center}
		Minimax Regret ranking (lower is better):
		\[
		P\,(10)>R\,(25)>Q\,(55).
		\]

		Step 5. Staged decision tree and final choice.
		Filtering rule: keep best 2 EV options, then choose smaller Minimax Regret.
		\begin{center}
			\begin{tabular}{l}
				\toprule
				Decision tree \\
				\midrule
				$D_0$: $\{P,Q,R\}$ \\

			\end{tabular}
		\end{center}
		Final explanation: $P$ is first in EV and Maximin, and also has the smallest maximum regret, so the tree process consistently ends at $P$.
