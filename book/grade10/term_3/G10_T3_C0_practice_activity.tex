		\subsection*{Problem 1 — App Notifications per Hour}
		
		\textbf{Problem.}
		Let $X$ be the number of app notifications a student receives in one hour. The probability mass function is
		\[
		\begin{array}{c|cccc}
			x & 0 & 1 & 2 & 3\\
			\hline
			P(X=x) & 0.2 & 0.3 & k & 0.1
		\end{array}
		\]
		
		\textbf{Question.}
		(a) Determine $k$ so that this is a valid probability distribution.
		(b) Compute: (i) $P(X=2)$, (ii) $P(X\le 2)$, and (iii) $P(1\le X\le 2)$.
		(c) Interpret these results in terms of percentage probability.
		
		\textbf{Solution.}
		For a valid pmf,
		\[
		0.2+0.3+k+0.1=1\Rightarrow k=0.4.
		\]
		\[
		P(X=2)=0.4,\quad
		P(X\le 2)=0.2+0.3+0.4=0.9,\quad
		P(1\le X\le 2)=0.3+0.4=0.7.
		\]
		\textbf{Interpretation.} There is a $40\%$ probability of exactly 2 notifications, a $90\%$ probability of at most 2 notifications, and a $70\%$ probability of between 1 and 2 notifications.

		\[

		\[

		\]
		\[
		P(Y=1)=0.2,\quad
		P(1\le Y\le 2)=0.2+0.2=0.4,\quad
		P(Y\ge 1)=0.2+0.2+0.1=0.5.
		\]
		\textbf{Interpretation.} There is a $20\%$ probability of exactly 1 defective item, a $40\%$ probability of 1 or 2 defective items, and a $50\%$ probability of at least 1 defective item.

		\textbf{Problem.}
		Let $Z$ be the number of lost pens found in a small box. The probability mass function is
		\[
		\begin{array}{c|cccc}
			z & 0 & 1 & 2 & 3\\
			\hline
			P(Z=z) & 0.4 & 0.5 & a & 0.2
		\end{array}
		\]
		
		\textbf{Question.}
		(a) Solve for $a$ so the probabilities sum to 1.
		(b) Decide whether the resulting distribution is valid.
		(c) Using the computed $a$, compute $P(Z\ge 2)$.
		
		\textbf{Solution.}
		From total probability,
		\[
		0.4+0.5+a+0.2=1\Rightarrow 1.1+a=1\Rightarrow a=-0.1.
		\]
		\textbf{Not a valid probability distribution} because $a<0$.
		\[
		P(Z\ge 2)=P(Z=2)+P(Z=3)=a+0.2=-0.1+0.2=0.1.
		\]

		\subsection*{Problem 10 — Goals in a Match Summary}
		
		\textbf{Problem.}
		Let $T$ be the number of goals scored by a team in a match summary. The probability mass function is
		\[
		P(T=t)=\begin{cases}
			p, & t=0,\\
			\frac{3}{4}-2p, & t=1,\\
			p, & t=2,\\
			\frac{1}{4}, & t=3.
		\end{cases}
		\]
		
		\textbf{Question.}
		(a) Determine all valid values of $p$.
		(b) Compute: (i) $P(T=1)$, (ii) $P(T\in\{0,2,3\})$, and (iii) $P(1\le T\le 3)$.
		(c) Interpret these results in terms of percentage probability.
		
		\textbf{Solution.}
		The sum condition is
		\[
		p+\left(\frac{3}{4}-2p\right)+p+\frac{1}{4}=1,
		\]
		which is true for all $p$.
		\[
		P(T=1)=\frac{3}{4}-2p,\quad
		P(T\in\{0,2,3\})=p+p+\frac{1}{4}=2p+\frac{1}{4},\quad
		P(1\le T\le 3)=1-p.
		\]
		\textbf{Interpretation.} The percentage probabilities are $100\left(\frac{3}{4}-2p\right)\%$ for exactly 1 goal, $100\left(2p+\frac{1}{4}\right)\%$ for 0, 2, or 3 goals, and $100(1-p)\%$ for between 1 and 3 goals.

		\textbf{Problem.}
		Let $X$ be the number of late buses observed in a morning. The probability mass function is
		\[
		\begin{array}{c|cccc}
			x & 0 & 1 & 2 & 3\\
			\hline
			P(X=x) & a & b & 2a & b
		\end{array}
		\]
		and $P(X=0)=P(X=3)$.
		
		\textbf{Question.}
		(a) Determine $a$ and $b$ so that the model is valid.
		(b) Compute: (i) $P(X=2)$, (ii) $P(X>1)$, and (iii) $P(1\le X\le 3)$.

		\textbf{Solution.}
		From the extra condition $P(X=0)=P(X=3)$,
		\[
		a=b.
		\]
		From total probability,
		\[
		a+b+2a+b=1\Rightarrow 3a+2b=1.
		\]
		Substitute $b=a$:
		\[
		5a=1\Rightarrow a=\frac{1}{5},\quad b=\frac{1}{5}.
		\]
		\[
		P(X=2)=2a=\frac{2}{5},\quad
		P(X>1)=2a+b=\frac{3}{5},\quad
		P(1\le X\le 3)=b+2a+b=\frac{4}{5}.
		\]
		\textbf{Interpretation.} There is a $40\%$ probability of exactly 2 late buses, a $60\%$ probability of more than 1 late bus, and an $80\%$ probability of between 1 and 3 late buses.

		\textbf{Problem.}
		Let $X$ be the number of yellow cards shown in a match summary. The probability mass function is
		\[

		\[
		2b=2a\Rightarrow b=a.
		\]

		\[
		2a=\frac{1}{4}\Rightarrow a=\frac{1}{8},\quad b=\frac{1}{8}.
		\]
		\[
		P(X=4)=2b=\frac{1}{4},\quad
		P(X\le 1)=a+b=\frac{1}{4},\quad
		P(2\le X\le 4)=(a+b)+2a+2b=\frac{3}{4}.
		\]
		\textbf{Interpretation.} There is a $25\%$ probability of exactly 4 yellow cards, a $25\%$ probability of at most 1 yellow card, and a $75\%$ probability of between 2 and 4 yellow cards.

		(c) Interpret these results in terms of percentage probability.
		
		\textbf{Solution.}
		Check total probability:
		\[
		\pi+2\pi+3\pi+\left(\frac{1}{2}-2\pi\right)+\frac{1}{4}=1
		\Rightarrow 4\pi+\frac{3}{4}=1
		\Rightarrow \pi=\frac{1}{16}.
		\]
		\[
		P(W=4)=\frac{1}{4},\quad
		P(W>2)=\left(\frac{1}{2}-2\pi\right)+\frac{1}{4}=\frac{5}{8},\quad
		P(1\le W\le 3)=2\pi+3\pi+\left(\frac{1}{2}-2\pi\right)=\frac{11}{16}.
		\]
		\textbf{Interpretation.} There is a $25\%$ probability of exactly 4 missing submissions, a $62.5\%$ probability of more than 2 missing submissions, and a $68.75\%$ probability of between 1 and 3 missing submissions.

		\subsection*{Problem 20 — Returns Processed at a Counter}
		
		\textbf{Problem.}
		Let $T$ be the number of product returns processed at a counter in a short period. Suppose
		\[
		\begin{array}{c|ccccc}
			t & 0 & 1 & 2 & 3 & 4\\
			\hline
			P(T=t) & k & 2k & 2k & k & k
		\end{array}
		\]
		
		\textbf{Question.}
		(a) Find $k$.
		(b) Compute: (i) $P(T=2)$, (ii) $P(T\ne 2)$ using a complement, and (iii) $P(1\le T\le 3)$.
		(c) Interpret these results in terms of percentage probability.
		
		\textbf{Solution.}
		Use the total probability rule:
		\[
		k+2k+2k+k+k=7k=1\Rightarrow k=\frac{1}{7}.
		\]
		\[
		P(T=2)=2k=\frac{2}{7},\quad
		P(T\ne 2)=1-2k=\frac{5}{7},\quad
		P(1\le T\le 3)=2k+2k+k=5k=\frac{5}{7}.
		\]
		\textbf{Interpretation.} There is a $28.57\%$ probability of exactly 2 returns, a $71.43\%$ probability of a number of returns different from 2, and a $71.43\%$ probability of between 1 and 3 returns.
