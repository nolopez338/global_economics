		\item
		\subsection*{Problem 1 — App Notifications per Hour}
		
		\textbf{Problem.}
		Let $X$ be the number of app notifications a student receives in one hour. The probability mass function is
		\[
		\begin{array}{c|cccc}
		x & 0 & 1 & 2 & 3\\
		\hline
		P(X=x) & \frac{1}{10} & \frac{3}{10} & \frac{2}{5} & \frac{1}{5}
		\end{array}
		\]
		
		\textbf{Question.}
		(a) Verify this is a valid probability distribution.
		(b) Compute $P(X\le 2)$.
		(c) Interpret the probability in context.
		
		\textbf{Solution.}
		Check the total probability:
		\[
		\frac{1}{10}+\frac{3}{10}+\frac{2}{5}+\frac{1}{5}=\frac{1}{10}+\frac{3}{10}+\frac{4}{10}+\frac{2}{10}=1.
		\]
		So it is a valid probability distribution.
		\[
		P(X\le 2)=P(X=0)+P(X=1)+P(X=2)=\frac{1}{10}+\frac{3}{10}+\frac{2}{5}=\frac{4}{5}.
		\]
		\textbf{Interpretation.} There is a $\frac{4}{5}$ chance the student gets at most 2 notifications in an hour.

		\subsection*{Problem 2 — Defective Items in a Batch of 4}
		
		\textbf{Problem.}
		A quality check records $X$, the number of defective items in a batch of 4. Suppose
		\[
		\begin{array}{c|ccccc}
		x & 0 & 1 & 2 & 3 & 4\\
		\hline
		P(X=x) & \frac{2}{5} & \frac{3}{10} & \frac{1}{5} & \frac{1}{10} & 0
		\end{array}
		\]
		
		\textbf{Question.}
		(a) Verify this is a valid probability distribution.
		(b) Compute $P(1\le X\le 3)$.
		(c) Interpret the probability in context.
		
		\textbf{Solution.}
		First, add all probabilities:
		\[
		\frac{2}{5}+\frac{3}{10}+\frac{1}{5}+\frac{1}{10}+0
		=\frac{4}{10}+\frac{3}{10}+\frac{2}{10}+\frac{1}{10}=1.
		\]
		So the distribution is valid.
		\[
		P(1\le X\le 3)=P(X=1)+P(X=2)+P(X=3)=\frac{3}{10}+\frac{1}{5}+\frac{1}{10}=\frac{3}{5}.
		\]
		\textbf{Interpretation.} The probability of getting between 1 and 3 defective items is $\frac{3}{5}$.

		\subsection*{Problem 3 — Late Deliveries in One Day}
		
		\textbf{Problem.}
		Let $X$ be the number of late deliveries made by a shop in one day. The probability mass function is
		\[
		\begin{array}{c|ccccc}
		x & 0 & 1 & 2 & 3 & 4\\
		\hline
		P(X=x) & \frac{1}{4} & \frac{3}{10} & \frac{1}{5} & \frac{3}{20} & \frac{1}{10}
		\end{array}
		\]
		
		\textbf{Question.}
		(a) Verify this is a valid probability distribution.
		(b) Compute $P(X\ge 2)$.
		(c) Interpret the probability in context.
		
		\textbf{Solution.}
		Check the sum of probabilities:
		\[
		\frac{1}{4}+\frac{3}{10}+\frac{1}{5}+\frac{3}{20}+\frac{1}{10}
		=\frac{5}{20}+\frac{6}{20}+\frac{4}{20}+\frac{3}{20}+\frac{2}{20}=1.
		\]
		Hence it is valid.
		\[
		P(X\ge 2)=P(X=2)+P(X=3)+P(X=4)=\frac{1}{5}+\frac{3}{20}+\frac{1}{10}=\frac{9}{20}.
		\]
		\textbf{Interpretation.} There is a $\frac{9}{20}$ chance of having at least 2 late deliveries in a day.

		\subsection*{Problem 4 — Customers in a 10-Minute Window}
		
		\textbf{Problem.}
		A café tracks $X$, the number of customers arriving in a 10-minute window.
		\[
		\begin{array}{c|cccc}
		x & 1 & 2 & 3 & 4\\
		\hline
		P(X=x) & \frac{1}{5} & \frac{7}{20} & \frac{3}{10} & \frac{3}{20}
		\end{array}
		\]
		
		\textbf{Question.}
		(a) Verify this is a valid probability distribution.
		(b) Compute $P(X\ne 4)$ using a complement.
		(c) Interpret the probability in context.
		
		\textbf{Solution.}
		Add the probabilities:
		\[
		\frac{1}{5}+\frac{7}{20}+\frac{3}{10}+\frac{3}{20}
		=\frac{4}{20}+\frac{7}{20}+\frac{6}{20}+\frac{3}{20}=1.
		\]
		So this is a valid probability distribution.
		\[
		P(X\ne 4)=1-P(X=4)=1-\frac{3}{20}=\frac{17}{20}.
		\]
		\textbf{Interpretation.} The chance that the café does not receive exactly 4 customers in that window is $\frac{17}{20}$.

		\subsection*{Problem 6 — Messages Received in an Hour}
		
		\textbf{Problem.}
		Let $M$ be the number of messages received in one hour. Suppose
		\[
		\begin{array}{c|ccccc}
		x & 0 & 1 & 2 & 3 & 4\\
		\hline
		P(M=x) & k & 2k & 3k & 2k & k
		\end{array}
		\]
		
		\textbf{Question.}
		(a) Find $k$ so this is a valid probability distribution.
		(b) Compute $P(M\ge 3)$.
		(c) Interpret the probability in context.
		
		\textbf{Solution.}
		Use the sum-to-1 rule:
		\[
		k+2k+3k+2k+k=9k=1\Rightarrow k=\frac{1}{9}.
		\]
		Now
		\[
		P(M\ge 3)=P(M=3)+P(M=4)=2k+k=3k=\frac{1}{3}.
		\]
		\textbf{Interpretation.} There is a $\frac{1}{3}$ chance of receiving at least 3 messages in one hour.

		\subsection*{Problem 7 — Goals in a Match Summary}
		
		\textbf{Problem.}
		Let $G$ be the number of goals scored by a team in a match summary. The probability mass function is
		\[
		P(G=x)=\begin{cases}
		k, & x=0,\\
		2k, & x=1,\\
		3k, & x=2,\\
		2k, & x=3.
		\end{cases}
		\]
		This corresponds to
		\[
		\begin{array}{c|cccc}
		x & 0 & 1 & 2 & 3\\
		\hline
		P(G=x) & k & 2k & 3k & 2k
		\end{array}
		\]
		
		\textbf{Question.}
		(a) Find $k$.
		(b) Compute $P(G\in\{0,2,3\})$.
		(c) Interpret the probability in context.
		
		\textbf{Solution.}
		Add all probabilities:
		\[
		k+2k+3k+2k=8k=1\Rightarrow k=\frac{1}{8}.
		\]
		Then
		\[
		P(G\in\{0,2,3\})=P(G=0)+P(G=2)+P(G=3)
		=k+3k+2k=6k=\frac{3}{4}.
		\]
		\textbf{Interpretation.} The chance of scoring 0, 2, or 3 goals is $\frac{3}{4}$.

		\textbf{Problem.}
		Let $N$ be the number of phone calls to an office in a short period. The probability mass function is
		\[
		P(N=x)=\begin{cases}
		\frac{k}{2}, & x=0,\\

		\frac{3k}{2}, & x=2,\\

		\end{cases}

		\end{array}
		\]
		
		\textbf{Question.}
		(a) Determine $k$.
		(b) Compute $P(N\le 1)$.

		\frac{k}{2}+k+\frac{3k}{2}+k=4k=1\Rightarrow k=\frac{1}{4}.
		\]
		Then
		\[
		P(N\le 1)=P(N=0)+P(N=1)=\frac{k}{2}+k=\frac{3k}{2}=\frac{3}{8}.

		\textbf{Interpretation.} There is a $\frac{3}{8}$ chance of receiving at most 1 phone call.

		\subsection*{Problem 15 — Returns Processed at a Counter}
		
		\textbf{Problem.}
		Let $R$ be the number of product returns processed at a counter in a short period. Suppose
		\[
		\begin{array}{c|ccccc}
		x & 0 & 1 & 2 & 3 & 4\\
		\hline
		P(R=x) & k & 2k & 2k & k & k
		\end{array}
		\]
		
		\textbf{Question.}
		(a) Find $k$.
		(b) Compute $P(R\ne 2)$ using a complement.
		(c) Interpret the probability in context.
		
		\textbf{Solution.}
		Use the total probability rule:
		\[
		k+2k+2k+k+k=7k=1\Rightarrow k=\frac{1}{7}.
		\]
		Then
		\[
		P(R\ne 2)=1-P(R=2)=1-2k=1-\frac{2}{7}=\frac{5}{7}.
		\]
		\textbf{Interpretation.} There is a $\frac{5}{7}$ chance that the counter processes a number of returns different from 2.
