		\item
		\subsection*{Problem 1 — App Notifications per Hour}
		
		\textbf{Problem.}
		Let $X$ be the number of app notifications a student receives in one hour. The probability mass function is
		\[
		\begin{array}{c|cccc}
		x & 0 & 1 & 2 & 3\\
		\hline
		P(X=x) & 0.2 & 0.3 & k & 0.1
		\end{array}
		\]
		
		\textbf{Question.}
		(a) Determine $k$ so that this is a valid probability distribution.
		(b) State the nonnegativity restriction on $k$.
		(c) Using your value of $k$, compute $P(X\le 2)$.
		(d) Interpret this probability in context.
		
		\textbf{Solution.}
		For a valid pmf,
		\[
		0.2+0.3+k+0.1=1\Rightarrow k=0.4.
		\]
		Nonnegativity requires $k\ge 0$, and $k=0.4$ satisfies this.
		\[
		P(X\le 2)=P(X=0)+P(X=1)+P(X=2)=0.2+0.3+0.4=0.9.
		\]
		\textbf{Interpretation.} There is a $0.9$ probability the student gets at most 2 notifications in an hour.

		\subsection*{Problem 2 — Defective Items in a Batch}
		
		\textbf{Problem.}
		A quality check records $X$, the number of defective items in a small batch. Suppose
		\[
		\begin{array}{c|cccc}
		x & 0 & 1 & 2 & 3\\
		\hline
		P(X=x) & 0.5 & k & 0.2 & 0.1
		\end{array}
		\]
		
		\textbf{Question.}
		(a) Determine $k$ so that the distribution is valid.
		(b) State the nonnegativity restriction on $k$.
		(c) Using your value of $k$, compute $P(1\le X\le 2)$.
		(d) Interpret this probability in context.
		
		\textbf{Solution.}
		Sum-to-1 gives
		\[
		0.5+k+0.2+0.1=1\Rightarrow k=0.2.
		\]
		Nonnegativity requires $k\ge 0$, and $k=0.2$ is valid.
		\[
		P(1\le X\le 2)=P(X=1)+P(X=2)=0.2+0.2=0.4.
		\]
		\textbf{Interpretation.} The probability of getting 1 or 2 defective items is $0.4$.

		\subsection*{Problem 5 — Correct Answers on a Short Quiz}
		
		\textbf{Problem.}
		Let $X$ be the number of correct answers on a 4-question quiz. The pmf is
		\[
		P(X=x)=\begin{cases}
		k, & x=0,\\
		2k, & x=1,\\
		3k, & x=2,\\
		2k, & x=3,\\
		1-8k, & x=4.
		\end{cases}
		\]
		
		\textbf{Question.}
		(a) Determine all values of $k$ that make this a valid distribution.
		(b) State the nonnegativity restrictions.
		(c) Compute $P(X\ge 3)$ in terms of $k$.
		(d) Interpret this probability in context.
		
		\textbf{Solution.}
		From the total probability condition,
		\[
		k+2k+3k+2k+(1-8k)=1,
		\]
		which is true for all $k$.
		Nonnegativity requires
		\[
		k\ge 0,\quad 1-8k\ge 0\Rightarrow 0\le k\le \frac{1}{8}.
		\]
		Now
		\[
		P(X\ge 3)=P(X=3)+P(X=4)=2k+(1-8k)=1-6k.
		\]
		\textbf{Interpretation.} The probability of at least 3 correct answers is $1-6k$, valid for $0\le k\le\frac{1}{8}$.

		\[
		P(M\ge 3)=P(M=3)+P(M=4)=2k+k=3k=\frac{1}{3}.
		\]
		\textbf{Interpretation.} There is a $\frac{1}{3}$ chance of receiving at least 3 messages in one hour.

		\subsection*{Problem 7 — Goals in a Match Summary}
		
		\textbf{Problem.}
		Let $G$ be the number of goals scored by a team in a match summary. The probability mass function is
		\[
		P(G=x)=\begin{cases}
		p, & x=0,\\
		\frac{3}{4}-2p, & x=1,\\
		p, & x=2,\\
		\frac{1}{4}, & x=3.
		\end{cases}
		\]
		
		\textbf{Question.}
		(a) Determine all valid values of $p$.
		(b) State nonnegativity restrictions.
		(c) Compute $P(G\in\{0,2,3\})$.
		(d) Interpret this probability in context.
		
		\textbf{Solution.}
		The sum condition is
		\[
		p+\left(\frac{3}{4}-2p\right)+p+\frac{1}{4}=1,
		\]
		which is true for all $p$.
		Nonnegativity gives
		\[
		p\ge 0,\quad \frac{3}{4}-2p\ge 0\Rightarrow 0\le p\le \frac{3}{8}.
		\]
		Then
		\[
		P(G\in\{0,2,3\})=p+p+\frac{1}{4}=2p+\frac{1}{4}.
		\]
		\textbf{Interpretation.} The chance of scoring 0, 2, or 3 goals is $2p+\frac{1}{4}$, with $0\le p\le\frac{3}{8}$.

		\subsection*{Problem 8 — Late Buses in a Morning}
		
		\textbf{Problem.}
		Let $B$ be the number of late buses observed in a morning. The probability mass function is
		\[
		\begin{array}{c|cccc}
		x & 0 & 1 & 2 & 3\\
		\hline
		P(B=x) & a & b & 2a & b
		\end{array}
		\]
		and $P(B=0)=P(B=3)$.
		
		\textbf{Question.}
		(a) Determine $a$ and $b$ so that the model is valid.
		(b) State nonnegativity restrictions.
		(c) Compute $P(B>1)$.
		(d) Interpret this probability in context.
		
		\textbf{Solution.}
		From the extra condition $P(B=0)=P(B=3)$,
		\[
		a=b.
		\]
		From total probability,
		\[
		a+b+2a+b=1\Rightarrow 3a+2b=1.
		\]
		Substitute $b=a$:
		\[
		5a=1\Rightarrow a=\frac{1}{5},\quad b=\frac{1}{5}.
		\]
		Nonnegativity requires $a\ge 0$ and $b\ge 0$, satisfied.
		\[
		P(B>1)=P(B=2)+P(B=3)=2a+b=\frac{2}{5}+\frac{1}{5}=\frac{3}{5}.
		\]
		\textbf{Interpretation.} The probability of observing more than 1 late bus is $\frac{3}{5}$.

		\subsection*{Problem 10 — Correct Answers on a 3-Question Check}
		
		\textbf{Problem.}
		Let $Q$ be the number of correct answers on a 3-question check.
		\[
		\begin{array}{c|cccc}
		x & 0 & 1 & 2 & 3\\
		\hline
		P(Q=x) & 2p & p & \frac{1}{2}-p & \frac{1}{4}
		\end{array}
		\]
		
		\textbf{Question.}
		(a) Determine $p$ so that this is a valid distribution.
		(b) State nonnegativity restrictions.
		(c) Compute $P(Q\in\{0,2\})$.
		(d) Interpret this probability in context.
		
		\textbf{Solution.}
		Sum-to-1 gives
		\[
		2p+p+\left(\frac{1}{2}-p\right)+\frac{1}{4}=1
		\Rightarrow 2p+\frac{3}{4}=1
		\Rightarrow p=\frac{1}{8}.
		\]
		Nonnegativity gives
		\[
		2p\ge 0,\quad p\ge 0,\quad \frac{1}{2}-p\ge 0,
		\]
		so $0\le p\le \frac{1}{2}$, and $p=\frac{1}{8}$ is valid.
		\[
		P(Q\in\{0,2\})=P(Q=0)+P(Q=2)=2p+\left(\frac{1}{2}-p\right)=\frac{5}{8}.
		\]
		\textbf{Interpretation.} The probability of getting either 0 or 2 correct answers is $\frac{5}{8}$.

		\subsection*{Problem 11 — Yellow Cards in a Match}
		
		\textbf{Problem.}
		Let $Y$ be the number of yellow cards shown in a match summary. The probability mass function is
		\[
		\begin{array}{c|ccccc}
		x & 0 & 1 & 2 & 3 & 4\\
		\hline
		P(Y=x) & a & b & a+b & 2a & 2b
		\end{array}
		\]
		and it is known that $P(Y=4)=2P(Y=0)$.
		
		\textbf{Question.}
		(a) Find $a$ and $b$ so the model is valid.
		(b) State nonnegativity restrictions.
		(c) Compute $P(Y\le 1)$.
		(d) Interpret this probability in context.
		
		\textbf{Solution.}
		From total probability,
		\[
		a+b+(a+b)+2a+2b=1\Rightarrow 4a+4b=1\Rightarrow a+b=\frac{1}{4}.
		\]
		From $P(Y=4)=2P(Y=0)$,
		\[
		2b=2a\Rightarrow b=a.
		\]
		Substitute into $a+b=\frac{1}{4}$:
		\[
		2a=\frac{1}{4}\Rightarrow a=\frac{1}{8},\quad b=\frac{1}{8}.
		\]
		Nonnegativity requires $a\ge 0$ and $b\ge 0$, satisfied.
		\[
		P(Y\le 1)=P(Y=0)+P(Y=1)=a+b=\frac{1}{4}.
		\]
		\textbf{Interpretation.} The probability of at most 1 yellow card is $\frac{1}{4}$ for this valid parameter choice.

		\textbf{Problem.}
		Let $D$ be the number of defects found in a final check. Suppose

		P(D=x)=\begin{cases}

		(c) Compute $P(D\in\{0,2,4\})$.
		(d) Interpret this probability in context.
		
		\textbf{Solution.}

		\[
		p+2p+\left(\frac{1}{2}-2p\right)+p+\frac{1}{4}=1

		\Rightarrow p=\frac{1}{8}.

		\[

		Nonnegativity requires $k\ge 0$, satisfied.
		\[
		P(N\le 1)=P(N=0)+P(N=1)=\frac{k}{2}+k=\frac{3k}{2}=\frac{3}{8}.
		\]
		\textbf{Interpretation.} There is a $\frac{3}{8}$ chance of receiving at most 1 phone call.

		\subsection*{Problem 14 — Homework Submissions Missing}
		
		\textbf{Problem.}
		A teacher records $H$, the number of missing homework submissions in a class group.
		\[
		\begin{array}{c|ccccc}
		x & 0 & 1 & 2 & 3 & 4\\
		\hline
		P(H=x) & p & 2p & 3p & \frac{1}{2}-2p & \frac{1}{4}
		\end{array}
		\]
		
		\textbf{Question.}
		(a) Determine $p$ so this is a valid distribution.
		(b) State nonnegativity restrictions.
		(c) Compute $P(H>2)$.
		(d) Interpret this probability in context.
		
		\textbf{Solution.}
		Check total probability:
		\[
		p+2p+3p+\left(\frac{1}{2}-2p\right)+\frac{1}{4}=1
		\Rightarrow 4p+\frac{3}{4}=1
		\Rightarrow p=\frac{1}{16}.
		\]
		Nonnegativity requires
		\[
		p\ge 0,\quad \frac{1}{2}-2p\ge 0\Rightarrow 0\le p\le \frac{1}{4},
		\]
		and $p=\frac{1}{16}$ is valid.
		\[
		P(H>2)=P(H=3)+P(H=4)=\left(\frac{1}{2}-2p\right)+\frac{1}{4}=\frac{5}{8}.
		\]
		\textbf{Interpretation.} The probability of more than 2 missing submissions is $\frac{5}{8}$.

		\subsection*{Problem 15 — Returns Processed at a Counter}
		
		\textbf{Problem.}
		Let $R$ be the number of product returns processed at a counter in a short period. Suppose
		\[
		\begin{array}{c|ccccc}
		x & 0 & 1 & 2 & 3 & 4\\
		\hline
		P(R=x) & k & 2k & 2k & k & k
		\end{array}
		\]
		
		\textbf{Question.}
		(a) Find $k$.
		(b) State the nonnegativity restriction.
		(c) Compute $P(R\ne 2)$ using a complement.
		(d) Interpret this probability in context.
		
		\textbf{Solution.}
		Use the total probability rule:
		\[
		k+2k+2k+k+k=7k=1\Rightarrow k=\frac{1}{7}.
		\]
		Nonnegativity requires $k\ge 0$, and this is satisfied.
		\[
		P(R\ne 2)=1-P(R=2)=1-2k=1-\frac{2}{7}=\frac{5}{7}.
		\]
		\textbf{Interpretation.} There is a $\frac{5}{7}$ chance that the counter processes a number of returns different from 2.
