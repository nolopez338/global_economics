		\item
		\subsection*{Problem 1 — App Notifications per Hour}
		
		\textbf{Problem.}
		Let $X$ be the number of app notifications a student receives in one hour. The probability mass function is
		\[
		\begin{array}{c|cccc}
		x & 0 & 1 & 2 & 3\\
		\hline
		P(X=x) & k & 2k & 3k & 4k
		\end{array}
		\]
		
		\textbf{Question.}
		(a) Determine $k$ so that this is a valid probability distribution.
		(b) State the nonnegativity restriction on $k$.
		(c) Using your value of $k$, compute $P(X\le 2)$.
		
		\textbf{Solution.}
		For a valid pmf,
		\[
		k+2k+3k+4k=1\Rightarrow 10k=1\Rightarrow k=\frac{1}{10}.
		\]
		Nonnegativity requires $k\ge 0$, and $k=\frac{1}{10}$ satisfies this.
		\[
		P(X\le 2)=P(X=0)+P(X=1)+P(X=2)=k+2k+3k=6k=\frac{3}{5}.
		\]
		\textbf{Interpretation.} There is a $\frac{3}{5}$ chance the student gets at most 2 notifications in an hour.

		\subsection*{Problem 2 — Defective Items in a Batch of 4}
		
		\textbf{Problem.}
		A quality check records $X$, the number of defective items in a batch of 4. Suppose
		\[
		\begin{array}{c|ccccc}
		x & 0 & 1 & 2 & 3 & 4\\
		\hline
		P(X=x) & p & 2p & 1-4p & p & 0
		\end{array}
		\]
		
		\textbf{Question.}
		(a) Determine all $p$ values that make the distribution valid.
		(b) State all nonnegativity restrictions.
		(c) For valid $p$, compute $P(1\le X\le 3)$.
		
		\textbf{Solution.}
		Sum-to-1 gives
		\[
		p+2p+(1-4p)+p+0=1,
		\]
		which is true for all $p$.
		Now impose nonnegativity:
		\[
		p\ge 0,\quad 2p\ge 0,\quad 1-4p\ge 0,\quad p\ge 0
		\]
		so
		\[
		0\le p\le \frac{1}{4}.
		\]
		Then
		\[
		P(1\le X\le 3)=2p+(1-4p)+p=1-p.
		\]
		For valid $p$, this ranges from $\frac{3}{4}$ to $1$.
		\textbf{Interpretation.} The probability of getting between 1 and 3 defects is $1-p$, where $0\le p\le\frac{1}{4}$.

		\subsection*{Problem 3 — Late Deliveries in One Day}
		
		\textbf{Problem.}
		Let $X$ be the number of late deliveries made by a shop in one day. The probability mass function is
		\[
		\begin{array}{c|cccc}
		x & 0 & 1 & 2 & 3\\
		\hline
		P(X=x) & a & b & 2a & 3b
		\end{array}
		\]
		and it is given that $P(X\ge 2)=\frac{2}{3}$.
		
		\textbf{Question.}
		(a) Determine $a$ and $b$ so that the distribution is valid.
		(b) State nonnegativity restrictions.
		(c) Compute $P(X\le 1)$.
		
		\textbf{Solution.}
		From total probability,
		\[
		a+b+2a+3b=1\Rightarrow 3a+4b=1.
		\]
		From $P(X\ge 2)=P(X=2)+P(X=3)$,
		\[
		2a+3b=\frac{2}{3}.
		\]
		Solve the system:
		\[
		\begin{aligned}
		3a+4b&=1,\\
		2a+3b&=\frac{2}{3}
		\end{aligned}
		\Rightarrow a=\frac{1}{3},\ b=0.
		\]
		Nonnegativity requires $a\ge 0$ and $b\ge 0$, satisfied here.
		\[
		P(X\le 1)=P(X=0)+P(X=1)=a+b=\frac{1}{3}.
		\]
		\textbf{Interpretation.} The chance of at most 1 late delivery is $\frac{1}{3}$.

		\subsection*{Problem 5 — Correct Answers on a Short Quiz}
		
		\textbf{Problem.}
		Let $X$ be the number of correct answers on a 4-question quiz. The pmf is
		\[
		P(X=x)=\begin{cases}
		k, & x=0,\\
		2k, & x=1,\\
		3k, & x=2,\\
		2k, & x=3,\\
		1-8k, & x=4.
		\end{cases}
		\]
		
		\textbf{Question.}
		(a) Determine all values of $k$ that make this a valid distribution.
		(b) State the nonnegativity restrictions.
		(c) Compute $P(X\ge 3)$ in terms of $k$.
		
		\textbf{Solution.}
		From the total probability condition,
		\[
		k+2k+3k+2k+(1-8k)=1,
		\]
		which is always true.
		Nonnegativity requires
		\[
		k\ge 0,\quad 1-8k\ge 0\Rightarrow 0\le k\le \frac{1}{8}.
		\]
		Now
		\[
		P(X\ge 3)=P(X=3)+P(X=4)=2k+(1-8k)=1-6k.
		\]
		\textbf{Interpretation.} The probability of at least 3 correct answers is $1-6k$, valid for $0\le k\le\frac{1}{8}$.

		\subsection*{Problem 8 — Late Buses in a Morning}
		
		\textbf{Problem.}
		Let $B$ be the number of late buses observed in a morning. The probability mass function is
		\[
		\begin{array}{c|cccc}
		x & 0 & 1 & 2 & 3\\
		\hline
		P(B=x) & a & b & 2a & b
		\end{array}
		\]
		and $P(B=0)=P(B=3)$.
		
		\textbf{Question.}
		(a) Determine $a$ and $b$ so that the model is valid.
		(b) State nonnegativity restrictions.
		(c) Compute $P(B>1)$.
		
		\textbf{Solution.}
		From the extra condition $P(B=0)=P(B=3)$,
		\[
		a=b.
		\]
		From total probability,
		\[
		a+b+2a+b=1\Rightarrow 3a+2b=1.
		\]
		Substitute $b=a$:
		\[
		5a=1\Rightarrow a=\frac{1}{5},\quad b=\frac{1}{5}.
		\]
		Nonnegativity requires $a\ge 0$ and $b\ge 0$, satisfied.
		\[
		P(B>1)=P(B=2)+P(B=3)=2a+b=\frac{2}{5}+\frac{1}{5}=\frac{3}{5}.
		\]
		\textbf{Interpretation.} The probability of observing more than 1 late bus is $\frac{3}{5}$.

		\subsection*{Problem 11 — Yellow Cards in a Match}
		
		\textbf{Problem.}
		Let $Y$ be the number of yellow cards shown in a match summary. The probability mass function is
		\[
		\begin{array}{c|ccccc}
		x & 0 & 1 & 2 & 3 & 4\\
		\hline
		P(Y=x) & a & b & a+b & 2a & 2b
		\end{array}
		\]
		and it is known that $P(Y=4)=2P(Y=0)$.
		
		\textbf{Question.}
		(a) Find $a$ and $b$ so the model is valid.
		(b) State nonnegativity restrictions.
		(c) Compute $P(Y\le 1)$.
		
		\textbf{Solution.}
		From total probability,
		\[
		a+b+(a+b)+2a+2b=1\Rightarrow 4a+4b=1\Rightarrow a+b=\frac{1}{4}.
		\]
		From $P(Y=4)=2P(Y=0)$,
		\[
		2b=2a\Rightarrow b=a.
		\]
		Substitute into $a+b=\frac{1}{4}$:
		\[
		2a=\frac{1}{4}\Rightarrow a=\frac{1}{8},\quad b=\frac{1}{8}.
		\]
		Nonnegativity requires $a\ge 0$ and $b\ge 0$, satisfied.
		\[
		P(Y\le 1)=P(Y=0)+P(Y=1)=a+b=\frac{1}{4}.
		\]
		\textbf{Interpretation.} The probability of at most 1 yellow card is $\frac{1}{4}$ for this valid parameter choice.

		\textbf{Problem.}

		\[
		\frac{k}{2}+k+\frac{3k}{2}+k=4k=1\Rightarrow k=\frac{1}{4}.
		\]
		Nonnegativity requires $k\ge 0$, satisfied.
		\[
		P(N\le 1)=P(N=0)+P(N=1)=\frac{k}{2}+k=\frac{3k}{2}=\frac{3}{8}.
		\]
		\textbf{Interpretation.} There is a $\frac{3}{8}$ chance of receiving at most 1 phone call.

		\subsection*{Problem 15 — Returns Processed at a Counter}
		
		\textbf{Problem.}
		Let $R$ be the number of product returns processed at a counter in a short period. Suppose
		\[
		\begin{array}{c|ccccc}
		x & 0 & 1 & 2 & 3 & 4\\
		\hline
		P(R=x) & k & 2k & 2k & k & k
		\end{array}
		\]
		
		\textbf{Question.}
		(a) Find $k$.
		(b) State the nonnegativity restriction.
		(c) Compute $P(R\ne 2)$ using a complement.
		
		\textbf{Solution.}
		Use the total probability rule:
		\[
		k+2k+2k+k+k=7k=1\Rightarrow k=\frac{1}{7}.
		\]
		Nonnegativity requires $k\ge 0$, and this is satisfied.
		\[
		P(R\ne 2)=1-P(R=2)=1-2k=1-\frac{2}{7}=\frac{5}{7}.
		\]
		\textbf{Interpretation.} There is a $\frac{5}{7}$ chance that the counter processes a number of returns different from 2.
