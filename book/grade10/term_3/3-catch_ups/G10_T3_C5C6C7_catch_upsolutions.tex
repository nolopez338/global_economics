	\section{(14) Standard Normal Table Problems (C5)}
	\label{sec:standard-normal-table-problems-c5}
	
	Let \(Z\sim N(0,1)\). Let \(\Phi(z)=P(Z<z)\).
	
	This section trains students to compute standard-normal probabilities using the table.
	
	\textbf{Questions/Tasks.}
	\begin{enumerate}[label=\textbf{\arabic*)}]
		\item \TaskHyperref{subsec:standard-normal-table-problems-c5-1}{\textbf{(6) Direct and complement-rule probabilities.}}
		\begin{enumerate}[label=\textbf{\alph*)}]
			\item Find \(P(Z<1.24)\).
			\item Find \(P(Z>-0.67)\).
			\item Find \(P(Z>1.18)\).
			\item Find \(P(Z<-0.52)\).
			\item Find \(P(Z<0.36)\).
			\item Find \(P(Z>2.05)\).
			\item Find \(P(Z<-1.41)\).
			\item Find \(P(Z>-1.09)\).
			\item Find \(P(Z>0.73)\).
			\item Find \(P(Z<1.96)\).

		\item \TaskHyperref{subsec:standard-normal-table-problems-c5-2}{\textbf{(4) Interval probabilities.}}
		\begin{enumerate}[label=\textbf{\alph*)}]

	\subsection*{1. (6) Direct and complement-rule probabilities}
	\phantomsection
	\label{subsec:standard-normal-table-problems-c5-1}
	Find the following probabilities.
	
	\[
	\textbf{Problem 1}
	\qquad
	P(Z<1.24)
	\qquad
	P(Z>-0.67)
	\]
	
	\[
	P(Z<1.24)=\Phi(1.24)=0.8925
	\]
	
	\[
	P(Z>-0.67)=1-\Phi(-0.67)
	\]
	
	\[
	\Phi(-0.67)=1-\Phi(0.67)
	\]
	
	\[
	P(Z>-0.67)=1-\left(1-\Phi(0.67)\right)=\Phi(0.67)=0.7486
	\]
	
	\[
	\textbf{Problem 2}
	\qquad
	P(Z>1.18)
	\qquad
	P(Z<-0.52)
	\]
	
	\[
	P(Z>1.18)=1-\Phi(1.18)=1-0.8810=0.1190
	\]
	
	\[
	\Phi(-x)=1-\Phi(x)
	\]
	
	\[
	P(Z<-0.52)=\Phi(-0.52)=1-\Phi(0.52)=1-0.6985=0.3015
	\]
	
	\[
	\textbf{Problem 3}
	\qquad
	P(Z<0.36)
	\qquad
	P(Z>2.05)
	\]
	
	\[
	P(Z<0.36)=\Phi(0.36)=0.6406
	\]
	
	\[
	P(Z>2.05)=1-\Phi(2.05)=1-0.9798=0.0202
	\]
	
	\[
	\textbf{Problem 4}
	\qquad
	P(Z<-1.41)
	\qquad
	P(Z>-1.09)
	\]
	
	\[
	P(Z<-1.41)=\Phi(-1.41)=1-\Phi(1.41)=1-0.9207=0.0793
	\]
	
	\[
	P(Z>-1.09)=1-\Phi(-1.09)
	\]
	
	\[
	\Phi(-1.09)=1-\Phi(1.09)
	\]
	
	\[
	P(Z>-1.09)=1-\left(1-\Phi(1.09)\right)=\Phi(1.09)=0.8621
	\]
	
	\[
	\textbf{Problem 5}
	\qquad
	P(Z>0.73)
	\qquad
	P(Z<1.96)
	\]
	
	\[
	P(Z>0.73)=1-\Phi(0.73)=1-0.7673=0.2327
	\]
	
	\[
	P(Z<1.96)=\Phi(1.96)=0.9750
	\]
	
	\[
	\textbf{Problem 6}
	\qquad
	P(Z<-0.88)
	\qquad
	P(Z>-2.17)
	\]
	
	\[
	P(Z<-0.88)=\Phi(-0.88)=1-\Phi(0.88)=1-0.8106=0.1894
	\]
	
	\[
	P(Z>-2.17)=1-\Phi(-2.17)
	\]
	
	\[
	\Phi(-2.17)=1-\Phi(2.17)
	\]
	
	\[
	P(Z>-2.17)=1-\left(1-\Phi(2.17)\right)=\Phi(2.17)=0.9850
	\]

	\subsection*{3. (2) Tail-union probabilities}
	\phantomsection
	\label{subsec:standard-normal-table-problems-c5-3}
	Find the following tail-union probabilities.
	
	\[
	\textbf{Problem 9}
	\qquad
	P(Z<-1.25\text{ or }Z>1.80)
	\qquad
	P(Z<0.45\text{ or }Z>1.70)
	\]
	
	\[
	P(Z<-1.25\text{ or }Z>1.80)=P(Z<-1.25)+P(Z>1.80)
	\]
	
	\[
	P(Z<-1.25)=\Phi(-1.25)=1-\Phi(1.25)=1-0.8944=0.1056
	\]
	
	\[
	P(Z>1.80)=1-\Phi(1.80)=1-0.9641=0.0359
	\]
	
	\[
	P(Z<-1.25\text{ or }Z>1.80)=0.1056+0.0359=0.1415
	\]
	
	\[
	P(Z<0.45\text{ or }Z>1.70)=P(Z<0.45)+P(Z>1.70)
	\]
	
	\[
	P(Z<0.45)=\Phi(0.45)=0.6736
	\]
	
	\[
	P(Z>1.70)=1-\Phi(1.70)=1-0.9554=0.0446
	\]
	
	\[
	P(Z<0.45\text{ or }Z>1.70)=0.6736+0.0446=0.7182
	\]
	
	\[
	\textbf{Problem 10}
	\qquad
	P(Z<-2.00\text{ or }Z>-0.60)
	\qquad
	P(Z<-0.30\text{ or }Z>2.25)
	\]
	
	\[
	P(Z<-2.00\text{ or }Z>-0.60)=P(Z<-2.00)+P(Z>-0.60)
	\]
	
	\[
	P(Z<-2.00)=\Phi(-2.00)=1-\Phi(2.00)=1-0.9772=0.0228
	\]
	
	\[
	P(Z>-0.60)=1-\Phi(-0.60)
	\]
	
	\[
	\Phi(-0.60)=1-\Phi(0.60)=1-0.7257=0.2743
	\]
	
	\[
	P(Z>-0.60)=1-0.2743=0.7257
	\]
	
	\[
	P(Z<-2.00\text{ or }Z>-0.60)=0.0228+0.7257=0.7485
	\]
	
	\[
	P(Z<-0.30\text{ or }Z>2.25)=P(Z<-0.30)+P(Z>2.25)
	\]
	
	\[
	P(Z<-0.30)=\Phi(-0.30)=1-\Phi(0.30)=1-0.6179=0.3821
	\]
	
	\[
	P(Z>2.25)=1-\Phi(2.25)=1-0.9878=0.0122
	\]
	
	\[
	P(Z<-0.30\text{ or }Z>2.25)=0.3821+0.0122=0.3943
	\]

	\subsection*{1. (1) Exam scores}
	\phantomsection
	\label{subsec:contextual-normal-distribution-problems-c6-1}
	Exam scores follow
	\[
	X\sim N(75,10^2)
	\]
	
	\textbf{First probability.} Find \(P(X<87.4)\).
	\[
	P(X<87.4)=P\left(Z<\frac{87.4-75}{10}\right)

	\]
	\[
	P(X<87.4)=\Phi(1.24)=0.8925
	\]
	
	\textbf{Second probability.} Find \(P(X>68.3)\).
	\[

	=P(Z>-0.67)
	\]
	\[
	P(X>68.3)=1-\Phi(-0.67)=1-0.2514=0.7486
	\]
	
	This uses the left-tail and complement-rule logic from C5.
