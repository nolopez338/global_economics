\section*{General Problem}

\textbf{Problem.}
Let $X$ be a continuous random variable with bounded support $[0,8]$. A piecewise linear model for its density is
\[
f(x)=\begin{cases}
kx, & 0\le x\le 2,\\
2k, & 2\le x\le 5,\\
\dfrac{2k}{3}(8-x), & 5\le x\le 8,\\
0, & \text{otherwise.}
\end{cases}
\]
This graph combines common linear-PDF features used in this activity (increasing segment, constant segment, and decreasing segment).

\textbf{Question.}
(a) Determine the parameter $k$ so that $f$ is a valid PDF. (b) Justify that the resulting function is a valid density on its support. (c) Use geometric areas to compute $P(1\le X\le 7)$.

\textbf{Solution.}
\emph{Step 1: Use total area $=1$ to determine the parameter.}
The graph on $[0,8]$ splits into three basic shapes:
\begin{itemize}
  \item left triangle on $[0,2]$: base $2$, height $2k$, area $\frac12(2)(2k)=2k$;
  \item middle rectangle on $[2,5]$: width $3$, height $2k$, area $3(2k)=6k$;
  \item right triangle on $[5,8]$: base $3$, height $2k$, area $\frac12(3)(2k)=3k$.
\end{itemize}
So
\[
\int_0^8 f(x)\,dx=2k+6k+3k=11k=1\quad\Rightarrow\quad k=\frac1{11}.
\]

\emph{Step 2: Check validity as a PDF.}
With $k=\frac1{11}$,
\[
f(x)=\begin{cases}
\dfrac{x}{11}, & 0\le x\le 2,\\
\dfrac{2}{11}, & 2\le x\le 5,\\
\dfrac{2}{33}(8-x), & 5\le x\le 8,\\
0, & \text{otherwise.}
\end{cases}
\]
Each piece is nonnegative on its interval, and the total area is $1$, so this is a valid PDF.

\emph{Step 3: Compute interval probability by geometric subareas.}
Use complement areas:
\[
P(1\le X\le 7)=1-P(0\le X\le 1)-P(7\le X\le 8).
\]
On $[0,1]$, area is a triangle with base $1$ and height $f(1)=\frac1{11}$:
\[
P(0\le X\le 1)=\frac12(1)\left(\frac1{11}\right)=\frac1{22}.
\]
On $[7,8]$, area is a triangle with base $1$ and height $f(7)=\frac{2}{33}$:
\[
P(7\le X\le 8)=\frac12(1)\left(\frac{2}{33}\right)=\frac1{33}.
\]
Therefore
\[
P(1\le X\le 7)=1-\frac1{22}-\frac1{33}=1-\frac5{66}=\frac{61}{66}\approx 0.924.
\]
\PiecewiseProbGraph{0}{8}{x <= 2 ? x/11 : (x <= 5 ? 2/11 : (2.0/33)*(8-x))}{1}{7}{0,1,2,5,7,8}

\textbf{Interpretation.}
This is the template method for piecewise linear density problems: verify nonnegativity on the bounded support, determine any parameter from total area $1$, then compute event probabilities as geometric areas under the graph.

\section{Phone Battery Life Ramp-Up}

\textbf{Problem.}
A phone repair shop models battery lifetime $X$ (hours) on $[0,8]$ with a linear density
\[
f(x)=a+bx,\quad 0\le x\le 8,
\]
and $f(0)=0$.

\textbf{Question.}
(a) Determine $a$ and $b$ and write $f(x)$ explicitly. (b) Compute $P(X\le 4)$. (c) Interpret the result.

\textbf{Solution.}
Because $f(0)=0$, we get $a=0$, so the graph is an increasing triangle: $f(x)=bx$ on $[0,8]$.
The full triangle has base $8$ and height $f(8)=8b$, so
\[
\frac{1}{2}(8)(8b)=1\Rightarrow 32b=1\Rightarrow b=\frac{1}{32}.
\]
Hence
\[
f(x)=\frac{x}{32},\quad 0\le x\le 8.
\]
Now $P(X\le 4)$ is the area of the smaller triangle from $0$ to $4$.
Its height is $f(4)=\frac{4}{32}=\frac18$, so
\[
P(X\le 4)=\frac12(4)\left(\frac18\right)=\frac14=0.25.
\]
\LinearProbGraph{0}{8}{x/32}{0}{4}{0,4,8}

\textbf{Interpretation.}
There is a $\frac14$ chance (25\%) that a battery lasts at most 4 hours.

\section{Delivery Weight Taper}

\textbf{Problem.}
A courier company models parcel weight $W$ (kg) on $[0,6]$ by
\[
f(w)=a-bw,\quad 0\le w\le 6,
\]
with $f(6)=0$.

\textbf{Question.}
(a) Find $a$ and $b$. (b) Compute $P(2\le W\le 5)$. (c) Interpret the result.

\textbf{Solution.}
Since $f(6)=0$, $a-6b=0\Rightarrow a=6b$.
The graph is a decreasing triangle with base $6$ and height $a$, so
\[
\frac12(6)(a)=1\Rightarrow 3a=1\Rightarrow a=\frac13.
\]
Then $b=\frac{a}{6}=\frac{1}{18}$, so
\[
f(w)=\frac13-\frac{w}{18}.
\]
For $P(2\le W\le 5)$, use trapezoid area on $[2,5]$.
Heights: $f(2)=\frac{2}{9}$ and $f(5)=\frac{1}{18}$, width $=3$:
\[
P(2\le W\le 5)=\frac{\left(\frac{2}{9}+\frac{1}{18}\right)}{2}\cdot 3
=\frac{\frac{5}{18}}{2}\cdot 3=\frac{5}{12}\approx 0.417.
\]
\LinearProbGraph{0}{6}{1/3 - x/18}{2}{5}{0,2,5,6}

\textbf{Interpretation.}
About $\frac{5}{12}$ of parcels are expected to weigh between 2 kg and 5 kg.

\section{Sensor Drift Build-Up}

\textbf{Problem.}
A lab models sensor drift $D$ (units) on $[0,10]$ with
\[
f(d)=kd,\quad 0\le d\le 10.
\]

\textbf{Question.}
(a) Determine $k$. (b) Compute $P(6\le D\le 10)$. (c) Interpret the probability.

\textbf{Solution.}
The graph is an increasing triangle from $(0,0)$ to $(10,10k)$.
Total area is
\[
\frac12(10)(10k)=50k=1\Rightarrow k=\frac{1}{50}.
\]
So $f(d)=\frac{d}{50}$.
Now $P(6\le D\le 10)$ is a trapezoid with width $4$ and heights
\[
f(6)=\frac{6}{50}=\frac{3}{25},\qquad f(10)=\frac{10}{50}=\frac15.
\]
Therefore
\[
P(6\le D\le 10)=\frac{\left(\frac{3}{25}+\frac15\right)}{2}\cdot 4
=\frac{\frac{8}{25}}{2}\cdot 4=\frac{16}{25}=0.64.
\]
\LinearProbGraph{0}{10}{x/50}{6}{10}{0,6,10}

\textbf{Interpretation.}
There is a 64\% chance the drift is between 6 and 10 units, so larger drift values are more common.

\section{Charging Time Drop-Off}

\textbf{Problem.}
Charging time $T$ (minutes) for quick top-ups is modeled on $[0,5]$ by
\[
f(t)=a-bt,\quad 0\le t\le 5,
\]
with $f(5)=0$.

\textbf{Question.}
(a) Find $a$ and $b$. (b) Compute $P(T<2)$. (c) Interpret the result.

\textbf{Solution.}
From $f(5)=0$, $a-5b=0\Rightarrow a=5b$.
The graph is a decreasing triangle with base $5$ and height $a$.
Using area $=1$:
\[
\frac12(5)(a)=1\Rightarrow \frac{5a}{2}=1\Rightarrow a=\frac{2}{5}.
\]
Then $b=\frac{a}{5}=\frac{2}{25}$, hence
\[
f(t)=\frac25-\frac{2}{25}t.
\]
Now $P(T<2)$ is a trapezoid on $[0,2]$ with heights
\[
f(0)=\frac25,\qquad f(2)=\frac25-\frac{4}{25}=\frac{6}{25},
\]
width $2$:
\[
P(T<2)=\frac{\left(\frac25+\frac{6}{25}\right)}{2}\cdot 2
=\frac{16}{25}=0.64.
\]
\LinearProbGraph{0}{5}{2/5 - 2*x/25}{0}{2}{0,2,5}

\textbf{Interpretation.}
Short charging times (under 2 minutes in this model) occur with probability $\frac{16}{25}$.

\section{Commute Delay with Nonzero Ends}

\textbf{Problem.}
Commute delay $C$ (minutes) is modeled on $[0,10]$ by a linear density
\[
f(c)=a+bc,\quad 0\le c\le 10,
\]
with endpoint heights $f(0)=\frac{1}{20}$ and $f(10)=\frac{3}{20}$.

\textbf{Question.}
(a) Find $a$ and $b$. (b) Compute $P(2\le C\le 6)$. (c) Interpret the result.

\textbf{Solution.}
From $f(0)=\frac{1}{20}$, we get $a=\frac{1}{20}$.
Using $f(10)=\frac{3}{20}$:
\[
\frac{1}{20}+10b=\frac{3}{20}\Rightarrow 10b=\frac{2}{20}=\frac{1}{10}\Rightarrow b=\frac{1}{100}.
\]
So
\[
f(c)=\frac{1}{20}+\frac{c}{100}.
\]
The full graph is a trapezoid (nonzero endpoints) with heights $\frac{1}{20}$ and $\frac{3}{20}$, width $10$:
\[
\frac{\left(\frac{1}{20}+\frac{3}{20}\right)}{2}\cdot 10=1,
\]
so it is valid.
For $[2,6]$, heights are
\[
f(2)=\frac{7}{100},\qquad f(6)=\frac{11}{100},
\]
width $4$:
\[
P(2\le C\le 6)=\frac{\left(\frac{7}{100}+\frac{11}{100}\right)}{2}\cdot 4
=\frac{18}{100}\cdot 2=\frac{9}{25}=0.36.
\]
\LinearProbGraph{0}{10}{1/20 + x/100}{2}{6}{0,2,6,10}

\textbf{Interpretation.}
A delay between 2 and 6 minutes has probability $\frac{9}{25}$.

\section{App Session Fade-Out}

\textbf{Problem.}
An app's session time $S$ (minutes) is modeled on $[0,12]$ by
\[
f(s)=k(12-s),\quad 0\le s\le 12.
\]

\textbf{Question.}
(a) Determine $k$. (b) Compare $P(0\le S\le 3)$ and $P(9\le S\le 12)$. (c) Interpret.

\textbf{Solution.}
This is a decreasing triangle ending at 0 when $s=12$.
Base is $12$, height is $f(0)=12k$:
\[
\frac12(12)(12k)=72k=1\Rightarrow k=\frac{1}{72}.
\]
So
\[
f(s)=\frac{12-s}{72}.
\]
For $[0,3]$, heights are $f(0)=\frac16$ and $f(3)=\frac18$, width $3$:
\[
P(0\le S\le 3)=\frac{\left(\frac16+\frac18\right)}{2}\cdot 3
=\frac{7}{16}.
\]
\LinearProbGraph{0}{12}{(12 - x)/72}{0}{3}{0,3,9,12}
For $[9,12]$, heights are $f(9)=\frac{1}{24}$ and $f(12)=0$, width $3$:
\[
P(9\le S\le 12)=\frac{\left(\frac{1}{24}+0\right)}{2}\cdot 3
=\frac{1}{16}.
\]
\LinearProbGraph{0}{12}{(12 - x)/72}{9}{12}{0,3,9,12}
Hence $P(0\le S\le 3)>P(9\le S\le 12)$.

\textbf{Interpretation.}
Very short sessions are much more likely than very long sessions in this model.

\section{Streaming Buffer Time with Baseline}

\textbf{Problem.}
A streaming app models buffer wait time $B$ (seconds) on $[0,8]$ with
\[
f(b)=a-b_1 b,\quad 0\le b\le 8,
\]
and endpoint conditions $f(0)=\frac14$, $f(8)=\frac{1}{20}$.

\textbf{Question.}
(a) Determine $a$ and $b_1$. (b) Find $P(2\le B\le 6)$. (c) Interpret the result.

\textbf{Solution.}
From $f(0)=\frac14$, $a=\frac14$.
Using $f(8)=\frac{1}{20}$:
\[
\frac14-8b_1=\frac{1}{20}\Rightarrow 8b_1=\frac14-\frac{1}{20}=\frac15
\Rightarrow b_1=\frac{1}{40}.
\]
Thus
\[
f(b)=\frac14-\frac{b}{40}.
\]
The full graph is a trapezoid with heights $\frac14$ and $\frac{1}{20}$ and width $8$:
\[
\frac{\left(\frac14+\frac{1}{20}\right)}{2}\cdot 8=1.
\]
For $[2,6]$, heights are
\[
f(2)=\frac15,\qquad f(6)=\frac{1}{10},
\]
width $4$:
\[
P(2\le B\le 6)=\frac{\left(\frac15+\frac{1}{10}\right)}{2}\cdot 4
=\frac{3}{20}\cdot 4=\frac{3}{5}=0.60.
\]
\LinearProbGraph{0}{8}{1/4 - x/40}{2}{6}{0,2,6,8}

\textbf{Interpretation.}
There is a 60\% chance that buffering lasts between 2 and 6 seconds.

\section{Packaging Fill Level with Two Linear Segments}

\textbf{Problem.}
A filling machine outputs fill level $F$ (litres) on $[0,8]$ with piecewise linear density
\[
f(x)=\begin{cases}
\dfrac{x}{16}, & 0\le x\le 4,\\
\dfrac{8-x}{16}, & 4\le x\le 8.
\end{cases}
\]

\textbf{Question.}
(a) Verify it is a valid density using area geometry. (b) Compute $P(2\le F\le 6)$. (c) Interpret.

\textbf{Solution.}
The graph is a triangle made from two linear segments, peaking at $x=4$ with height $f(4)=\frac14$.
Total area:
\[
\frac12(8)\left(\frac14\right)=1,
\]
so it is a valid density.
For $P(2\le F\le 6)$, use symmetry or subtract side triangles.
The full area is 1. The left excluded triangle on $[0,2]$ has base $2$, height $f(2)=\frac18$:
\[
A_L=\frac12(2)\left(\frac18\right)=\frac18.
\]
The right excluded triangle on $[6,8]$ has base $2$, height $f(6)=\frac18$:
\[
A_R=\frac12(2)\left(\frac18\right)=\frac18.
\]
Hence
\[
P(2\le F\le 6)=1-\frac18-\frac18=\frac34=0.75.
\]
\PiecewiseProbGraph{0}{8}{x <= 4 ? x/16 : (8-x)/16}{2}{6}{0,2,4,6,8}

\textbf{Interpretation.}
There is a 75\% chance the fill level is between 2 L and 6 L.

\section{Drone Delivery Delay with Uneven Peak}

\textbf{Problem.}
Drone delay $Z$ (minutes) is modeled on $[0,8]$ by
\[
f(z)=\begin{cases}
kz, & 0\le z\le 4,\\
\dfrac{k}{2}(8-z), & 4\le z\le 8.
\end{cases}
\]

\textbf{Question.}
(a) Determine $k$. (b) Find $P(2\le Z\le 6)$. (c) Interpret the probability.

\textbf{Solution.}
The graph is piecewise linear: rising from 0 to $4k$ at $z=4$, then falling to 0 at $z=8$ with a gentler slope.
Use area of two triangles:
\[
\text{Area on }[0,4]=\frac12(4)(4k)=8k,
\]
\[
\text{Area on }[4,8]=\frac12(4)(2k)=4k.
\]
Total area $8k+4k=12k=1$, so
\[
k=\frac{1}{12}.
\]
Now compute $P(2\le Z\le 6)$ by splitting the interval.
On $[2,4]$, heights are
\[
f(2)=\frac{1}{6},\qquad f(4)=\frac{1}{3},
\]
width $2$, so
\[
A_1=\frac{\left(\frac16+\frac13\right)}{2}\cdot 2=\frac12.
\]
On $[4,6]$, heights are
\[
f(4)=\frac{1}{6},\qquad f(6)=\frac{1}{12},
\]
width $2$, so
\[
A_2=\frac{\left(\frac16+\frac{1}{12}\right)}{2}\cdot 2=\frac14.
\]
Therefore
\[
P(2\le Z\le 6)=A_1+A_2=\frac34=0.75.
\]
\PiecewiseProbGraph{0}{8}{x <= 4 ? x/12 : (8-x)/24}{2}{6}{0,2,4,6,8}

\textbf{Interpretation.}
About 75\% of deliveries have delays between 2 and 6 minutes.

\section{Calibrated Piecewise Ramp with Parameter}

A random variable $Y$ on $[0,6]$ has density
\[
f(y)=\begin{cases}
ky, & 0\le y\le 2,\\

\end{cases}

(a) Determine $k$ so that $f$ is a valid PDF. (b) Then compute $P(1\le Y\le 5)$.

Compute total area as three simple shapes.
On $[0,2]$, triangle area is
\[
\frac12(2)(2k)=2k.
\]
On $[2,4]$, rectangle area is
\[
(2)(2k)=4k.
\]
On $[4,6]$, triangle area is
\[
\frac12(2)(2k)=2k.
\]

\[
k=\frac18.
\]
Now evaluate $P(1\le Y\le 5)$ by pieces.
From $1$ to $2$: heights $f(1)=\frac18$, $f(2)=\frac14$, width $1$:
\[
A_1=\frac{\left(\frac18+\frac14\right)}{2}\cdot 1=\frac{3}{16}.
\]
From $2$ to $4$: rectangle area
\[
A_2=2\cdot\frac14=\frac12.
\]
From $4$ to $5$: heights $f(4)=\frac14$, $f(5)=\frac18$, width $1$:
\[
A_3=\frac{\left(\frac14+\frac18\right)}{2}\cdot 1=\frac{3}{16}.
\]
Therefore
\[
P(1\le Y\le 5)=A_1+A_2+A_3=\frac{3}{16}+\frac12+\frac{3}{16}=\frac78=0.875.
\]
\PiecewiseProbGraph{0}{6}{x <= 2 ? x/8 : (x <= 4 ? 1/4 : (6-x)/8)}{1}{5}{0,1,2,4,5,6}

\textbf{Interpretation.}
Most of the mass lies in the central part; the probability from 1 to 5 is $\frac78$.

\section{Multi-Breakpoint Service-Time Density}

\textbf{Problem.}
Service time $T$ (minutes) on $[0,6]$ is modeled by
\[
f(t)=\begin{cases}
\dfrac{t}{6}, & 0\le t\le 1,\\
\dfrac{t+1}{12}, & 1\le t\le 3,\\
\dfrac{5-t}{6}, & 3\le t\le 4,\\
\dfrac{6-t}{12}, & 4\le t\le 6.
\end{cases}
\]

\textbf{Question.}
(a) Verify this is a valid PDF. (b) Compute $P\!\left(\frac12\le T\le \frac{9}{2}\right)$.

\textbf{Solution.}
Check area piece by piece.
On $[0,1]$, triangle with height $\frac16$ gives area $\frac{1}{12}$.
On $[1,3]$, trapezoid with heights $\frac16$ and $\frac13$, width $2$:
\[
A_2=\frac{\left(\frac16+\frac13\right)}{2}\cdot 2=\frac12.
\]
On $[3,4]$, trapezoid with heights $\frac13$ and $\frac16$, width $1$:
\[
A_3=\frac{\left(\frac13+\frac16\right)}{2}\cdot 1=\frac14.
\]
On $[4,6]$, triangle with base $2$, height $\frac16$ gives area $\frac16$.
Total area:
\[
\frac{1}{12}+\frac12+\frac14+\frac16=1,
\]
so it is a valid PDF.
Now compute $P\!\left(\frac12\le T\le \frac{9}{2}\right)$ as four subareas.
\[
A_{[1/2,1]}=\frac{\left(f\!\left(\frac12\right)+f(1)\right)}{2}\cdot\frac12
=\frac{\left(\frac{1}{12}+\frac16\right)}{2}\cdot\frac12=\frac{1}{16}.
\]
\[
A_{[1,3]}=\frac12.
\]
\[
A_{[3,4]}=\frac14.
\]
On $\left[4,\frac{9}{2}\right]$, heights are $f(4)=\frac16$ and $f\!\left(\frac{9}{2}\right)=\frac18$, width $\frac12$:
\[
A_{[4,9/2]}=\frac{\left(\frac16+\frac18\right)}{2}\cdot\frac12=\frac{7}{96}.
\]
Therefore
\[
P\!\left(\frac12\le T\le \frac{9}{2}\right)=\frac{1}{16}+\frac12+\frac14+\frac{7}{96}
=\frac{85}{96}\approx 0.885.
\]
\PiecewiseProbGraph{0}{6}{x <= 1 ? x/6 : (x <= 3 ? (x+1)/12 : (x <= 4 ? (5-x)/6 : (6-x)/12))}{0.5}{4.5}{0,0.5,1,3,4,4.5,6}

\textbf{Interpretation.}
The interval from $0.5$ to $4.5$ minutes captures most outcomes in this model.

\section{Full Validation and Probability Across Several Segments}

\textbf{Problem.}
A bounded model for response time $R$ (seconds) on $[0,10]$ is
\[
f(r)=\begin{cases}
kr, & 0\le r\le 2,\\
2k, & 2\le r\le 5,\\
2k-\dfrac{k}{2}(r-5), & 5\le r\le 7,\\
\dfrac{k}{3}(10-r), & 7\le r\le 10.
\end{cases}
\]

\textbf{Question.}
(a) Determine $k$ so that $f$ is a valid PDF and verify nonnegativity on $[0,10]$. (b) Compute $P(1\le R\le 8)$.

\textbf{Solution.}
All pieces are nonnegative on their stated intervals when $k>0$.
Find $k$ from total area $=1$.
\[
A_{[0,2]}=\frac12(2)(2k)=2k,
\quad
A_{[2,5]}=(3)(2k)=6k,
\]
\[
A_{[5,7]}=\frac{(2k+k)}{2}\cdot 2=3k,
\quad
A_{[7,10]}=\frac{(k+0)}{2}\cdot 3=\frac{3k}{2}.
\]
So
\[
2k+6k+3k+\frac{3k}{2}=\frac{25k}{2}=1
\Rightarrow
k=\frac{2}{25}.
\]
Now compute $P(1\le R\le 8)$.
On $[1,2]$: heights $f(1)=\frac{2}{25}$, $f(2)=\frac{4}{25}$, width $1$:
\[
A_1=\frac{\left(\frac{2}{25}+\frac{4}{25}\right)}{2}\cdot 1=\frac{3}{25}.
\]
On $[2,5]$: rectangle area
\[
A_2=3\cdot\frac{4}{25}=\frac{12}{25}.
\]
On $[5,7]$: trapezoid area
\[
A_3=\frac{\left(\frac{4}{25}+\frac{2}{25}\right)}{2}\cdot 2=\frac{6}{25}.
\]
On $[7,8]$: heights $f(7)=\frac{2}{25}$ and $f(8)=\frac{4}{75}$, width $1$:
\[
A_4=\frac{\left(\frac{2}{25}+\frac{4}{75}\right)}{2}\cdot 1=\frac{1}{15}.
\]
Hence
\[
P(1\le R\le 8)=A_1+A_2+A_3+A_4
=\frac{3}{25}+\frac{12}{25}+\frac{6}{25}+\frac{1}{15}
=\frac{68}{75}\approx 0.907.
\]
\PiecewiseProbGraph{0}{10}{x <= 2 ? 2*x/25 : (x <= 5 ? 4/25 : (x <= 7 ? 4/25 - (x-5)/25 : 2*(10-x)/75))}{1}{8}{0,1,2,5,7,8,10}

\textbf{Interpretation.}
The interval from 1 to 8 seconds contains most of the probability mass in this calibrated model.
