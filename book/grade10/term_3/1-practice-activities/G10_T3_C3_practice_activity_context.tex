\section{Problem 1: Rising Export-Order Size Density}

\textbf{Problem.}
A port authority tracks the number of export containers booked each day on a short-sea corridor, measured in tens of containers. Let $X$ denote this daily booked size. Because the lane has fixed vessel slots, values below $0$ and above $6$ tens are impossible, so $0\le x\le 6$. Small bookings are less common than near-capacity bookings during the current contract season, so the density rises linearly across the interval:
\[
f(x)=\frac{x}{18},\quad 0\le x\le 6,
\]
and $f(x)=0$ otherwise.

\textbf{Questions/Tasks.}
\begin{enumerate}[label=(\alph*)]
\item State the support of $X$.
\item Verify geometrically that $f$ is a valid PDF.
\item Compute $P(0\le X\le 3)$.
\end{enumerate}

\subsection*{C3}
(a) The support is $[0,6]$.

(b) The graph is a single straight line from $(0,0)$ to $(6,\tfrac13)$.
So total area is triangle area:
\[
\frac12(6)\left(\frac13\right)=1.
\]
Hence $f$ is valid.

(c) On $[0,3]$, the area is a smaller triangle with height $f(3)=\frac16$:
\[
P(0\le X\le 3)=\frac12(3)\left(\frac16\right)=\frac14.
\]
\LinearProbGraph{0}{6}{x/18}{0}{3}{0,3,6}

\textbf{Interpretation.}
As export batches approach the upper support limit, they become more likely than very small dispatches, matching a ramp-up trade pattern.

\section{Problem 2: Symmetric Customs-Clearance Time Density}

\textbf{Problem.}
A customs agency records end-to-end clearance time for standard-document shipments on a priority trade lane. Let $Y$ be clearance time in hours. Operational rules guarantee completion between $0$ and $8$ hours for this lane, so the support is $[0,8]$. Cases around $4$ hours are most common because that is the usual queue-and-review cycle, while both unusually fast and unusually slow cases are rarer, giving a symmetric triangular profile:
\[
f(y)=\begin{cases}
\dfrac{y}{16}, & 0\le y\le 4,\\
\dfrac{8-y}{16}, & 4\le y\le 8,\\
0, & \text{otherwise.}
\end{cases}
\]

\textbf{Questions/Tasks.}
\begin{enumerate}[label=(\alph*)]
\item Verify this is a valid PDF.
\item Compute $P(2\le Y\le 6)$.
\item Identify the most likely region and explain why.
\end{enumerate}

\subsection*{C3}
(a) The graph is a triangle with base $8$ and peak height $f(4)=\frac14$.
\[
\text{Area}=\frac12(8)\left(\frac14\right)=1.
\]
So it is valid.

(b) Use complement:
\[
P(2\le Y\le 6)=1-P(0\le Y\le 2)-P(6\le Y\le 8).
\]
Each side area is a triangle with base $2$ and height $\frac18$:
\[
\frac12(2)\left(\frac18\right)=\frac18.
\]
Hence
\[
P(2\le Y\le 6)=1-\frac18-\frac18=\frac34.
\]
\PiecewiseProbGraph{0}{8}{x <= 4 ? x/16 : (8-x)/16}{2}{6}{0,2,4,6,8}

\textbf{Interpretation.}
Clearance durations near $y=4$ hours are most likely, while very short and very long cases are less likely.

\section{Problem 3: Asymmetric Port-Delay Density}

\textbf{Problem.}
A grain-import terminal measures arrival delay relative to schedule for inbound commodity vessels. Let $Z$ be this delay in days. For this contract set, delays are capped at $10$ days by diversion policy, and negative delay is not recorded, so $0\le z\le 10$. Delays near $4$ days are most frequent (typical berth congestion), but the right tail is longer because weather and inspection disruptions can extend delays beyond the mode:
\[
f(z)=\begin{cases}
\dfrac{z}{20}, & 0\le z\le 4,\\
\dfrac{10-z}{30}, & 4\le z\le 10,\\
0, & \text{otherwise.}
\end{cases}
\]

\textbf{Questions/Tasks.}
\begin{enumerate}[label=(\alph*)]
\item Verify the model is a valid PDF.
\item Compute $P(1\le Z\le 7)$.
\item Where is $Z$ most likely to be found?
\end{enumerate}

\subsection*{C3}
(a) Peak at $z=4$ has height $\frac15$.
Left triangle area: $\frac12(4)(\frac15)=\frac25$.
Right triangle area: $\frac12(6)(\frac15)=\frac35$.
Total $=1$.

(b) Compute by complement:
\[
P(1\le Z\le 7)=1-P(0\le Z\le 1)-P(7\le Z\le 10).
\]
\[
P(0\le Z\le 1)=\frac12(1)\left(\frac1{20}\right)=\frac1{40}.
\]
At $z=7$, $f(7)=\frac{1}{10}$, so
\[
P(7\le Z\le 10)=\frac12(3)\left(\frac{1}{10}\right)=\frac{3}{20}.
\]
Therefore
\[
P(1\le Z\le 7)=1-\frac1{40}-\frac3{20}=\frac{33}{40}=0.825.
\]
\PiecewiseProbGraph{0}{10}{x <= 4 ? x/20 : (10-x)/30}{1}{7}{0,1,4,7,10}

\textbf{Interpretation.}
The most likely delay is around $z=4$ days, but the longer right side reflects occasional extended disruptions.

\section{Problem 4: Triangular Density for Regional Freight Costs}

\textbf{Problem.}
A freight forwarder publishes a normalized surcharge index for regional shipping contracts, where $0$ means no surcharge and $6$ is the contractual cap allowed this quarter. Let $T$ be that index. Mid-range surcharge conditions near $3$ are most common when fuel and capacity pressures are balanced, while both very low and very high surcharge regimes are less frequent, producing:
\[
f(t)=\begin{cases}
\dfrac{t}{9}, & 0\le t\le 3,\\
\dfrac{6-t}{9}, & 3\le t\le 6,\\
0, & \text{otherwise.}
\end{cases}
\]

\textbf{Questions/Tasks.}
\begin{enumerate}[label=(\alph*)]
\item Verify validity.
\item Compute $P(2\le T\le 5)$ by splitting at the peak.
\item Compare $P(1\le T\le 2)$ and $P(4\le T\le 5)$.
\end{enumerate}

\subsection*{C3}
(a) Triangle base $6$, height $\frac13$ gives area
\[
\frac12(6)\left(\frac13\right)=1.
\]

(b) Split $[2,5]$ into $[2,3]$ and $[3,5]$.
\[
A_{[2,3]}=\frac{\left(\frac29+\frac13\right)}{2}\cdot1=\frac{5}{18},
\quad
A_{[3,5]}=\frac{\left(\frac13+\frac19\right)}{2}\cdot2=\frac{4}{9}.
\]
So
\[
P(2\le T\le 5)=\frac{5}{18}+\frac{4}{9}=\frac{13}{18}.
\]

(c) By symmetry around $t=3$, these two one-unit intervals have equal probability.
\PiecewiseProbGraph{0}{6}{x <= 3 ? x/9 : (6-x)/9}{2}{5}{0,1,2,3,4,5,6}

\textbf{Interpretation.}
Mid-range surcharge levels around $t=3$ are most probable, consistent with a central market-cost band.

\section{Problem 5: Piecewise Density for Import Lead Times}

\textbf{Problem.}
A manufacturer tracks lead time for imported semiconductor components, from purchase order to warehouse receipt. Let $W$ be lead time in weeks. For these suppliers, contractual service windows place outcomes between $0$ and $8$ weeks. Around $5$ weeks is most typical, but the decline toward $8$ is steeper because orders approaching that limit are expedited or rerouted, so very late arrivals become quickly less likely:
\[
f(w)=\begin{cases}
\dfrac{w}{20}, & 0\le w\le 5,\\
\dfrac{8-w}{12}, & 5\le w\le 8,\\
0, & \text{otherwise.}
\end{cases}
\]

\textbf{Questions/Tasks.}
\begin{enumerate}[label=(\alph*)]
\item Verify this is a valid PDF.
\item Compute $P(3\le W\le 7)$ by splitting at $w=5$.
\item Which is larger: $P(2\le W\le 4)$ or $P(6\le W\le 8)$? Explain from the graph.
\end{enumerate}

\subsection*{C3}
(a) Peak height is $f(5)=\frac14$.
\[
\text{Area}=\frac12(5)\left(\frac14\right)+\frac12(3)\left(\frac14\right)=\frac58+\frac38=1.
\]

(b)
\[
A_{[3,5]}=\frac{\left(\frac3{20}+\frac14\right)}{2}\cdot2=\frac25,
\quad
A_{[5,7]}=\frac{\left(\frac14+\frac1{12}\right)}{2}\cdot2=\frac13.
\]
Hence
\[
P(3\le W\le 7)=\frac25+\frac13=\frac{11}{15}.
\]

(c)
\[
P(2\le W\le 4)=\frac{\left(\frac1{10}+\frac15\right)}{2}\cdot2=\frac3{10},
\quad
P(6\le W\le 8)=\frac12(2)\left(\frac16\right)=\frac16.
\]
So $P(2\le W\le 4)>P(6\le W\le 8)$.
\PiecewiseProbGraph{0}{8}{x <= 5 ? x/20 : (8-x)/12}{3}{7}{0,2,3,4,5,6,7,8}

\textbf{Interpretation.}
Because the peak is near week 5 and the left side is wider, moderate lead times can carry more probability than late-arrival extremes.

\section{Problem 6: Unknown-Scale Density for Trade-Finance Processing}

\textbf{Problem.}
A multinational bank models the business-day turnaround for standard trade-finance applications after complete documentation is received. Let $X$ denote this processing time. Internal service policy keeps completion between $0$ and $6$ business days. The most common completion point is around day $3$, with equally plausible variation on either side because early/late deviations are driven by similar workflow factors, so:
\[
f(x)=\begin{cases}
kx, & 0\le x\le 3,\\
k(6-x), & 3\le x\le 6,\\
0, & \text{otherwise.}
\end{cases}
\]

\textbf{Questions/Tasks.}
\begin{enumerate}[label=(\alph*)]
\item Determine $k$.
\item Compute $P(2\le X\le 4)$.
\item Explain why the peak corresponds to the most likely region.
\end{enumerate}

\subsection*{C3}
(a) The graph is a symmetric triangle with base $6$ and height $3k$:
\[
\frac12(6)(3k)=9k=1\Rightarrow k=\frac19.
\]

(b) On $[2,3]$ and $[3,4]$:
\[
A_{[2,3]}=\frac{\left(\frac29+\frac13\right)}{2}\cdot1=\frac{5}{18},
\quad
A_{[3,4]}=\frac{\left(\frac13+\frac29\right)}{2}\cdot1=\frac{5}{18}.
\]
So
\[
P(2\le X\le 4)=\frac{5}{9}.
\]
\PiecewiseProbGraph{0}{6}{x <= 3 ? x/9 : (6-x)/9}{2}{4}{0,2,3,4,6}

\textbf{Interpretation.}
Processing durations around $x=3$ days form the most likely region in this symmetric workflow model.

\section{Problem 7: Unknown-Scale Asymmetric Density for Delivery Backlogs}

\textbf{Problem.}
A global courier estimates the time needed to clear a temporary cross-border parcel backlog. Let $Y$ be resolution time in days. Under contingency plans, the process finishes between $0$ and $10$ days. Around day $4$ is most likely under normal recovery staffing, but the right side extends farther because customs bottlenecks and flight-capacity shortages can prolong the clean-up period:
\[
f(y)=\begin{cases}
ky, & 0\le y\le 4,\\
\dfrac{2k}{3}(10-y), & 4\le y\le 10,\\
0, & \text{otherwise.}
\end{cases}
\]

\textbf{Questions/Tasks.}
\begin{enumerate}[label=(\alph*)]
\item Find $k$.
\item Compute $P(3\le Y\le 8)$.
\item Identify the most likely region from the shape.
\end{enumerate}

\subsection*{C3}
(a) Peak is $f(4)=4k$ (continuous at $4$).
Total area:
\[
\frac12(4)(4k)+\frac12(6)(4k)=8k+12k=20k=1\Rightarrow k=\frac1{20}.
\]

(b)
\[
A_{[3,4]}=\frac{\left(\frac3{20}+\frac15\right)}{2}\cdot1=\frac7{40}.
\]
For $[4,8]$, heights are $\frac15$ and $\frac1{15}$:
\[
A_{[4,8]}=\frac{\left(\frac15+\frac1{15}\right)}{2}\cdot4=\frac{8}{15}.
\]
Therefore
\[
P(3\le Y\le 8)=\frac7{40}+\frac{8}{15}=\frac{17}{24}.
\]
\PiecewiseProbGraph{0}{10}{x <= 4 ? x/20 : (10-x)/30}{3}{8}{0,3,4,8,10}

\textbf{Interpretation.}
The most likely backlog duration is near $y=4$, but the longer right side captures extended congestion periods.

\section{Problem 8: Three-Segment Density for Global Platform Transactions}

\textbf{Problem.}
A fintech platform models daily cross-border transaction volume using a scaled index $B$. Platform capacity and risk controls bound the index between $0$ and $12$. Volume usually builds up to a peak around $5$, remains high through the core settlement window up to about $9$, then falls more sharply toward $12$ as anti-fraud throttles and end-of-day cutoffs reduce extreme-volume days:
\[
f(b)=\begin{cases}
\dfrac{k}{5}b, & 0\le b\le 5,\\
k-\dfrac{k}{12}(b-5), & 5\le b\le 9,\\
\dfrac{2k}{9}(12-b), & 9\le b\le 12,\\
0, & \text{otherwise.}
\end{cases}
\]

\textbf{Questions/Tasks.}
\begin{enumerate}[label=(\alph*)]
\item Determine $k$.
\item Compute $P(4\le B\le 10)$.
\item Explain how the peak location affects where most mass sits.
\end{enumerate}

\subsection*{C3}
(a) Areas:
\[
A_{[0,5]}=\frac12(5)(k)=\frac{5k}{2},
\quad
A_{[5,9]}=\frac{(k+\frac{2k}{3})}{2}\cdot4=\frac{10k}{3},
\quad
A_{[9,12]}=\frac12(3)\left(\frac{2k}{3}\right)=k.
\]
So
\[
\frac{5k}{2}+\frac{10k}{3}+k=\frac{41k}{6}=1
\Rightarrow k=\frac{6}{41}.
\]

(b) Split into $[4,5]$, $[5,9]$, and $[9,10]$.
\[
A_{[4,5]}=\frac{\left(\frac{4k}{5}+k\right)}{2}\cdot1=\frac{9k}{10},
\quad
A_{[5,9]}=\frac{10k}{3},
\]
\[
A_{[9,10]}=\frac{\left(\frac{2k}{3}+\frac{4k}{9}\right)}{2}\cdot1=\frac{5k}{9}.
\]
Hence
\[
P(4\le B\le 10)=\frac{9k}{10}+\frac{10k}{3}+\frac{5k}{9}=\frac{431}{615}\approx0.701.
\]
\PiecewiseProbGraph{0}{12}{x <= 5 ? (6*x)/205 : (x <= 9 ? (6.0/41)*(1-(x-5)/12) : (4.0/123)*(12-x))}{4}{10}{0,4,5,9,10,12}

\textbf{Interpretation.}
The high-density middle range (about $5$ to $9$) represents the most probable transaction activity window.

\section{Problem 9: Comparing Two Trade-Route Throughput Models}

\textbf{Problem.}
An infrastructure planner compares two candidate probability models for a weekly throughput index on a rail-port corridor. The index is bounded between $0$ and $8$ by physical slot capacity. Model $A$ assumes the most likely week is near the center ($x=4$), while Model $B$ assumes peak throughput is later and higher ($x=6$) because dispatching improvements shift probability toward larger weekly outcomes:
\[
f_A(x)=\begin{cases}
\dfrac{x}{16}, & 0\le x\le 4,\\
\dfrac{8-x}{16}, & 4\le x\le 8,
\end{cases}
\qquad
f_B(x)=\begin{cases}
\dfrac{x}{24}, & 0\le x\le 6,\\
\dfrac{8-x}{8}, & 6\le x\le 8.
\end{cases}
\]

\textbf{Questions/Tasks.}
\begin{enumerate}[label=(\alph*)]
\item Show each is a valid PDF.
\item Compare $P_A(3\le X\le 5)$ and $P_B(3\le X\le 5)$.
\item Explain how moving the peak changes the most likely region.
\end{enumerate}

\subsection*{C3}
(a) Each has base $8$ and height $\frac14$, so each total area is
\[
\frac12(8)\left(\frac14\right)=1.
\]

(b) For model $A$:
\[
P_A(3\le X\le 5)=2\cdot\frac{\left(\frac{3}{16}+\frac14\right)}{2}\cdot1=\frac{7}{16}.
\]
For model $B$, on $[3,5]$ we are on the rising side only:
\[
P_B(3\le X\le 5)=\frac{\left(\frac18+\frac{5}{24}\right)}{2}\cdot2=\frac13.
\]
So $P_A(3\le X\le 5)>P_B(3\le X\le 5)$.

(c) Peak $4$ (model $A$) centers mass near the middle, while peak $6$ (model $B$) shifts more mass to larger values.
\PiecewiseProbGraph{0}{8}{x <= 4 ? x/16 : (8-x)/16}{3}{5}{0,3,4,5,8}

\textbf{Interpretation.}
Moving the peak to the right shifts likely throughput toward higher-index weeks and lowers mid-range likelihood.

\section{Problem 10: Variable Peak for Tariff-Adjusted Shipment Sizes}

\textbf{Problem.}
An export consortium uses a shipment-size index $X$ after tariff adjustments, where policy and vessel limits keep values between $0$ and $10$. The parameter $p$ marks the most likely shipment size under a given tariff regime: moving $p$ left or right reflects whether smaller or larger consignments are operationally preferred. For $0<p<10$, define
\[
f_p(x)=\begin{cases}
\dfrac{x}{5p}, & 0\le x\le p,\\
\dfrac{10-x}{5(10-p)}, & p\le x\le 10,\\
0, & \text{otherwise.}
\end{cases}
\]

\textbf{Questions/Tasks.}
\begin{enumerate}[label=(\alph*)]
\item Show $f_p$ is a valid PDF for any $0<p<10$.
\item Express $P(X\le p)$ in terms of $p$.
\item Explain how changing $p$ affects symmetry and which side carries more probability.
\end{enumerate}

\subsection*{C3}
(a) Peak value is always $f_p(p)=\frac15$.
Total area is triangle area with base $10$ and height $\frac15$:
\[
\frac12(10)\left(\frac15\right)=1.
\]

(b) Left side is a triangle with base $p$ and height $\frac15$:
\[
P(X\le p)=\frac12 p\left(\frac15\right)=\frac{p}{10}.
\]

(c) If $p=5$, the density is symmetric.
If $p<5$, the peak is left-shifted and more mass lies on the right.
If $p>5$, the peak is right-shifted and more mass lies on the left.
\PiecewiseProbGraph{0}{10}{x <= 6 ? x/30 : (10-x)/20}{0}{6}{0,3,5,6,8,10}

\textbf{Interpretation.}
The tariff-driven peak location determines whether smaller or larger shipment sizes capture more probability mass.

\section{Problem 11: Symbolic Height for Labor-Market Vacancy Duration}

\textbf{Problem.}
A labor economist models how many weeks export manufacturers take to fill specialized vacancies after posting. Let $X$ be vacancy-fill duration. Hiring policy cycles and reporting windows bound outcomes to $[0,8]$ weeks. Fills near week $3$ are most common (typical screening-to-offer timeline), while longer durations taper with a different slope because escalated recruitment channels activate after week $3$:
\[
f(x)=\begin{cases}
\dfrac{h}{3}x, & 0\le x\le 3,\\
h-\dfrac{h}{5}(x-3), & 3\le x\le 8,\\
0, & \text{otherwise.}
\end{cases}
\]
where $h>0$.

\textbf{Questions/Tasks.}
\begin{enumerate}[label=(\alph*)]
\item Determine $h$ so $f$ is a valid PDF.
\item Find $P(2\le X\le 6)$.
\item Compare $P(0\le X\le 2)$ with $P(6\le X\le 8)$ and interpret asymmetry.
\end{enumerate}

\subsection*{C3}
(a) The graph is triangular with base $8$ and height $h$, so
\[
\frac12(8)h=1\Rightarrow h=\frac14.
\]

(b) With $h=\frac14$:
\[
A_{[2,3]}=\frac{\left(\frac16+\frac14\right)}{2}\cdot1=\frac{5}{24},
\quad
A_{[3,6]}=\frac{\left(\frac14+\frac1{10}\right)}{2}\cdot3=\frac{21}{40}.
\]
Hence
\[
P(2\le X\le 6)=\frac{5}{24}+\frac{21}{40}=\frac{11}{15}.
\]

(c)
\[
P(0\le X\le 2)=\frac12(2)\left(\frac16\right)=\frac16,
\quad
P(6\le X\le 8)=\frac12(2)\left(\frac1{10}\right)=\frac1{10}.
\]
So $P(0\le X\le 2)>P(6\le X\le 8)$.
\PiecewiseProbGraph{0}{8}{x <= 3 ? x/12 : 1/4 - (x-3)/20}{2}{6}{0,2,3,6,8}

\textbf{Interpretation.}
A single peak can still produce uneven labor-duration probabilities when the two sides have different widths.

\section{Problem 12: Parameterized Triangular Model for Inflation-Adjusted Costs}

\textbf{Problem.}
A regional trade observatory models an inflation-adjusted logistics cost index $X$ that is normalized between $0$ and $12$ by design of its reporting scale. The parameter $p$ is the cost level most likely in a given quarter; shifting $p$ captures whether market pressure is concentrated in lower or higher cost bands while keeping the same total support. For $0<p<12$, define
\[
f(x)=\begin{cases}
kx, & 0\le x\le p,\\
\dfrac{kp}{12-p}(12-x), & p\le x\le 12,\\
0, & \text{otherwise.}
\end{cases}
\]

\textbf{Questions/Tasks.}
\begin{enumerate}[label=(\alph*)]
\item Determine $k$ in terms of $p$.
\item Show that the peak height is independent of $p$.
\item Find $P(X\le 6)$ for the case $p=4$.
\item Compare $P(X\le 6)$ for $p=4$ and $p=8$.
\item Explain clearly how moving the peak changes the most likely region and concentration of probability mass.
\end{enumerate}

\subsection*{C3}
(a) The peak value is $f(p)=kp$.
Total area is sum of two triangles:
\[
\frac12 p(kp)+\frac12(12-p)(kp)=6kp=1
\Rightarrow
k=\frac{1}{6p}.
\]

(b) Peak height is
\[
f(p)=kp=\frac{1}{6},
\]
so it does not depend on $p$.

(c) For $p=4$, $k=\frac{1}{24}$ and
\[
f(x)=\begin{cases}
\dfrac{x}{24}, & 0\le x\le 4,\\
\dfrac{12-x}{48}, & 4\le x\le 12.
\end{cases}
\]
Then
\[
P(X\le 6)=\underbrace{\frac12(4)\left(\frac16\right)}_{[0,4]}+\underbrace{\frac{\left(\frac16+\frac18\right)}{2}\cdot2}_{[4,6]}=\frac13+\frac{7}{24}=\frac58.
\]

(d) For $p=8$, by symmetry of the setup around $x=6$, we get
\[
P(X\le 6)=\frac38.
\]
So shifting the peak right from $4$ to $8$ lowers left-half probability.

(e) Moving the peak changes which region is most likely: left-shifted peaks concentrate area earlier, right-shifted peaks concentrate area later.
\PiecewiseProbGraph{0}{12}{x <= 4 ? x/24 : (12-x)/48}{0}{6}{0,4,6,8,12}

\textbf{Interpretation.}
The most likely cost band is at the peak, while actual probabilities still come from geometric area under the curve.

\section{Problem 13: Multi-Parameter Density for Export Contract Sizes}

\textbf{Problem.}
An export credit agency tracks a normalized contract-size index $X$ for approved trade deals. Values run from $0$ to $p$ because $p$ is the maximum index used in that reporting cycle. The mode occurs at $q$ ($0<q<p$), representing the most frequently negotiated contract scale; values farther from $q$ are less likely on both sides. Define
\[
f(x)=\begin{cases}
\dfrac{h}{q}x, & 0\le x\le q,\\
\dfrac{h}{p-q}(p-x), & q\le x\le p,\\
0, & \text{otherwise.}
\end{cases}
\]

\textbf{Questions/Tasks.}
\begin{enumerate}[label=(\alph*)]
\item Determine $h$ so that $f$ is a valid PDF.
\item Find $P(X\le q)$ in terms of $p$ and $q$.
\item Find $P\!\left(\frac q2\le X\le q\right)$ and $P\!\left(q\le X\le\frac{p+q}{2}\right)$.
\item For fixed $p$, explain how increasing $q$ changes the shape and concentration of mass.
\end{enumerate}

\subsection*{C3}
(a) The graph is triangular with base $p$ and peak height $h$, so
\[
\frac12(p)(h)=1\Rightarrow h=\frac{2}{p}.
\]

(b) Left area up to the peak is a triangle:
\[
P(X\le q)=\frac12(q)(h)=\frac12(q)\left(\frac{2}{p}\right)=\frac{q}{p}.
\]

(c) On $\left[\frac q2,q\right]$, heights are $\frac h2$ and $h$:
\[
P\!\left(\frac q2\le X\le q\right)=\frac{\left(\frac h2+h\right)}{2}\cdot\frac q2=\frac{3hq}{8}=\frac{3q}{4p}.
\]
On $\left[q,\frac{p+q}{2}\right]$, heights are $h$ and $\frac h2$:
\[
P\!\left(q\le X\le\frac{p+q}{2}\right)=\frac{\left(h+\frac h2\right)}{2}\cdot\frac{p-q}{2}=\frac{3h(p-q)}{8}=\frac{3(p-q)}{4p}.
\]

(d) With $p$ fixed, increasing $q$ moves the peak right. The left side widens and the right side narrows, so a larger fraction of total area lies on the left of the peak ($\frac{q}{p}$ increases).
\PiecewiseProbGraph{0}{10}{x <= 4 ? x/20 : (10-x)/30}{2}{7}{0,2,4,7,10}

\textbf{Interpretation.}
The most likely contract-size region is around the peak, but probability comparisons remain area-based.

\section{Problem 14: Two-Setting Comparison for Regional Supply Performance}

\textbf{Problem.}
A supply-chain analyst uses the family below for a monthly supply-performance index $X$, where $0$ is complete underperformance and $p$ is the best score achievable under the scoring protocol. The peak at $q$ marks the most likely performance level, and changing $(p,q)$ shifts both spread and asymmetry:
\[
f_{p,q}(x)=\begin{cases}
\dfrac{2x}{pq}, & 0\le x\le q,\\
\dfrac{2(p-x)}{p(p-q)}, & q\le x\le p,\\
0, & \text{otherwise.}
\end{cases}
\]
Consider two parameter settings:
\[
\text{Model A: }(p,q)=(10,4),\qquad \text{Model B: }(p,q)=(12,8).
\]

\textbf{Questions/Tasks.}
\begin{enumerate}[label=(\alph*)]
\item Verify that this is a valid PDF for all $0<q<p$.
\item For $0<r\le q$, show that $P(X\le r)=\dfrac{r^2}{pq}$.
\item Compare $P_A(0\le X\le 4)$ and $P_B(0\le X\le 4)$.
\item Explain geometrically how changing both $p$ and $q$ changes peak location, asymmetry, spread, and concentration of mass.
\end{enumerate}

\subsection*{C3}
(a) Peak height is
\[
f_{p,q}(q)=\frac{2}{p}.
\]
Total area is triangular area:
\[
\frac12(p)\left(\frac{2}{p}\right)=1,
\]
so it is valid.

(b) If $0<r\le q$, the area from $0$ to $r$ is a triangle with height $f(r)=\frac{2r}{pq}$:
\[
P(X\le r)=\frac12(r)\left(\frac{2r}{pq}\right)=\frac{r^2}{pq}.
\]

(c) Model A: $r=4=q$, so
\[
P_A(0\le X\le 4)=\frac{4}{10}=\frac25.
\]
Model B: $r=4<q=8$, so use part (b):
\[
P_B(0\le X\le 4)=\frac{4^2}{(12)(8)}=\frac{16}{96}=\frac16.
\]
Hence
\[
P_A(0\le X\le 4)>P_B(0\le X\le 4).
\]

(d) From A to B, support gets wider ($p$ increases) and peak moves farther right ($q$ increases). So the distribution becomes more right-skewed in where mass is concentrated, and early-left intervals (like $[0,4]$) capture much less area.

\textbf{Model A graph}
\PiecewiseProbGraph{0}{10}{x <= 4 ? x/20 : (10-x)/30}{0}{4}{0,4,10}

\textbf{Model B graph}
\PiecewiseProbGraph{0}{12}{x <= 8 ? x/48 : (12-x)/24}{0}{4}{0,4,8,12}

\textbf{Interpretation.}
Changing both peak position and support width can strongly shift where supply-performance outcomes concentrate.

\section{Problem 15: Capstone Model for Cross-Border Service Demand}

\textbf{Problem.}
A digital trade platform models a cross-border service-demand index $X$ between contractual minimum activity $a$ and infrastructure limit $b$. Demand is most likely near operating point $p$ ($a<p<b$): values away from $p$ are less frequent, with slope on each side determined by how quickly demand drops toward the lower and upper bounds. The density is
\[
f(x)=\begin{cases}
C(x-a), & a\le x\le p,\\
\dfrac{C(p-a)}{b-p}(b-x), & p\le x\le b,\\
0, & \text{otherwise.}
\end{cases}
\]

\textbf{Questions/Tasks.}
\begin{enumerate}[label=(\alph*)]
\item Determine $C$ in terms of $a,p,b$.
\item Find the peak height $f(p)$ and show how it depends on the support width.
\item Find $P(a\le X\le p)$.
\item Compute $P\!\left(\frac{a+p}{2}\le X\le\frac{p+b}{2}\right)$.
\item For $0<d\le\min\{p-a,b-p\}$, compare $P(p-d\le X\le p)$ and $P(p\le X\le p+d)$, and explain how $p-a$ and $b-p$ jointly control which side near the peak carries more mass.
\end{enumerate}

\subsection*{C3}
(a) Peak height is
\[
h=f(p)=C(p-a).
\]
Total area is one triangle with base $b-a$ and height $h$:
\[
\frac12(b-a)h=1
\Rightarrow
h=\frac{2}{b-a}.
\]
Thus
\[
C=\frac{h}{p-a}=\frac{2}{(b-a)(p-a)}.
\]

(b) From above,
\[
f(p)=h=\frac{2}{b-a},
\]
so the peak height depends on total support width $b-a$ (wider support gives lower peak).

(c) Left-of-peak probability is left triangle area:
\[
P(a\le X\le p)=\frac12(p-a)h=\frac{p-a}{b-a}.
\]

(d) Split at $p$.
On $\left[\frac{a+p}{2},p\right]$, heights are $\frac h2$ and $h$:
\[
A_L=\frac{\left(\frac h2+h\right)}{2}\cdot\frac{p-a}{2}=\frac{3h(p-a)}{8}.
\]
On $\left[p,\frac{p+b}{2}\right]$, heights are $h$ and $\frac h2$:
\[
A_R=\frac{\left(h+\frac h2\right)}{2}\cdot\frac{b-p}{2}=\frac{3h(b-p)}{8}.
\]
So
\[
P\!\left(\frac{a+p}{2}\le X\le\frac{p+b}{2}\right)=A_L+A_R=\frac{3h(b-a)}{8}=\frac34.
\]

(e) Left near-peak strip:
\[
P(p-d\le X\le p)=\frac{\left(h\left(1-\frac{d}{p-a}\right)+h\right)}{2}\cdot d
=hd\left(1-\frac{d}{2(p-a)}\right).
\]
Right near-peak strip:
\[
P(p\le X\le p+d)=\frac{\left(h+h\left(1-\frac{d}{b-p}\right)\right)}{2}\cdot d
=hd\left(1-\frac{d}{2(b-p)}\right).
\]
If $p-a<b-p$, then $\frac{1}{p-a}>\frac{1}{b-p}$ so the left expression is smaller; more near-peak mass is on the right. If $p-a>b-p$, the opposite occurs. If equal, the two probabilities are equal.
\PiecewiseProbGraph{1}{10}{x <= 4 ? 2*(x-1)/27 : (10-x)/27}{2.5}{7}{1,4,10}

\textbf{Interpretation.}
This demand model shows how $p-a$ and $b-p$ jointly control asymmetry: demand is most likely near $p$, but nearby left/right probabilities differ with slope steepness.
