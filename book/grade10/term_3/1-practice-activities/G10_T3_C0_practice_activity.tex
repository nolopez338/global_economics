		\section*{General Problem}
		
		\textbf{Problem.}
		Let $X$ be a discrete random variable that can take values $x_1,x_2,\dots,x_n$ with probability mass function
		\[
		\begin{array}{c|cccc}
			x & x_1 & x_2 & \cdots & x_n\\
			\hline
			P(X=x) & p_1 & p_2 & \cdots & p_n
		\end{array}
		\]
		where one or more probabilities may be unknown parameters.
		
		\textbf{Question.}
		(a) Use the pmf conditions to determine the unknown parameter(s):
		\[
		\sum_{i=1}^{n}p_i=1\quad\text{and}\quad p_i\ge 0\text{ for all }i.
		\]
		(b) Compute required probabilities such as $P(X=a)$, $P(X\le a)$, $P(X\ge a)$, or $P(a\le X\le b)$ by adding the relevant pmf values.
		(c) Interpret each result as a percentage in context.
		
		\textbf{Solution.}
		
		\textbf{(a)} Set up the total-probability equation
		\[
		p_1+p_2+\cdots+p_n=1
		\]
		and solve for the unknown parameter(s). Then check nonnegativity of every probability.
		
		\textbf{(b)} For any event $A$ on the support,
		\[
		P(X\in A)=\sum_{x_i\in A}P(X=x_i).
		\]
		In a pmf stem graph, each horizontal value $x_i$ is a possible value of $X$, and the vertical stem height is $P(X=x_i)$. The full graph shows all stems $\{(x_i,p_i)\}_{i=1}^n$, while highlighted stems show the values included in an event such as $P(X=a)$, $P(X\le b)$, or $P(a\le X\le b)$.
		\[
		P(X=x)=\{(x_1,p_1),(x_2,p_2),\dots,(x_n,p_n)\}.
		\]

		\section{App Notifications per Hour}
		
		\textbf{Problem.}
		Let $X$ be the number of app notifications a student receives in one hour. The probability mass function is
		\[
		\begin{array}{c|cccc}
			x & 0 & 1 & 2 & 3\\
			\hline
			P(X=x) & 0.2 & 0.3 & k & 0.1
		\end{array}
		\]
		
		\textbf{Question.}
		(a) Determine $k$ so that this is a valid probability distribution.
		(b) Compute: (i) $P(X=2)$, (ii) $P(X\le 2)$, and (iii) $P(1\le X\le 2)$.
		(c) Interpret these results in terms of percentage probability.
		
		\textbf{Solution.}

		\textbf{(a)} For a valid pmf,
		\[
		0.2+0.3+k+0.1=1\Rightarrow k=0.4.
		\]

		
\[
		P(X=x)=\{(0,0.2),(1,0.3),(2,0.4),(3,0.1)\}.
		\]

		\section{Lost Pens in a Small Box}
		
		\textbf{Problem.}
		Let $Z$ be the number of lost pens found in a small box. The probability mass function is
		\[
		\begin{array}{c|cccc}
			z & 0 & 1 & 2 & 3\\
			\hline
			P(Z=z) & 0.4 & 0.5 & a & 0.2
		\end{array}
		\]
		
		\textbf{Question.}
		(a) Solve for $a$ so the probabilities sum to 1.
		(b) Decide whether the resulting distribution is valid.
		(c) Using the computed $a$, compute $P(Z\ge 2)$.
		
		\textbf{Solution.}

		\textbf{(a)} From total probability,
		\[
		0.4+0.5+a+0.2=1\Rightarrow 1.1+a=1\Rightarrow a=-0.1.
		\]
		\[
		P(Z=z)=\{(0,0.4),(1,0.5),(2,-0.1),(3,0.2)\}.
		\]

		\section{Customers in a 10-Minute Window}
		
		\textbf{Problem.}
		A café tracks $T$, the number of customers arriving in a 10-minute window.
		\[
		\begin{array}{c|cccc}
			t & 1 & 2 & 3 & 4\\
			\hline
			P(T=t) & \lambda & 0.7-\lambda & 0.2 & 0.1
		\end{array}
		\]
		
		\textbf{Question.}
		(a) Determine $\lambda$ so that this is a valid probability distribution.
		(b) Compute: (i) $P(T=2)$, (ii) $P(T\ne 4)$, and (iii) $P(2\le T\le 3)$.
		(c) Interpret these results in terms of percentage probability.

		\textbf{(a)} Sum-to-1 gives
		\[
		\lambda+(0.7-\lambda)+0.2+0.1=1,
		\]
		which is true for all $\lambda$. For a valid pmf we need nonnegative probabilities:
		\[
		\lambda\ge 0,\qquad 0.7-\lambda\ge 0
		\Rightarrow 0\le \lambda\le 0.7.
		\]

		
\[
		P(T=t)=\{(1,\lambda),(2,0.7-\lambda),(3,0.2),(4,0.1)\},\quad 0\le\lambda\le 0.7.
		\]

		\section{Defects Found in Final Check}
		
			\textbf{Problem.}
		Let $Y$ be the number of defects found in a final check. Suppose
		\[
		P(Y=y)=\begin{cases}
			r, & y=0,\\
			2r, & y=1,\\
			\frac{1}{2}-2r, & y=2,\\
			r, & y=3,\\
			\frac{1}{4}, & y=4.
		\end{cases}
		\]
		
		\textbf{Question.}
		(a) Determine the range of valid values of $r$ and verify the value of $r$ obtained from the sum-to-1 condition.
		(b) Compute: (i) $P(Y=4)$, (ii) $P(Y\in\{0,2,4\})$, and (iii) $P(1\le Y\le 3)$.
		(c) Interpret these results in terms of percentage probability.
		
		\textbf{Solution.}

		\textbf{(a)} For the distribution to be valid, all probabilities must be nonnegative:
		\[
		r\ge 0,\qquad 2r\ge 0,\qquad \frac{1}{2}-2r\ge 0,\qquad \frac{1}{4}\ge 0.
		\]
		Hence the valid range is
		\[
		0\le r\le \frac{1}{4}.
		\]
		
		The sum condition is
		\[
		r+2r+\left(\frac{1}{2}-2r\right)+r+\frac{1}{4}=1
		\Rightarrow 2r+\frac{3}{4}=1
		\Rightarrow r=\frac{1}{8}.
		\]
		Since
		\[
		0\le \frac{1}{8}\le \frac{1}{4},
		\]
		the value $r=\frac{1}{8}$ is valid.
