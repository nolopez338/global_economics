		\subsection*{Problem 3 — Lost Pens in a Small Box}
		
		\textbf{Problem.}
		Let $Z$ be the number of lost pens found in a small box. The probability mass function is
		\[
		\begin{array}{c|cccc}
			z & 0 & 1 & 2 & 3\\
			\hline
			P(Z=z) & 0.4 & 0.5 & a & 0.2
		\end{array}
		\]
		
		\textbf{Question.}
		(a) Solve for $a$ so the probabilities sum to 1.
		(b) Decide whether the resulting distribution is valid.
		(c) Using the computed $a$, compute $P(Z\ge 2)$.
		
		\textbf{Solution.}

		\textbf{(a)} From total probability,
		\[
		0.4+0.5+a+0.2=1\Rightarrow 1.1+a=1\Rightarrow a=-0.1.
		\]
		

		\textbf{(b)}
		\[
		P(Z\ge 2)=P(Z=2)+P(Z=3)=a+0.2=-0.1+0.2=0.1.
		\]
		\textbf{(c)} \textbf{Not a valid probability distribution} because $a<0$.

		A café tracks $X$, the number of customers arriving in a short checkout window.
		\[
		\begin{array}{c|cccc}
			x & 1 & 2 & 3 & 4\\
			\hline
			P(X=x) & 0.6 & d & 0.5 & d
		\end{array}
		\]
		
		\textbf{Question.}
		(a) Solve for $d$ so the probabilities sum to 1.
		(b) Decide whether the resulting distribution is valid.
		(c) Using the computed $d$, compute $P(2\le X\le 4)$.
		
		\textbf{Solution.}

		\textbf{(a)} Using total probability,
		\[
		0.6+d+0.5+d=1\Rightarrow 1.1+2d=1\Rightarrow 2d=-0.1\Rightarrow d=-0.05.
		\]
		

		\textbf{(b)}
		\[
		P(2\le X\le 4)=P(X=2)+P(X=3)+P(X=4)=d+0.5+d=-0.05+0.5-0.05=0.4.
		\]
		\textbf{(c)} \textbf{Not a valid probability distribution} because $d<0$.

		\subsection*{Problem 7 — Correct Answers on a Short Quiz}
		
		\textbf{Problem.}
		Let $Y$ be the number of correct answers on a 4-question quiz. The pmf is
		\[
		P(Y=y)=\begin{cases}
			r, & y=0,\\
			2r, & y=1,\\
			3r, & y=2,\\
			2r, & y=3,\\
			1-8r, & y=4.
		\end{cases}
		\]
		
		\textbf{Question.}
		(a) Determine all values of $r$ that make this a valid distribution.
		(b) Compute: (i) $P(Y=4)$, (ii) $P(Y\ge 3)$, and (iii) $P(1\le Y\le 3)$.
		(c) Interpret these results in terms of percentage probability.
		
		\textbf{Solution.}

		\textbf{(a)} From the total probability condition,
		\[
		r+2r+3r+2r+(1-8r)=1,
		\]
		which is true for all $r$.
		
		For the distribution to be valid, all probabilities must be nonnegative:
		\[
		r\ge 0,\qquad 1-8r\ge 0.
		\]
		Hence,
		\[
		0\le r\le \frac{1}{8}.
		\]
		

		\textbf{(b)}
		\[
		P(Y=4)=1-8r,\quad
		P(Y\ge 3)=2r+(1-8r)=1-6r,\quad
		P(1\le Y\le 3)=2r+3r+2r=7r.
		\]
		\textbf{(c)} The percentage probabilities are $100(1-8r)\%$ for exactly 4 correct answers, $100(1-6r)\%$ for at least 3 correct answers, and $700r\%$ for between 1 and 3 correct answers.

		Let $W$ be the number of calls received in a busy hour, with
		\[
		\begin{array}{c|ccccc}
			w & 0 & 1 & 2 & 3 & 4\\

		(a) Verify that the probabilities sum to 1.
		(b) Find the values of $h$ for which this is a valid distribution.
		(c) For valid $h$, compute $P(W\ge 2)$.
		
		\textbf{Solution.}
		
		\textbf{(a)}
		\[
		h+0.4+2h+0.3+(0.3-3h)=1,
		\]
		so the probabilities sum to 1 for all $h$.
		
		\textbf{(b)} Validity requires nonnegative probabilities:
		\[
		h\ge 0,\qquad 2h\ge 0,\qquad 0.3-3h\ge 0.
		\]
		From $0.3-3h\ge 0$, we get $0.3\ge 3h$, so $0.1\ge h$, that is, $h\le 0.1$. Hence
		\[
		0\le h\le 0.1.
		\]
		
		\textbf{(c)}
		\[
		P(W\ge 2)=P(W=2)+P(W=3)+P(W=4)=2h+0.3+(0.3-3h)=0.6-h.
		\]
		Thus,
		\[
		P(W\ge 2)=0.6-h,\qquad 0\le h\le 0.1.
		\]

		\subsection*{Problem 10 — Goals in a Match Summary}
		
		\textbf{Problem.}
		Let $T$ be the number of goals scored by a team in a match summary. The probability mass function is
		\[
		P(T=t)=\begin{cases}
			p, & t=0,\\
			\frac{3}{4}-2p, & t=1,\\
			p, & t=2,\\
			\frac{1}{4}, & t=3.
		\end{cases}
		\]
		
		\textbf{Question.}
		(a) Determine all valid values of $p$. Since all listed probabilities must satisfy nonnegativity, require
		\[
		p\ge 0,\qquad \frac{3}{4}-2p\ge 0,
		\]
		so the feasible interval is
		\[
		0\le p\le \frac{3}{8}.
		\]
		(b) Compute: (i) $P(T=1)$, (ii) $P(T\in\{0,2,3\})$, and (iii) $P(1\le T\le 3)$.
		(c) Interpret these results in terms of percentage probability.
		
		\textbf{Solution.}

		\textbf{(a)} The sum condition is
		\[
		p+\left(\frac{3}{4}-2p\right)+p+\frac{1}{4}=1,
		\]
		which is true for all $p$.
		For the pmf to be valid, each probability must be nonnegative:
		\[
		p\ge 0,\qquad \frac{3}{4}-2p\ge 0.
		\]
		Hence
		\[
		0\le p\le \frac{3}{8}.
		\]

		\textbf{(b)}
		\[
		P(T=1)=\frac{3}{4}-2p,\quad
		P(T\in\{0,2,3\})=p+p+\frac{1}{4}=2p+\frac{1}{4},\quad
		P(1\le T\le 3)=1-p.
		\]
		\textbf{(c)} The percentage probabilities are $100\left(\frac{3}{4}-2p\right)\%$ for exactly 1 goal, $100\left(2p+\frac{1}{4}\right)\%$ for 0, 2, or 3 goals, and $100(1-p)\%$ for between 1 and 3 goals.
