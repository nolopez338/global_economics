\section{Problem 1: Introductory Uniform Distribution}

\textbf{Problem.}
Let $X\sim \text{Uniform}(0,10)$.

\textbf{Question.}
Find:
\begin{itemize}
    \item[(a)] the density value $f(x)$ on the support,
    \item[(b)] the support interval of $X$,
    \item[(c)] $P(2\le X\le 7)$,
    \item[(d)] $P(X>8)$.
\end{itemize}

\textbf{Solution.}
For $X\sim \text{Uniform}(0,10)$,
\[
f(x)=\frac{1}{10-0}=\frac{1}{10},\qquad 0\le x\le 10.
\]
So the support is $[0,10]$.
\[
P(2\le X\le 7)=\frac{7-2}{10}=\frac{1}{2}.
\]
\[
P(X>8)=\frac{10-8}{10}=\frac{1}{5}.
\]
\CtwoProbGraph{0}{10}{1/10}{2}{7}{0,2,7,8,10}{0.18}

\textbf{Interpretation.}
For a simple uniform model, probability is interval length divided by total length.

\section{Problem 2: Two-Interval Piecewise Uniform (Basic)}

\textbf{Problem.}
Let $X$ have density
\[
f(x)=
\begin{cases}
\dfrac{1}{5}, & 0\le x<4,\\
\dfrac{1}{30}, & 4\le x\le 10,\\
0, & \text{otherwise.}
\end{cases}
\]

\textbf{Question.}
Find:
\begin{itemize}
    \item[(a)] $P(X<3)$,
    \item[(b)] $P(2\le X\le 8)$,
    \item[(c)] $P(X\ge 4)$.
\end{itemize}

\textbf{Solution.}
Check total area:
\[
4\left(\frac{1}{5}\right)+6\left(\frac{1}{30}\right)=\frac{4}{5}+\frac{1}{5}=1.
\]
\[
P(X<3)=3\left(\frac{1}{5}\right)=\frac{3}{5}.
\]
\[
P(2\le X\le 8)=2\left(\frac{1}{5}\right)+4\left(\frac{1}{30}\right)=\frac{2}{5}+\frac{2}{15}=\frac{8}{15}.
\]
\[
P(X\ge 4)=6\left(\frac{1}{30}\right)=\frac{1}{5}.
\]
\CtwoProbGraph{0}{10}{x < 4 ? 1/5 : 1/30}{2}{8}{0,2,3,4,8,10}{0.26}

\textbf{Interpretation.}
Most probability mass is on $[0,4)$ because its density height is much larger.

\section{Problem 3: Two-Interval Piecewise Uniform (Basic)}

\textbf{Problem.}
Let $X$ have density
\[
f(x)=
\begin{cases}
\dfrac{1}{6}, & -2\le x<1,\\
\dfrac{1}{10}, & 1\le x\le 6,\\
0, & \text{otherwise.}
\end{cases}
\]

\textbf{Question.}
Find:
\begin{itemize}
    \item[(a)] $P(-1\le X\le 4)$,
    \item[(b)] $P(X>3)$,
    \item[(c)] $P(X<1)$.
\end{itemize}

\textbf{Solution.}
Check total area:
\[
3\left(\frac{1}{6}\right)+5\left(\frac{1}{10}\right)=\frac{1}{2}+\frac{1}{2}=1.
\]
\[
P(-1\le X\le 4)=2\left(\frac{1}{6}\right)+3\left(\frac{1}{10}\right)=\frac{1}{3}+\frac{3}{10}=\frac{19}{30}.
\]
\[
P(X>3)=3\left(\frac{1}{10}\right)=\frac{3}{10}.
\]
\[
P(X<1)=3\left(\frac{1}{6}\right)=\frac{1}{2}.
\]
\CtwoProbGraph{-2}{6}{x < 1 ? 1/6 : 1/10}{-1}{4}{-2,-1,1,3,4,6}{0.24}

\textbf{Interpretation.}
To find each probability, add the rectangular areas over the relevant subintervals.

\section{Problem 4: Three-Interval Piecewise Uniform (Combining Regions)}

\textbf{Problem.}
Let $X$ have density
\[
f(x)=
\begin{cases}
\dfrac{1}{10}, & 0\le x<2,\\
\dfrac{1}{5}, & 2\le x<5,\\
\dfrac{1}{10}, & 5\le x\le 7,\\
0, & \text{otherwise.}
\end{cases}
\]

\textbf{Question.}
Find:
\begin{itemize}
    \item[(a)] $P(X<2\ \text{or}\ X>5)$,
    \item[(b)] $P(2\le X\le 4)$,
    \item[(c)] Which is more likely: $X\in[0,2)$ or $X\in[2,5)$? Explain.
\end{itemize}

\textbf{Solution.}
Check total area:
\[
2\left(\frac{1}{10}\right)+3\left(\frac{1}{5}\right)+2\left(\frac{1}{10}\right)=\frac{1}{5}+\frac{3}{5}+\frac{1}{5}=1.
\]
\[
P(X<2\ \text{or}\ X>5)=2\left(\frac{1}{10}\right)+2\left(\frac{1}{10}\right)=\frac{2}{5}.
\]
\[
P(2\le X\le 4)=2\left(\frac{1}{5}\right)=\frac{2}{5}.
\]
Also,
\[
P(0\le X<2)=\frac{1}{5},\qquad P(2\le X<5)=\frac{3}{5}.
\]
So $[2,5)$ is more likely.
\CtwoProbGraph{0}{7}{x < 2 ? 1/10 : (x < 5 ? 1/5 : 1/10)}{2}{4}{0,2,4,5,7}{0.26}

\textbf{Interpretation.}
Even with equal widths, a taller density rectangle gives larger probability.

\section{Problem 5: Piecewise Uniform with Complement Reasoning}

\textbf{Problem.}
Let $X$ have density
\[
f(x)=
\begin{cases}
\dfrac{3}{20}, & 10\le x<14,\\
\dfrac{1}{10}, & 14\le x<18,\\
0, & \text{otherwise.}
\end{cases}
\]

\textbf{Question.}
Find:
\begin{itemize}
    \item[(a)] $P(11\le X\le 13\ \text{or}\ 16\le X\le 18)$,
    \item[(b)] $P(X<12\ \text{or}\ X>17)$,
    \item[(c)] $P(12\le X\le 17)$ using a complement.
\end{itemize}

\textbf{Solution.}
Check total area:
\[
4\left(\frac{3}{20}\right)+4\left(\frac{1}{10}\right)=\frac{3}{5}+\frac{2}{5}=1.
\]
So densities are $\frac{2}{11}$ and $\frac{1}{11}$, both nonnegative.
\[
P(11\le X\le 13\ \text{or}\ 16\le X\le 18)=2\left(\frac{3}{20}\right)+2\left(\frac{1}{10}\right)=\frac{3}{10}+\frac{1}{5}=\frac{1}{2}.
\]
\[
P(X<12\ \text{or}\ X>17)=2\left(\frac{3}{20}\right)+1\left(\frac{1}{10}\right)=\frac{3}{10}+\frac{1}{10}=\frac{2}{5}.
\]
\[
P(12\le X\le 17)=1-P(X<12\ \text{or}\ X>17)=1-\frac{2}{5}=\frac{3}{5}.
\]
\CtwoProbGraph{10}{18}{x < 14 ? 3/20 : 1/10}{12}{17}{10,11,12,13,14,16,17,18}{0.22}

\textbf{Interpretation.}
Complement reasoning avoids repeating long area sums.

\section{Problem 6: Piecewise Uniform with Unknown $k$}

\textbf{Problem.}
Let $X$ have density
\[
f(x)=
\begin{cases}
2k, & 0\le x<3,\\
k, & 3\le x\le 8,\\
0, & \text{otherwise,}
\end{cases}
\]
where $k>0$.

\textbf{Question.}
Find:
\begin{itemize}
    \item[(a)] the value of $k$ so that $f$ is a valid PDF,
    \item[(b)] $P(X<5)$,
    \item[(c)] $P(6\le X\le 8)$.
\end{itemize}

\textbf{Solution.}
For validity, total area must be 1:
\[
3(2k)+5(k)=1\ \Rightarrow\ 11k=1\ \Rightarrow\ k=\frac{1}{11}.
\]
So densities are $\frac{2}{11}$ and $\frac{1}{11}$, both nonnegative.
\[
P(X<5)=3\left(\frac{2}{11}\right)+2\left(\frac{1}{11}\right)=\frac{8}{11}.
\]
\[
P(6\le X\le 8)=2\left(\frac{1}{11}\right)=\frac{2}{11}.
\]
\CtwoProbGraph{0}{8}{x < 3 ? 2/11 : 1/11}{0}{5}{0,3,5,6,8}{0.26}

\textbf{Interpretation.}
First solve for $k$ using total area, then compute probabilities by area of rectangles.

\section{Problem 7: Piecewise Uniform with Unknown $k$ and Validity Check}

\textbf{Problem.}
Let $X$ have density
\[
f(x)=
\begin{cases}
k, & 0\le x<2,\\
2k, & 2\le x<5,\\
\dfrac{k}{2}, & 5\le x\le 9,\\
0, & \text{otherwise,}
\end{cases}
\]
where $k$ is a constant.

\textbf{Question.}
Find:
\begin{itemize}
    \item[(a)] $k$,
    \item[(b)] verify that all density values are valid for a PDF,
    \item[(c)] $P(1\le X\le 6)$.
\end{itemize}

\textbf{Solution.}
Total area condition:
\[
2(k)+3(2k)+4\left(\frac{k}{2}\right)=1
\Rightarrow 2k+6k+2k=1
\Rightarrow 10k=1
\Rightarrow k=\frac{1}{10}.
\]
Then heights are
\[
\frac{1}{10},\quad \frac{1}{5},\quad \frac{1}{20},
\]
all nonnegative, so valid.
\[
P(1\le X\le 6)=1\left(\frac{1}{10}\right)+3\left(\frac{1}{5}\right)+1\left(\frac{1}{20}\right)
=\frac{2}{20}+\frac{12}{20}+\frac{1}{20}=\frac{3}{4}.
\]
\CtwoProbGraph{0}{9}{x < 2 ? 1/10 : (x < 5 ? 1/5 : 1/20)}{1}{6}{0,1,2,5,6,9}{0.26}

\textbf{Interpretation.}
A valid PDF needs both nonnegative heights and total area exactly 1.

\section{Problem 8: Four-Interval Piecewise Uniform (Organizing Across Many Intervals)}

\textbf{Problem.}
Let $X$ have density
\[
f(x)=
\begin{cases}
\dfrac{3}{10}, & 0\le x<1,\\
\dfrac{1}{5}, & 1\le x<3,\\
\dfrac{1}{10}, & 3\le x<5,\\
\dfrac{1}{10}, & 5\le x\le 6,\\
0, & \text{otherwise.}
\end{cases}
\]

\textbf{Question.}
Find:
\begin{itemize}
    \item[(a)] $P(0.5\le X\le 5.5)$,
    \item[(b)] compare $P(X<3)$ and $P(X\ge 3)$,
    \item[(c)] identify where most of the probability mass is concentrated.
\end{itemize}

\textbf{Solution.}
Check total area:
\[
1\left(\frac{3}{10}\right)+2\left(\frac{1}{5}\right)+2\left(\frac{1}{10}\right)+1\left(\frac{1}{10}\right)
=\frac{3}{10}+\frac{4}{10}+\frac{2}{10}+\frac{1}{10}=1.
\]
\[
P(0.5\le X\le 5.5)=0.5\left(\frac{3}{10}\right)+2\left(\frac{1}{5}\right)+2\left(\frac{1}{10}\right)+0.5\left(\frac{1}{10}\right)
=\frac{3}{20}+\frac{8}{20}+\frac{4}{20}+\frac{1}{20}=\frac{4}{5}.
\]
\[
P(X<3)=\frac{3}{10}+\frac{4}{10}=\frac{7}{10},
\qquad
P(X\ge 3)=\frac{3}{10}.
\]
Most mass is below 3.
\CtwoProbGraph{0}{6}{x < 1 ? 3/10 : (x < 3 ? 1/5 : (x < 5 ? 1/10 : 1/10))}{0.5}{5.5}{0,0.5,1,3,5,5.5,6}{0.38}

\textbf{Interpretation.}
Careful partitioning of a long interval keeps multi-step area sums accurate.

\section{Problem 9: Structured Piecewise Uniform with Parameter $k$}

\textbf{Problem.}
Let $X$ have density
\[
f(x)=
\begin{cases}
k, & -3\le x<0,\\
2k, & 0\le x<2,\\
\dfrac{1}{12}, & 2\le x\le 6,\\
0, & \text{otherwise.}
\end{cases}
\]

\textbf{Question.}
Find:
\begin{itemize}
    \item[(a)] $k$,
    \item[(b)] $P(-1\le X\le 4)$,
    \item[(c)] which is larger: $P(X<0)$ or $P(X\ge 2)$?
\end{itemize}

\textbf{Solution.}
Use total area:
\[
3k+2(2k)+4\left(\frac{1}{12}\right)=1
\Rightarrow 7k+\frac{1}{3}=1
\Rightarrow 7k=\frac{2}{3}
\Rightarrow k=\frac{2}{21}.
\]
\[
P(-1\le X\le 4)=1\left(\frac{2}{21}\right)+2\left(\frac{4}{21}\right)+2\left(\frac{1}{12}\right)
=\frac{2}{21}+\frac{8}{21}+\frac{1}{6}
=\frac{13}{21}+\frac{1}{6}
=\frac{11}{14}.
\]
\[
P(X<0)=3\left(\frac{2}{21}\right)=\frac{2}{7},
\qquad
P(X\ge 2)=4\left(\frac{1}{12}\right)=\frac{1}{3}.
\]
So $P(X\ge 2)$ is larger.
\CtwoProbGraph{-3}{6}{x < 0 ? 2/21 : (x < 2 ? 4/21 : 1/12)}{-1}{4}{-3,-1,0,2,4,6}{0.26}

\textbf{Interpretation.}
A wider interval with a moderate height can beat a narrower interval with a higher height.

\section{Problem 10: Symbolic Endpoint (Conceptual Effect of Width)}

\textbf{Problem.}
Let $m>0$ and define
\[
f(x)=
\begin{cases}
\dfrac{1}{3m}, & 0\le x<m,\\
\dfrac{2}{3m}, & m\le x\le 2m,\\
0, & \text{otherwise.}
\end{cases}
\]

\textbf{Question.}
Find:
\begin{itemize}
    \item[(a)] verify that $f$ is a valid PDF,
    \item[(b)] $P\left(X<\dfrac{3m}{2}\right)$,
    \item[(c)] explain what happens to this probability if $m$ is doubled.
\end{itemize}

\textbf{Solution.}
\[
m\left(\frac{1}{3m}\right)+m\left(\frac{2}{3m}\right)=\frac{1}{3}+\frac{2}{3}=1,
\]
so $f$ is valid and nonnegative.
\[
P\left(X<\frac{3m}{2}\right)=m\left(\frac{1}{3m}\right)+\frac{m}{2}\left(\frac{2}{3m}\right)=\frac{1}{3}+\frac{1}{3}=\frac{2}{3}.
\]
If $m$ is doubled, both widths scale by the same factor and densities scale down accordingly, so this probability stays $\frac{2}{3}$.

\section{Problem 11: Infer Missing Endpoint from a Probability Condition}

\textbf{Problem.}
Let $b>2$ and let
\[
f(x)=
\begin{cases}
c, & 0\le x<2,\\
\dfrac{c}{2}, & 2\le x\le b,\\
0, & \text{otherwise.}
\end{cases}
\]
Suppose $P(X>2)=\dfrac{1}{3}$.

\textbf{Question.}
Find:
\begin{itemize}
    \item[(a)] $c$,
    \item[(b)] $b$,
    \item[(c)] $P(1\le X\le 3)$.
\end{itemize}

\textbf{Solution.}
Given
\[
P(X>2)=(b-2)\left(\frac{c}{2}\right)=\frac{1}{3}. \tag{1}
\]
Total area is 1:
\[
2c+(b-2)\left(\frac{c}{2}\right)=1.
\]
Using (1):
\[
2c+\frac{1}{3}=1 \Rightarrow 2c=\frac{2}{3} \Rightarrow c=\frac{1}{3}.
\]
Now substitute into (1):
\[
(b-2)\left(\frac{1}{6}\right)=\frac{1}{3}
\Rightarrow b-2=2
\Rightarrow b=4.
\]
Finally,
\[
P(1\le X\le 3)=1\left(\frac{1}{3}\right)+1\left(\frac{1}{6}\right)=\frac{1}{2}.
\]
\CtwoProbGraph{0}{4}{x < 2 ? 1/3 : 1/6}{1}{3}{0,1,2,3,4}{0.42}

\textbf{Interpretation.}
A probability condition plus total-area condition can determine unknown parameters.

\section{Problem 12: Rich Piecewise Uniform Design Problem}

\textbf{Problem.}
Let $X$ have density
\[
f(x)=
\begin{cases}
k, & 0\le x<1,\\
2k, & 1\le x<4,\\
m, & 4\le x<7,\\
k, & 7\le x\le 10,\\
0, & \text{otherwise,}
\end{cases}
\]
where $k,m>0$. Assume
\[
P(4\le X<7)=P(0\le X<1)+P(7\le X\le 10).
\]

\textbf{Question.}
Find:
\begin{itemize}
    \item[(a)] $k$ and $m$,
    \item[(b)] $P(2\le X\le 8)$,
    \item[(c)] compare $P(1\le X<4)$ and $P(4\le X\le 10)$,
    \item[(d)] explain where the distribution places most of its mass and why.
\end{itemize}

\textbf{Solution.}
From the condition:
\[
3m=1\cdot k+3\cdot k=4k
\Rightarrow m=\frac{4k}{3}. \tag{1}
\]
Total area:
\[
1(k)+3(2k)+3(m)+3(k)=1
\Rightarrow 10k+3m=1.
\]
Using (1):
\[
10k+4k=1 \Rightarrow 14k=1 \Rightarrow k=\frac{1}{14},
\qquad
m=\frac{4}{3}\cdot\frac{1}{14}=\frac{2}{21}.
\]
Now
\[
P(2\le X\le 8)=2\left(\frac{1}{7}\right)+3\left(\frac{2}{21}\right)+1\left(\frac{1}{14}\right)
=\frac{2}{7}+\frac{2}{7}+\frac{1}{14}=\frac{9}{14}.
\]
\[
P(1\le X<4)=3\left(\frac{1}{7}\right)=\frac{3}{7},
\qquad
P(4\le X\le 10)=3\left(\frac{2}{21}\right)+3\left(\frac{1}{14}\right)=\frac{4}{7}.
\]
So $P(4\le X\le 10)$ is larger.
\CtwoProbGraph{0}{10}{x < 1 ? 1/14 : (x < 4 ? 1/7 : (x < 7 ? 2/21 : 1/14))}{2}{8}{0,1,2,4,7,8,10}{0.22}

\textbf{Interpretation.}
The right side $[4,10]$ gets more mass because it combines substantial width with moderate heights across two intervals.
