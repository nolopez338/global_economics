\section{Problem 1: Introductory Uniform Distribution}

\textbf{Problem.}
Let $X\sim \text{Uniform}(0,10)$.
\begin{center}
\begin{tikzpicture}
\pgfmathsetmacro{\supportA}{0}
\pgfmathsetmacro{\supportB}{10}
\pgfmathsetmacro{\pad}{0.12*(\supportB-\supportA)}
\begin{axis}[
  width=9cm,
  height=5cm,
  axis lines=left,
  xlabel={$x$},
  ylabel={$f(x)$},
  xmin={\supportA-\pad},
  xmax={\supportB+\pad},
  ymin=0,
  ymax=0.18,
  grid=both,
  xtick={0,2,4,6,8,10}
]
\addplot[domain=0:10, samples=2, draw=none, fill=blue!20, fill opacity=0.35] {1/10} \closedcycle;
\addplot[domain=0:10, samples=2] {1/10};
\end{axis}
\end{tikzpicture}
\end{center}

\textbf{Question.}
Find:
\begin{itemize}
    \item[(a)] the density value $f(x)$ on the support,
    \item[(b)] the support interval of $X$,
    \item[(c)] $P(2\le X\le 7)$,
    \item[(d)] $P(X>8)$.
\end{itemize}

\textbf{Solution.}
For $X\sim \text{Uniform}(0,10)$,
\[
f(x)=\frac{1}{10-0}=\frac{1}{10},\qquad 0\le x\le 10.
\]
So the support is $[0,10]$.
\[
P(2\le X\le 7)=\frac{7-2}{10}=\frac{1}{2}.
\]
\[
P(X>8)=\frac{10-8}{10}=\frac{1}{5}.
\]

\textbf{Interpretation.}
For a simple uniform model, probability is interval length divided by total length.

\textbf{Question.}

\begin{itemize}
    \item[(a)] $P(X<3)$,
    \item[(b)] $P(2\le X\le 8)$,
    \item[(c)] $P(X\ge 4)$.
\end{itemize}

\textbf{Solution.}
Check total area:
\[
4\left(\frac{1}{5}\right)+6\left(\frac{1}{30}\right)=\frac{4}{5}+\frac{1}{5}=1.
\]
\[
P(X<3)=3\left(\frac{1}{5}\right)=\frac{3}{5}.
\]
\[
P(2\le X\le 8)=2\left(\frac{1}{5}\right)+4\left(\frac{1}{30}\right)=\frac{2}{5}+\frac{2}{15}=\frac{8}{15}.
\]
\[
P(X\ge 4)=6\left(\frac{1}{30}\right)=\frac{1}{5}.
\]

\textbf{Interpretation.}
Most probability mass is on $[0,4)$ because its density height is much larger.

\textbf{Solution.}
Check total area:
\[
2\left(\frac{1}{10}\right)+3\left(\frac{1}{5}\right)+2\left(\frac{1}{10}\right)=\frac{1}{5}+\frac{3}{5}+\frac{1}{5}=1.

\]
Also,
\[
P(0\le X<2)=\frac{1}{5},\qquad P(2\le X<5)=\frac{3}{5}.
\]
So $[2,5)$ is more likely.

\textbf{Interpretation.}
Even with equal widths, a taller density rectangle gives larger probability.

\textbf{Question.}
Find:
\begin{itemize}
    \item[(a)] $P(11\le X\le 13\ \text{or}\ 16\le X\le 18)$,
    \item[(b)] $P(X<12\ \text{or}\ X>17)$,
    \item[(c)] $P(12\le X\le 17)$ using a complement.
\end{itemize}

\textbf{Solution.}
Check total area:
\[
4\left(\frac{3}{20}\right)+4\left(\frac{1}{10}\right)=\frac{3}{5}+\frac{2}{5}=1.
\]
\[
P(11\le X\le 13\ \text{or}\ 16\le X\le 18)=2\left(\frac{3}{20}\right)+2\left(\frac{1}{10}\right)=\frac{3}{10}+\frac{1}{5}=\frac{1}{2}.
\]
\[
P(X<12\ \text{or}\ X>17)=2\left(\frac{3}{20}\right)+1\left(\frac{1}{10}\right)=\frac{3}{10}+\frac{1}{10}=\frac{2}{5}.
\]
\[
P(12\le X\le 17)=1-P(X<12\ \text{or}\ X>17)=1-\frac{2}{5}=\frac{3}{5}.
\]

\textbf{Interpretation.}
Complement reasoning avoids repeating long area sums.

\section{Problem 6: Piecewise Uniform with Unknown $k$}

\textbf{Problem.}
Let $X$ have density
\[
f(x)=
\begin{cases}
2k, & 0\le x<3,\\
k, & 3\le x\le 8,\\
0, & \text{otherwise,}
\end{cases}
\]
where $k>0$.
\begin{center}
\begin{tikzpicture}
\begin{axis}[
  width=9cm,
  height=5cm,
  axis lines=left,
  xlabel={$x$},
  ylabel={$f(x)$},
  xmin=-1,

  ymin=0,
  ymax=0.26,
  grid=both,
  xtick={0,3,5,8}
]
\addplot[draw=none, fill=blue!20, fill opacity=0.35] coordinates {(0,0) (0,2/11) (3,2/11) (3,0)} \closedcycle;
\addplot[draw=none, fill=blue!20, fill opacity=0.35] coordinates {(3,0) (3,1/11) (8,1/11) (8,0)} \closedcycle;
\addplot[thick] coordinates {(0,2/11) (3,2/11)};
\addplot[thick] coordinates {(3,1/11) (8,1/11)};
\addplot[thick] coordinates {(3,1/11) (3,2/11)};
\end{axis}
\end{tikzpicture}
\end{center}

\textbf{Question.}
Find:
\begin{itemize}
    \item[(a)] the value of $k$ so that $f$ is a valid PDF,
    \item[(b)] $P(X<5)$,
    \item[(c)] $P(6\le X\le 8)$.
\end{itemize}

\textbf{Solution.}
For validity, total area must be 1:
\[
3(2k)+5(k)=1\ \Rightarrow\ 11k=1\ \Rightarrow\ k=\frac{1}{11}.
\]
So densities are $\frac{2}{11}$ and $\frac{1}{11}$, both nonnegative.
\[
P(X<5)=3\left(\frac{2}{11}\right)+2\left(\frac{1}{11}\right)=\frac{8}{11}.
\]
\[
P(6\le X\le 8)=2\left(\frac{1}{11}\right)=\frac{2}{11}.
\]

\textbf{Interpretation.}
First solve for $k$ using total area, then compute probabilities by area of rectangles.

Find:
\begin{itemize}
    \item[(a)] $k$,
    \item[(b)] verify that all density values are valid for a PDF,
    \item[(c)] $P(1\le X\le 6)$.
\end{itemize}

\textbf{Solution.}
Total area condition:
\[
2(k)+3(2k)+4\left(\frac{k}{2}\right)=1
\Rightarrow 2k+6k+2k=1
\Rightarrow 10k=1
\Rightarrow k=\frac{1}{10}.
\]
Then heights are
\[
\frac{1}{10},\quad \frac{1}{5},\quad \frac{1}{20},
\]
all nonnegative, so valid.
\[
P(1\le X\le 6)=1\left(\frac{1}{10}\right)+3\left(\frac{1}{5}\right)+1\left(\frac{1}{20}\right)
=\frac{2}{20}+\frac{12}{20}+\frac{1}{20}=\frac{3}{4}.
\]

\textbf{Interpretation.}
A valid PDF needs both nonnegative heights and total area exactly 1.

\]
Most mass is below 3.

\begin{itemize}

    \item[(b)] $P(-1\le X\le 4)$,
    \item[(c)] which is larger: $P(X<0)$ or $P(X\ge 2)$?
\end{itemize}

\textbf{Solution.}
Use total area:
\[
3k+2(2k)+4\left(\frac{1}{12}\right)=1
\Rightarrow 7k+\frac{1}{3}=1
\Rightarrow 7k=\frac{2}{3}
\Rightarrow k=\frac{2}{21}.
\]
\[
P(-1\le X\le 4)=1\left(\frac{2}{21}\right)+2\left(\frac{4}{21}\right)+2\left(\frac{1}{12}\right)
=\frac{2}{21}+\frac{8}{21}+\frac{1}{6}
=\frac{13}{21}+\frac{1}{6}
=\frac{11}{14}.
\]
\[
P(X<0)=3\left(\frac{2}{21}\right)=\frac{2}{7},
\qquad
P(X\ge 2)=4\left(\frac{1}{12}\right)=\frac{1}{3}.
\]
So $P(X\ge 2)$ is larger.

\textbf{Interpretation.}
A wider interval with a moderate height can beat a narrower interval with a higher height.

\section{Problem 10: Symbolic Endpoint (Conceptual Effect of Width)}

f(x)=

\dfrac{2}{3m}, & m\le x\le 2m,\\
0, & \text{otherwise.}
\end{cases}

\addplot[thick] coordinates {(1,1/3) (1,2/3)};
\end{axis}
\end{tikzpicture}
\end{center}

\textbf{Question.}
Find:
\begin{itemize}
    \item[(a)] verify that $f$ is a valid PDF,
    \item[(b)] $P\left(X<\dfrac{3m}{2}\right)$,
    \item[(c)] explain what happens to this probability if $m$ is doubled.
\end{itemize}

\textbf{Solution.}
\[
m\left(\frac{1}{3m}\right)+m\left(\frac{2}{3m}\right)=\frac{1}{3}+\frac{2}{3}=1,
\]
so $f$ is valid and nonnegative.
\[
P\left(X<\frac{3m}{2}\right)=m\left(\frac{1}{3m}\right)+\frac{m}{2}\left(\frac{2}{3m}\right)=\frac{1}{3}+\frac{1}{3}=\frac{2}{3}.
\]
If $m$ is doubled, both widths scale by the same factor and densities scale down accordingly, so this probability stays $\frac{2}{3}$.

\textbf{Interpretation.}
Rescaling the horizontal axis can leave certain relative probabilities unchanged.

\section{Problem 11: Infer Missing Endpoint from a Probability Condition}

\textbf{Problem.}
Let $b>2$ and let
\[
f(x)=
\begin{cases}
c, & 0\le x<2,\\
\dfrac{c}{2}, & 2\le x\le b,\\
0, & \text{otherwise.}
\end{cases}
\]
Suppose $P(X>2)=\dfrac{1}{3}$.

\addplot[draw=none, fill=blue!20, fill opacity=0.35] coordinates {(2,0) (2,1/6) (4,1/6) (4,0)} \closedcycle;
\addplot[thick] coordinates {(0,1/3) (2,1/3)};
\addplot[thick] coordinates {(2,1/6) (4,1/6)};
\addplot[thick] coordinates {(2,1/6) (2,1/3)};
\end{axis}
\end{tikzpicture}
\end{center}

\textbf{Question.}
Find:
\begin{itemize}
    \item[(a)] $c$,
    \item[(b)] $b$,
    \item[(c)] $P(1\le X\le 3)$.
\end{itemize}

\textbf{Solution.}
Given
\[
P(X>2)=(b-2)\left(\frac{c}{2}\right)=\frac{1}{3}. \tag{1}
\]
Total area is 1:
\[
2c+(b-2)\left(\frac{c}{2}\right)=1.
\]
Using (1):
\[
2c+\frac{1}{3}=1 \Rightarrow 2c=\frac{2}{3} \Rightarrow c=\frac{1}{3}.
\]
Now substitute into (1):
\[
(b-2)\left(\frac{1}{6}\right)=\frac{1}{3}
\Rightarrow b-2=2
\Rightarrow b=4.
\]
Finally,
\[
P(1\le X\le 3)=1\left(\frac{1}{3}\right)+1\left(\frac{1}{6}\right)=\frac{1}{2}.
\]

\textbf{Interpretation.}
A probability condition plus total-area condition can determine unknown parameters.

\section{Problem 12: Rich Piecewise Uniform Design Problem}

\textbf{Problem.}
Let $X$ have density
\[
f(x)=
\begin{cases}
k, & 0\le x<1,\\
2k, & 1\le x<4,\\
m, & 4\le x<7,\\
k, & 7\le x\le 10,\\
0, & \text{otherwise,}
\end{cases}
\]
where $k,m>0$. Assume
\[
P(4\le X<7)=P(0\le X<1)+P(7\le X\le 10).
\]
\begin{center}
\begin{tikzpicture}
\begin{axis}[
  width=9cm,
  height=5cm,
  axis lines=left,
  xlabel={$x$},
  ylabel={$f(x)$},

\addplot[draw=none, fill=blue!20, fill opacity=0.35] coordinates {(4,0) (4,2/21) (7,2/21) (7,0)} \closedcycle;
\addplot[draw=none, fill=blue!20, fill opacity=0.35] coordinates {(7,0) (7,1/14) (10,1/14) (10,0)} \closedcycle;
\addplot[thick] coordinates {(0,1/14) (1,1/14)};
\addplot[thick] coordinates {(1,1/7) (4,1/7)};

\addplot[thick] coordinates {(7,1/14) (10,1/14)};
\addplot[thick] coordinates {(1,1/14) (1,1/7)};
\addplot[thick] coordinates {(4,2/21) (4,1/7)};
\addplot[thick] coordinates {(7,1/14) (7,2/21)};
\end{axis}
\end{tikzpicture}
\end{center}

\textbf{Question.}
Find:
\begin{itemize}
    \item[(a)] $k$ and $m$,
    \item[(b)] $P(2\le X\le 8)$,
    \item[(c)] compare $P(1\le X<4)$ and $P(4\le X\le 10)$,
    \item[(d)] explain where the distribution places most of its mass and why.
\end{itemize}

\textbf{Solution.}
From the condition:
\[
3m=1\cdot k+3\cdot k=4k
\Rightarrow m=\frac{4k}{3}. \tag{1}
\]
Total area:
\[
1(k)+3(2k)+3(m)+3(k)=1
\Rightarrow 10k+3m=1.
\]
Using (1):
\[
10k+4k=1 \Rightarrow 14k=1 \Rightarrow k=\frac{1}{14},
\qquad
m=\frac{4}{3}\cdot\frac{1}{14}=\frac{2}{21}.
\]
Now
\[
P(2\le X\le 8)=2\left(\frac{1}{7}\right)+3\left(\frac{2}{21}\right)+1\left(\frac{1}{14}\right)
=\frac{2}{7}+\frac{2}{7}+\frac{1}{14}=\frac{9}{14}.
\]
\[
P(1\le X<4)=3\left(\frac{1}{7}\right)=\frac{3}{7},
\qquad
P(4\le X\le 10)=3\left(\frac{2}{21}\right)+3\left(\frac{1}{14}\right)=\frac{4}{7}.
\]
So $P(4\le X\le 10)$ is larger.

\textbf{Interpretation.}
The right side $[4,10]$ gets more mass because it combines substantial width with moderate heights across two intervals.
