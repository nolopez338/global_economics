\newcommand{\COneDensityGraph}[5]{
\begin{center}
\begin{tikzpicture}
\pgfmathsetmacro{\supportA}{#1}
\pgfmathsetmacro{\supportB}{#2}
\pgfmathsetmacro{\pad}{0.12*(\supportB-\supportA)}
\begin{axis}[
  width=9cm,
  height=5cm,
  axis lines=left,
  xlabel={$x$},
  ylabel={$f(x)$},
  xmin={\supportA-\pad},
  xmax={\supportB+\pad},
  ymin=0,
  ymax=#4,
  grid=both,
  xtick={#5}
]
\addplot[domain=#1:#2, samples=200, draw=none, fill=blue!20, fill opacity=0.35] {#3} \closedcycle;
\addplot[smooth, domain=#1:#2, samples=200] {#3};
\end{axis}
\end{tikzpicture}
\end{center}
}

\section{Water Storage Level (Uniform)}

\textbf{Problem.}
Let $X$ be the water storage level (in m) with support $[0,4]$ and density
\[
f(x)=k\quad\text{for }0\le x\le 4.
\]
\COneDensityGraph{0}{4}{1/4}{0.35}{0,1,2,3,4}

\textbf{Question.}
\begin{itemize}
  \item \textbf{C1 Calculations:} Find $k$ so $f(x)$ is a valid PDF.
  \item \textbf{C1 Interpretation:} Explain what this model says about equal-length intervals.
  \item \textbf{Extra analysis:} Find $P(1\le X\le 3)$.
\end{itemize}
