\section{Water Storage Level (Uniform)}

\textbf{Problem.}
Let $X$ be the water storage level (in m) with support $[0,4]$ and density
\[
f(x)=k\quad\text{for }0\le x\le 4.
\]
\COneDensityGraph{0}{4}{1/4}{0.35}{0,1,2,3,4}

\textbf{Question.}
\begin{itemize}
  \item \textbf{C1 Calculations:} Find $k$ so $f(x)$ is a valid PDF.
  \item \textbf{C1 Interpretation:} Explain what this model says about equal-length intervals.
  \item \textbf{Extra analysis:} Find $P(1\le X\le 3)$.
\end{itemize}



\textbf{Solution.}
\begin{itemize}
  \item Support: $0\le x\le 4$. Since $f(x)=k$ is constant and $k>0$, the density is nonnegative on its support.
  \item Valid PDF condition:
  \[
  \int_0^4 k\,dx=4k=1\implies k=\frac14.
  \]
  \item Probability:
  \[
  P(1\le X\le 3)=\int_1^3 \frac14\,dx=\frac{3-1}{4}=\frac12.
  \]
  Geometrically, this is the area of a rectangle of width $2$ and height $\tfrac14$.
\end{itemize}
\textbf{Final answer:} $k=\dfrac14$ and $P(1\le X\le 3)=\dfrac12$. Equal-length intervals have equal probability because the density height is constant.

\section{Startup Delay (Linear, Endpoint at 0)}

\textbf{Problem.}
Let $T$ be a startup delay (in s) with support $[0,6]$ and density
\[
f(t)=kt\quad\text{for }0\le t\le 6.
\]
\COneDensityGraph{0}{6}{x/18}{0.40}{0,2,4,6}

\textbf{Question.}
\begin{itemize}
  \item \textbf{C1 Calculations:} Find $k$ so $f(t)$ is a valid PDF.
  \item \textbf{C1 Interpretation:} Describe how probability changes as $t$ increases.
  \item \textbf{Extra analysis:} Find $P(T\ge 4)$.
\end{itemize}



\textbf{Solution.}
\begin{itemize}
  \item Support: $0\le t\le 6$. On this support, $f(t)=kt\ge 0$ for $k>0$.
  \item Valid PDF condition:
  \[
  \int_0^6 kt\,dt=k\cdot\frac{6^2}{2}=18k=1\implies k=\frac1{18}.
  \]
  \item Probability:
  \[
  P(T\ge 4)=\int_4^6 \frac{t}{18}\,dt=\left.\frac{t^2}{36}\right|_4^6=\frac{36-16}{36}=\frac59.
  \]
\end{itemize}
\textbf{Final answer:} $k=\dfrac1{18}$ and $P(T\ge 4)=\dfrac59$. As $t$ increases, density increases linearly, so larger delays are more likely than equal-width smaller delays.

\section{Packing Score (Linear, Both Endpoints Nonzero)}

\textbf{Problem.}
Let $Q$ be a packing score with support $[2,8]$ and density
\[
f(q)=k(q-1)\quad\text{for }2\le q\le 8.
\]
\COneDensityGraph{2}{8}{(x-1)/27}{0.35}{2,4,6,8}

\textbf{Question.}
\begin{itemize}
  \item \textbf{C1 Calculations:} Find $k$ so $f(q)$ is a valid PDF.
  \item \textbf{C1 Interpretation:} Compare density near $q=3$ and $q=7$.
  \item \textbf{Extra analysis:} Find $P(4\le Q\le 6)$.
\end{itemize}



\textbf{Solution.}
\begin{itemize}
  \item Support: $2\le q\le 8$. Since $q-1>0$ on this support, $f(q)=k(q-1)\ge 0$ when $k>0$.
  \item Valid PDF condition:
  \[
  \int_2^8 k(q-1)\,dq
  =k\left.\frac{(q-1)^2}{2}\right|_2^8
  =k\cdot\frac{49-1}{2}=24k=1
  \implies k=\frac1{24}.
  \]
  \item Probability:
  \[
  P(4\le Q\le 6)=\int_4^6 \frac{q-1}{24}\,dq
  =\frac1{24}\left.\frac{(q-1)^2}{2}\right|_4^6
  =\frac1{24}\cdot\frac{25-9}{2}=\frac13.
  \]
\end{itemize}
\textbf{Final answer:} $k=\dfrac1{24}$ and $P(4\le Q\le 6)=\dfrac13$. Density near $q=7$ is larger than near $q=3$ because the line rises with $q$.

\[
f(a)=k(5-a)\quad\text{for }0\le a\le 5.

\COneDensityGraph{0}{5}{2*(5-x)/25}{0.45}{0,1,2,3,4,5}

\textbf{Question.}
\begin{itemize}
  \item \textbf{C1 Calculations:} Find $k$ so $f(a)$ is a valid PDF.
  \item \textbf{C1 Interpretation:} Explain what the decreasing line says about short vs long times.
  \item \textbf{Extra analysis:} Find $P(1\le A\le 4)$.
\end{itemize}



\textbf{Solution.}
\begin{itemize}
  \item Support: $0\le a\le 5$. Since $5-a\ge0$ on this support, $f(a)=k(5-a)\ge0$ if $k>0$.
  \item Valid PDF condition:
  \[
  \int_0^5 k(5-a)\,da=k\left(25-\frac{25}{2}\right)=\frac{25k}{2}=1

  \[
  P(1\le A\le 4)=\int_1^4 \frac{2}{25}(5-a)\,da
  =\frac{2}{25}\left.\left(5a-\frac{a^2}{2}\right)\right|_1^4
  =\frac35.
  \]
\end{itemize}
\textbf{Final answer:} $k=\dfrac{2}{25}$ and $P(1\le A\le 4)=\dfrac35$. The decreasing line means shorter times are more probable than equally wide longer-time intervals.

\section{Quality Index (Linear, Both Endpoints Nonzero)}

\textbf{Problem.}
Let $R$ be a quality index with support $[1,7]$ and density
\[
f(r)=k(r+1)\quad\text{for }1\le r\le 7.
\]
\COneDensityGraph{1}{7}{(x+1)/30}{0.30}{1,2,3,4,5,6,7}

\textbf{Question.}
\begin{itemize}
  \item \textbf{C1 Calculations:} Find $k$ so $f(r)$ is a valid PDF.
  \item \textbf{C1 Interpretation:} Explain why right-side intervals have more probability than equal-width left-side intervals.
  \item \textbf{Extra analysis:} Find $P(R\le 4)$.
\end{itemize}



\textbf{Solution.}
\begin{itemize}
  \item Support: $1\le r\le 7$. Since $r+1>0$ on this support, $f(r)=k(r+1)\ge0$ for $k>0$.
  \item Valid PDF condition:
  \[
  \int_1^7 k(r+1)\,dr
  =k\left.\left(\frac{r^2}{2}+r\right)\right|_1^7
  =k(30)=1
  \implies k=\frac1{30}.
  \]
  \item Probability:
  \[
  P(R\le 4)=\int_1^4 \frac{r+1}{30}\,dr
  =\frac1{30}\left.\left(\frac{r^2}{2}+r\right)\right|_1^4
  =\frac{7}{20}.
  \]
\end{itemize}
\textbf{Final answer:} $k=\dfrac1{30}$ and $P(R\le 4)=\dfrac7{20}$. Equal-width intervals farther right have larger probability because the density height is larger there.

\end{itemize}
\textbf{Final answer:} $k=\dfrac1{p-3}$ and $P(4.5\le M\le p-1)=\dfrac{p-\tfrac{11}{2}}{p-3}$ (for $p\ge\tfrac{11}{2}$). Increasing $p$ widens support and lowers the rectangle height.

\section{Battery Discharge (Piecewise Linear with One Variable Endpoint)}

\textbf{Problem.}
Let $B$ be battery discharge measure with support $[0,p]$ and density
\[
f(b)=\begin{cases}
k\left(1+\dfrac{b}{3}\right),&0\le b\le 3,\\[2mm]
k\left(2-\dfrac{b-3}{p-3}\right),&3<b\le p,
\end{cases}
\]
where $p>3$ and $k>0$.

The graph is shown below.
\begin{center}
\begin{tikzpicture}[scale=0.88]
  \draw[->] (-0.3,0) -- (10.4,0) node[right] {$b$};
  \draw[->] (0,-0.2) -- (0,2.9) node[above] {$f(b)$};
  \draw[thick] (0,1.0) -- (3,2.0) -- (9.6,1.0);
  \draw[dashed] (3,0) -- (3,2.0);
  \draw[dashed] (9.6,0) -- (9.6,1.0);
  \node[below] at (0,0) {$0$};
  \node[below] at (3,0) {$3$};
  \node[below] at (9.6,0) {$p$};
  \node[left] at (0,1.0) {$k$};
  \node[left] at (0,2.0) {$2k$};
\end{tikzpicture}
\end{center}

\textbf{Question.}
\begin{itemize}
  \item \textbf{C1 Calculations:} Use total area to find $k$ as a function of $p$.
  \item \textbf{C1 Interpretation:} Explain how increasing $p$ changes the length of the falling segment and the distribution of area.
  \item \textbf{Extra analysis:} Find $P(2\le B\le p-1)$ in terms of $p$.
\end{itemize}



\textbf{Solution.}
\begin{itemize}
  \item Support: $0\le b\le p$, $p>3$. Each branch is linear and nonnegative on its interval when $k>0$.
  \item Normalization by geometry:
  \[
  \text{Area}_{[0,3]}=\frac{3(k+2k)}2=\frac{9k}{2},\qquad
  \text{Area}_{[3,p]}=\frac{(p-3)(2k+k)}2=\frac{3k(p-3)}2.
  \]
  So
  \[
  \frac{9k}{2}+\frac{3k(p-3)}2=\frac{3k(p+3)}2=1
  \implies k=\frac{2}{3(p+3)}.
  \]
  \item Probability:
  \[
  P(2\le B\le p-1)=\int_2^3 k\left(1+\frac b3\right)db+\int_3^{p-1}k\left(2-\frac{b-3}{p-3}\right)db.
  \]
  This gives
  \[
  P(2\le B\le p-1)=\frac{9p^2-49p+63}{9(p^2-9)}.
  \]
\end{itemize}
\textbf{Final answer:} $k=\dfrac{2}{3(p+3)}$ and $P(2\le B\le p-1)=\dfrac{9p^2-49p+63}{9(p^2-9)}$. Increasing $p$ stretches the falling segment, redistributing area toward the right side.

\section{Model Comparison with a Variable Right Endpoint}

\textbf{Problem.}
Two density models are given:
\[
\text{Model A: }f_A(x)=\begin{cases}
\dfrac{x}{8},&0\le x\le 4,\\[2mm]
\dfrac{8-x}{8},&4<x\le 8,
\end{cases}
\]
\[
\text{Model B: }f_B(x)=\begin{cases}
kx,&0\le x\le 6,\\[2mm]
\dfrac{6k(p-x)}{p-6},&6<x\le p,
\end{cases}
\]
where $p>7$.

The two graphs are shown below.

  \node[below] at (6,0) {$6$};
  \node[below] at (8,0) {$8$};
  \node[below] at (10,0) {$p$};
  \node[blue] at (4.8,2.3) {Model A};
  \node[red] at (7.3,2.1) {Model B};
\end{tikzpicture}
\end{center}

\textbf{Question.}
\begin{itemize}
  \item \textbf{C1 Validity check:} Verify geometrically that Model A is a valid PDF, then use total area to find $k$ for Model B in terms of $p$.
  \item \textbf{C1 Calculations:} Find $P_A(5\le X\le 7)$ and $P_B(5\le X\le 7)$ in terms of $p$.
  \item \textbf{C1 Interpretation:} Decide for which values of $p$ Model B gives larger probability on $[5,7]$ than Model A.
  \item \textbf{Extra analysis:} Explain how shifting the right endpoint $p$ changes the right-tail concentration in Model B.
\end{itemize}



\textbf{Solution.}
\begin{itemize}
  \item Model A check (as written):
  \[
  \int_0^4\frac{x}{8}dx+\int_4^8\frac{8-x}{8}dx=1+1=2\neq1.
  \]
  So Model A is nonnegative but not normalized as written.
  \item Model B normalization:
  \[
  \int_0^6 kx\,dx+\int_6^p\frac{6k(p-x)}{p-6}\,dx
  =18k+3k(p-6)=3kp=1,
  \]
  hence
  \[
  k=\frac1{3p}.
  \]
  \item Interval probabilities:
  \[
  P_A(5\le X\le 7)=\int_5^7\frac{8-x}{8}dx=\frac12.
  \]
  \[
  P_B(5\le X\le 7)=\int_5^6 kx\,dx+\int_6^7\frac{6k(p-x)}{p-6}dx
  =\frac{23p-144}{6p(p-6)}.
  \]
  \item Comparison:
  \[
  P_B>P_A\iff \frac{23p-144}{6p(p-6)}>\frac12
  \iff 3p^2-41p+144<0,
  \]
  but $3p^2-41p+144>0$ for all real $p$, so no $p>7$ satisfies this.
\end{itemize}
\textbf{Final answer:} Model A is not a valid PDF as written; for Model B, $k=\dfrac1{3p}$ and $P_B(5\le X\le 7)=\dfrac{23p-144}{6p(p-6)}<\dfrac12=P_A(5\le X\le 7)$ for all $p>7$.

\section{Delivery Score Generalization (Two Variable Endpoints)}

\textbf{Problem.}
Let $D$ have support $[a,b]$ and density
\[
f(d)=k(d-a)\quad\text{for }a\le d\le b,
\]
where $a<b$.
\begin{center}
\begin{tikzpicture}[scale=0.9]
  \draw[->] (-0.3,0) -- (8.8,0) node[right] {$d$};
  \draw[->] (0,-0.2) -- (0,2.6) node[above] {$f(d)$};
  \draw[thick] (1,0) -- (7.8,2.0);
  \draw[dashed] (1,0) -- (1,0.02);
  \draw[dashed] (7.8,0) -- (7.8,2.0);
  \node[below] at (1,0) {$a$};
  \node[below] at (7.8,0) {$b$};
  \node[left] at (0,2.0) {$k(b-a)$};
\end{tikzpicture}
\end{center}

\textbf{Question.}
\begin{itemize}
  \item \textbf{C1 Calculations:} Find $k$ in terms of $a$ and $b$ so $f(d)$ is a valid PDF.
  \item \textbf{C1 Symbolic probability:} Find $P\!\left(a\le D\le \dfrac{a+b}{2}\right)$.
  \item \textbf{C1 Interpretation:} Explain how changing $a$ and $b$ (while keeping $b-a$ fixed, or changing $b-a$) affects the graph and probabilities.
\end{itemize}



\textbf{Solution.}
\begin{itemize}
  \item Support: $a\le d\le b$ with $a<b$. Since $d-a\ge0$, $f(d)=k(d-a)\ge0$ for $k>0$.
  \item Normalize:
  \[
  \int_a^b k(d-a)\,dd
  =k\left.\frac{(d-a)^2}{2}\right|_a^b
  =k\frac{(b-a)^2}{2}=1
  \implies k=\frac{2}{(b-a)^2}.
  \]
  \item Symbolic probability:
  \[
  P\!\left(a\le D\le\frac{a+b}{2}\right)
  =\int_a^{(a+b)/2}k(d-a)\,dd
  =k\frac{(b-a)^2}{8}=\frac14.
  \]
\end{itemize}
\textbf{Final answer:} $k=\dfrac{2}{(b-a)^2}$ and $P\!\left(a\le D\le\frac{a+b}{2}\right)=\dfrac14$. Shifting $a,b$ together translates the graph; changing $(b-a)$ changes spread and peak height inversely.

\section{Clinic Waiting Time (Algebra + Geometry with Two Variable Endpoints)}

\textbf{Problem.}
Let $W$ have support $[a,b]$ and density
\[
f(w)=\begin{cases}
\dfrac{2h}{b-a}(w-a),&a\le w\le \dfrac{a+b}{2},\\[2mm]
h-\dfrac{h}{b-a}\left(w-\dfrac{a+b}{2}\right),&\dfrac{a+b}{2}<w\le b,
\end{cases}
\]
where $a<b$ and $h>0$.

The graph is shown below.
\begin{center}
\begin{tikzpicture}[scale=0.88]
  \draw[->] (-0.3,0) -- (9.4,0) node[right] {$w$};
  \draw[->] (0,-0.2) -- (0,2.9) node[above] {$f(w)$};
  \draw[thick] (1,0) -- (5,2.2) -- (8.8,1.1);
  \draw[dashed] (5,0) -- (5,2.2);
  \draw[dashed] (8.8,0) -- (8.8,1.1);
  \node[below] at (1,0) {$a$};
  \node[below] at (5,0) {$\dfrac{a+b}{2}$};
  \node[below] at (8.8,0) {$b$};
  \node[left] at (0,2.2) {$h$};
  \node[left] at (0,1.1) {$\dfrac{h}{2}$};
\end{tikzpicture}
\end{center}

\textbf{Question.}
\begin{itemize}
  \item \textbf{C1 Calculations:} Use geometry (triangle + trapezoid) to find $h$ in terms of $a$ and $b$.
  \item \textbf{C1 Interval probability:} Find $P\!\left(W\ge \dfrac{a+b}{2}\right)$.
  \item \textbf{C1 Interpretation:} Compare the area on $\left[a,\dfrac{a+b}{2}\right]$ and $\left[\dfrac{a+b}{2},b\right]$ and explain what this says about early vs later waiting times.
\end{itemize}



\textbf{Solution.}
\begin{itemize}
  \item Let $L=b-a$. The graph has a left triangle on $[a,\frac{a+b}{2}]$ and a right trapezoid on $[\frac{a+b}{2},b]$.
  \item Normalize by area:
  \[
  A_{\text{left}}=\frac12\cdot\frac{L}{2}\cdot h=\frac{hL}{4},
  \qquad
  A_{\text{right}}=\frac{\frac{L}{2}(h+\frac h2)}{2}=\frac{3hL}{8}.
  \]
  Total area $=1$ gives
  \[
  \frac{hL}{4}+\frac{3hL}{8}=\frac{5hL}{8}=1
  \implies h=\frac{8}{5(b-a)}.
  \]
  \item Interval probability:
  \[
  P\!\left(W\ge\frac{a+b}{2}\right)=A_{\text{right}}=\frac{3hL}{8}=\frac35.
  \]
\end{itemize}
\textbf{Final answer:} $h=\dfrac{8}{5(b-a)}$ and $P\!\left(W\ge\frac{a+b}{2}\right)=\dfrac35$. The right half has larger area than the left half, so later waiting times are more likely here.

\section{Capstone: Two-Endpoint Ramp-Plateau-Ramp Family}

\textbf{Problem.}
Let $T$ have support $[a,b]$ and density
\[
f(t)=\begin{cases}
\dfrac{3k}{b-a}(t-a),&a\le t\le a+\dfrac{b-a}{3},\\[2mm]
k,&a+\dfrac{b-a}{3}<t\le a+\dfrac{2(b-a)}{3},\\[2mm]
\dfrac{3k}{b-a}(b-t),&a+\dfrac{2(b-a)}{3}<t\le b,
\end{cases}
\]
where $a<b$ and $k>0$.

The graph is shown below.
\begin{center}
\begin{tikzpicture}[scale=0.82]
  \draw[->] (-0.3,0) -- (10.8,0) node[right] {$t$};
  \draw[->] (0,-0.2) -- (0,3.1) node[above] {$f(t)$};
  \draw[thick] (1,0) -- (4.2,2.1) -- (7.4,2.1) -- (10.6,0);
  \draw[dashed] (4.2,0) -- (4.2,2.1);
  \draw[dashed] (7.4,0) -- (7.4,2.1);
  \node[below] at (1,0) {$a$};
  \node[below] at (4.2,0) {$a+\dfrac{b-a}{3}$};
  \node[below] at (7.4,0) {$a+\dfrac{2(b-a)}{3}$};
  \node[below] at (10.6,0) {$b$};
  \node[left] at (0,2.1) {$k$};
\end{tikzpicture}
\end{center}

\textbf{Question.}
\begin{itemize}
  \item \textbf{C1 Parameter solving:} Use total area to find $k$ in terms of $a$ and $b$.
  \item \textbf{C1 Symbolic probability:} Find $P\!\left(T\le a+\dfrac{2(b-a)}{3}\right)$.
  \item \textbf{C1 Region comparison:} Compare $P\!\left(a\le T\le a+\dfrac{b-a}{3}\right)$ and $P\!\left(a+\dfrac{2(b-a)}{3}\le T\le b\right)$.
  \item \textbf{C1 Capstone interval:} Find $P\!\left(a+\dfrac{b-a}{6}\le T\le a+\dfrac{5(b-a)}{6}\right)$.
  \item \textbf{Extra analysis (capstone):} Explain, using area arguments, how increasing $(b-a)$ changes the graph height and spread while preserving total probability.
\end{itemize}



\textbf{Solution.}
\begin{itemize}
  \item Let $L=b-a$. Areas: left triangle $\frac12\cdot\frac L3\cdot k=\frac{kL}{6}$, middle rectangle $\frac L3\cdot k=\frac{kL}{3}$, right triangle $\frac{kL}{6}$.
  \item Normalize:
  \[
  \frac{kL}{6}+\frac{kL}{3}+\frac{kL}{6}=\frac{2kL}{3}=1
  \implies k=\frac{3}{2(b-a)}.
  \]
  \item Probability to the second breakpoint:
  \[
  P\!\left(T\le a+\frac{2(b-a)}{3}\right)=\frac{kL}{6}+\frac{kL}{3}=\frac{kL}{2}=\frac34.
  \]
  \item Region comparison:
  \[
  P\!\left(a\le T\le a+\frac{b-a}{3}\right)=\frac{kL}{6}=\frac14,
  \quad
  P\!\left(a+\frac{2(b-a)}{3}\le T\le b\right)=\frac{kL}{6}=\frac14.
  \]
  \item Capstone interval:
  The excluded tiny triangles on both ends each have area $\frac{kL}{24}$. Hence
  \[
  P\!\left(a+\frac{b-a}{6}\le T\le a+\frac{5(b-a)}{6}\right)
  =1-2\cdot\frac{kL}{24}
  =1-\frac18
  =\frac78.
  \]
\end{itemize}
\textbf{Final answer:} $k=\dfrac{3}{2(b-a)}$, $P\!\left(T\le a+\dfrac{2(b-a)}{3}\right)=\dfrac34$, first-third and last-third probabilities are equal ($\dfrac14$ each), and $P\!\left(a+\dfrac{b-a}{6}\le T\le a+\dfrac{5(b-a)}{6}\right)=\dfrac78$.

\section{Two-Parameter Ramp-and-Cap Model (Advanced Piecewise Linear)}

\textbf{Problem.}
Let $X$ have support $[0,6]$ and density
\[
f(x)=\begin{cases}
a+bx,&0\le x\le 2,\\
4a-ax,&2<x\le 4,\\
a,&4<x\le 6,
\end{cases}
\]
where $a>0$ and $b$ are constants.

The graph is shown below.
\begin{center}
\begin{tikzpicture}[scale=0.9]
  \draw[->] (-0.3,0) -- (6.7,0) node[right] {$x$};
  \draw[->] (0,-0.2) -- (0,2.7) node[above] {$f(x)$};
  \draw[thick] (0,0.6) -- (2,2.0);
  \draw[thick] (2,2.0) -- (4,0.6);
  \draw[thick] (4,0.6) -- (6,0.6);
  \draw[dashed] (2,0) -- (2,2.0);
  \draw[dashed] (4,0) -- (4,0.6);
  \node[below] at (0,0) {$0$};
  \node[below] at (2,0) {$2$};
  \node[below] at (4,0) {$4$};
  \node[below] at (6,0) {$6$};
  \node[left] at (0,0.6) {$a$};
  \node[left] at (0,2.0) {$2a$};
\end{tikzpicture}
\end{center}

\textbf{Question.}
\begin{itemize}
  \item \textbf{C1 Calculations:} Use continuity at $x=2$ to write $b$ in terms of $a$, then use total area $=1$ to find both $a$ and $b$.
  \item \textbf{C1 Symbolic probability:} Find $P(1\le X\le 5)$ exactly.
  \item \textbf{C1 Comparison:} Compare $P(0\le X\le 2)$ and $P(4\le X\le 6)$.
  \item \textbf{Extra analysis:} If the middle peak is changed from $2a$ to $pa$ (with $p>1$) while keeping the same breakpoints and right plateau height $a$, describe how increasing $p$ redistributes probability mass.
\end{itemize}



\textbf{Solution.}
\begin{itemize}
  \item Continuity at $x=2$:
  \[
  a+2b=4a-2a=2a\implies b=\frac a2.
  \]
  \item Normalize by areas (or piecewise integrals):
  \[
  A_{[0,2]}=3a,\quad A_{(2,4]}=3a,\quad A_{(4,6]}=2a,
  \]
  so $8a=1\implies a=\frac18$, hence $b=\frac1{16}$.
  \item Probability:
  \[
  P(1\le X\le 5)=\int_1^2(a+bx)dx+\int_2^4(4a-ax)dx+\int_4^5 a\,dx
  =\frac{19}{32}.
  \]
  \item Comparison:
  \[
  P(0\le X\le 2)=3a=\frac38,
  \qquad
  P(4\le X\le 6)=2a=\frac14,
  \]
  so the first interval is larger.
\end{itemize}
\textbf{Final answer:} $a=\dfrac18$, $b=\dfrac1{16}$, $P(1\le X\le 5)=\dfrac{19}{32}$, and $P(0\le X\le 2)>P(4\le X\le 6)$.

\section{Model Comparison: Left-Heavy vs Right-Heavy Triangular PDFs}

\textbf{Problem.}
Two density models are defined on the same support $[0,8]$:
\[
\text{Model A: }f_A(x)=\begin{cases}
\dfrac{x}{12},&0\le x\le 4,\\[2mm]
\dfrac{8-x}{12},&4<x\le 8,
\end{cases}
\]
\[
\text{Model B: }f_B(x)=\begin{cases}
\dfrac{x}{16},&0\le x\le 6,\\[2mm]
\dfrac{8-x}{16/3},&6<x\le 8.
\end{cases}
\]

The two graphs are shown below.
\begin{center}
\begin{tikzpicture}[scale=0.85]
  \draw[->] (-0.3,0) -- (8.8,0) node[right] {$x$};
  \draw[->] (0,-0.2) -- (0,2.9) node[above] {$f(x)$};
  \draw[thick,blue] (0,0) -- (4,2.0) -- (8,0);
  \draw[thick,red] (0,0) -- (6,1.5) -- (8,0);
  \draw[dashed,blue] (4,0) -- (4,2.0);
  \draw[dashed,red] (6,0) -- (6,1.5);
  \node[below] at (0,0) {$0$};
  \node[below] at (4,0) {$4$};
  \node[below] at (6,0) {$6$};
  \node[below] at (8,0) {$8$};
  \node[blue] at (4.6,2.3) {Model A};
  \node[red] at (6.8,1.8) {Model B};
\end{tikzpicture}
\end{center}

\textbf{Question.}
\begin{itemize}
  \item \textbf{C1 Validity check:} Verify geometrically that each model is a valid PDF.
  \item \textbf{C1 Calculations:} Compute $P_A(5\le X\le 7)$ and $P_B(5\le X\le 7)$.
  \item \textbf{C1 Interpretation:} Decide which model gives larger probability on $[5,7]$ and explain using peak location, slope, and concentration of area.
  \item \textbf{Extra analysis:} Compare $P_A(X\ge 6)$ and $P_B(X\ge 6)$ and explain why the ordering is consistent (or not) with your previous comparison.
\end{itemize}



\textbf{Solution.}
\begin{itemize}
  \item Validity check from total area (as written):
  \[
  \int_0^8 f_A(x)dx=\frac{4}{3},\qquad
  \int_0^8 f_B(x)dx=\frac32,
  \]
  so neither is normalized to $1$ as written.
  \item Requested interval computations:
  \[
  P_A(5\le X\le 7)=\int_5^7\frac{8-x}{12}dx=\frac13.
  \]
  \[
  P_B(5\le X\le 7)=\int_5^6\frac{x}{16}dx+\int_6^7\frac{3(8-x)}{16}dx
  =\frac{11}{32}+\frac{9}{32}=\frac58.
  \]
  \item Right-tail comparison:
  \[
  P_A(X\ge 6)=\int_6^8\frac{8-x}{12}dx=\frac16,
  \qquad
  P_B(X\ge 6)=\int_6^8\frac{3(8-x)}{16}dx=\frac38.
  \]
\end{itemize}
\textbf{Final answer:} On $[5,7]$, Model B gives the larger value ($\frac58$ vs $\frac13$), and also has larger right-tail mass beyond $6$ ($\frac38$ vs $\frac16$), consistent with its peak being shifted right.

\section{Capstone: Symbolic Trapezoid-Triangle Family}

\textbf{Problem.}
Let $T$ have support $[0,10]$ and density
\[
f(t)=\begin{cases}
kt,&0\le t\le 2,\\
2k,&2<t\le p,\\
\dfrac{2k(10-t)}{10-p},&p<t\le 10,
\end{cases}
\]
where $2<p<10$ and $k>0$.

The graph is shown below.
\begin{center}
\begin{tikzpicture}[scale=0.82]
  \draw[->] (-0.3,0) -- (10.8,0) node[right] {$t$};
  \draw[->] (0,-0.2) -- (0,3.1) node[above] {$f(t)$};
  \draw[thick] (0,0) -- (2,2.0) -- (6.6,2.0) -- (10,0);
  \draw[dashed] (2,0) -- (2,2.0);
  \draw[dashed] (6.6,0) -- (6.6,2.0);
  \node[below] at (0,0) {$0$};
  \node[below] at (2,0) {$2$};
  \node[below] at (6.6,0) {$p$};
  \node[below] at (10,0) {$10$};
  \node[left] at (0,2.0) {$2k$};
\end{tikzpicture}
\end{center}

\textbf{Question.}
\begin{itemize}
  \item \textbf{C1 Parameter solving:} Use total area to show that $k$ is a function of $p$, and simplify your result.
  \item \textbf{C1 Interval probability:} Find $P(T\le p)$ in terms of $p$.
  \item \textbf{C1 Region comparison:} For which values of $p$ is $P(T\le p)>P(T>p)$?
  \item \textbf{C1 Piecewise split:} Find $P(1\le T\le 8)$ as a function of $p$, carefully splitting by region if needed.
  \item \textbf{Extra analysis (capstone):} Describe how increasing $p$ changes the shape (ramp, plateau, and right tail) and explain, with area arguments, how this shifts probability mass between early, middle, and late values.
\end{itemize}



\textbf{Solution.}
\begin{itemize}
  \item Normalize with geometry:
  \[
  A_{[0,2]}=\frac12\cdot2\cdot2k=2k,
  \quad
  A_{(2,p]}=2k(p-2),
  \quad
  A_{(p,10]}=\frac12(10-p)(2k)=k(10-p).
  \]
  So
  \[
  2k+2k(p-2)+k(10-p)=k(p+8)=1
  \implies k=\frac1{p+8}.
  \]
  \item Probability up to $p$:
  \[
  P(T\le p)=2k+2k(p-2)=2k(p-1)=\frac{2(p-1)}{p+8}.
  \]
  \item Comparison with right side:
  \[
  P(T\le p)>P(T>p)\iff P(T\le p)>\frac12
  \iff \frac{2(p-1)}{p+8}>\frac12
  \iff p>4.
  \]
  \item Piecewise probability $P(1\le T\le 8)$:
  \[
  P(1\le T\le 8)=
  \begin{cases}
  \dfrac{-2p^2+5p+142}{2(p+8)(10-p)}, & 2<p<8,\\[2mm]
  \dfrac{27}{2(p+8)}, & 8\le p<10.
  \end{cases}
  \]
\end{itemize}
\textbf{Final answer:} $k=\dfrac1{p+8}$, $P(T\le p)=\dfrac{2(p-1)}{p+8}$, $P(T\le p)>P(T>p)$ when $p>4$, and the $P(1\le T\le 8)$ expression is piecewise as above. Larger $p$ extends the plateau and delays right-tail decay, shifting area toward middle/late values.
