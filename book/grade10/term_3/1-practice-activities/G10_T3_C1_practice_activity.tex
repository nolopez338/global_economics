	\section{Checking Two Candidate Density Functions I}
	
	\textbf{Problem.}
	A continuous random variable is modeled on the interval shown in each case.
	
	Distribution A
	\[
	f(x)=0.25\quad\text{for }0\le x\le 4
	\]
	
	Distribution B
	\[
	f(x)=0.40\quad\text{for }0\le x\le 4
	\]
	
	\textbf{Question.}
	\begin{itemize}
		\item \textbf{C1 Calculations:} Decide which distribution is a valid PDF and which one is not.
		\item \textbf{C1 Interpretation:} Justify your answer using the total area under each graph.
		\item \textbf{Extra analysis:} For the valid distribution, find $P(1\le X\le 3)$.
	\end{itemize}

	\textbf{Solution.}
	For a valid PDF, the density must be nonnegative and total area on its support must be $1$.
	\begin{itemize}
		\item Distribution A: rectangle on $[0,4]$ with height $0.25$, so area $=4(0.25)=1$. Valid PDF.
		\item Distribution B: rectangle on $[0,4]$ with height $0.40$, so area $=4(0.40)=1.6\ne1$. Not a valid PDF.
	\end{itemize}
	For the valid distribution (A),
	\[
	P(1\le X\le 3)=\text{area of rectangle of width }(3-1)=2\text{ and height }0.25=2(0.25)=0.5.
	\]
	\textbf{Final answer:} A is valid, B is not valid, and $P(1\le X\le 3)=0.5$.

	\section{Checking Two Candidate Density Functions II}
	
	\textbf{Problem.}
	A continuous random variable is modeled on the interval shown in each case.
	
	Distribution A
	\[
	f(x)=\frac{x}{8}\quad\text{for }0\le x\le 4
	\]
	
	Distribution B
	\[
	f(x)=\frac{x}{6}\quad\text{for }0\le x\le 4
	\]
	
	\textbf{Question.}
	\begin{itemize}
		\item \textbf{C1 Calculations:} Decide which distribution is a valid PDF and which one is not.
		\item \textbf{C1 Interpretation:} Justify your answer using geometric reasoning about the area under each line.
		\item \textbf{Extra analysis:} For the valid distribution, find $P(2\le X\le 4)$.
	\end{itemize}

	\textbf{Solution.}
	Use geometric area under each line on $0\le x\le4$.
	\begin{itemize}
		\item Distribution A: $f(x)=\frac{x}{8}$ is a triangle with base $4$ and height $\frac{4}{8}=\frac12$.
		\[
		\text{Area}=\frac12(4)\left(\frac12\right)=1.
		\]
		So A is a valid PDF.
		\item Distribution B: $f(x)=\frac{x}{6}$ is a triangle with base $4$ and height $\frac{4}{6}=\frac23$.
		\[
		\text{Area}=\frac12(4)\left(\frac23\right)=\frac43\ne1.
		\]
		So B is not valid.
	\end{itemize}
	For A,
	\[
	P(2\le X\le4)=1-P(0\le X\le2).
	\]
	On $[0,2]$, area is a triangle with base $2$ and height $\frac{2}{8}=\frac14$:
	\[
	P(0\le X\le2)=\frac12(2)\left(\frac14\right)=\frac14.
	\]
	Hence
	\[
	P(2\le X\le4)=1-\frac14=\frac34.
	\]
	\textbf{Final answer:} A is valid, B is not valid, and $P(2\le X\le4)=\frac34$.

	\section{Checking Two Candidate Density Functions III}
	
	\textbf{Problem.}
	A continuous random variable is modeled on the interval shown in each case.
	
	Distribution A
	\[
	f(x)=\frac{5-x}{8}\quad\text{for }1\le x\le 5
	\]
	
	Distribution B
	\[
	f(x)=\frac{5-x}{6}\quad\text{for }1\le x\le 5
	\]
	
	\textbf{Question.}
	\begin{itemize}
		\item \textbf{C1 Calculations:} Decide which distribution is a valid PDF and which one is not.
		\item \textbf{C1 Interpretation:} Justify your answer using geometric reasoning about the area under each line.
		\item \textbf{Extra analysis:} For the valid distribution, find $P(1\le X\le 3)$.
	\end{itemize}

	\textbf{Solution.}
	Check total area on $1\le x\le5$.
	\begin{itemize}
		\item Distribution A: $f(x)=\frac{5-x}{8}$. At $x=1$, height $=\frac{4}{8}=\frac12$; at $x=5$, height $=0$. So area is a triangle:
		\[
		\frac12(5-1)\left(\frac12\right)=\frac12(4)\left(\frac12\right)=1.
		\]
		Valid PDF.
		\item Distribution B: $f(x)=\frac{5-x}{6}$. At $x=1$, height $=\frac{4}{6}=\frac23$; at $x=5$, height $=0$.
		\[
		\text{Area}=\frac12(4)\left(\frac23\right)=\frac43\ne1.
		\]
		Not valid.
	\end{itemize}
	For valid distribution A,
	\[
	P(1\le X\le3)=\text{area of trapezium on }[1,3].
	\]
	Heights are $f(1)=\frac12$ and $f(3)=\frac{2}{8}=\frac14$, width $=2$:
	\[
	P(1\le X\le3)=\frac{\frac12+\frac14}{2}\cdot2=\frac34.
	\]
	\textbf{Final answer:} A is valid, B is not valid, and $P(1\le X\le3)=\frac34$.

	\section{Water Storage Level (Uniform)}
	
	\textbf{Problem.}
	Let $X$ be the water storage level (in m) with support $[0,4]$ and density
	\[
	f(x)=k\quad\text{for }0\le x\le 4.
	\]
	
	\textbf{Question.}
	\begin{itemize}
		\item \textbf{C1 Calculations:} Find $k$ so $f(x)$ is a valid PDF.
		\item \textbf{C1 Interpretation:} Explain what this model says about equal-length intervals.
		\item \textbf{Extra analysis:} Find $P(1\le X\le 3)$.
	\end{itemize}

	\textbf{Solution.}
	Since $f(x)=k$ on $[0,4]$, total area is rectangle area:
	\[
	4k=1\implies k=\frac14.
	\]
	A constant density means any two intervals with the same length have the same probability.
	Now
	\[
	P(1\le X\le3)=\text{width}\times\text{height}=(3-1)\left(\frac14\right)=\frac12.
	\]
	\textbf{Final answer:} $k=\frac14$ and $P(1\le X\le3)=\frac12$.

	\section{Startup Delay (Linear, Endpoint at 0)}
	
	\textbf{Problem.}
	Let $T$ be a startup delay (in s) with support $[0,6]$ and density
	\[
	f(t)=kt\quad\text{for }0\le t\le 6.
	\]
	
	\textbf{Question.}
	\begin{itemize}
		\item \textbf{C1 Calculations:} Find $k$ so $f(t)$ is a valid PDF.
		\item \textbf{C1 Interpretation:} Describe how probability changes as $t$ increases.
		\item \textbf{Extra analysis:} Find $P(T\ge 4)$.
	\end{itemize}

	\textbf{Solution.}
	For $f(t)=kt$ on $[0,6]$, area is a triangle with base $6$ and height $6k$:
	\[
	\frac12(6)(6k)=18k=1\implies k=\frac1{18}.
	\]
	Because the line increases with $t$, larger delays have higher density (are more likely per equal-width interval).
	For $P(T\ge4)$, use area from $4$ to $6$ (a trapezium): heights are $4k$ and $6k$, width $2$.
	\[
	P(T\ge4)=\frac{4k+6k}{2}\cdot2=10k=\frac{10}{18}=\frac59.
	\]
	\textbf{Final answer:} $k=\frac1{18}$ and $P(T\ge4)=\frac59$.

	\textbf{Problem.}
	Let $Q$ be a packing score with support $[2,8]$ and density
	\[
	f(q)=k(q-1)\quad\text{for }2\le q\le 8.

	\begin{itemize}
		\item \textbf{C1 Calculations:} Find $k$ so $f(q)$ is a valid PDF.
		\item \textbf{C1 Interpretation:} Compare density near $q=3$ and $q=7$.
		\item \textbf{Extra analysis:} Find $P(4\le Q\le 6)$.
	\end{itemize}

	\textbf{Solution.}
	On $[2,8]$, $f(q)=k(q-1)$ is linear and positive. Total area is a trapezium with width $6$, heights
	\[
	f(2)=k,\quad f(8)=7k.
	\]
	So
	\[
	\frac{k+7k}{2}\cdot6=24k=1\implies k=\frac1{24}.
	\]

	\[
	f(3)=2k,\quad f(7)=6k,
	\]
	so density near $q=7$ is three times density near $q=3$.
	Now
	\[
	P(4\le Q\le6)=\text{trapezium area on }[4,6],
	\]
	with heights $f(4)=3k$, $f(6)=5k$, width $2$:
	\[
	P(4\le Q\le6)=\frac{3k+5k}{2}\cdot2=8k=\frac13.
	\]
	\textbf{Final answer:} $k=\frac1{24}$ and $P(4\le Q\le6)=\frac13$.

	\textbf{Problem.}
	Let $R$ be a quality index with support $[1,7]$ and density
	\[
	f(r)=k(r+1)\quad\text{for }1\le r\le 7.
	\]
	
	\textbf{Question.}
	\begin{itemize}
		\item \textbf{C1 Calculations:} Find $k$ so $f(r)$ is a valid PDF.
		\item \textbf{C1 Interpretation:} Explain why right-side intervals have more probability than equal-width left-side intervals.
		\item \textbf{Extra analysis:} Find $P(R\le 4)$.
	\end{itemize}

	\textbf{Solution.}
	On $[1,7]$, $f(r)=k(r+1)$ has
	\[
	f(1)=2k,\quad f(7)=8k.

	\[
	\frac{2k+8k}{2}\cdot6=30k=1\implies k=\frac1{30}.

	Now
	\[
	P(R\le4)=P(1\le R\le4)=\frac{f(1)+f(4)}{2}\cdot(4-1)
	\]
	with $f(4)=5k$:
	\[
	P(R\le4)=\frac{2k+5k}{2}\cdot3=\frac{21k}{2}=\frac7{20}.
	\]
	\textbf{Final answer:} $k=\frac1{30}$ and $P(R\le4)=\frac7{20}$.

	\[
	P(2\le B\le6)=\frac{f(2)+f(6)}{2}\cdot4=\frac{\frac{3k}{2}+\frac{5k}{2}}{2}\cdot4=8k=\frac12.
	\]
	\textbf{Final answer:} $k=\frac1{16}$ and $P(2\le B\le6)=\frac12$.

	\section{Traffic Flow Index (Linear with Variable Domain)}
	
	\textbf{Problem.}
	Let $F$ be a traffic flow index with support $[0,b]$, where $b>0$, and density
	\[
	f(f)=kf\quad\text{for }0\le f\le b.
	\]
	
	\textbf{Question.}
	\begin{itemize}
		\item \textbf{C1 Calculations:} Find $k$ in terms of $b$ so $f(f)$ is a valid PDF.
		\item \textbf{C1 Interpretation:} Explain why larger values of $f$ are more likely than smaller values in equal-width intervals.
		\item \textbf{Extra analysis:} Find $P\!\left(\frac{b}{2}\le F\le b\right)$.
	\end{itemize}

	\textbf{Solution.}
	For $f(f)=kf$ on $[0,b]$ ($b>0$), area is a triangle with base $b$ and height $kb$:
	\[
	\frac12(b)(kb)=\frac{kb^2}{2}=1\implies k=\frac{2}{b^2}.
	\]
	Since the line increases with $f$, equal-width intervals further right have larger area, so larger values are more likely.
	Now
	\[
	P\!\left(\frac{b}{2}\le F\le b\right)=1-P\!\left(0\le F\le\frac{b}{2}\right).
	\]
	Left part is a triangle with base $\frac b2$ and height $k\frac b2$:
	\[
	P\!\left(0\le F\le\frac{b}{2}\right)=\frac12\left(\frac b2\right)\left(k\frac b2\right)=\frac{kb^2}{8}=\frac14.
	\]
	Hence
	\[
	P\!\left(\frac{b}{2}\le F\le b\right)=1-\frac14=\frac34.
	\]
	\textbf{Final answer:} $k=\frac{2}{b^2}$ and $P\!\left(\frac{b}{2}\le F\le b\right)=\frac34$.

	\section{Delivery Deviation (Shifted Linear with Two Variable Endpoints)}
	
	\textbf{Problem.}
	Let $D$ be a delivery deviation with support $[p,q]$, where $p<q$, and density
	\[
	f(d)=k(d-p+1)\quad\text{for }p\le d\le q.
	\]
	
	\textbf{Question.}
	\begin{itemize}
		\item \textbf{C1 Calculations:} Find $k$ in terms of $p$ and $q$ so $f(d)$ is a valid PDF.
		\item \textbf{C1 Interpretation:} Explain why the density stays positive throughout the support.
		\item \textbf{Extra analysis:} Find $P\!\left(p+\frac{q-p}{3}\le D\le q\right)$.
	\end{itemize}

	\textbf{Solution.}
	For $f(d)=k(d-p+1)$ on $[p,q]$ ($p<q$):
	\[
	d-p+1\in[1,\,q-p+1],
	\]
	so density is positive on the whole support.
	Let $t=q-p>0$. Then heights are $k$ and $k(t+1)$, width $t$, so
	\[
	\frac{k+k(t+1)}{2}\cdot t=\frac{k(t+2)t}{2}=1
	\implies
	k=\frac{2}{(q-p)(q-p+2)}.
	\]
	Now compute
	\(
	P\!\left(p+\frac{q-p}{3}\le D\le q\right)
	\).
	Using $y=d-p$ ($0\le y\le t$), this is $y\in\left[\frac t3,t\right]$ with density $k(y+1)$:
	\[
	P=\int_{t/3}^{t}k(y+1)\,dy
	=\frac{k}{2}\left[(y+1)^2\right]_{t/3}^{t}
	=\frac{k}{2}\left((t+1)^2-\left(\frac t3+1\right)^2\right).
	\]
	Substitute $k=\frac{2}{t(t+2)}$ and simplify:
	\[
	P=\frac{4(2t+3)}{9(t+2)}=\frac{4(2(q-p)+3)}{9(q-p+2)}.
	\]
	\textbf{Final answer:} $k=\frac{2}{(q-p)(q-p+2)}$ and
	\[
	P\!\left(p+\frac{q-p}{3}\le D\le q\right)=\frac{4(2(q-p)+3)}{9(q-p+2)}.
	\]
