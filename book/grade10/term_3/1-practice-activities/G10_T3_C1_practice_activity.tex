	\section{Checking Two Candidate Density Functions I}
	
	\textbf{Problem.}
	A continuous random variable is modeled on the interval shown in each case.
	
	Distribution A
	\[
	f(x)=0.25\quad\text{for }0\le x\le 4
	\]
	
	Distribution B
	\[
	f(x)=0.40\quad\text{for }0\le x\le 4
	\]
	
	\textbf{Question.}
	\begin{itemize}
		\item \textbf{C1 Calculations:} Decide which distribution is a valid PDF and which one is not.
		\item \textbf{C1 Interpretation:} Justify your answer using the total area under each graph.
		\item \textbf{Extra analysis:} For the valid distribution, find $P(1\le X\le 3)$.
	\end{itemize}

	\textbf{Solution.}

\textbf{C1 calculations.}
For a valid PDF, $f(x)\ge 0$ on its support and total area is $1$.
\begin{itemize}
	\item Distribution A: support $[0,4]$, constant height $0.25$.
	\[
	\text{Area}=4(0.25)=1,
	\]
	so A is valid.
	\item Distribution B: support $[0,4]$, constant height $0.40$.
	\[
	\text{Area}=4(0.40)=1.6\ne1,
	\]
	so B is not valid.
\end{itemize}

\textbf{C2 interpretation.}
\COneGraph{0}{4}{0.25}{0}{4}{0,1,2,3,4}{0.5}
The valid PDF is uniform on $[0,4]$: it is constant, nonnegative, and gives equal probability to equal-width intervals.

\textbf{Extra analysis.}
For the valid distribution A,
\[
P(1\le X\le 3)=(3-1)(0.25)=0.5.
\]
\COneGraph{0}{4}{0.25}{1}{3}{0,1,2,3,4}{0.5}
\textbf{Final answer:} A is valid, B is not valid, and $P(1\le X\le 3)=0.5$.

	\section{Checking Two Candidate Density Functions II}
	
	\textbf{Problem.}
	A continuous random variable is modeled on the interval shown in each case.
	
	Distribution A
	\[
	f(x)=\frac{x}{8}\quad\text{for }0\le x\le 4
	\]
	
	Distribution B
	\[
	f(x)=\frac{x}{6}\quad\text{for }0\le x\le 4
	\]
	
	\textbf{Question.}
	\begin{itemize}
		\item \textbf{C1 Calculations:} Decide which distribution is a valid PDF and which one is not.
		\item \textbf{C1 Interpretation:} Justify your answer using geometric reasoning about the area under each line.
		\item \textbf{Extra analysis:} For the valid distribution, find $P(2\le X\le 4)$.
	\end{itemize}

	\textbf{Solution.}

\textbf{C1 calculations.}
Check nonnegativity and total area on support $[0,4]$.
\begin{itemize}
	\item Distribution A: $f(x)=\frac{x}{8}$ forms a triangle with base $4$ and height $\frac{4}{8}=\frac12$.
	\[
	\text{Area}=\frac12(4)\left(\frac12\right)=1,
	\]
	so A is valid.
	\item Distribution B: $f(x)=\frac{x}{6}$ has base $4$ and height $\frac{4}{6}=\frac23$.
	\[
	\text{Area}=\frac12(4)\left(\frac23\right)=\frac43\ne1,
	\]
	so B is not valid.
\end{itemize}

\textbf{C2 interpretation.}
\COneGraph{0}{4}{x/8}{0}{4}{0,1,2,3,4}{0.6}
The valid PDF increases linearly on $[0,4]$, so larger $x$ values have higher density than smaller ones.

\textbf{Extra analysis.}
For A,
\[
P(2\le X\le4)=1-P(0\le X\le2).
\]
On $[0,2]$, area is a triangle with base $2$ and height $\frac{2}{8}=\frac14$:
\[
P(0\le X\le2)=\frac12(2)\left(\frac14\right)=\frac14.
\]
Hence
\[
P(2\le X\le4)=1-\frac14=\frac34.
\]
\COneGraph{0}{4}{x/8}{2}{4}{0,1,2,3,4}{0.6}
\textbf{Final answer:} A is valid, B is not valid, and $P(2\le X\le4)=\frac34$.

	\section{Checking Two Candidate Density Functions III}
	
	\textbf{Problem.}
	A continuous random variable is modeled on the interval shown in each case.
	
	Distribution A
	\[
	f(x)=\frac{6-x}{12}\quad\text{for }1\le x\le 5
	\]
	
	Distribution B
	\[
	f(x)=\frac{6-x}{10}\quad\text{for }1\le x\le 5
	\]
	
	\textbf{Question.}
	\begin{itemize}
		\item \textbf{C1 Calculations:} Decide which distribution is a valid PDF and which one is not.
		\item \textbf{C1 Interpretation:} Justify your answer using geometric reasoning about the area under each line.
		\item \textbf{Extra analysis:} For the valid distribution, find $P(1\le X\le 3)$.
	\end{itemize}
	
	\textbf{Solution.}
	
	\textbf{C1 calculations.}
	On support $[1,5]$, both densities are nonnegative. Check total area:
	\begin{itemize}
		\item Distribution A: $f(x)=\frac{6-x}{12}$ gives a trapezium with base $4$, left height $f(1)=\frac{5}{12}$, and right height $f(5)=\frac{1}{12}$.
		\[
		\text{Area}=\frac{\frac{5}{12}+\frac{1}{12}}{2}\cdot 4=1,
		\]
		so A is valid.
		\item Distribution B: $f(x)=\frac{6-x}{10}$ gives base $4$, left height $f(1)=\frac{5}{10}=\frac12$, and right height $f(5)=\frac{1}{10}$.
		\[
		\text{Area}=\frac{\frac12+\frac{1}{10}}{2}\cdot 4=\frac65\ne1,
		\]
		so B is not valid.
	\end{itemize}
	
	\textbf{C2 interpretation.}
	\COneGraph{1}{5}{(6-x)/12}{1}{5}{1,2,3,4,5}{0.5}
	The valid PDF is decreasing on $[1,5]$, so values near $1$ are more likely than values near $5$, but values near $5$ still have positive density.
	
	\textbf{Extra analysis.}
	For valid distribution A,
	\[
	P(1\le X\le3)=\text{area of trapezium on }[1,3].
	\]
	With heights $f(1)=\frac{5}{12}$ and $f(3)=\frac14$, width $2$:
	\[
	P(1\le X\le3)=\frac{\frac{5}{12}+\frac14}{2}\cdot 2=\frac23.
	\]
	\COneGraph{1}{5}{(6-x)/12}{1}{3}{1,2,3,4,5}{0.5}
	\textbf{Final answer:} A is valid, B is not valid, and $P(1\le X\le3)=\frac23$.

\section{Battery Discharge (Linear, Endpoint at 0)}

\textbf{Problem.}
Let $B$ be battery discharge measure with support $[0,c]$ and density
\[
f(b)=\frac{1}{16}+\frac{b}{8c}\quad\text{for }0\le b\le c.
\]

\textbf{Question.}
\begin{itemize}
	\item \textbf{C1 Calculations:} Find $c$ so $f(b)$ is a valid PDF.
	\item \textbf{C1 Interpretation:} Compare likelihood of values in $[0,2]$ and $[6,8]$.
	\item \textbf{Extra analysis:} Find $P(2\le B\le 6)$.
\end{itemize}

\textbf{Solution.}

\textbf{C1 calculations.}
The endpoint values are fixed:
\[
f(0)=\frac{1}{16},\quad f(c)=\frac{1}{16}+\frac{c}{8c}=\frac{3}{16}.
\]
Total area must be $1$. Using the trapezium formula,
\[
\frac{\frac{1}{16}+\frac{3}{16}}{2}\cdot c=1.
\]
So
\[
\frac{4/16}{2}\cdot c=1
\]
\[
\frac{1}{8}c=1\implies c=8.
\]

Now compare the two intervals using the resolved density
\[
f(b)=\frac{1}{16}+\frac{b}{64}\quad\text{for }0\le b\le 8.
\]

For $[0,2]$:
\[
f(0)=\frac{1}{16},\quad f(2)=\frac{1}{16}+\frac{2}{64}=\frac{3}{32}.
\]
Thus
\[
P(0\le B\le2)=\frac{\frac{1}{16}+\frac{3}{32}}{2}\cdot2=\frac{5}{32}.
\]

For $[6,8]$:
\[
f(6)=\frac{1}{16}+\frac{6}{64}=\frac{5}{32},\quad f(8)=\frac{3}{16}.
\]
Thus
\[
P(6\le B\le8)=\frac{\frac{5}{32}+\frac{3}{16}}{2}\cdot2=\frac{11}{32}.
\]
So $[6,8]$ is more likely.

For $2\le B\le6$:
\[
P(2\le B\le6)=\frac{f(2)+f(6)}{2}\cdot4
=\frac{\frac{3}{32}+\frac{5}{32}}{2}\cdot4
=\frac12.
\]

\textbf{C2 interpretation.}

	\section{Cooling Level (Decreasing Linear with Two Variable Endpoints)}
	
	\textbf{Problem.}
	Let $C$ be a cooling level with support $[r,s]$, where $r<s$, and density
	\[
	f(c)=k(s-c)\quad\text{for }r\le c\le s.
	\]
	
	\textbf{Question.}
	\begin{itemize}
		\item \textbf{C1 Calculations:} Find $k$ in terms of $r$ and $s$ so $f(c)$ is a valid PDF.
		\item \textbf{C1 Interpretation:} Explain why values near $r$ are more likely than values near $s$.
		\item \textbf{Extra analysis:} Find $P\!\left(r\le C\le \frac{r+s}{2}\right)$ using geometric reasoning.
	\end{itemize}

	\textbf{Solution.}

\textbf{C1 calculations.}
For $f(c)=k(s-c)$ on $[r,s]$ ($r<s$), heights are
	\[
	f(r)=k(s-r),\quad f(s)=0,
	\]
	so total area (triangle) is
	\[
	\frac12(s-r)\,k(s-r)=\frac{k(s-r)^2}{2}=1
	\implies
	k=\frac{2}{(s-r)^2}.
	\]
	Because the density decreases toward $s$, values near $r$ are more likely than values near $s$.
	For $r\le C\le\frac{r+s}{2}$, width is $\frac{s-r}{2}$ and heights are $k(s-r)$ and $k\frac{s-r}{2}$.
	\[
	P\!\left(r\le C\le\frac{r+s}{2}\right)
	=\frac{k(s-r)+k\frac{s-r}{2}}{2}\cdot\frac{s-r}{2}
	=\frac{3k(s-r)^2}{8}

	\section{Delivery Deviation (Shifted Linear with Two Variable Endpoints)}
	
	\textbf{Problem.}
	Let $D$ be a delivery deviation with support $[p,q]$, where $p<q$, and density
	\[
	f(d)=k(d-p+1)\quad\text{for }p\le d\le q.
	\]
	
	\textbf{Question.}
	\begin{itemize}
		\item \textbf{C1 Calculations:} Find $k$ in terms of $p$ and $q$ so $f(d)$ is a valid PDF.
		\item \textbf{C1 Interpretation:} Explain why the density stays positive throughout the support.
		\item \textbf{Extra analysis:} Find $P\!\left(p+\frac{q-p}{3}\le D\le q\right)$.
	\end{itemize}

	\textbf{Solution.}

\textbf{C1 calculations.}
For $f(d)=k(d-p+1)$ on $[p,q]$ ($p<q$):
	\[
	d-p+1\in[1,\,q-p+1],
	\]
	so density is positive on the whole support.
	Let $t=q-p>0$. Then heights are $k$ and $k(t+1)$, width $t$, so
	\[
	\frac{k+k(t+1)}{2}\cdot t=\frac{k(t+2)t}{2}=1
	\implies
	k=\frac{2}{(q-p)(q-p+2)}.
	\]
	Now compute
	\(
	P\!\left(p+\frac{q-p}{3}\le D\le q\right)
	\).
	Using $y=d-p$ ($0\le y\le t$), this is $y\in\left[\frac t3,t\right]$ with density $k(y+1)$:
	\[
	P=\int_{t/3}^{t}k(y+1)\,dy
	=\frac{k}{2}\left[(y+1)^2\right]_{t/3}^{t}
	=\frac{k}{2}\left((t+1)^2-\left(\frac t3+1\right)^2\right).
	\]
	Substitute $k=\frac{2}{t(t+2)}$ and simplify:
	\[
	P=\frac{4(2t+3)}{9(t+2)}=\frac{4(2(q-p)+3)}{9(q-p+2)}.
	\]
