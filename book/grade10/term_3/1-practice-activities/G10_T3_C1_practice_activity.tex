		\subsection*{Problem 1 — Water Tank Fill Level}
		
		\textbf{Problem.}
		A sensor records the fill level $X$ (in meters) of a water tank during a short period.
		Assume $X$ is continuous on $[0,5]$ with density
		\[
		f(x)=k,\quad 0\le x\le 5.
		\]
		
		\textbf{Question.}
		\begin{itemize}
			\item \textbf{C1 Calculations:} Determine $k$ so the model is a valid density.
			\item \textbf{C1 Interpretation:} Explain what the C1 calculations show about why the density model is valid.
			\item \textbf{Extra analysis:} Compute $P(1\le X\le 4)$ and represent this event on a probability-region graph.
		\end{itemize}
		
		\textbf{Solution.}
		\textbf{C1 Calculations.}
		The graph is a rectangle with width $5$ and height $k$, so total area is
		\[
		5k=1\Rightarrow k=\frac{1}{5}.
		\]
		\[
		f(x)=\frac{1}{5},\quad 0\le x\le 5.
		\]
		\begin{center}
		\begin{tikzpicture}
		\begin{axis}[
		  width=11cm,
		  height=6cm,
		  axis lines=left,
		  xlabel={$x$},
		  ylabel={$f(x)$},
		  xmin=-1,
		  xmax=6,
		  ymin=0.15,
		  ymax=0.25,
		  grid=both
		]
		\addplot[domain=0:5, samples=200, draw=none, fill=blue!20, fill opacity=0.35] {1/5} \closedcycle;
		\addplot[smooth, domain=0:5, samples=200] {1/5};
		\end{axis}
		\end{tikzpicture}
		\end{center}
		
		\textbf{C1 Interpretation.}
		The value $k=\frac{1}{5}$ makes the total area over all possible tank fill levels from $0$ to $5$ meters equal to $1$, so this is a valid probability model for $X$. Because the density is constant on the interval, the model gives the same density to each fill level in that range.
		
		\textbf{Extra analysis.}
		Now $P(1\le X\le 4)$ is a rectangle of width $3$ and height $\frac{1}{5}$:
		\[
		P(1\le X\le 4)=3\cdot\frac{1}{5}=\frac{3}{5}=0.6.
		\]

		\item
		\subsection*{Problem 2 — Ramp Travel Time}
		
		\textbf{Problem.}
		The travel time $T$ (in seconds) for a robot to cross a ramp is modeled on $[2,8]$ by
		\[
		f(t)=k(t-2),\quad 2\le t\le 8.
		\]
		
		\textbf{Question.}
		\begin{itemize}
			\item \textbf{C1 Calculations:} Determine $k$ so $f(t)$ is a valid density.
			\item \textbf{C1 Interpretation:} Explain what the C1 calculations show about why the density model is valid.
			\item \textbf{Extra analysis:} Compute $P(T>6)$ and represent this event on a probability-region graph.
		\end{itemize}
		
		\textbf{Solution.}
		\textbf{C1 Calculations.}
		The graph from $2$ to $8$ is a triangle with base $6$ and height $f(8)=6k$.
		So total area is
		\[
		\frac{1}{2}(6)(6k)=18k=1\Rightarrow k=\frac{1}{18}.
		\]
		\[
		f(t)=\frac{1}{18}(t-2),\quad 2\le t\le 8.
		\]
		\begin{center}
		\begin{tikzpicture}
		\begin{axis}[
		  width=11cm,
		  height=6cm,
		  axis lines=left,
		  xlabel={$x$},
		  ylabel={$f(x)$},

		  xmax=9,
		  ymin=-0.05,
		  ymax=0.383,
		  grid=both
		]
		\addplot[domain=2:8, samples=200, draw=none, fill=blue!20, fill opacity=0.35] {(x-2)/18} \closedcycle;
		\addplot[smooth, domain=2:8, samples=200] {(x-2)/18};
		\end{axis}
		\end{tikzpicture}
		\end{center}
		
		\textbf{C1 Interpretation.}
		The value $k=\frac{1}{18}$ makes the full area across robot travel times from $2$ to $8$ seconds equal to $1$, and $f(t)\ge 0$ on that interval, so this is a valid density model for $T$. Since the line increases with $t$, larger times in this range are modeled with greater density than smaller times.
		
		\textbf{Extra analysis.}
		For $T>6$, use the trapezoid on $[6,8]$.
		Its left height is $f(6)=\frac{1}{18}(4)=\frac{2}{9}$, right height is $f(8)=\frac{1}{3}$, width is $2$:
		\[
		P(T>6)=\frac{\left(\frac{2}{9}+\frac{1}{3}\right)}{2}\cdot 2
		=\frac{2}{9}+\frac{1}{3}=\frac{5}{9}.
		\]

		\subsection*{Problem 3 — Air Quality Index Block Model}
		
		\textbf{Problem.}
		An air-quality indicator $A$ is modeled on $[0,4]$ with piecewise-constant density
		\[
		f(a)=\begin{cases}
		k, & 0\le a<1,\\
		2k, & 1\le a<3,\\
		k, & 3\le a\le 4.
		\end{cases}
		\]
		
		\textbf{Question.}
		\begin{itemize}
			\item \textbf{C1 Calculations:} Find $k$.
			\item \textbf{C1 Interpretation:} Explain what the C1 calculations show about why the density model is valid.
			\item \textbf{Extra analysis:} Compute $P(0.5\le A\le 2.5)$ and represent this event on a probability-region graph.
		\end{itemize}

		Add rectangle areas for total area 1:
		\[
		(1\cdot k)+(2\cdot 2k)+(1\cdot k)=6k=1\Rightarrow k=\frac{1}{6}.
		\]
		\[
		f(a)=\begin{cases}
		\frac{1}{6}, & 0\le a<1,\\
		\frac{1}{3}, & 1\le a<3,\\
		\frac{1}{6}, & 3\le a\le 4.
		\end{cases}
		\]

		\subsection*{Problem 4 — Symmetric Delivery Window}
		
		\textbf{Problem.}
		A delivery offset $Y$ (hours from schedule) is modeled on $[-1,3]$ with a triangular density
		that rises to a peak at $y=1$:
		\[
		f(y)=\begin{cases}
		k(y+1), & -1\le y\le 1,\\
		k(3-y), & 1< y\le 3.
		\end{cases}
		\]
		
		\textbf{Question.}
		\begin{itemize}
			\item \textbf{C1 Calculations:} Determine $k$.
			\item \textbf{C1 Interpretation:} Explain what the C1 calculations show about why the density model is valid.
			\item \textbf{Extra analysis:} Compute $P(0\le Y\le 2)$ and represent this event on a probability-region graph.
		\end{itemize}
		
		\textbf{Solution.}
		\textbf{C1 Calculations.}
		The whole graph is one triangle with base $4$ (from $-1$ to $3$) and height $f(1)=2k$.
		Thus
		\[
		\frac{1}{2}(4)(2k)=4k=1\Rightarrow k=\frac{1}{4}.
		\]
		\[
		f(y)=\begin{cases}
		\frac{1}{4}(y+1), & -1\le y\le 1,\\
		\frac{1}{4}(3-y), & 1< y\le 3.
		\end{cases}
		\]

		  width=11cm,

		\end{tikzpicture}
		\end{center}
		
		\textbf{C1 Interpretation.}
		The constant $k=\frac{1}{4}$ makes the total area across all delivery offsets from $-1$ to $3$ hours equal to $1$, so this is a valid density model for $Y$. Its triangular shape means offsets near $y=1$ have greater density, while offsets near the endpoints have lower density, and the model is symmetric around $y=1$.
		
		\textbf{Extra analysis.}
		For $0\le Y\le 2$, use two equal trapezoids:
		on $[0,1]$ heights are $f(0)=\frac{1}{4}$ and $f(1)=\frac{1}{2}$,
		so area is $\frac{\left(\frac{1}{4}+\frac{1}{2}\right)}{2}\cdot 1=\frac{3}{8}$.
		By symmetry, area on $[1,2]$ is also $\frac{3}{8}$.
		Hence
		\[
		P(0\le Y\le 2)=\frac{3}{8}+\frac{3}{8}=\frac{3}{4}.
		\]

		\subsection*{Problem 7 — Traffic Gap with Flat Peak}
		
		\textbf{Problem.}
		Let $G$ (seconds) be a traffic gap modeled on $[0,6]$ by
		\[
		f(g)=\begin{cases}
		kg, & 0\le g<2,\\
		2k, & 2\le g<4,\\
		k(6-g), & 4\le g\le 6.
		\end{cases}
		\]
		
		\textbf{Question.}
		\begin{itemize}
			\item \textbf{C1 Calculations:} Find $k$.
			\item \textbf{C1 Interpretation:} Explain what the C1 calculations show about why the density model is valid.
			\item \textbf{Extra analysis:} Compute $P(1\le G\le 5)$ and represent this event on a probability-region graph.
		\end{itemize}
		
		\textbf{Solution.}
		\textbf{C1 Calculations.}
		Compute total area by pieces:
		left triangle on $[0,2]$: $\frac{1}{2}(2)(2k)=2k$,
		middle rectangle on $[2,4]$: $(2)(2k)=4k$,
		right triangle on $[4,6]$: $\frac{1}{2}(2)(2k)=2k$.
		So
		\[
		2k+4k+2k=8k=1\Rightarrow k=\frac{1}{8}.
		\]
		\[
		f(g)=\begin{cases}
		\frac{g}{8}, & 0\le g<2,\\
		\frac{1}{4}, & 2\le g<4,\\
		\frac{6-g}{8}, & 4\le g\le 6.
		\end{cases}
		\]

		  width=11cm,

		  ylabel={$f(x)$},

		\item
		\subsection*{Problem 8 — Quality Score Trapezoid}
		
		\textbf{Problem.}
		A quality score $Q$ is modeled on $[2,8]$ by a line that starts above zero:
		\[
		f(q)=k\left(1+\frac{q-2}{3}\right),\quad 2\le q\le 8.
		\]
		This is linear, with $f(2)=k$ and $f(8)=3k$.
		
		\textbf{Question.}
		\begin{itemize}
			\item \textbf{C1 Calculations:} Determine $k$.
			\item \textbf{C1 Interpretation:} Explain what the C1 calculations show about why the density model is valid.
			\item \textbf{Extra analysis:} Compute $P(4\le Q\le 7)$ and represent this event on a probability-region graph.
		\end{itemize}
		
		\textbf{Solution.}
		\textbf{C1 Calculations.}
		The full graph on $[2,8]$ is a trapezoid with parallel sides $k$ and $3k$, width $6$:
		\[
		\text{Area}=\frac{k+3k}{2}\cdot 6=12k=1\Rightarrow k=\frac{1}{12}.
		\]
		\[
		f(q)=\frac{1}{12}\left(1+\frac{q-2}{3}\right),\quad 2\le q\le 8.
		\]

		\textbf{Extra analysis.}
		For $[4,7]$, heights are
		\[

		\subsection*{Problem 9 — Centered Error Triangle}
		
		\textbf{Problem.}
		A measurement error $Z$ is modeled on $[-2,2]$ by
		\[
		f(z)=\begin{cases}
		k(z+2), & -2\le z\le 0,\\
		k(2-z), & 0< z\le 2.
		\end{cases}
		\]
		
		\textbf{Question.}
		\begin{itemize}
			\item \textbf{C1 Calculations:} Find $k$.
			\item \textbf{C1 Interpretation:} Explain what the C1 calculations show about why the density model is valid.
			\item \textbf{Extra analysis:} Compute $P(-1\le Z\le 1)$ and represent this event on a probability-region graph.
		\end{itemize}
		
		\textbf{Solution.}
		\textbf{C1 Calculations.}
		The graph is a symmetric triangle with base $4$ and peak height $f(0)=2k$.
		So
		\[
		\frac{1}{2}(4)(2k)=4k=1\Rightarrow k=\frac{1}{4}.
		\]
		\[
		f(z)=\begin{cases}
		\frac{1}{4}(z+2), & -2\le z\le 0,\\
		\frac{1}{4}(2-z), & 0< z\le 2.
		\end{cases}
		\]

		\textbf{Problem.}
		A service time variable $S$ (minutes) is modeled on $[0,10]$ by
		\[
		f(s)=\begin{cases}
		k, & 0\le s<4,\\
		\frac{k}{2}, & 4\le s<8,\\
		\frac{3k}{2}, & 8\le s\le 10.
		\end{cases}
		\]
		
		\textbf{Question.}
		\begin{itemize}
			\item \textbf{C1 Calculations:} Determine $k$.
			\item \textbf{C1 Interpretation:} Explain what the C1 calculations show about why the density model is valid.
			\item \textbf{Extra analysis:} Compute $P(S<7)$ and represent this event on a probability-region graph.
		\end{itemize}
		
		\textbf{Solution.}
		\textbf{C1 Calculations.}
		Use rectangle areas for total area 1:
		\[
		(4)(k)+(4)\left(\frac{k}{2}\right)+(2)\left(\frac{3k}{2}\right)
		=4k+2k+3k=9k=1,
		\]
		so
		\[
		k=\frac{1}{9}.
		\]
		\[
		f(s)=\begin{cases}
		\frac{1}{9}, & 0\le s<4,\\
		\frac{1}{18}, & 4\le s<8,\\
		\frac{1}{6}, & 8\le s\le 10.
		\end{cases}
