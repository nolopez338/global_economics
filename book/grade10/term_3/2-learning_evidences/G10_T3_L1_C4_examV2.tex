		\SubsectionBox{Uniform Distribution}\vspace{-0.35cm}
		For a uniform distribution $U[a,b]$:
		\[
		\mu=\frac{a+b}{2}
		\qquad \qquad
		\sigma=\frac{b-a}{\sqrt{12}}
		\]

		\SubsectionBox{Triangular Distribution}\vspace{-0.35cm}
		For a triangular distribution with left endpoint $a$, mode $c$, and right endpoint $b$:
		\[
		\mu=\frac{a+b+c}{3}
		\qquad \qquad
		\sigma=\sqrt{\frac{a^2+b^2+c^2-ab-ac-bc}{18}}
		\]

		\SubsectionBox{Normal Distribution}\vspace{-0.35cm}
		For a normal distribution $N(\mu,\sigma^2)$:
		\[
		E[X]=\mu
		\qquad \qquad
		\mathrm{SD}(X)=\sigma
		\]
