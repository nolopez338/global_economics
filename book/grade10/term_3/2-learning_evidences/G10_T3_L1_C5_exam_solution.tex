\section{Problem 1 --- Left area and right tail}
\textbf{Problem.}
Let \(Z\sim N(0,1)\).

\textbf{Tasks.}
Calculate \(P(Z<0.48)\) and \(P(Z>1.25)\).

\subsection*{Solution}
From the table,
\[
\Phi(0.48)=0.6844,
\qquad
\Phi(1.25)=0.8944.
\]
So
\[
P(Z<0.48)=0.6844,
\]
\[
P(Z>1.25)=1-\Phi(1.25)=1-0.8944=0.1056.
\]

\section{Problem 2 --- Interval crossing zero}
\textbf{Problem.}
Let \(Z\sim N(0,1)\).

\textbf{Task.}
Find \(P(-0.85\le Z\le 1.30)\).

\subsection*{Solution}
From the table,
\[
\Phi(1.30)=0.9032,
\qquad
\Phi(0.85)=0.8023.
\]
By symmetry,
\[
\Phi(-0.85)=1-\Phi(0.85)=1-0.8023=0.1977.
\]
Therefore,
\[
P(-0.85\le Z\le 1.30)=\Phi(1.30)-\Phi(-0.85)=0.9032-0.1977=0.7055.
\]

\section{Problem 3 --- Two-tail union}
\textbf{Problem.}
Let \(Z\sim N(0,1)\).

\textbf{Task.}
Determine \(P(Z\le -1.35\ \text{or}\ Z\ge 0.75)\).

\subsection*{Solution}
From the table,
\[
\Phi(1.35)=0.9115,
\qquad
\Phi(0.75)=0.7734.
\]
Thus,
\[
P(Z\le -1.35)=1-\Phi(1.35)=1-0.9115=0.0885,
\]
\[
P(Z\ge 0.75)=1-\Phi(0.75)=1-0.7734=0.2266.
\]
The events are disjoint, so
\[
P(Z\le -1.35\ \text{or}\ Z\ge 0.75)=0.0885+0.2266=0.3151.
\]

\section{Problem 4 --- Absolute value and complement}
\textbf{Problem.}
Let \(Z\sim N(0,1)\).

\textbf{Tasks.}
Compute \(P(|Z|\le 1.50)\) and \(P(|Z|>1.50)\).

\subsection*{Solution}
From the table,
\[
\Phi(1.50)=0.9332,
\qquad
\Phi(-1.50)=1-0.9332=0.0668.
\]
Then
\[
P(|Z|\le 1.50)=P(-1.50\le Z\le 1.50)=0.9332-0.0668=0.8664.
\]
So
\[
P(|Z|>1.50)=1-0.8664=0.1336.
\]

\section{Problem 6 --- Symmetry and decomposition}
\textbf{Problem.}
Let \(Z\sim N(0,1)\).

\textbf{Tasks.}
Find \(P(Z<-1.10)\) and \(P(-1.10\le Z\le 0)\).

\subsection*{Solution}
From the table,
\[
\Phi(1.10)=0.8643,
\qquad \Phi(0)=0.5000.
\]
By symmetry,
\[
P(Z<-1.10)=\Phi(-1.10)=1-\Phi(1.10)=1-0.8643=0.1357.
\]
Then
\[
P(-1.10\le Z\le 0)=\Phi(0)-\Phi(-1.10)=0.5000-0.1357=0.3643.
\]
