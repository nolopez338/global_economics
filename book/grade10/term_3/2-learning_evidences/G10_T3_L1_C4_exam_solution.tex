\section*{Evaluated criterion}
\textbf{C4.} Interprets the concept of expected value, variance, and standard deviation of a probability function.

\bigskip
\textbf{Interpretation focus used in this solution set:}
\begin{itemize}
    \item The \textbf{expected value} is interpreted as the horizontal \emph{balance point/center} of the graph.
    \item The \textbf{standard deviation} is interpreted as the typical horizontal \emph{spread} around the center.
    \item \textbf{Variance} is interpreted as spread squared; larger variance means a more dispersed graph.
    \item The \textbf{shape} (flat, slanted, triangular, bell-shaped) helps explain where probability mass is concentrated.
\end{itemize}

\section{Problem 1 (Type A) --- Uniform: match graph, mean, and spread}
Three uniform densities are shown.
\begin{center}
\begin{tikzpicture}
\begin{axis}[
    width=12cm,height=6cm,
    axis lines=left,
    xlabel={$x$}, ylabel={$f(x)$},
    xmin=-1,xmax=11,ymin=0,ymax=0.55,
    grid=both,
    legend style={at={(0.98,0.98)},anchor=north east}
]
    % A: U[0,4]
    \addplot[very thick,blue,domain=0:4,samples=2] {0.25};
    \addplot[blue,dashed] coordinates {(2,0) (2,0.28)};
    % B: U[2,8]
    \addplot[very thick,red,domain=2:8,samples=2] {1/6};
    \addplot[red,dashed] coordinates {(5,0) (5,0.20)};
    % C: U[6,10]
    \addplot[very thick,teal!70!black,domain=6:10,samples=2] {0.25};
    \addplot[teal!70!black,dashed] coordinates {(8,0) (8,0.28)};
    \legend{Graph A, $\mu_A$, Graph B, $\mu_B$, Graph C, $\mu_C$}
\end{axis}
\end{tikzpicture}
\end{center}

Match each graph with one option from each list.
\begin{itemize}
    \item Mean options: $\{2,\ 5,\ 8\}$
    \item Standard deviation options (approximately): $\{1.15,\ 1.15,\ 1.73\}$
\end{itemize}

\subsection*{Solution to Problem 1}
\textbf{Final match.}
\begin{itemize}
    \item Graph A $\to \mu=2$, $\sigma\approx1.15$
    \item Graph B $\to \mu=5$, $\sigma\approx1.73$
    \item Graph C $\to \mu=8$, $\sigma\approx1.15$
\end{itemize}

\textbf{Reasoning (conceptual).}
For a uniform graph, the center is the midpoint of the interval.
\begin{itemize}
    \item Graph A runs from $0$ to $4$, midpoint $2$.
    \item Graph B runs from $2$ to $8$, midpoint $5$.
    \item Graph C runs from $6$ to $10$, midpoint $8$.
\end{itemize}
Spread depends on interval width: wider interval $\Rightarrow$ larger standard deviation.
\begin{itemize}
    \item Widths: A = $4$, B = $6$, C = $4$.
    \item So B has largest spread ($\sigma\approx1.73$), while A and C have equal smaller spread ($\sigma\approx1.15$).
\end{itemize}

\section{Problem 2 (Type B) --- Same mean, different spread (uniform \& normal)}
The following four graphs all have center at $x=4$.
\begin{center}
\begin{tikzpicture}
\begin{axis}[
    width=12.5cm,height=7cm,
    axis lines=left,
    xlabel={$x$}, ylabel={$f(x)$},
    xmin=-2,xmax=10,ymin=0,ymax=0.85,
    grid=both,
    legend style={at={(0.02,0.98)},anchor=north west}
]
    % Uniform narrow [3,5]
    \addplot[very thick,blue,domain=3:5,samples=2] {0.5};
    % Uniform wide [1,7]
    \addplot[very thick,red,domain=1:7,samples=2] {1/6};
    % Normal narrow mu=4 sigma=0.7
    \addplot[very thick,teal!70!black,domain=-2:10,samples=250] {1/(0.7*sqrt(2*pi))*exp(-((x-4)^2)/(2*0.7^2))};
    % Normal wide mu=4 sigma=1.6
    \addplot[very thick,orange,dashed,domain=-2:10,samples=250] {1/(1.6*sqrt(2*pi))*exp(-((x-4)^2)/(2*1.6^2))};
    \addplot[black,densely dashed] coordinates {(4,0) (4,0.82)};
    \legend{$U[3,5]$, $U[1,7]$, $N(4,0.7^2)$, $N(4,1.6^2)$, center $x=4$}
\end{axis}
\end{tikzpicture}
\end{center}

(a) Which graph has the largest standard deviation? \\
(b) Which graph has the smallest standard deviation? \\
(c) Explain why all means are equal even though shapes differ.

\subsection*{Solution to Problem 2}
\textbf{Final answers.}
\begin{itemize}
    \item (a) Largest standard deviation: $N(4,1.6^2)$ (orange dashed).
    \item (b) Smallest standard deviation: $U[3,5]$ (blue).
    \item (c) All means are $4$ because each graph is centered at $x=4$ (symmetry/balance point).
\end{itemize}

\textbf{Reasoning.}
All four graphs are centered at $x=4$, but spread is different.
\begin{itemize}
    \item Wider support or flatter bell $\Rightarrow$ probability is distributed farther from 4 $\Rightarrow$ larger $\sigma$.
    \item Narrow support or sharper peak around 4 $\Rightarrow$ values stay closer to 4 $\Rightarrow$ smaller $\sigma$.
\end{itemize}
Thus mean (location) can stay fixed while standard deviation (dispersion) changes.

\section{Problem 3 (Type C) --- Different mean, same spread (normal shifts)}
Consider three normal distributions with the same standard deviation $\sigma=1$:
\[
X\sim N(1,1^2),\qquad Y\sim N(4,1^2),\qquad Z\sim N(7,1^2).
\]

\begin{center}
\begin{tikzpicture}
\begin{axis}[
    width=12cm,height=6.5cm,
    axis lines=left,
    xlabel={$x$}, ylabel={$f(x)$},
    xmin=-2,xmax=10,ymin=0,ymax=0.45,
    grid=both,
    legend style={at={(0.98,0.98)},anchor=north east}
]
    \addplot[very thick,blue,domain=-2:10,samples=250] {1/sqrt(2*pi)*exp(-((x-1)^2)/2)};
    \addplot[very thick,red,domain=-2:10,samples=250] {1/sqrt(2*pi)*exp(-((x-4)^2)/2)};
    \addplot[very thick,teal!70!black,domain=-2:10,samples=250] {1/sqrt(2*pi)*exp(-((x-7)^2)/2)};
    \addplot[blue,dashed] coordinates {(1,0) (1,0.43)};
    \addplot[red,dashed] coordinates {(4,0) (4,0.43)};
    \addplot[teal!70!black,dashed] coordinates {(7,0) (7,0.43)};
    \legend{$N(1,1^2)$,$N(4,1^2)$,$N(7,1^2)$}
\end{axis}
\end{tikzpicture}
\end{center}

(a) Order the expected values from smallest to largest. \\
(b) Compare the standard deviations. \\
(c) Explain why horizontal shift changes mean but not spread here.

\subsection*{Solution to Problem 3}
\textbf{Final answers.}
\begin{itemize}
    \item (a) $\mu_X<\mu_Y<\mu_Z$, i.e. $1<4<7$.
    \item (b) All standard deviations are equal: $\sigma_X=\sigma_Y=\sigma_Z=1$.
    \item (c) Each curve has exactly the same shape and width; only the center moves left or right.
\end{itemize}

\textbf{Reasoning.}
A pure horizontal translation moves the balance point (mean), but does not stretch or compress the graph. Therefore dispersion is unchanged.

Also, shifting $A$ right by 2 units gives $B$: mean changes by $+2$, spread unchanged.

\end{axis}
\end{tikzpicture}
\end{center}

Match each graph to a mean option and a spread description.
\begin{itemize}
    \item Mean options: $\left\{3,\ \frac{13}{3},\ \frac{17}{3}\right\}$
    \item Spread description options:
    \begin{itemize}
        \item \textbf{S1:} medium spread (support length 6, symmetric)
        \item \textbf{S2:} larger spread with long right tail
        \item \textbf{S3:} smaller spread, concentrated near the right side
    \end{itemize}
\end{itemize}

\subsection*{Solution to Problem 5}
\textbf{Final match.}
\begin{itemize}
    \item $T_1\to \mu=3$, spread \textbf{S1}
    \item $T_2\to \mu=\frac{13}{3}$, spread \textbf{S2}
    \item $T_3\to \mu=\frac{17}{3}$, spread \textbf{S3}
\end{itemize}

\textbf{Reasoning (shape-based).}
\begin{itemize}
    \item $T_1$ is symmetric around 3, so its mean is exactly 3.
    \item $T_2$ has much of its support extending to the right, so its balance point is pulled rightward; it also has a long right tail, so spread is larger.
    \item $T_3$ places most mass near larger $x$ values (peak near 7 on support $[2,8]$), so mean is further right; compared with $T_2$, values are less spread to the far right, so spread is smaller.
\end{itemize}
(Formula check when needed: for triangular $(a,c,b)$, $\mu=(a+b+c)/3$, which confirms the three means above.)

\section{Problem 6 (Type E) --- Linear increasing vs decreasing densities}
Two linear densities are defined on $[0,6]$:
\[
f_1(x)=\frac{x}{18},\qquad f_2(x)=\frac{6-x}{18},\qquad 0\le x\le 6.
\]

\begin{center}
\begin{tikzpicture}
\begin{axis}[
    width=12cm,height=6.5cm,
    axis lines=left,
    xlabel={$x$}, ylabel={$f(x)$},
    xmin=-0.5,xmax=6.5,ymin=0,ymax=0.38,
    grid=both,
    legend style={at={(0.98,0.98)},anchor=north east}
]
    \addplot[very thick,blue,domain=0:6,samples=150] {x/18};
    \addplot[very thick,red,dashed,domain=0:6,samples=150] {(6-x)/18};
    \addplot[blue,dotted] coordinates {(4,0) (4,0.34)};
    \addplot[red,dotted] coordinates {(2,0) (2,0.34)};
    \legend{$f_1(x)=x/18$,$f_2(x)=(6-x)/18$}
\end{axis}
\end{tikzpicture}
\end{center}

(a) Which density has larger expected value? \\
(b) Compare their standard deviations. \\
(c) Explain using shape, not only formula.

\subsection*{Solution to Problem 6}
\textbf{Final answers.}
\begin{itemize}
    \item (a) $f_1$ has larger expected value.
    \item (b) The standard deviations are equal.
    \item (c) $f_1$ and $f_2$ are mirror images around $x=3$.
\end{itemize}

\textbf{Reasoning.}
\begin{itemize}
    \item For $f_1$, density increases with $x$, so more weight is on larger values; this pushes the mean to the right ($\mu_1=4$).
    \item For $f_2$, density decreases with $x$, so more weight is on smaller values; this pulls the mean left ($\mu_2=2$).
    \item Because the two shapes are reflections of each other on the same interval length, their spread around their own centers is the same (equal $\sigma$).
\end{itemize}
This is a strong example of \emph{different mean, same spread} produced by a shape reflection.

\subsection*{Solution to Problem 7}
\textbf{Final answers.}
\begin{itemize}
    \item (a) P $\to \sigma=0.6$, Q $\to \sigma=1.0$, R $\to \sigma=1.8$.
    \item (b) Largest variance: Curve R, because variance is $\sigma^2$ and R has the largest $\sigma$.
    \item (c) With fixed total area 1, a narrower distribution must be taller; a wider one is flatter.
\end{itemize}

\textbf{Reasoning.}
All centers are at 5 (same mean), but widths differ. Visual width and one-standard-deviation intervals directly show spread size.

\section{Problem 8 (Cross-family interpretation)}
Three distributions are drawn on the same axes:
\begin{itemize}
    \item Uniform $U[1,9]$
    \item Triangular $(1,5,9)$ (symmetric)
    \item Normal $N(5,1.3^2)$
\end{itemize}

\begin{center}
\begin{tikzpicture}
\begin{axis}[
    width=12.5cm,height=7cm,
    axis lines=left,
    xlabel={$x$}, ylabel={$f(x)$},
    xmin=0,xmax=10,ymin=0,ymax=0.55,
    grid=both,
    legend style={at={(0.98,0.98)},anchor=north east}
]
    \addplot[very thick,blue,domain=1:9,samples=2] {1/8};
    \addplot[very thick,red,dashed] coordinates {(1,0) (5,0.25) (9,0)};
    \addplot[very thick,teal!70!black,domain=0:10,samples=300] {1/(1.3*sqrt(2*pi))*exp(-((x-5)^2)/(2*1.3^2))};
    \addplot[black,dotted] coordinates {(5,0) (5,0.53)};
    \legend{$U[1,9]$, Triangular $(1,5,9)$, $N(5,1.3^2)$}
\end{axis}
\end{tikzpicture}
\end{center}

(a) Which distributions clearly have expected value 5 from symmetry? \\
(b) Which is most concentrated near the center? \\
(c) Which places relatively more probability toward the outer parts of the interval $[1,9]$?

\subsection*{Solution to Problem 8}
\textbf{Final answers.}
\begin{itemize}
    \item (a) All three have expected value 5 (each is centered symmetrically at $x=5$).
    \item (b) Most concentrated near center: the normal curve $N(5,1.3^2)$.
    \item (c) Relatively more probability in outer parts: the uniform $U[1,9]$.
\end{itemize}

\textbf{Reasoning.}
\begin{itemize}
    \item The normal curve is highest near 5 and drops quickly, so many values are close to the mean (smaller spread among these three).
    \item The triangular curve still favors center values, but less strongly than the normal here.
    \item Uniform gives equal height everywhere in $[1,9]$, so edge values are not down-weighted; this corresponds to greater dispersion toward extremes.
\end{itemize}

\bigskip
\hrule
\bigskip
\textbf{Conclusion.} Across uniform, linear, triangular, and normal families, interpretation of graphs follows the same principle: \emph{mean describes center; standard deviation/variance describe spread; shape explains where probability mass is concentrated.}
