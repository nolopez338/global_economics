\section{Matching Problems}

\subsection*{Problem 1. Uniform: match graphs with their means}
Correct matches: \(A\to -1,\ B\to 2,\ C\to 5,\ D\to 9\).
\[

\]

\subsection*{Problem 2. Uniform: match graph with their spread}
Correct matches: \(A\to 0.58,\ B\to 1.15,\ C\to 1.73,\ D\to 2.89\).
\[
\sigma=\frac{b-a}{\sqrt{12}}
\]

\subsection*{Problem 3. Match graph, mean, and spread for three uniform distributions}
Correct matches:
\[
A:(\mu,\sigma)=(2,1.15),\quad
B:(\mu,\sigma)=(5,1.73),\quad
C:(\mu,\sigma)=(8,1.15).
\]

\subsection*{Problem 4. Match graph, mean, and spread for four uniform distributions}
Correct matches:
\[
A:(\mu,\sigma)=(-1,1.15),\quad
B:(\mu,\sigma)=(2,1.73),\quad
C:(\mu,\sigma)=(6,2.31),\quad
D:(\mu,\sigma)=(9,1.15).
\]

\subsection*{Problem 7. Triangular: same spread, different mean}
Correct matches: \(A\to -\frac{4}{3},\ B\to \frac{7}{3},\ C\to \frac{14}{3},\ D\to \frac{28}{3}\).
\[
\mu=\frac{a+b+c}{3}
\]

\subsection*{Problem 8. Triangular: same mean, different spread}
Correct matches: \(A\to 0.41,\ B\to 1.08,\ C\to 1.47,\ D\to 2.04\).
\[
\sigma=\sqrt{\frac{a^2+b^2+c^2-ab-ac-bc}{18}}
\]

\subsection*{Problem 9. Match graph, mean, and spread for three triangular distributions}
Correct matches:
\[
A:(\mu,\sigma)=(0,0.82),\quad
B:(\mu,\sigma)=(4,1.22),\quad

\]

\subsection*{Problem 10. Match graph, mean, and spread for four triangular distributions}
Correct matches:
\[
A:(\mu,\sigma)=\left(-\frac{4}{3},0.85\right),\quad

C:(\mu,\sigma)=\left(\frac{17}{3},1.65\right),\quad
D:(\mu,\sigma)=\left(\frac{28}{3},0.85\right).
\]

\subsection*{Problem 13. Normal I: match two graphs with their mean and standard deviation}

\]

\subsection*{Problem 14. Normal II: match three graphs with their mean and standard deviation}
Correct matches:
\[
A:(\mu,\sigma)=(2.6,1.1),\quad B:(\mu,\sigma)=(3.3,0.9),\quad C:(\mu,\sigma)=(4.0,1.3).
