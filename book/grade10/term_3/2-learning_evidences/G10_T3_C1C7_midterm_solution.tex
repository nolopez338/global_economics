	\section{(9) Online Coffee Sales Investment (C1)}
	\label{sec:online-coffee-sales-investment-c1}
	
	A small entrepreneur runs a specialty coffee resale business. Let $V$ be the end-of-quarter value, measured in million pesos. The final value cannot be below $20$ and cannot exceed $68$. Lower values are possible, but higher values become gradually more likely across this interval, so the final value is modeled with an increasing density. The initial investment is $32$ million pesos. The probability of the extreme value of 20 million pesos can be considered to be 0.
	
	\textbf{Questions/Tasks.}
	\begin{enumerate}[label=\textbf{\alph*)}]
		\item \TaskHyperref{subsec:online-coffee-sales-investment-c1-a}{\textbf{(3)} Determine the density constant and write the pdf.}
		\item \TaskHyperref{subsec:online-coffee-sales-investment-c1-b}{\textbf{(2)} Find the probability of loss and the probability of earning.}
		\item \TaskHyperref{subsec:online-coffee-sales-investment-c1-c}{\textbf{(4)} Compute the probability that the money obtained falls between 32 and 56 million.}
%		\item \TaskHyperref{subsec:online-coffee-sales-investment-c1-d}{\textbf{(2)} Compute the probability that the money obtained is above 56 million.}
	\end{enumerate}
	
	\subsection*{a) \textbf{(3)} Determine the density constant and write the pdf.}
	\phantomsection
	\label{subsec:online-coffee-sales-investment-c1-a}
	Let $V$ denote the end-of-quarter value, measured in million pesos. Since the density rises linearly from the lower endpoint $20$ to the upper endpoint $68$, we model it with a linear function.
	
	One possible general form for a line is
	\[
	f(v)=m(v+c).
	\]
	
	We know that the density starts at $0$ when $v=20$, so
	\[
	\begin{aligned}
		f(20)=m(20+c)=0.
	\end{aligned}
	\]
	%
	Since the density is increasing, \(m\neq 0\). Therefore,
	\[
	20+c=0
	\qquad \longrightarrow \qquad
	c=-20
	\qquad \longrightarrow \qquad
	f(v)=m(v-20)
	\]
	%
	Since the region under the density must have total area \(1\), and the graph forms a triangle with base and height:
	\[
	B = 68-20=48
	\qquad \qquad  \qquad
	h = f(68)
	\]
	we have
	\[
	\begin{aligned}
		A_{\triangle}=\frac{1}{2}Bh
		=\frac{1}{2}(48)h
		=1
		&\qquad \longrightarrow \qquad
		24h=1
		\qquad \longrightarrow \qquad
		h=\frac{1}{24}.
	\end{aligned}
	\]
	
	Since \(f(68)=h\), we now use \(h=\frac{1}{24}\). Using \(f(v)=m(v-20)\), substitute \(v=68\):
	\[
	f(68)=m(68-20)
	\qquad \longrightarrow \qquad
	\frac{1}{24}=h=f(68)=48m
	\qquad \longrightarrow \qquad
	\frac{1}{24}=48m
	\qquad \longrightarrow \qquad
	m=\frac{1}{24 \cdot 48}=\frac{1}{1152}
	\]

	%
	Hence the pdf is
	\[
	f(v)=
	\begin{cases}
		\dfrac{1}{1152}(v-20), & 20\le v\le 68,\\[0.2cm]
		0, & \text{otherwise.}
	\end{cases}
	\]

	\subsection*{b) \textbf{(2)} Find the probability of loss and the probability of earning.}
	\phantomsection
	\label{subsec:online-coffee-sales-investment-c1-b}
	\textbf{Probability of loss.} A loss occurs when the final value is below the purchase price of $32$ million pesos. Therefore, we need to find
	\[
	P(\text{loss})=P(V<32)
	\]
	
	On the interval from $20$ to $32$, the graph forms a smaller triangle. Its base and height are
	\[
	B=32-20=12
	\qquad \qquad \qquad
	h=f(32)
	\]
	
	Using the pdf from part a, we calculate the height:
	\[
	\begin{aligned}
		h=f(32)
		&=\frac{1}{1152}(32-20)
		=\frac{12}{1152}
		=\frac{1}{96}
	\end{aligned}
	\]
	
	Since the probability is the area under the density curve, we have
	\[
	\begin{aligned}
		P(\text{loss})=P(V<32)
		&=A_{\triangle}
		=\frac{1}{2}Bh
		=\frac{1}{2}(12)\left(\frac{1}{96}\right)
		=\frac{6}{96}
		=\frac{1}{16}
	\end{aligned}
	\]
	
	\textbf{Probability of earning.} An earning occurs when the final value is above the purchase price of $32$ million pesos. Therefore, we need to find $P(\text{earning})$, we proceed using complement rule
	\[
	P(\text{earning})=P(V>32)=1-P(V<32)=1-\frac{1}{16}
	=\frac{15}{16}
	\]
	%
	Therefore,
	\[
	\text{Loss} \rightarrow P(V<32)=\frac{1}{16}
	\qquad\qquad
	\text{Earning} \rightarrow P(V>32)=\frac{15}{16}
	\]

	\subsection*{c) \textbf{(4)} Compute the probability that the money obtained falls between 32 and 56 million.}
	\phantomsection
	\label{subsec:online-coffee-sales-investment-c1-c}
	Now compute the probability that the money obtained falls between $32$ and $56$ million pesos. Therefore, we need to find
	\[
	P(32\le V\le 56)
	\]
	
	\textbf{Solution 1: Trapezoid method.}
	
	On the interval from $32$ to $56$, the region under the density forms a trapezoid. Its base and heights are
	\[
	B=56-32=24
	\qquad \qquad \qquad
	h_1=f(32)
	\qquad \qquad \qquad
	h_2=f(56)
	\]
	
	Using the pdf from part a, we calculate the first height:
	\[
	\begin{aligned}
		h_1=f(32)
		&=\frac{1}{1152}(32-20)
		=\frac{12}{1152}
		=\frac{1}{96}
	\end{aligned}
	\]
	
	Using the same pdf, we calculate the second height:
	\[
	\begin{aligned}
		h_2=f(56)
		&=\frac{1}{1152}(56-20)
		=\frac{36}{1152}
		=\frac{1}{32}
	\end{aligned}
	\]
	
	Since the probability is the area under the density curve, we have
	\[
	\begin{aligned}
		P(32\le V\le 56)
		&=A_{\text{trapezoid}}
		=\frac{1}{2}B(h_1+h_2)\\
		&=\frac{1}{2}(24)\left(\frac{1}{96}+\frac{1}{32}\right)
		=12\left(\frac{1}{96}+\frac{3}{96}\right)\\
		&=12\left(\frac{4}{96}\right)
		=\frac{48}{96}
		=\frac{1}{2}
	\end{aligned}
	\]
	
	\textbf{Solution 2: Larger triangle minus smaller triangle.}
	
	First, compute the probability below $56$. On the interval from $20$ to $56$, the graph forms a triangle. Its base and height are
	\[
	B_1=56-20=36
	\qquad \qquad \qquad
	h_1=f(56)
	\]
	
	Using the pdf from part a), we calculate the height:
	\[
	\begin{aligned}
		h_1=f(56)
		&=\frac{1}{1152}(56-20)
		=\frac{36}{1152}
		=\frac{1}{32}
	\end{aligned}
	\]
	
	Therefore,
	\[
	\begin{aligned}
		P(V<56)
		&=A_{\triangle}
		=\frac{1}{2}B_1h_1
		=\frac{1}{2}(36)\left(\frac{1}{32}\right)
		=\frac{18}{32}
		=\frac{9}{16}
	\end{aligned}
	\]
	
	The probability below $32$ was already calculated in part b), so we use
	
	\[P(V<32)=\frac{1}{16}\]
	
	Since the region from $32$ to $56$ is obtained by subtracting the smaller triangle from the larger triangle, we have
	\[
	P(32\le V\le 56)=P(V<56)-P(V<32)=\frac{9}{16}-\frac{1}{16}
	=\frac{8}{16}
	=\frac{1}{2}
	\]
	
	Thus, the probability that the money obtained falls between $32$ and $56$ million pesos is $50\%$

\section{(9) Three-Scenario Inventory Investment (C2)}
\label{sec:three-scenario-inventory-investment-c2}

\textbf{Problem description.}
A seller invests in a short-term inventory deal to resell packaged foods in nearby cities. For the quarter, they consider 3 realistic market scenarios. In Scenario 1, transport bottlenecks continue and restocking stays slow; this has probability $0.2$. In Scenario 2, transport returns to normal and inventory rotates faster; this has probability $0.5$. In Scenario 3, demand increases because restaurants and small stores begin buying in bulk; this has probability $0.3$. If Scenario 1 occurs, the final value of the investment is expected to end between $48$ and $68$ million pesos, with all values in that interval treated as equally plausible. If Scenario 2 occurs, the final value is expected to end between $54$ and $104$ million pesos, again with all values in that interval treated as equally plausible. If Scenario 3 occurs, the final value is expected to end between $60$ and $90$ million pesos, again with all values in that interval treated as equally plausible. The initial investment is $60$ million pesos.

\textbf{Questions/Tasks.}
\begin{enumerate}[label=\textbf{\alph*)}]
	\item \TaskHyperref{subsec:three-scenario-inventory-investment-c2-a}{\textbf{(2)} Write the conditional density for each scenario.}
	\item \TaskHyperref{subsec:three-scenario-inventory-investment-c2-b}{\textbf{(3)} Build the unconditional pdf of the final value as a piecewise function.}
	\item \TaskHyperref{subsec:three-scenario-inventory-investment-c2-c}{\textbf{(4)} Find the probability of loss and the probability of earning.}
\end{enumerate}

A conditional density function describes how the final value is distributed once one specific scenario is assumed to be true. In this problem, each scenario has its own density because the investment behaves differently depending on market conditions.

The unconditional density function combines the scenario-specific densities into one overall model. It is obtained by weighting each conditional density by the probability of its scenario. The overlap region is important because it contains values that are plausible under at least two market scenarios.

\subsection*{a) \textbf{(2)} Write the conditional density for each scenario.}
\phantomsection
\label{subsec:three-scenario-inventory-investment-c2-a}
Let $V$ be the final value of the investment, measured in million pesos.

Under Scenario 1, the value is uniformly distributed on $[48,68]$. Therefore, the conditional density is
\[
f_{V\mid S_1}(v)=
\begin{cases}
	\dfrac{1}{20}, & 48\le v\le 68,\\[4pt]
	0, & \text{otherwise.}
\end{cases}
\]

Under Scenario 2, the value is uniformly distributed on $[54,104]$. Therefore, the conditional density is
\[
f_{V\mid S_2}(v)=
\begin{cases}
	\dfrac{1}{50}, & 54\le v\le 104,\\[4pt]
	0, & \text{otherwise.}
\end{cases}
\]

Under Scenario 3, the value is uniformly distributed on $[60,90]$. Therefore, the conditional density is
\[
f_{V\mid S_3}(v)=
\begin{cases}
	\dfrac{1}{30}, & 60\le v\le 90,\\[4pt]
	0, & \text{otherwise.}
\end{cases}
\]

The scenario probabilities are
\[
P(S_1)=0.2
\qquad
P(S_2)=0.5
\qquad
P(S_3)=0.3.
\]

So the unconditional density is the mixture
\[
f_V(v)=0.2\,f_{V\mid S_1}(v)+0.5\,f_{V\mid S_2}(v)+0.3\,f_{V\mid S_3}(v).
\]

\subsection*{b) \textbf{(3)} Build the unconditional pdf of the final value as a piecewise function.}
\phantomsection
\label{subsec:three-scenario-inventory-investment-c2-b}
Now build it by intervals.

For $48\le v<54$, only Scenario 1 contributes, so
\[
f_V(v)=0.2\left(\frac{1}{20}\right)=\frac{1}{100}.
\]

For $54\le v<60$, Scenarios 1 and 2 contribute, so
\[
f_V(v)=0.2\left(\frac{1}{20}\right)+0.5\left(\frac{1}{50}\right)
=\frac{1}{100}+\frac{1}{100}
=\frac{1}{50}.
\]

For $60\le v\le 68$, all three scenarios contribute, so
\[
f_V(v)=0.2\left(\frac{1}{20}\right)+0.5\left(\frac{1}{50}\right)+0.3\left(\frac{1}{30}\right)
=\frac{1}{100}+\frac{1}{100}+\frac{1}{100}
=\frac{3}{100}.
\]

For $68<v\le 90$, Scenarios 2 and 3 contribute, so
\[
f_V(v)=0.5\left(\frac{1}{50}\right)+0.3\left(\frac{1}{30}\right)
=\frac{1}{100}+\frac{1}{100}
=\frac{1}{50}.
\]

For $90<v\le 104$, only Scenario 2 contributes, so
\[
f_V(v)=0.5\left(\frac{1}{50}\right)=\frac{1}{100}.
\]

Therefore, the unconditional pdf is
\[
f_V(v)=
\begin{cases}
	\dfrac{1}{100}, & 48\le v<54,\\[6pt]
	\dfrac{1}{50}, & 54\le v<60,\\[6pt]
	\dfrac{3}{100}, & 60\le v\le 68,\\[6pt]
	\dfrac{1}{50}, & 68<v\le 90,\\[6pt]
	\dfrac{1}{100}, & 90<v\le 104,\\[6pt]
	0, & \text{otherwise.}
\end{cases}
\]

\subsection*{c) \textbf{(4)} Find the probability of loss and the probability of earning.}
\phantomsection
\label{subsec:three-scenario-inventory-investment-c2-c}
\textbf{Probability of loss.} A loss occurs when the final value is below the initial investment of $60$ million pesos. This includes the interval from $48$ to $54$, plus the interval from $54$ to $60$.

From $48$ to $54$, the area is
\[
6\left(\frac{1}{100}\right)=\frac{3}{50}.
\]

From $54$ to $60$, the width is $6$ and the height is $\dfrac{1}{50}$, so the area is
\[
6\left(\frac{1}{50}\right)=\frac{3}{25}.
\]

Therefore,
\[
P(\text{loss})=P(V<60)=\frac{3}{50}+\frac{3}{25}=\frac{9}{50}.
\]

\textbf{Probability of earning.}
An earning occurs when the final value is above the initial investment of $60$ million pesos. Since the total probability is $1$ and there is no probability mass at exactly $60$, use the complement rule
\[
P(\text{earning})=P(V>60)=1-P(V<60).
\]
From the previous result,
\[
P(V<60)=\frac{9}{50}.
\]
Therefore,
\[
P(V>60)=1-\frac{9}{50}\qquad \longrightarrow \qquad P(V>60)=\frac{41}{50}.
\]

	\section{(9) Small-Business Investment (C3)}
	\label{sec:small-business-investment-c3}
	
	An investor invests in a short-term small-business opportunity: buying low-cost clothing bundles to resell online and at weekend markets. They estimate the final value cannot be below $0$ million pesos because, in this simplified model, the end-of-period value cannot be negative. They also estimate it cannot exceed $40$ million pesos because their first-round sales capacity is limited. Their most likely final value is $16$ million pesos, based on expected demand from regular customers. Values below $16$ million pesos are possible, but become less likely as results move farther down from that level. Values above $16$ million pesos are also possible, but after that point, high outcomes become gradually less likely. Since the model is continuous, the probability of obtaining any exact extreme value can be considered $0$, so the probabilities of ending at exactly $0$ million pesos or exactly $40$ million pesos are treated as $0$. The initial investment is $10$ million pesos. A loss occurs if the final value is below that amount, and an earning occurs if the final value is above it.
	
	\textbf{Questions/Tasks.}
	\begin{enumerate}[label=\textbf{\alph*)}]
		\item \TaskHyperref{subsec:small-business-investment-c3-a}{\textbf{(3)} Write the pdf as a piecewise function.}
		\item \TaskHyperref{subsec:small-business-investment-c3-b}{\textbf{(3)} Find the probability of loss and the probability of earning.}
		\item \TaskHyperref{subsec:small-business-investment-c3-c}{\textbf{(3)} Find the specific probability of earning more than 50\% of the investment.}
	\end{enumerate}
	
	\subsection*{a) \textbf{(3)} Write the pdf as a piecewise function.}
	\phantomsection
	\label{subsec:small-business-investment-c3-a}
	Let $V$ be the final value of the investment, measured in million pesos.
	
	The support is from $0$ to $40$, and the most likely value is $16$. Since the model is triangular, the graph rises linearly from $(0,0)$ to the peak at $(16,h)$, then falls linearly from $(16,h)$ to $(40,0)$.
	
	Since the region under the density must have total area \(1\), and the graph forms a triangle with base and height:
	\[
	B=40-0=40
	\qquad \qquad \qquad
	h=f(16)
	\]
	we have
	\[
	\begin{aligned}
		A_{\triangle}=\frac{1}{2}Bh
		=\frac{1}{2}(40)h
		=1
		&\qquad \longrightarrow \qquad
		20h=1
		\qquad \longrightarrow \qquad
		h=\frac{1}{20}.
	\end{aligned}
	\]
	
	\textbf{Left-hand piece.}
	
	On the interval from $0$ to $16$, the graph is linear. One possible general form for a line is
	\[
	f(v)=m(v+c)
	\]
	
	We know that the density starts at $0$ when $v=0$, so
	\[
	f(0)=m(0+c)=0
	\qquad \longrightarrow \qquad
	c=0
	\qquad \longrightarrow \qquad
	f(v)=mv
	\]
	%
	Since \(f(16)=h\), we now use \(h=\frac{1}{20}\). Substituting \(v=16\):
	\[
	f(16)=16m
	\qquad \longrightarrow \qquad
	\frac{1}{20}=h=f(16)=16m
	\qquad \longrightarrow \qquad
	m=\frac{1}{20\cdot 16}=\frac{1}{320}.
	\]
	
	Thus, the left-hand piece is
	\[
	f(v)=\frac{v}{320}
	\qquad \text{for }0\le v\le 16.
	\]
	
	\textbf{Right-hand piece.}
	
	On the interval from $16$ to $40$, the graph is also linear. We use again
	\[
	f(v)=m(v+c).
	\]
	
	We know that the density ends at $0$ when \(v=40\), so
	\[
	f(40)=m(40+c)=0
	\qquad \longrightarrow \qquad
	40+c=0
	\qquad \longrightarrow \qquad
	c=-40
	\qquad \longrightarrow \qquad
	f(v)=m(v-40)
	\]
	%
	Since \(f(16)=h\), we now use \(h=\frac{1}{20}\). Substituting \(v=16\):
	\[
	f(16)=m(16-40)
	\qquad \longrightarrow \qquad
	\frac{1}{20}=h=f(16)=-24m
	\qquad \longrightarrow \qquad
	m=-\frac{1}{20\cdot 24}=-\frac{1}{480}.
	\]
	
	Thus, the right-hand piece is
	\[
	f(v)=-\frac{1}{480}(v-40)=\frac{40-v}{480}
	\qquad \text{for }16\le v\le 40.
	\]
	
	Therefore, the pdf is
	\[
	f(v)=
	\begin{cases}
		\dfrac{v}{320}, & 0\le v\le 16,\\[6pt]
		\dfrac{40-v}{480}, & 16\le v\le 40,\\[6pt]
		0, & \text{otherwise.}
	\end{cases}
	\]

	\subsection*{b) \textbf{(3)} Find the probability of loss and the probability of earning.}
	\phantomsection
	\label{subsec:small-business-investment-c3-b}
	\textbf{Probability of loss.} A loss occurs when the final value is below the purchase price of $10$ million pesos. Therefore, we need to find
	\[
	P(\text{loss})=P(V<10).
	\]
	
	On the interval from $0$ to $10$, the graph forms a triangle. Its base and height are
	\[
	B=10-0=10
	\qquad \qquad \qquad
	h=f(10)
	\]
	
	Using the pdf from part a, we calculate the height:
	\[
	\begin{aligned}
		h=f(10)
		&=\frac{10}{320}
		=\frac{1}{32}.
	\end{aligned}
	\]
	
	Since the probability is the area under the density curve, we have
	\[
	\begin{aligned}
		P(\text{loss})=P(V<10)
		&=A_{\triangle}
		=\frac{1}{2}Bh
		=\frac{1}{2}(10)\left(\frac{1}{32}\right)
		=\frac{5}{32}.
	\end{aligned}
	\]
	
	\textbf{Probability of earning.} An earning occurs when the final value is above the purchase price of $10$ million pesos. Therefore, we need to find $P(\text{earning})$, we proceed using complement rule:
	\[
	P(\text{earning})=P(V>10)=1-P(V<10) =1-\frac{5}{32}
	=\frac{27}{32}
	\]
	Therefore,
	\[
	\text{Loss} \rightarrow P(V<10)=\frac{5}{32}
	\qquad\qquad
	\text{Earning} \rightarrow P(V>10)=\frac{27}{32}.
	\]

	\subsection*{c) \textbf{(3)} Find the specific probability of earning more than 50\% of the investment.}
	\phantomsection
	\label{subsec:small-business-investment-c3-c}
	The investment amount is $10$ million pesos, so earning more than $50\%$ means ending above
	\[
	10(1.5)=15.
	\]
	
	Therefore, we need to find
	\[
	P(\text{earning more than }50\%)=P(V>15).
	\]
	
	Since \(15\) lies to the left of the most likely value \(16\), the complement rule is convenient. We first calculate the probability below \(15\):
	\[
	P(V>15)=1-P(V\le 15).
	\]
	
	On the interval from $0$ to $15$, the graph forms a triangle. Its base and height are
	\[
	B=15-0=15
	\qquad \qquad \qquad
	h=f(15)
	\]
	
	Using the pdf from part a, we calculate the height:
	\[
	\begin{aligned}
		h=f(15)
		&=\frac{15}{320}
		=\frac{3}{64}.
	\end{aligned}
	\]
	
	Since the probability is the area under the density curve, we have
	\[
	\begin{aligned}
		P(V\le 15)
		&=A_{\triangle}
		=\frac{1}{2}Bh
		=\frac{1}{2}(15)\left(\frac{3}{64}\right)
		=\frac{45}{128}.
	\end{aligned}
	\]
	
	Now use the complement rule:
	\[
	\begin{aligned}
		P(V>15)
		&=1-P(V\le 15)
		=1-\frac{45}{128}
		=\frac{83}{128}.
	\end{aligned}
	\]
	
	Thus, the probability of earning more than \(50\%\) of the investment is
	\[
	P(\text{earning more than }50\%)=P(V>15)=\frac{83}{128}.
	\]
