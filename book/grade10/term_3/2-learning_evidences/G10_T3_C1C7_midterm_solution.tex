	\section{(11) Online Coffee Sales Investment (C1)}
	\label{sec:online-coffee-sales-investment-c1}
	
	A small entrepreneur runs a specialty coffee resale business. Let $V$ be the end-of-quarter value, measured in million pesos. The final value cannot be below $20$ and cannot exceed $68$. Lower values are possible, but higher values become gradually more likely across this interval, so the final value is modeled with an increasing density on $[20,68]$. The initial investment is $32$ million pesos. The probability of the extreme value of 20 million pesos can be considered to be 0.
	
	\textbf{Questions/Tasks.}
	\begin{enumerate}[label=\textbf{\alph*)}]
		\item \TaskHyperref{subsec:online-coffee-sales-investment-c1-a}{\textbf{(3)} Determine the density constant and write the pdf.}
		\item \TaskHyperref{subsec:online-coffee-sales-investment-c1-b}{\textbf{(2)} Find the probability of loss and the probability of earning.}
		\item \TaskHyperref{subsec:online-coffee-sales-investment-c1-c}{\textbf{(4)} Compute the probability that the money obtained falls between 32 and 56 million.}
		\item \TaskHyperref{subsec:online-coffee-sales-investment-c1-d}{\textbf{(2)} Compute the probability that the money obtained is above 56 million.}
	\end{enumerate}
	
	\subsection*{a) \textbf{(3)} Determine the density constant and write the pdf.}
	\phantomsection
	\label{subsec:online-coffee-sales-investment-c1-a}
	Let $V$ denote the end-of-quarter value, measured in million pesos. Since the density rises linearly from the lower endpoint $20$ to the upper endpoint $68$, we model it by
	\[
	f(v)=k(v-20)\qquad \text{for }20\le v\le 68,
	\]
	and
	\[
	f(v)=0\qquad \text{otherwise.}
	\]
	
	To determine $k$, use total area $=1$. The graph is a straight line from $(20,0)$ up to $(68,48k)$, so the region under the density is a triangle with base $48$ and height $48k$. Therefore,
	\[
	\frac12(48)(48k)=1.
	\]
	So
	\[
	\begin{aligned}
		\frac12(48)(48k)=1
		&\qquad \longrightarrow \qquad 1152k=1
		\qquad \longrightarrow \qquad k=\frac{1}{1152}.
	\end{aligned}
	\]
	
	Hence the pdf is
	\[
	f(v)=\frac{1}{1152}(v-20)\qquad \text{for }20\le v\le 68.
	\]
	\LinearProbGraph{20}{68}{(x-20)/1152}{20}{68}{20,32,56,68}

	\subsection*{b) \textbf{(2)} Find the probability of loss and the probability of earning.}
	\phantomsection
	\label{subsec:online-coffee-sales-investment-c1-b}
	\textbf{Probability of loss.} A loss occurs when the final value is below the purchase price of $32$ million pesos. On the interval from $20$ to $32$, the graph forms a smaller triangle.
	
	Its base is
	\[
	32-20=12,
	\]
	and its height is
	\[
	f(32)=\frac{1}{1152}(32-20)=\frac{12}{1152}=\frac{1}{96}.
	\]
	So
	\[
	P(\text{loss})=P(V<32)=\frac12(12)\left(\frac{1}{96}\right)\qquad \longrightarrow \qquad P(\text{loss})=\frac{6}{96}\qquad \longrightarrow \qquad P(\text{loss})=\frac{1}{16}.
	\]
	\textbf{Probability of earning.}
	An earning occurs when the final value is above the purchase price of $32$ million pesos. Since the total probability is $1$ and a loss occurs when the final value is below $32$, we use the complement rule
	\[
	P(\text{earning})=P(V>32)=1-P(V<32).
	\]
	\[
	\text{Loss} \rightarrow P(V<32)=\frac{1}{16}
	\qquad\qquad
	\text{Earning} \rightarrow P(V>32)=1-\frac{1}{16}=\frac{15}{16}
	\]

	\subsection*{c) \textbf{(4)} Compute the probability that the money obtained falls between 32 and 56 million.}
	\phantomsection
	\label{subsec:online-coffee-sales-investment-c1-c}
	Now compute the probability that the money obtained falls between $32$ and $56$ million pesos.
	
	From part b),
	\[
	P(V>32)=\frac{15}{16}.
	\]
	
	From part d),
	\[
	P(V>56)=\frac{7}{16}.
	\]
	
	Therefore, using the difference of these two events,
	\[
	\begin{aligned}
		P(32\le V\le 56)=P(V>32)-P(V>56)
		&\qquad \longrightarrow \qquad P(32\le V\le 56)=\frac{15}{16}-\frac{7}{16}\\
		&\qquad \longrightarrow \qquad P(32\le V\le 56)=\frac{8}{16}\qquad \longrightarrow \qquad P(32\le V\le 56)=\frac{1}{2}.
	\end{aligned}
	\]
	
	Thus, the probability that the money obtained falls between $32$ and $56$ million pesos is
	\[
	\frac{1}{2}.
	\]

	\subsection*{d) \textbf{(2)} Compute the probability that the money obtained is above 56 million.}
	\phantomsection
	\label{subsec:online-coffee-sales-investment-c1-d}
	Now compute the probability that the money obtained is above $56$ million pesos.
	
	Instead of calculating the small region to the right of $56$, it is easier to first calculate the probability that the money obtained is less than $56$ million pesos. On the interval from $20$ to $56$, the graph forms a triangle.
	
	Its base is
	\[
	56-20=36,
	\]
	and its height is
	\[
	f(56)=\frac{1}{1152}(56-20)=\frac{36}{1152}=\frac{1}{32}.
	\]
	
	So
	\[
	P(V<56)=\frac12(36)\left(\frac{1}{32}\right)\qquad \longrightarrow \qquad P(V<56)=\frac{18}{32}\qquad \longrightarrow \qquad P(V<56)=\frac{9}{16}.
	\]
	
	Using the complement rule,
	\[
	P(V>56)=1-P(V<56)\qquad \longrightarrow \qquad P(V>56)=1-\frac{9}{16}\qquad \longrightarrow \qquad P(V>56)=\frac{7}{16}.
	\]
	
	Thus, the probability that the money obtained is above $56$ million pesos is
	\[
	\frac{7}{16}.
	\]

	\section{(12) Two-Scenario Inventory Investment (C2)}
	\label{sec:two-scenario-inventory-investment-c2}
	
	\textbf{Problem description:}
	A seller invests in a short-term inventory deal to resell packaged foods in nearby cities. For the quarter, they consider two realistic market scenarios. In Scenario 1, transport bottlenecks continue and restocking stays slow; this has probability $0.3$. In Scenario 2, transport returns to normal and inventory rotates faster; this has probability $0.7$. If Scenario 1 occurs, the final value of the investment is expected to end between forty-eight and sixty-six million pesos, with all values in that interval treated as equally plausible. If Scenario 2 occurs, the final value is expected to end between fifty-four and ninety-six million pesos, again with all values in that interval treated as equally plausible. The initial investment is sixty million pesos. A loss occurs if the final value is below that amount, and an earning occurs if the final value is above it.
	
	\textbf{Questions/Tasks.}
	\begin{enumerate}[label=\textbf{\alph*)}]
		\item \TaskHyperref{subsec:two-scenario-inventory-investment-c2-a}{\textbf{(2)} Write the conditional density for each scenario.}
		\item \TaskHyperref{subsec:two-scenario-inventory-investment-c2-b}{\textbf{(3)} Build the unconditional pdf of the final value as a piecewise function.}
		\item \TaskHyperref{subsec:two-scenario-inventory-investment-c2-c}{\textbf{(2)} Verify that the unconditional pdf integrates to one.}
		\item \TaskHyperref{subsec:two-scenario-inventory-investment-c2-d}{\textbf{(3)} Find the probability of loss and the probability of earning.}
		\item \TaskHyperref{subsec:two-scenario-inventory-investment-c2-e}{\textbf{(2)} Find the probability that the final value falls inside the overlap interval.}
	\end{enumerate}
	
	A conditional density function describes how the final value is distributed once one specific scenario is assumed to be true. In this problem, each scenario has its own density because the investment behaves differently depending on whether transport problems continue or normal logistics return. So, the conditional density tells us how the final value is distributed under that particular market condition.
	
	The general, or unconditional, density function combines those scenario-specific densities into one overall model. It describes the distribution of the final value before knowing which scenario actually occurs, and it is obtained by weighting the conditional densities by the probabilities of the scenarios. The overlap interval is especially important because it contains values that are plausible in both market scenarios. For a small investor, this shared middle range is often more useful for planning than extreme values that appear only under one scenario, since it represents outcomes that remain feasible whether logistics stay difficult or improve.
	
	\subsection*{a) \textbf{(2)} Write the conditional density for each scenario.}
	\phantomsection
	\label{subsec:two-scenario-inventory-investment-c2-a}
	Let $V$ be the final value of the investment, measured in million pesos.
	
	Under the delayed-clearance scenario, the value is uniformly distributed on $[48,66]$. Therefore, the conditional density is
	\[
	f_{V\mid S_1}(v)=
	\begin{cases}
		\dfrac{1}{18}, & 48\le v\le 66,\\[4pt]
		0, & \text{otherwise.}
	\end{cases}
	\]
	
	Under the normal-clearance scenario, the value is uniformly distributed on $[54,96]$. Therefore, the conditional density is
	\[
	f_{V\mid S_2}(v)=
	\begin{cases}
		\dfrac{1}{42}, & 54\le v\le 96,\\[4pt]
		0, & \text{otherwise.}
	\end{cases}
	\]
	
	The scenario probabilities are
	\[
	P(S_1)=0.3
	\qquad \text{and} \qquad
	P(S_2)=0.7.
	\]
	\CtwoProbGraph{48}{96}{x < 54 ? 1/60 : (x <= 66 ? 1/30 : 1/60)}{48}{96}{48,54,66,96}{6e-2}
	
	So the unconditional density is the mixture
	\[
	f_V(v)=0.3\,f_{V\mid S_1}(v)+0.7\,f_{V\mid S_2}(v).
	\]

	\subsection*{b) \textbf{(3)} Build the unconditional pdf of the final value as a piecewise function.}
	\phantomsection
	\label{subsec:two-scenario-inventory-investment-c2-b}
	Now build it by intervals.
	
	For $48\le v<54$, only Scenario 1 contributes, so
	\[
	f_V(v)=0.3\left(\frac{1}{18}\right)=\frac{1}{60}.
	\]
	
	For $54\le v\le 66$, both scenarios contribute, so
	\[
	f_V(v)=0.3\left(\frac{1}{18}\right)+0.7\left(\frac{1}{42}\right)
	=\frac{1}{60}+\frac{1}{60}
	=\frac{1}{30}.
	\]
	
	For $66<v\le 96$, only Scenario 2 contributes, so
	\[
	f_V(v)=0.7\left(\frac{1}{42}\right)=\frac{1}{60}.
	\]
	
	Therefore, the unconditional pdf is
	\[
	f_V(v)=
	\begin{cases}
		\dfrac{1}{60}, & 48\le v<54,\\[6pt]
		\dfrac{1}{30}, & 54\le v\le 66,\\[6pt]
		\dfrac{1}{60}, & 66<v\le 96,\\[6pt]
		0, & \text{otherwise.}
	\end{cases}
	\]
	\CtwoProbGraph{48}{96}{x < 54 ? 1/60 : (x <= 66 ? 1/30 : 1/60)}{54}{66}{48,54,66,96}{6e-2}
	
	\secInicio[sec:two-scenario-inventory-investment-c2][sec:uniform-distribution-i-c2]
	
	\subsection*{c) \textbf{(2)} Verify that the unconditional pdf integrates to one.}
	\phantomsection
	\label{subsec:two-scenario-inventory-investment-c2-c}
	Use the total area of the three rectangles.
	
	From $48$ to $54$, the width is $6$ and the height is $\dfrac{1}{60}$, so the area is
	\[
	6\left(\frac{1}{60}\right)=\frac{1}{10}.
	\]
	
	From $54$ to $66$, the width is $12$ and the height is $\dfrac{1}{30}$, so the area is
	\[
	12\left(\frac{1}{30}\right)=\frac{2}{5}.
	\]
	
	From $66$ to $96$, the width is $30$ and the height is $\dfrac{1}{60}$, so the area is
	\[
	30\left(\frac{1}{60}\right)=\frac{1}{2}.
	\]
	
	Adding the three pieces,
	\[
	\frac{1}{10}+\frac{2}{5}+\frac{1}{2}=1.
	\]
	So the unconditional density is valid.

	\subsection*{d) \textbf{(3)} Find the probability of loss and the probability of earning.}
	\phantomsection
	\label{subsec:two-scenario-inventory-investment-c2-d}
	\textbf{Probability of loss.} A loss occurs when the final value is below the purchase price of $60$ million pesos. This includes the whole interval from $48$ to $54$, plus the interval from $54$ to $60$.
	
	From $48$ to $54$, the area is
	\[
	6\left(\frac{1}{60}\right)=\frac{1}{10}.
	\]
	
	From $54$ to $60$, the width is $6$ and the height is $\dfrac{1}{30}$, so the area is
	\[
	6\left(\frac{1}{30}\right)=\frac{1}{5}.
	\]
	
	Therefore,
	\[
	P(\text{loss})=P(V<60)=\frac{1}{10}+\frac{1}{5}=\frac{3}{10}.
	\]
	\textbf{Probability of earning.}
	An earning occurs when the final value is above the purchase price of $60$ million pesos. Since the total probability is $1$ and a loss occurs when the final value is below $60$, we use the complement rule
	\[
	P(\text{earning})=P(V>60)=1-P(V<60).
	\]
	From the previous result,
	\[
	P(V<60)=\frac{3}{10}.
	\]
	Therefore,
	\[
	P(V>60)=1-\frac{3}{10}\qquad \longrightarrow \qquad P(V>60)=\frac{7}{10}.
	\]
	\CtwoThresholdGraph{48}{96}{x < 54 ? 1/60 : (x <= 66 ? 1/30 : 1/60)}{60}{48,54,60,66,96}{6e-2}{1/30}
	
	\secInicio[sec:two-scenario-inventory-investment-c2][sec:uniform-distribution-i-c2]
	
	\subsection*{e) \textbf{(2)} Find the probability that the final value falls inside the overlap interval.}
	\phantomsection
	\label{subsec:two-scenario-inventory-investment-c2-e}
	The overlap interval is from $54$ to $66$. Over this region, the density is constant at $\dfrac{1}{30}$, so
	\[
	P(54\le V\le 66)=12\left(\frac{1}{30}\right)=\frac{2}{5}.
	\]
	\CtwoProbGraph{48}{96}{x < 54 ? 1/60 : (x <= 66 ? 1/30 : 1/60)}{54}{66}{48,54,66,96}{6e-2}

	\section{(12) Small-Business Investment(C3)}
	\label{sec:small-business-investment-c3}
	
	An investor invests in a short-term small-business opportunity: buying low-cost clothing bundles to resell online and at weekend markets. They estimate the final value cannot be below zero million pesos because, in this simplified model, the end-of-period value cannot be negative. They also estimate it cannot exceed forty million pesos because their first-round sales capacity is limited. Their most likely final value is twenty million pesos, based on expected demand from regular customers. Values below twenty million pesos are possible, but become less likely as results move farther down from that level. Values above twenty million pesos are also possible, but after that point, very high outcomes become gradually less likely. Since the model is continuous, the probability of obtaining any exact extreme value can be considered $0$, so the probabilities of ending at exactly zero million pesos or exactly forty million pesos are treated as $0$. The initial investment is ten million pesos. A loss occurs if the final value is below that amount, and an earning occurs if the final value is above it.
	
	\textbf{Questions/Tasks.}
	\begin{enumerate}[label=\textbf{\alph*)}]
		\item \TaskHyperref{subsec:small-business-investment-c3-a}{\textbf{(4)} Write the pdf as a piecewise function.}
		\item \TaskHyperref{subsec:small-business-investment-c3-b}{\textbf{(1)} Verify that the total area under the density is one.}
		\item \TaskHyperref{subsec:small-business-investment-c3-c}{\textbf{(3)} Find the probability of loss and the probability of earning.}
		\item \TaskHyperref{subsec:small-business-investment-c3-d}{\textbf{(4)} Find the specific probability of earning more than 50\% of the investment.}
	\end{enumerate}
	
	\subsection*{a) \textbf{(4)} Write the pdf as a piecewise function.}
	\phantomsection
	\label{subsec:small-business-investment-c3-a}
	Let $V$ be the final value of the investment, measured in million pesos.
	
	The support is from $0$ to $40$, and the most likely value is $20$. Since the model is triangular, the graph rises linearly from $(0,0)$ to the peak at $(20,h)$, then falls linearly from $(20,h)$ to $(40,0)$.
	
	First determine the peak height $h$ from total area $=1$. The whole graph is one triangle with base
	\[
	40-0=40.
	\]
	So
	\[
	\frac12(40)(h)=1.
	\]
	Hence
	\[
	20h=1
	\qquad \Rightarrow \qquad
	h=\frac{1}{20}.
	\]
	
	Now write each linear piece.
	
	On the interval from $0$ to $20$, the line goes through $(0,0)$ and $(20,\tfrac{1}{20})$. Its slope is
	\[
	\frac{\tfrac{1}{20}-0}{20-0}=\frac{1}{400}.
	\]
	So the left-hand piece is
	\[
	f(v)=\frac{v}{400}\qquad \text{for }0\le v\le 20.
	\]
	
	On the interval from $20$ to $40$, the line goes through $(20,\tfrac{1}{20})$ and $(40,0)$. Its slope is
	\[
	\frac{0-\tfrac{1}{20}}{40-20}=-\frac{1}{400}.
	\]
	So the right-hand piece is
	\[
	f(v)=\frac{40-v}{400}\qquad \text{for }20\le v\le 40.
	\]
	
	Therefore, the pdf is
	\[
	f(v)=
	\begin{cases}
		\dfrac{v}{400}, & 0\le v\le 20,\\[6pt]
		\dfrac{40-v}{400}, & 20\le v\le 40,\\[6pt]
		0, & \text{otherwise.}
	\end{cases}
	\]
	\PiecewiseProbGraph{0}{40}{x <= 20 ? x/400 : (40-x)/400}{0}{40}{0,10,20,40}
	
	\secInicio[sec:small-business-investment-c3][sec:basic-symmetric-triangular-density-c3]
	
	\subsection*{b) \textbf{(1)} Verify that the total area under the density is one.}
	\phantomsection
	\label{subsec:small-business-investment-c3-b}
	The graph is a single triangle with base $40$ and height $\dfrac{1}{20}$. Therefore,
	\[
	\text{Total area}=\frac12(40)\left(\frac{1}{20}\right)=1.
	\]
	So this is a valid pdf.

	\subsection*{c) \textbf{(3)} Find the probability of loss and the probability of earning.}
	\phantomsection
	\label{subsec:small-business-investment-c3-c}
	\textbf{Probability of loss.} A loss occurs when the final value is below the purchase price of $10$ million pesos. On the graph, this is the small triangle from $0$ to $10$.
	
	Its base is
	\[
	10-0=10.
	\]
	Its height is
	\[
	f(10)=\frac{10}{400}=\frac{1}{40}.
	\]
	So
	\[
	P(\text{loss})=P(V<10)=\frac12(10)\left(\frac{1}{40}\right)=\frac{10}{80}=\frac{1}{8}.
	\]
	\textbf{Probability of earning.}
	An earning occurs when the final value is above the purchase price of $10$ million pesos. Since the total probability is $1$ and a loss occurs when the final value is below $10$, we use the complement rule
	\[
	P(\text{earning})=P(V>10)=1-P(V<10).
	\]
	From the previous result,
	\[
	P(V<10)=\frac{1}{8}.
	\]
	Therefore,
	\[
	P(V>10)=1-\frac{1}{8}=\frac{7}{8}.
	\]
	\LinearThresholdGraph{0}{40}{x <= 20 ? x/400 : (40-x)/400}{10}{0,10,20,40}{0.075}{1/40}

	\subsection*{d) \textbf{(4)} Find the specific probability of earning more than 50\% of the investment.}
	\phantomsection
	\label{subsec:small-business-investment-c3-d}
	The investment amount is $10$ million pesos, so earning more than $50\%$ means ending above
	\[
	10(1.5)=15.
	\]
	Therefore, we need
	\[
	P(\text{earning more than }50\%)=P(V>15).
	\]
	
	Notice that $15$ lies to the left of the most probable value $20$. For that reason, the complement rule is convenient here, because the excluded region from $0$ to $15$ is a simple triangle.
	
	So
	\[
	P(V>15)=1-P(V\le 15).
	\]
	
	On the interval from $0$ to $15$, the graph forms a triangle.
	
	Its base is
	\[
	15-0=15,
	\]
	and its height is
	\[
	f(15)=\frac{15}{400}=\frac{3}{80}.
	\]
	
	Hence
	\[
	P(V\le 15)=\frac12(15)\left(\frac{3}{80}\right)=\frac{45}{160}=\frac{9}{32}.
	\]
	
	Therefore,
	\[
	P(V>15)=1-\frac{9}{32}=\frac{23}{32}.
	\]
	\PiecewiseProbGraph{0}{40}{x <= 20 ? x/400 : (40-x)/400}{15}{40}{0,10,15,20,40}

	\section{Contextual Normal Distribution Problems (C6)}
	\label{sec:contextual-normal-distribution-problems-c6}
	
	Let \(Z\sim N(0,1)\). Let \(\Phi(z)=P(Z<z)\).
	
	Each contextual problem gives a normal variable \(X\sim N(\mu,\sigma^2)\). Since the standard normal table is written for \(Z\), each value of \(X\) must first be converted into a standard normal value.
	
	Use
	\[
	Z=\frac{X-\mu}{\sigma}
	\]
	
	This formula measures how many standard deviations the value of \(X\) is from the mean \(\mu\). After converting \(X\) into \(Z\), the probability can be found using \(\Phi(z)\).
	
	For left-tail probabilities, use
	\[
	P(X<x)=P\left(Z<\frac{x-\mu}{\sigma}\right)=\Phi\left(\frac{x-\mu}{\sigma}\right)
	\]
	
	For right-tail probabilities, use
	\[
	P(X>x)=P\left(Z>\frac{x-\mu}{\sigma}\right)=1-\Phi\left(\frac{x-\mu}{\sigma}\right)
	\]
	
	For interval probabilities, use
	\[
	P(a<X<b)=P\left(\frac{a-\mu}{\sigma}<Z<\frac{b-\mu}{\sigma}\right)
	\]
	\[
	P(a<X<b)=\Phi\left(\frac{b-\mu}{\sigma}\right)-\Phi\left(\frac{a-\mu}{\sigma}\right)
	\]

	\section{Inverse Normal Problems (C7)}
	\label{sec:inverse-normal-problems-c7}
	
	\textbf{Questions/Tasks.}
	
	Let \(Z\sim N(0,1)\). Let \(\Phi(z)=P(Z<z)\). Let \(z_p=\Phi^{-1}(p)\), so that
	\[
	\Phi(z_p)=p
	\]
	
	Here, \(z_p\) is the standard normal value with left-tail probability \(p\).
	
	For each normal variable \(X\sim N(\mu,\sigma^2)\), use
	\[
	Z=\frac{X-\mu}{\sigma}
	\]
	
	Solving for \(X\),
	\[
	X=\mu+\sigma Z
	\]
	
	Therefore, after finding \(z_p\), convert it to the original variable using
	\[
	x=\mu+\sigma z_p
	\]
	
	\subsection*{Problem 1}

	\[

	\]

	\[

	\]
	\[
	P(X<x)=0.95

	\[
	\Phi(z)=0.95
	\]
	\[
	z_{0.95}=1.645
	\]

	\[
	x=72+9(1.645)
	\]
	\[
	x=72+14.805=86.805
	\]
	\[
	x\approx 86.81
	\]
