\section{Problem 1: Left-tail inverse normal (positive and negative z)}
Exam scores follow $X\sim N(68,8^2)$.
\subsection*{C7}
\[P(X<x_1)=0.8023,\qquad P(X<x_2)=0.1151.\]
\XNormalLeft{0.85}{x_1}{P(X<x_1)=0.8023}{}
\XNormalLeft{-1.20}{x_2}{P(X<x_2)=0.1151}{}
\[\Phi(z)=0.8023\Rightarrow z_1=0.85,\quad \Phi(z)=0.1151\Rightarrow z_2=-1.20.\]
\[Z=\frac{X-68}{8},\qquad x=68+z(8).\]
\[x_1=74.8,\qquad x_2=58.4.\]
\ZNormalLeft{0.85}{P(Z<0.85)=0.8023}{}
\textbf{Interpretation:} About 80.23\% score below 74.8; 11.51\% score below 58.4.

\section{Problem 2: Right-tail inverse normal (positive and negative z)}
Delivery times follow $X\sim N(40,6^2)$.
\subsection*{C7}
\[P(X>x_1)=0.0885,\qquad P(X>x_2)=0.7580.\]
\XNormalRight{1.35}{x_1}{P(X>x_1)=0.0885}{}
\[P(X\le x_1)=0.9115\Rightarrow z_1=\Phi^{-1}(0.9115)=1.35.\]
\XNormalRight{-0.70}{x_2}{P(X>x_2)=0.7580}{}
\[P(X\le x_2)=0.2420\Rightarrow z_2=\Phi^{-1}(0.2420)=-0.70.\]
\[Z=\frac{X-40}{6},\quad x=40+z(6).\]
\[x_1=48.1,\qquad x_2=35.8.\]
\ZNormalRight{1.35}{P(Z>1.35)=0.0885}{}
\textbf{Interpretation:} 8.85\% exceed 48.1 min; 75.80\% exceed 35.8 min.
