		\item C5. Uses the normal standard distribution and its tables to calculate probabilities in contextual situations.
	\end{itemize}
	
	
	\noindent\textbf{Meaning of the notation.}
	The expression \(P(Z<x)\) means the probability that the standard normal random variable \(Z\) is less than the value \(x\). In the graph of the standard normal distribution, this corresponds to the area under the curve to the left of \(x\).
	
	For example $P(Z<0.85)$ means the area under the standard normal curve to the left of \(x=0.85\). Since the table gives values of \(\Phi(z)=P(Z<x)\), this value can be read directly from the table as
	\[
	P(Z<0.85)=\Phi(0.85)=0.8023.
	\]
	Here are five examples of left-tail probabilities read directly from the table.
	\begin{center}
		\begin{tabular}{c c c}
			\toprule
			\(x\) & \(P(Z<x)=\Phi(x)\) & Value \\
			\midrule
			\(0.25\) & \(P(Z<0.25)=\Phi(0.25)\) & \(0.5987\) \\
			\(0.70\) & \(P(Z<0.70)=\Phi(0.70)\) & \(0.7580\) \\
			\(0.85\) & \(P(Z<0.85)=\Phi(0.85)\) & \(0.8023\) \\
			\(1.20\) & \(P(Z<1.20)=\Phi(1.20)\) & \(0.8849\) \\
			\(1.65\) & \(P(Z<1.65)=\Phi(1.65)\) & \(0.9505\) \\
			\bottomrule
		\end{tabular}
	\end{center}
	
	If the value is negative, such as $	P(Z<-1.20)$, the expression means the area to the left of \(z=-1.20\). Since the table only shows positive \(z\)-values, symmetry and the complement rule can be used:
	\[
	P(Z<-1.20)=P(Z>1.20)=1-P(Z<1.20)=1-\Phi(1.20).
	\]
	Since \(\Phi(1.20)=0.8849\),
	\[
	P(Z<-1.20)=1-0.8849=0.1151.
	\]
	
	The same idea can be summarized as follows.
	
	\begin{center}

	\end{center}

	\section{Problem 1: Left Tail with Positive and Negative Bounds}
	\textbf{Problem.} Let \(Z\sim N(0,1)\).
	
	\textbf{Questions/Tasks.}
	\begin{enumerate}[label=\textbf{\alph*)}]
		\item Find \(P(Z<0.85)\) using the table.
		\item Find \(P(Z<-1.20)\) using the table.
	\end{enumerate}
	
	\subsection*{Solution}
	
	\textbf{a)}
	From the table, \(\Phi(0.85)=0.8023\), so
	\[
	P(Z<0.85)=0.8023.
	\]
	
	\StandardNormalLeftTailGraph{0.85}{z=0.85}{P(Z<0.85)=0.8023}
	
	\textbf{b)}
	The goal is to find the left-tail probability
	\[
	P(Z<-1.20).
	\]
	
	\StandardNormalLeftTailGraph{-1.20}{z=-1.20}{P(Z<-1.20)}
	
	\[	\text{By Symmetry: } P(Z<-1.20)=P(Z>1.20).	\]
	
	\StandardNormalSymmetryGraph{1.20}{P(Z<-1.20)}{P(Z>1.20)}
	
	Now use the table value
	\[
	P(Z<1.20)=\Phi(1.20)=0.8849.
	\]
	Using the complement rule
	\[
	P(Z>1.20)=1-\Phi(1.20)=1-0.8849=0.1151.
	\]
	
	\StandardNormalComplementMirrorGraph
	{1.20}
	{P(Z<1.20)=0.8849}
	{P(Z>1.20)=0.1151}
	
	Therefore,
	\[
	P(Z<-1.20)=P(Z>1.20)=0.1151.
	\]
	
	\textbf{Interpretation.} About 80.2\% of values are below 0.85, while 11.5\% are below \(-1.20\).
	
	\section{Problem 2: Right Tail with Positive and Negative Bounds}
	\textbf{Problem.} Let \(Z\sim N(0,1)\).
	
	\textbf{Questions/Tasks.}
	\begin{enumerate}[label=\textbf{\alph*)}]
		\item Find \(P(Z>1.35)\) using the table.
		\item Find \(P(Z>-0.70)\) using the table.
	\end{enumerate}
	
	\subsection*{Solution}
	
	\textbf{a)}
	The goal is to find the right-tail probability
	\[
	P(Z>1.35).
	\]
	
	\StandardNormalRightTailGraph{1.35}{z=1.35}{P(Z>1.35)}
	
	Now use the table value
	\[
	P(Z<1.35)=\Phi(1.35)=0.9115.
	\]
	Using the complement rule,
	\[
	P(Z>1.35)=1-\Phi(1.35)=1-0.9115=0.0885.
	\]
	
	\StandardNormalComplementMirrorGraph
	{1.35}
	{P(Z<1.35)=0.9115}
	{P(Z>1.35)=0.0885}
	
	\textbf{b)}
	The goal is to find the right-tail probability
	\[
	P(Z>-0.70).
	\]
	
	The next two graphs show the wanted area and its mirrored left-tail image.
	
	\begin{center}
		\begin{minipage}{0.48\textwidth}
			\centering
			\StandardNormalRightTailGraphSmall{-0.70}{z=-0.70}{P(Z>-0.70)}
		\end{minipage}
		\hfill
		\begin{minipage}{0.48\textwidth}
			\centering
			\StandardNormalLeftTailOrangeGraphSmall{0.70}{z=0.70}{P(Z<0.70)}
		\end{minipage}
	\end{center}
	
	By symmetry of the standard normal distribution,
	\[
	P(Z>-0.70)=P(Z<0.70)=\Phi(0.70)=0.7580.
	\]
	
	\textbf{Interpretation.} The right tail above 1.35 is small (8.9\%), but above -0.70 it is much larger (75.8\%).
	
	\section{Problem 3: Mixed Left-Tail and Right-Tail Reasoning}
\textbf{Problem.} Let \(Z\sim N(0,1)\).

\textbf{Questions/Tasks.} Calculate \(P(Z<1.05)\) and \(P(Z>-0.55)\), and identify which is left-tail reasoning and which is right-tail reasoning.

\subsection*{C5}
For \(P(Z<1.05)\), this is a left-tail probability:
\StandardNormalLeftTailGraph{1.05}{z=1.05}{P(Z<1.05)=0.8531}
\[
P(Z<1.05)=\Phi(1.05)=0.8531.
\]
For \(P(Z>-0.55)\), this is a right-tail probability with a negative bound:

\begin{center}
\begin{minipage}{0.48\textwidth}
\centering
\StandardNormalRightTailGraphSmall{-0.55}{z=-0.55}{P(Z>-0.55)}
\end{minipage}
\hfill
\begin{minipage}{0.48\textwidth}
\centering
\StandardNormalLeftTailOrangeGraphSmall{0.55}{z=0.55}{P(Z<0.55)}
\end{minipage}
\end{center}

\[
P(Z>-0.55)=P(Z<0.55)=\Phi(0.55)=0.7088.
\]
\textbf{Interpretation.} Left-tail and right-tail setups use different formulas even when both answers are fairly large.

\section{Problem 4: Central Band with Positive Bounds}
\textbf{Problem.} Let \(Z\sim N(0,1)\).

\textbf{Questions/Tasks.} Compute \(P(0.35<Z<1.40)\).

\subsection*{C5}
The target central-band probability is
\[
P(0.35<Z<1.40).
\]
\StandardNormalCentralBandGraph{0.35}{1.40}{z=0.35}{z=1.40}{P(0.35<Z<1.40)}
Using \(P(a<Z<b)=\Phi(b)-\Phi(a)\), first read the cumulative areas:
\[
P(Z<1.40)=\Phi(1.40)=0.9192
\]
\[
P(Z<0.35)=\Phi(0.35)=0.6368.
\]
The central band is the difference of these cumulative areas:
\[
P(0.35<Z<1.40)=\Phi(1.40)-\Phi(0.35)=0.9192-0.6368=0.2824.
\]
\textbf{Interpretation.} About 28.2\% of values are between 0.35 and 1.40.

\section{Problem 5: Central Band with a Negative and a Positive Bound}
\textbf{Problem.} Let \(Z\sim N(0,1)\).

\textbf{Questions/Tasks.} Compute \(P(-0.95<Z<1.15)\).

\subsection*{C5}
The target central-band probability is
\[
P(-0.95<Z<1.15).
\]
\StandardNormalCentralBandGraph{-0.95}{1.15}{z=-0.95}{z=1.15}{P(-0.95<Z<1.15)}
\[
\Phi(-0.95)=1-\Phi(0.95)=1-0.8289=0.1711.
\]
\[
\Phi(1.15)=0.8749.
\]
\[
P(-0.95<Z<1.15)=0.8749-0.1711=0.7038.
\]
\textbf{Interpretation.} About 70.4\% of values lie between -0.95 and 1.15.

\section{Problem 7: Two Tails with Positive Bounds}
\textbf{Problem.} Let \(Z\sim N(0,1)\).

\textbf{Questions/Tasks.} Find \(P(Z<0.55\ \text{or}\ Z>1.65)\).

\subsection*{C5}
The target union probability is
\[
P(Z<0.55\ \text{or}\ Z>1.65).
\]
\StandardNormalTwoTailGraph{0.55}{1.65}{z=0.55}{z=1.65}{P(Z<0.55)}{P(Z>1.65)}
These two events are disjoint, so we add probabilities.
Note that \(Z<0.55\) is a left region with a \emph{positive} bound, not a negative tail.
\[
P(Z<0.55)=\Phi(0.55)=0.7088.
\]
\StandardNormalComplementMirrorGraph
{1.65}
{P(Z<1.65)=0.9505}
{P(Z>1.65)=0.0495}
\[
P(Z>1.65)=1-\Phi(1.65)=1-0.9505=0.0495.
\]
\[
P(Z<0.55\ \text{or}\ Z>1.65)=0.7088+0.0495=0.7583.
\]
\textbf{Interpretation.} Most values satisfy this union because all values below 0.55 are included plus a small far-right tail.

\section{Problem 8: Two Tails with One Negative and One Positive Bound}
\textbf{Problem.} Let \(Z\sim N(0,1)\).

\textbf{Questions/Tasks.} Find \(P(Z<-1.05\ \text{or}\ Z>1.20)\).

\subsection*{C5}
The target union probability is
\[
P(Z<-1.05\ \text{or}\ Z>1.20).
\]
\StandardNormalTwoTailGraph{-1.05}{1.20}{z=-1.05}{z=1.20}{P(Z<-1.05)}{P(Z>1.20)}
The events are disjoint, so add them.
\[
P(Z<-1.05)=P(Z>1.05)
\]
\[
P(Z<-1.05)=1-\Phi(1.05)=1-0.8531=0.1469.
\]
\[
P(Z>1.20)=1-\Phi(1.20)=1-0.8849=0.1151.
\]
\[
P(Z<-1.05\ \text{or}\ Z>1.20)=0.1469+0.1151=0.2620.
\]
\textbf{Interpretation.} About 26.2\% of values are in these two outer regions.

\section{Problem 9: Two Tails with Two Negative Bounds}
\textbf{Problem.} Let \(Z\sim N(0,1)\).

\textbf{Questions/Tasks.} Find \(P(Z<-1.90\ \text{or}\ Z>-0.35)\).

\subsection*{C5}
The target union probability is
\[
P(Z<-1.90\ \text{or}\ Z>-0.35).
\]
\StandardNormalTwoTailGraph{-1.90}{-0.35}{z=-1.90}{z=-0.35}{P(Z<-1.90)}{P(Z>-0.35)}
The two events are disjoint because one is far left of \(-1.90\) and the other is to the right of \(-0.35\).
\[
P(Z<-1.90)=1-\Phi(1.90)=1-0.9713=0.0287.
\]
For the right-tail bound at a negative value:

\begin{center}
\begin{minipage}{0.48\textwidth}
\centering
\StandardNormalRightTailGraphSmall{-0.35}{z=-0.35}{P(Z>-0.35)}
\end{minipage}
\hfill
\begin{minipage}{0.48\textwidth}
\centering
\StandardNormalLeftTailOrangeGraphSmall{0.35}{z=0.35}{P(Z<0.35)}
\end{minipage}
\end{center}

\[
P(Z>-0.35)=P(Z<0.35)=\Phi(0.35)=0.6368.
\]
So
\[
P(Z<-1.90\ \text{or}\ Z>-0.35)=0.0287+0.6368=0.6655.
\]
\textbf{Interpretation.} This probability is large because all values above -0.35 are included, plus a small far-left tail.
