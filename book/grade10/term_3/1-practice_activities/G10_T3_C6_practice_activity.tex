\section{Problem 1: Normal Distribution Left Tail with Positive and Negative z-values}
\textbf{Problem.} Exam scores follow \(X\sim N(70,8^2)\). Find \(P(X<82)\) and \(P(X<60.4)\).\\
\StdLeft{1.50}{P(Z<1.50)}
\StdSym{1.20}{P(Z<-1.20)}{P(Z>1.20)}
\[Z=\frac{X-70}{8}.\]
\[P(X<82)=P\!\left(Z<\frac{82-70}{8}\right)=P(Z<1.50)=0.9332.\]
\[P(X<60.4)=P\!\left(Z<\frac{60.4-70}{8}\right)=P(Z<-1.20)=1-0.8849=0.1151.\]
\textbf{Interpretation.} About 93.32\% score below 82, while 11.51\% score below 60.4.

\section{Problem 2: Normal Distribution Right Tail with Positive and Negative z-values}
\textbf{Problem.} Delivery times follow \(X\sim N(56,5^2)\) minutes. Find \(P(X>63.5)\) and \(P(X>52)\).\\
\StdRight{1.50}{P(Z>1.50)}
\StdComp{1.50}{P(Z<1.50)=0.9332}{P(Z>1.50)=0.0668}
\[Z=\frac{X-56}{5}.\]
\[P(X>63.5)=P\!\left(Z>\frac{63.5-56}{5}\right)=P(Z>1.50)=1-0.9332=0.0668.\]
\[P(X>52)=P\!\left(Z>\frac{52-56}{5}\right)=P(Z>-0.80)=\Phi(0.80)=0.7881.\]
\textbf{Interpretation.} 6.68\% take more than 63.5 min, and 78.81\% take more than 52 min.

\section{Problem 3: Normal Distribution Mixed Left-Tail and Right-Tail Reasoning}
\textbf{Problem.} Production weights follow \(X\sim N(500,20^2)\) g. Find \(P(X<514)\) and \(P(X>483)\).\\
\StdLeft{0.70}{P(Z<0.70)}
\StdRight{-0.85}{P(Z>-0.85)}
\[Z=\frac{X-500}{20}.\]
\[P(X<514)=P\!\left(Z<\frac{514-500}{20}\right)=P(Z<0.70)=0.7580.\]
\[P(X>483)=P\!\left(Z>\frac{483-500}{20}\right)=P(Z>-0.85)=\Phi(0.85)=0.8023.\]
\textbf{Interpretation.} 75.80\% are below 514 g, and 80.23\% are above 483 g.

\section{Problem 4: Normal Distribution Central Band with Positive z-bounds}
\textbf{Problem.} Weekly app usage follows \(X\sim N(14,4^2)\) hours. Find \(P(15<X<18.6)\).\\
\StdBand{0.25}{1.15}{P(0.25<Z<1.15)}
\[Z=\frac{X-14}{4},\quad P(15<X<18.6)=P(0.25<Z<1.15).\]
\[P=\Phi(1.15)-\Phi(0.25)=0.8749-0.5987=0.2762.\]
\textbf{Interpretation.} 27.62\% of students use the app between 15 h and 18.6 h weekly.

\section{Problem 5: Normal Distribution Central Band with One Negative and One Positive z-bound}
\textbf{Problem.} Commute times follow \(X\sim N(40,10^2)\) minutes. Find \(P(30.5<X<50.5)\).\\
\StdBand{-0.95}{1.05}{P(-0.95<Z<1.05)}
\[Z=\frac{X-40}{10},\quad P(30.5<X<50.5)=P(-0.95<Z<1.05).\]
\[P=\Phi(1.05)-\Phi(-0.95)=0.8531-(1-0.8289)=0.6820.\]
\textbf{Interpretation.} 68.20\% of commutes are between 30.5 and 50.5 minutes.

\section{Problem 6: Normal Distribution Central Band with Both Negative z-bounds}
\textbf{Problem.} Monthly sales follow \(X\sim N(120,20^2)\) units. Find \(P(84<X<103)\).\\
\StdBand{-1.80}{-0.85}{P(-1.80<Z<-0.85)}
\[Z=\frac{X-120}{20},\quad P(84<X<103)=P(-1.80<Z<-0.85).\]
\[P=\Phi(-0.85)-\Phi(-1.80)=(1-0.8023)-(1-0.9641)=0.1618.\]
\textbf{Interpretation.} 16.18\% of months have sales between 84 and 103 units.

\section{Problem 7: Normal Distribution Two Regions with Positive z-bounds}
\textbf{Problem.} Household electricity use follows \(X\sim N(400,50^2)\) kWh. Find \(P(X<450\ \text{or}\ X>495)\).\\
\StdTwo{1.00}{1.90}{P(Z<1.00\ \text{or}\ Z>1.90)}
\[Z=\frac{X-400}{50},\quad P(X<450\ \text{or}\ X>495)=P(Z<1.00\ \text{or}\ Z>1.90).\]
The regions are disjoint, so
\[P=\Phi(1.00)+\bigl(1-\Phi(1.90)\bigr)=0.8413+0.0287=0.8700.\]
\textbf{Interpretation.} 87.00\% of homes are below 450 kWh or above 495 kWh.

\section{Problem 8: Normal Distribution Two Tails with One Negative and One Positive z-bound}
\textbf{Problem.} Scholarship scores follow \(X\sim N(300,20^2)\). Find \(P(X<279\ \text{or}\ X>327)\).\\
\StdTwo{-1.05}{1.35}{P(Z<-1.05\ \text{or}\ Z>1.35)}
\[Z=\frac{X-300}{20},\quad P(X<279\ \text{or}\ X>327)=P(Z<-1.05\ \text{or}\ Z>1.35).\]
The regions are disjoint, so
\[P=\Phi(-1.05)+\bigl(1-\Phi(1.35)\bigr)=(1-0.8531)+(1-0.9115)=0.2354.\]
\textbf{Interpretation.} 23.54\% are either below 279 or above 327 points.

\section{Problem 9: Normal Distribution Two Regions with Both Negative z-bounds}
\textbf{Problem.} Customer waiting times follow \(X\sim N(90,20^2)\) seconds. Find \(P(X<66\ \text{or}\ X>73)\).\\
\StdTwo{-1.20}{-0.85}{P(Z<-1.20\ \text{or}\ Z>-0.85)}
\[Z=\frac{X-90}{20},\quad P(X<66\ \text{or}\ X>73)=P(Z<-1.20\ \text{or}\ Z>-0.85).\]
The regions are disjoint, so
\[P=\Phi(-1.20)+\bigl(1-\Phi(-0.85)\bigr)=0.1151+0.8023=0.9174.\]
\textbf{Interpretation.} 91.74\% wait below 66 s or above 73 s.
