
\pgfplotsset{
	SharedLegendPosition/.style={
		legend style={at={(1,0.83)},anchor=south east}
	}
}

% -------------------------------------------------
% Helper plot styles for uniform distributions
% -------------------------------------------------

\newcommand{\UniformPlotA}[2]{%
	\addplot[
	very thick,
	black,
	solid,
	fill=black,
	fill opacity=0.10,
	domain=#1:#2,
	samples=2,
	forget plot
	] {1/((#2)-(#1))} \closedcycle;
}

\newcommand{\UniformPlotB}[2]{%
	\addplot[
	very thick,
	black,
	solid,
	fill=black,
	fill opacity=0.18,
	pattern=horizontal lines,
	pattern color=black,
	domain=#1:#2,
	samples=2,
	forget plot
	] {1/((#2)-(#1))} \closedcycle;
}

\newcommand{\UniformPlotC}[2]{%
	\addplot[
	very thick,
	black,
	densely dotted,
	fill=black,
	fill opacity=0.26,
	domain=#1:#2,
	samples=2,
	forget plot
	] {1/((#2)-(#1))} \closedcycle;
}

\newcommand{\UniformPlotD}[2]{%
	\addplot[
	very thick,
	black,
	solid,
	fill=black,
	fill opacity=0.34,
	pattern=north east lines,
	pattern color=black,
	domain=#1:#2,
	samples=2,
	forget plot
	] {1/((#2)-(#1))} \closedcycle;
}

% -------------------------------------------------
% Main command for 3 uniform graphs
%
% #1 = width
% #2 = height
% #3 = xmin
% #4 = xmax
% #5 = ymax
% #6 = Graph A interval as {left,right}
% #7 = Graph B interval as {left,right}
% #8 = Graph C interval as {left,right}
% -------------------------------------------------

\NewDocumentCommand{\UniformThreeGraph}
{m m m m m >{\SplitArgument{1}{,}}m >{\SplitArgument{1}{,}}m >{\SplitArgument{1}{,}}m}{%
	\begin{center}
		\begin{tikzpicture}
			\begin{axis}[
				width=#1,height=#2,
				axis lines=left,
				xlabel={$x$}, ylabel={$f(x)$},
				xmin=#3,xmax=#4,ymin=0,ymax=#5,
				grid=both,
				SharedLegendPosition
				]
				
				\UniformPlotA#6
				\UniformPlotB#7
				\UniformPlotC#8
				
				\UniformLegendA{Graph A}
				\UniformLegendB{Graph B}
				\UniformLegendC{Graph C}
			\end{axis}
		\end{tikzpicture}
	\end{center}
}

{m m m m m >{\SplitArgument{1}{,}}m >{\SplitArgument{1}{,}}m}{%

				\NormalPlotB{#3}{#4}#7
				
				\NormalLegendA{Graph A}
				\NormalLegendB{Graph B}
			\end{axis}
		\end{tikzpicture}
	\end{center}
}
