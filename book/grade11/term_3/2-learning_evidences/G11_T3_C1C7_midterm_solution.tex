	Summary data:
	
	\[
	n_{25}=30

	\bar{x}_{25}=7.63

	Use \(\alpha=0.05\)
	
	\subsection*{C1}
	
	\textbf{Mean setting.}
	
	\[
	\mu_{25}=\text{true mean number of late arrivals per morning for Prom 2025.}
	\]
	
	\[
	\mu_{26}=\text{true mean number of late arrivals per morning for Prom 2026.}
	\]
	
	\[
	n_{25}=30,
	\qquad
	\bar{x}_{25}=7.63,
	\qquad
	\sigma_{25}=3.40.
	\]
	
	\[
	n_{26}=30,
	\qquad
	\bar{x}_{26}=8.60,
	\qquad
	\sigma_{26}=3.24.
	\]
	
	\textbf{Proportion setting.}
	
	\[
	p_{25}=\text{true proportion of high-lateness mornings for Prom 2025.}
	\]
	
	\[
	p_{26}=\text{true proportion of high-lateness mornings for Prom 2026.}
	\]
	
	\[
	x_{25}=7,
	\qquad
	n_{25}=30,
	\qquad
	\hat{p}_{25}=\frac{7}{30}\approx 0.2333.
	\]
	
	\[
	x_{26}=12,
	\qquad
	n_{26}=30,
	\qquad
	\hat{p}_{26}=\frac{12}{30}=0.40.
	\]
	
	\subsection*{Using Prom 2025 as benchmark}
	
	\subsubsection*{C2}
	
	\textbf{Proportion test.}
	
	\[
	H_0\colon p=p_0
	\qquad
	H_a\colon p>p_0
	\]
	
	\[
	p=p_{26},
	\qquad
	p_0=\hat{p}_{25}=\frac{7}{30}\approx 0.2333.
	\]
	
	\[
	H_0\colon p_{26}=0.2333
	\qquad
	H_a\colon p_{26}>0.2333.
	\]
	
	\textbf{Mean test.}
	
	\[
	H_0\colon \mu=\mu_0
	\qquad
	H_a\colon \mu>\mu_0
	\]
	
	\[
	\mu=\mu_{26},
	\qquad
	\mu_0=\bar{x}_{25}=7.63.
	\]
	
	\[
	H_0\colon \mu_{26}=7.63
	\qquad
	H_a\colon \mu_{26}>7.63.
	\]
	
	\subsubsection*{C3}
	
	Both tests are right-tailed.
	
	\[
	H_a\colon p_{26}>0.2333,
	\qquad
	H_a\colon \mu_{26}>7.63.
	\]
	
	\ZNormalRight{1.645}{Right tail, \(\alpha=0.05\)}
	
	\subsubsection*{C4}
	
	\[
	\hat{p}_{25}=0.2333,
	\qquad
	\hat{p}_{26}=0.40,
	\qquad
	n_{26}=30,
	\qquad
	k=2.20.
	\]
	
	\[
	z=
	\frac{\hat{p}_{26}-\hat{p}_{25}}{\sqrt{k\,\frac{\hat{p}_{25}(1-\hat{p}_{25})}{n_{26}}}}
	=
	\frac{0.40-0.2333}{\sqrt{2.20\,\frac{0.2333(0.7667)}{30}}}.
	\]
	
	\[
	z=
	\frac{0.1667}{0.1145}
	\approx 1.46.
	\]
	
	\[
	z_{\text{critical}}=1.645.
	\]
	
	\[
	1.46<1.645.
	\]
	
	\FailDecision{There is not enough evidence to conclude that \(p_{26}>0.2333\).}
	
	\subsubsection*{C5}
	
	\[
	\bar{x}_{25}=7.63,
	\qquad
	\bar{x}_{26}=8.60,
	\qquad
	\sigma_{25}=3.40,
	\qquad
	\sigma_{26}=3.24,
	\qquad
	n_{26}=30.
	\]
	
	The test sample is Prom 2026, so the statistic uses \(\sigma_{26}\).
	
	\[
	z=
	\frac{\bar{x}_{26}-\bar{x}_{25}}{\frac{\sigma_{26}}{\sqrt{n_{26}}}}
	=
	\frac{8.60-7.63}{\frac{3.24}{\sqrt{30}}}.
	\]
	
	\[
	z=
	\frac{0.97}{0.5916}
	\approx
	1.64.
	\]
	
	\[
	z_{\text{critical}}=1.645.
	\]
	
	\[
	1.64<1.645.
	\]
	
	\FailDecision{There is not enough evidence to conclude that \(\mu_{26}>7.63\).}
	
	\subsubsection*{C7}

	\textbf{Proportion test.}

	\[
	\frac{0.1667}{\sqrt{2.20\,\frac{0.2333(0.7667)}{n_{26}}}}>1.645.
	\]

	\[
	n_{26}>
	\left(
	\frac{1.645\sqrt{2.20\cdot 0.2333(0.7667)}}{0.1667}
	\right)^2
	\approx
	38.34.
	\]

	\[
	n_{26,\min}=39.
	\]

	\textbf{Mean test.}

	\[
	\frac{0.97}{\frac{3.24}{\sqrt{n_{26}}}}>1.645
	\Rightarrow
	n_{26}>
	\left(\frac{1.645(3.24)}{0.97}\right)^2
	\approx 30.18.
	\]

	\[
	n_{26,\min}=31.
	\]

	\subsubsection*{C6}

	With the original sample size \(n_{26}=30\), both tests failed to reject \(H_0\).

	Using the minimum sample sizes from C7:

	\[
	z_p=\frac{0.1667}{\sqrt{2.20\,\frac{0.2333(0.7667)}{39}}}\approx 1.66>1.645
	\]

	\[
	z_\mu=\frac{0.97}{\frac{3.24}{\sqrt{31}}}\approx 1.67>1.645
	\]

	Now both tests reject \(H_0\).

	For the original decision (fail to reject), the possible error is Type II error.

	For the recalculated decision (reject), the possible error is Type I error.

	\subsubsection*{C4}

	\bar{x}_{25}=7.63,

	\frac{\bar{x}_{25}-\bar{x}_{26}}{\frac{\sigma_{25}}{\sqrt{n_{25}}}}
	=
	\frac{7.63-8.60}{\frac{3.40}{\sqrt{30}}}.
	\]
	
	\[
	z=
	\frac{-0.97}{0.6207}

	\subsubsection*{C7}

	\textbf{Proportion test.}

	\[
	\frac{0.1667}{\sqrt{1.50\,\frac{0.40(0.60)}{n_{25}}}}>1.645
	\Rightarrow
	n_{25}>
	\left(
	\frac{1.645\sqrt{1.50\cdot 0.40(0.60)}}{0.1667}
	\right)^2
	\approx 44.54.
	\]

	\[
	n_{25,\min}=45.
	\]

	\textbf{Mean test.}

	\[
	\frac{0.97}{\frac{3.40}{\sqrt{n_{25}}}}>1.645
	\Rightarrow
	n_{25}>
	\left(\frac{1.645(3.40)}{0.97}\right)^2
	\approx 33.23.
	\]

	\[
	n_{25,\min}=34.
	\]

	\subsubsection*{C6}

	With the original sample size \(n_{25}=30\), both tests failed to reject \(H_0\).

	Using the minimum sample sizes from C7:

	\[
	z_p=\frac{-0.1667}{\sqrt{1.50\,\frac{0.40(0.60)}{45}}}\approx -1.66<-1.645
	\]

	\[
	z_\mu=\frac{-0.97}{\frac{3.40}{\sqrt{34}}}\approx -1.66<-1.645
	\]

	Now both tests reject \(H_0\).

	For the original decision (fail to reject), the possible error is Type II error.

	For the recalculated decision (reject), the possible error is Type I error.
