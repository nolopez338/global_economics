\section*{Evaluated criteria}
\begin{itemize}
    \item C6. Interprets decisions from hypothesis tests in terms of context, including p-value comparisons and error types.
    \item C7. Uses one-sample mean \(z\)-tests with known \(\sigma\) to find minimum sample sizes required to reject \(H_0\).
\end{itemize}

\section{(6) Fuel-Inflation Claim Decision (C6)}
\label{sec:fuel-inflation-claim-decision-c6}
A city transport authority tests whether mean monthly fuel inflation is above the 4.0\% planning benchmark:
\[
H_0:\mu=4.0,\qquad H_1:\mu>4.0,\qquad \alpha=0.05.
\]
The report provides \(z_{obs}=2.05\) and \(p\text{-value}=0.020\).

\textbf{Questions/Tasks.}
\begin{enumerate}[label=\textbf{\alph*)}]
    \item \TaskHyperref{subsec:fuel-inflation-claim-decision-c6-a}{\textbf{(2)} State the decision rule and determine whether to reject or fail to reject \(H_0\).}
    \item \TaskHyperref{subsec:fuel-inflation-claim-decision-c6-b}{\textbf{(2)} Interpret the statistical conclusion in context.}
    \item \TaskHyperref{subsec:fuel-inflation-claim-decision-c6-c}{\textbf{(2)} Identify the relevant error type and explain its practical implication.}
\end{enumerate}

\subsection*{a) \textbf{(2)} State the decision rule and determine whether to reject or fail to reject \(H_0\).}
\phantomsection
\label{subsec:fuel-inflation-claim-decision-c6-a}
Decision rule:
\[
\text{Reject }H_0\text{ if }p\text{-value}<\alpha,
\qquad
\text{Fail to reject }H_0\text{ if }p\text{-value}\ge \alpha.
\]
Here, \(p\text{-value}=0.020<0.05=\alpha\), so we \textbf{reject \(H_0\)}.

\secInicio[sec:fuel-inflation-claim-decision-c6]

\subsection*{b) \textbf{(2)} Interpret the statistical conclusion in context.}
\phantomsection
\label{subsec:fuel-inflation-claim-decision-c6-b}
Because we reject \(H_0\), the data provide sufficient evidence (at the 5\% significance level) that mean monthly fuel inflation is \textbf{greater than 4.0\%}.

\secInicio[sec:fuel-inflation-claim-decision-c6]

\subsection*{c) \textbf{(2)} Identify the relevant error type and explain its practical implication.}
\phantomsection
\label{subsec:fuel-inflation-claim-decision-c6-c}
Since the decision is to reject \(H_0\), the relevant risk is a \textbf{Type I error}. In context, this would mean concluding that average fuel inflation exceeds 4.0\% when, in reality, the true mean is still 4.0\%.

\textbf{Questions/Tasks.}
\begin{enumerate}[label=\textbf{\alph*)}]
    \item \TaskHyperref{subsec:unemployment-duration-claim-decision-c6-a}{\textbf{(2)} State the decision rule and determine whether to reject or fail to reject \(H_0\).}
    \item \TaskHyperref{subsec:unemployment-duration-claim-decision-c6-b}{\textbf{(2)} Interpret the statistical conclusion in context.}
    \item \TaskHyperref{subsec:unemployment-duration-claim-decision-c6-c}{\textbf{(2)} Identify the relevant error type and explain its practical implication.}

\subsection*{a) \textbf{(2)} State the decision rule and determine whether to reject or fail to reject \(H_0\).}
\phantomsection
\label{subsec:unemployment-duration-claim-decision-c6-a}
Decision rule:
\[
\text{Reject }H_0\text{ if }p\text{-value}<\alpha,
\qquad
\text{Fail to reject }H_0\text{ if }p\text{-value}\ge \alpha.
\]
Here, \(p\text{-value}=0.198\ge0.10=\alpha\), so we \textbf{fail to reject \(H_0\)}.

\phantomsection
\label{subsec:unemployment-duration-claim-decision-c6-b}

\section{(7) Minimum Sample Size for Export Price Evidence (C7)}
\label{sec:export-price-minimum-sample-size-c7}
For export prices, test:
\[
H_0:\mu=120,\qquad H_1:\mu>120,
\]
with known \(\sigma=20\), observed \(\bar x=124\), and \(\alpha=0.05\).

\textbf{Questions/Tasks.}
\begin{enumerate}[label=\textbf{\alph*)}]
    \item \TaskHyperref{subsec:export-price-minimum-sample-size-c7-a}{\textbf{(3)} Derive \(z(n)\), the rejection inequality, and find \(n_{\min}\).}
    \item \TaskHyperref{subsec:export-price-minimum-sample-size-c7-b}{\textbf{(2)} Interpret \(n_{\min}\) in context.}
\end{enumerate}

\subsection*{a) \textbf{(3)} Derive \(z(n)\), the rejection inequality, and find \(n_{\min}\).}
\phantomsection
\label{subsec:export-price-minimum-sample-size-c7-a}
Let \(\mu\) be the population mean export price.
\[
z(n)=\frac{\bar x-\mu_0}{\sigma/\sqrt n}=\frac{124-120}{20/\sqrt n}=0.2\sqrt n.
\]
For a right-tailed test at \(\alpha=0.05\), the critical value is \(z_\alpha=1.645\), and rejection requires:
\[
0.2\sqrt n\ge1.645
\Rightarrow
\sqrt n\ge\frac{1.645}{0.2}=8.225
\Rightarrow
n\ge67.65.
\]
Thus,
\[
n_{\min}=\left\lceil\left(\frac{1.645}{0.2}\right)^2\right\rceil=68.
\]

\secInicio[sec:export-price-minimum-sample-size-c7]

\subsection*{b) \textbf{(2)} Interpret \(n_{\min}\) in context.}
\phantomsection
\label{subsec:export-price-minimum-sample-size-c7-b}
At least \(68\) export-price observations are needed so that this sample mean difference is large enough (relative to \(\sigma\)) to reject \(H_0\) at the 5\% level.

\section{(7) Minimum Sample Size for Delivery-Time Evidence (C7)}
\label{sec:delivery-time-minimum-sample-size-c7}
For delivery-time records, test:
\[
H_0:\mu=18,\qquad H_1:\mu<18,
\]
with known \(\sigma=4\), observed \(\bar x=17\), and \(\alpha=0.10\).

\textbf{Questions/Tasks.}
\begin{enumerate}[label=\textbf{\alph*)}]
    \item \TaskHyperref{subsec:delivery-time-minimum-sample-size-c7-a}{\textbf{(3)} Derive \(z(n)\), the rejection inequality, and find \(n_{\min}\).}
    \item \TaskHyperref{subsec:delivery-time-minimum-sample-size-c7-b}{\textbf{(2)} Interpret \(n_{\min}\) in context.}
\end{enumerate}

\subsection*{a) \textbf{(3)} Derive \(z(n)\), the rejection inequality, and find \(n_{\min}\).}
\phantomsection
\label{subsec:delivery-time-minimum-sample-size-c7-a}
Let \(\mu\) be the population mean delivery time (days).
\[
z(n)=\frac{\bar x-\mu_0}{\sigma/\sqrt n}=\frac{17-18}{4/\sqrt n}=-0.25\sqrt n.
\]
For a left-tailed test at \(\alpha=0.10\), rejection requires \(z(n)\le -z_\alpha=-1.282\):
\[
-0.25\sqrt n\le-1.282
\Rightarrow
\sqrt n\ge\frac{1.282}{0.25}=5.128
\Rightarrow
n\ge26.30.
\]
Thus,
\[
n_{\min}=\left\lceil\left(\frac{1.282}{0.25}\right)^2\right\rceil=27.
\]

\secInicio[sec:delivery-time-minimum-sample-size-c7]

\subsection*{b) \textbf{(2)} Interpret \(n_{\min}\) in context.}
\phantomsection
\label{subsec:delivery-time-minimum-sample-size-c7-b}
At least \(27\) observations are required to obtain enough evidence to reject \(H_0\) and support that mean delivery time is below 18 days at the 10\% level.

\section{(7) Minimum Sample Size for Spending-Deviation Evidence (C7)}
\label{sec:spending-two-tailed-minimum-sample-size-c7}
For weekly household imported-food spending, test:
\[
H_0:\mu=75,\qquad H_1:\mu\ne75,
\]
with known \(\sigma=12\), observed \(\bar x=78\), and \(\alpha=0.05\).

\textbf{Questions/Tasks.}
\begin{enumerate}[label=\textbf{\alph*)}]
    \item \TaskHyperref{subsec:spending-two-tailed-minimum-sample-size-c7-a}{\textbf{(3)} Derive \(|z(n)|\), the two-tailed rejection inequality, and find \(n_{\min}\).}
    \item \TaskHyperref{subsec:spending-two-tailed-minimum-sample-size-c7-b}{\textbf{(2)} Interpret \(n_{\min}\) in context.}
\end{enumerate}

\subsection*{a) \textbf{(3)} Derive \(|z(n)|\), the two-tailed rejection inequality, and find \(n_{\min}\).}
\phantomsection
\label{subsec:spending-two-tailed-minimum-sample-size-c7-a}
Let \(\mu\) be the population mean weekly imported-food spending.
\[
|z(n)|=\left|\frac{\bar x-\mu_0}{\sigma/\sqrt n}\right|=\left|\frac{78-75}{12/\sqrt n}\right|=0.25\sqrt n.
\]
For a two-tailed test with \(\alpha=0.05\), the critical value is \(z_{\alpha/2}=z_{0.025}=1.96\), and rejection requires:
\[
0.25\sqrt n\ge1.96
\Rightarrow
\sqrt n\ge\frac{1.96}{0.25}=7.84
\Rightarrow
n\ge61.47.
\]
Thus,
\[
n_{\min}=\left\lceil\left(\frac{1.96}{0.25}\right)^2\right\rceil=62.
\]

\secInicio[sec:spending-two-tailed-minimum-sample-size-c7]

\subsection*{b) \textbf{(2)} Interpret \(n_{\min}\) in context.}
\phantomsection
\label{subsec:spending-two-tailed-minimum-sample-size-c7-b}
A minimum of \(62\) surveyed households is needed to reject \(H_0\) at the 5\% level and conclude that mean spending differs from 75 (in either direction).
