\item
\subsection*{Problem 1 — Daily app check-ins}

\textbf{Problem.} A school learning app reports that 0.62 of active users open the app every day. A class project tracks 200 active users and finds that 112 opened the app every day.
\begin{itemize}
    \item Population parameter: \(p\), the true proportion of active users who open the app every day.
    \item Significance level: \(\alpha=0.05\).
\end{itemize}

\textbf{Question.} Apply the general hypothesis-testing procedure to decide whether the daily check-in rate is lower than 0.62.

\textbf{Solution.}

\textbf{Step 1: Hypotheses.}
\[
\begin{aligned}
H_0&: p = 0.62 \\
H_a&: p < 0.62
\end{aligned}
\]

\textbf{Step 2: Type of test.} This is a left-tailed test because the claim is that the population proportion is below the benchmark.

\textbf{Step 3: Test statistic.}
\[
\hat{p}=\frac{x}{n}=\frac{112}{200}=0.56
\]
\[
SE=\sqrt{\frac{p_0(1-p_0)}{n}}=\sqrt{\frac{0.62(0.38)}{200}}\approx 0.0343
\]
\[
z=\frac{\hat{p}-p_0}{SE}=\frac{0.56-0.62}{0.0343}\approx -1.75
\]

\textbf{Step 4: Reference distribution.} Under \(H_0\), the test statistic follows \(Z\sim N(0,1)\).

\textbf{Step 5: P-value.}
\[
P\text{-value}=P(Z\le -1.75)\approx 0.0401
\]

\textbf{Step 6: Decision.} Since \(0.0401<0.05\), reject \(H_0\).

\textbf{Conclusion.} The data provide enough evidence that the true daily check-in rate is lower than 0.62.

\subsection*{Problem 2 — On-time bus arrivals}

\textbf{Problem.} A transportation office states that 0.78 of school buses arrive on time. A student team observes 250 bus arrivals and records 209 on-time arrivals.
\begin{itemize}
    \item Population parameter: \(p\), the true proportion of bus arrivals that are on time.
    \item Significance level: \(\alpha=0.01\).
\end{itemize}

\textbf{Question.} Apply the general hypothesis-testing procedure to decide whether the on-time rate is higher than 0.78.

\textbf{Solution.}

\textbf{Step 1: Hypotheses.}
\[
\begin{aligned}
H_0&: p = 0.78 \\
H_a&: p > 0.78
\end{aligned}
\]

\textbf{Step 2: Type of test.} This is a right-tailed test because the claim is that the population proportion is above the benchmark.

\textbf{Step 3: Test statistic.}
\[
\hat{p}=\frac{x}{n}=\frac{209}{250}=0.836
\]
\[
SE=\sqrt{\frac{0.78(0.22)}{250}}\approx 0.0262
\]
\[
z=\frac{0.836-0.78}{0.0262}\approx 2.14
\]

\textbf{Step 4: Reference distribution.} Under \(H_0\), the test statistic follows \(Z\sim N(0,1)\).

\textbf{Step 5: P-value.}
\[
P\text{-value}=P(Z\ge 2.14)\approx 0.0162
\]

\textbf{Step 6: Decision.} Since \(0.0162>0.01\), fail to reject \(H_0\).

\textbf{Conclusion.} The sample does not provide enough evidence at the 0.01 level that the true on-time arrival rate is higher than 0.78.

\item
\subsection*{Problem 3 — Defect rate in chargers}

\textbf{Problem.} A factory process is designed so that 0.06 of phone chargers are defective. In a quality check of 400 chargers, 34 are found defective.
\begin{itemize}
    \item Population parameter: \(p\), the true proportion of chargers that are defective.
    \item Significance level: \(\alpha=0.05\).
\end{itemize}

\textbf{Question.} Apply the general hypothesis-testing procedure to decide whether the defect proportion is different from 0.06.

\textbf{Solution.}

\textbf{Step 1: Hypotheses.}
\[
\begin{aligned}
H_0&: p = 0.06 \\
H_a&: p \ne 0.06
\end{aligned}
\]

\textbf{Step 2: Type of test.} This is a two-tailed test because the claim allows deviation in either direction from the benchmark.

\textbf{Step 3: Test statistic.}
\[
\hat{p}=\frac{x}{n}=\frac{34}{400}=0.085
\]
\[
SE=\sqrt{\frac{0.06(0.94)}{400}}\approx 0.0119
\]
\[
z=\frac{0.085-0.06}{0.0119}\approx 2.10
\]

\textbf{Step 4: Reference distribution.} Under \(H_0\), the test statistic follows \(Z\sim N(0,1)\).

\textbf{Step 5: P-value.}
\[
P\text{-value}=2\cdot P(Z\ge 2.10)\approx 2(0.0179)=0.0358
\]

\textbf{Step 6: Decision.} Since \(0.0358<0.05\), reject \(H_0\).

\textbf{Conclusion.} The data suggest that the true defect proportion is not equal to 0.06.

H_a&: p < 0.88
\end{aligned}

\textbf{Step 3: Test statistic.}
\[
\hat{p}=\frac{149}{180}\approx 0.8278
\]
\[
SE=\sqrt{\frac{0.88(0.12)}{180}}\approx 0.0242

\subsection*{Problem 6 — Survey response participation}

\textbf{Problem.} A district office expects that 0.45 of students complete a feedback survey. In a sample of 260 students, 132 submit completed surveys.
\begin{itemize}
    \item Population parameter: \(p\), the true proportion of students who complete the survey.
    \item Significance level: \(\alpha=0.05\).
\end{itemize}

\textbf{Question.} Apply the general hypothesis-testing procedure to decide whether the true survey completion proportion differs from 0.45.

\textbf{Solution.}

\textbf{Step 1: Hypotheses.}
\[
\begin{aligned}
H_0&: p = 0.45 \\
H_a&: p \ne 0.45
\end{aligned}
\]

\textbf{Step 2: Type of test.} This is a two-tailed test because the claim checks for either an increase or a decrease from the benchmark.

\textbf{Step 3: Test statistic.}
\[
\hat{p}=\frac{132}{260}\approx 0.5077
\]
\[
SE=\sqrt{\frac{0.45(0.55)}{260}}\approx 0.0309
\]
\[
z=\frac{0.5077-0.45}{0.0309}\approx 1.87
\]

\textbf{Step 4: Reference distribution.} Under \(H_0\), the test statistic follows \(Z\sim N(0,1)\).

\textbf{Step 5: P-value.}
\[
P\text{-value}=2\cdot P(Z\ge 1.87)\approx 2(0.0307)=0.0614
\]

\textbf{Step 6: Decision.} Since \(0.0614>0.05\), fail to reject \(H_0\).

\textbf{Conclusion.} There is not enough evidence that the true survey completion proportion is different from 0.45.
