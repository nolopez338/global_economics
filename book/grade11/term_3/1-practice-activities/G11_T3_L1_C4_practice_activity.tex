\section{General Problem (One-Sample Proportion Test Template)}
A benchmark claim concerns a population proportion. A random sample of size \(n\) has \(x\) successes, so \(\hat p=x/n\). At significance level \(\alpha\), test whether the true proportion is lower than, higher than, or different from the benchmark \(p_0\).

\textbf{Questions}
\begin{enumerate}
    \item[(a)] Identify the population parameter and write hypotheses for lower-tail, higher-tail, and two-tailed alternatives.
    \item[(b)] State the condition checks for a one-sample proportion \(z\)-test.
    \item[(c)] Write the one-sample proportion test statistic.
    \item[(d)] State how the p-value is computed for lower-tail, higher-tail, and two-tailed tests.
    \item[(e)] State the decision rule and write a contextual conclusion template.
\end{enumerate}

\subsection*{C4}
\begin{itemize}
    \item \textbf{Parameter:} \(p\) = true population proportion for the context.
    \item \textbf{Hypotheses:}
    \[
    H_0\colon p=p_0,\qquad
    H_a\colon p<p_0\ \text{or}\ p>p_0\ \text{or}\ p\ne p_0.
    \]
    \item \textbf{Conditions:} Check \(np_0\ge 10\) and \(n(1-p_0)\ge 10\).
    \item \textbf{Test statistic:}
    \[
    z=\frac{\hat p-p_0}{\sqrt{\frac{p_0(1-p_0)}{n}}}.
    \]
    \item \textbf{p-value by tail type:}
    \begin{itemize}
        \item Lower-tail: \(P(Z\le z_{obs})\)
        \item Higher-tail: \(P(Z\ge z_{obs})\)
        \item Two-tailed: \(2P(Z\ge |z_{obs}|)\)
    \end{itemize}
    \item \textbf{Decision rule:} Reject \(H_0\) if p-value \(\le \alpha\); otherwise fail to reject \(H_0\).
    \item \textbf{Conclusion template:} ``There is sufficient evidence'' (reject \(H_0\)) or ``There is not enough evidence'' (fail to reject \(H_0\)) for the contextual claim in \(H_a\).
\end{itemize}

\section{On-time assignment completion rate (introductory lower-tail)}
A school reports that 72\% of assignments are submitted on time. In a random sample of 250 assignments, 170 are submitted on time. At \(\alpha=5\%\), test whether the true on-time proportion is lower than 72\%.

\subsection*{C4}
\begin{itemize}
    \item \textbf{Parameter:} \(p\) = true on-time submission proportion.
    \item \textbf{Hypotheses:} \(H_0\colon p=0.72\), \(H_a\colon p<0.72\).
    \item \textbf{Conditions:} \(np_0=250(0.72)=180\ge10\), \(n(1-p_0)=250(0.28)=70\ge10\).
    \item \textbf{Sample proportion:} \(\hat p=170/250=0.68\).
    \item \textbf{Test statistic:}
    \[
    z=\frac{0.68-0.72}{\sqrt{\frac{0.72(0.28)}{250}}}\approx -1.41.
    \]
    \item \textbf{p-value:} \(P(Z\le -1.41)\approx 0.0793\).
    \item \textbf{Decision:} Fail to reject \(H_0\) since \(0.0793>0.05\).
    \item \textbf{Conclusion:} There is not enough evidence that the true on-time submission rate is below 72\%.
\end{itemize}

    \item \textbf{Conclusion:} Strong evidence in the higher direction; both higher-tail and two-tailed tests reject, while the opposite direction does not.
\end{itemize}
