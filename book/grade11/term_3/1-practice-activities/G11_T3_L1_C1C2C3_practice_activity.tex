	\ExamTitleBlock{11th grade}{Term 3 — Lesson 1 Practice Activity (C1–C3) with Solutions}{}

	\ExamSection{Evaluated criteria}
	\begin{ExamCriteria}
		\item C1: Describe the concept of a statistical hypothesis in context.
		\item C2: Identify the null and alternative hypotheses using correct parameter notation.
		\item C3: Determine whether a test is one tailed or two tailed based on the claim.
	\end{ExamCriteria}

	\ExamSection{Problems}

	\begin{ExamProblems}

		\item
		\subsection*{Case 1 — Daily app check-ins}
		A school learning app reports that the daily check-in proportion is \(p_0=0.62\). A class project samples \(n=200\) active users and records \(x=112\) daily check-ins.

		\textbf{Questions.}
		\begin{enumerate}
			\item[(a)] Using \(\alpha=0.05\), set up the hypotheses for the case \(H_a\colon p<0.62\).
			\item[(b)] Using \(\alpha=0.05\), set up the hypotheses for the case \(H_a\colon p>0.62\).
			\item[(c)] Using \(\alpha=0.05\), set up the hypotheses for the case \(H_a\colon p\ne 0.62\).
		\end{enumerate}

		\subsection*{C1}
		The population parameter is the proportion \(p\) of all active users who check in daily, and the hypothesis testing focus is this population proportion relative to the reported value. Parts (a)–(c) use the same parameter \(p\) and the same null value \(p_0=0.62\).

		\subsection*{C2}
		\textbf{(a)} \(H_0\colon p=0.62,\; H_a\colon p<0.62\), \(\alpha=0.05\).

		\textbf{(b)} \(H_0\colon p=0.62,\; H_a\colon p>0.62\), \(\alpha=0.05\).

		\textbf{(c)} \(H_0\colon p=0.62,\; H_a\colon p\ne 0.62\), \(\alpha=0.05\).

		\subsection*{C3}
		(a) Left-tailed test because the alternative uses \(<\).

		(b) Right-tailed test because the alternative uses \(>\).

		(c) Two-tailed test because the alternative uses \(\ne\).
		% --------------------------------------------------
