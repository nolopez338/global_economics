	\ExamSection{Evaluated criteria}
	\begin{ExamCriteria}
		\item C1: Describe the concept of a statistical hypothesis in context.
		\item C2: Identify the null and alternative hypotheses using correct parameter notation.
		\item C3: Determine whether a test is one tailed or two tailed based on the claim.
	\end{ExamCriteria}

	\ExamSection{Problems}

	\begin{ExamProblems}

		\item
		\subsection*{Case 1 — Daily app check-ins}
		A school learning app reports a benchmark daily check-in proportion of 0.62. A class project samples 200 active users and records 112 daily check-ins.

		\textbf{Questions.}
		\begin{enumerate}
			\item[(a)] Using a significance level of 0.05, write the null and alternative hypotheses for the case where the true daily check-in proportion is lower than 0.62.
			\item[(b)] Using a significance level of 0.05, write the null and alternative hypotheses for the case where the true daily check-in proportion is higher than 0.62.
			\item[(c)] Using a significance level of 0.05, write the null and alternative hypotheses for the case where the true daily check-in proportion is different from 0.62.
		\end{enumerate}

		\subsection*{C1}
		Let \(p\) be the true population proportion of active users who check in daily. Let \(p_0\) be the benchmark proportion from the app report, so \(p_0=0.62\). Let \(n\) be the sample size, with \(n=200\). Let \(x\) be the number of daily check-ins observed in the sample, with \(x=112\). Let \(\alpha\) be the significance level, with \(\alpha=0.05\).

		Parts (a)–(c) use the same parameter \(p\) and the same benchmark \(p_0\), but each part uses a different form of the alternative claim.

		\subsection*{C2}
		\textbf{(a)} \(H_0\colon p=0.62,\; H_a\colon p<0.62\), \(\alpha=0.05\).

		\textbf{(b)} \(H_0\colon p=0.62,\; H_a\colon p>0.62\), \(\alpha=0.05\).

		\textbf{(c)} \(H_0\colon p=0.62,\; H_a\colon p\ne 0.62\), \(\alpha=0.05\).

		\subsection*{C3}
		(a) Left-tailed test because the alternative claims the parameter is lower than the benchmark.

		(b) Right-tailed test because the alternative claims the parameter is higher than the benchmark.

		(c) Two-tailed test because the alternative claims the parameter is different from the benchmark.
