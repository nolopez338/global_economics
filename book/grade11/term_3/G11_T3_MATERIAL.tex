	\hypertarget{c1-ex1}{}
	\subsection*{Problem 1 — School Water Bottle Capacity}
	
	\textbf{Problem.} A school supplier claims that its reusable water bottles hold an average of $750$ mL. A quality team measures a sample of $n=36$ bottles and wants to evaluate the claim.
	\begin{itemize}
		\item Population parameter: mean capacity $\mu$ (mL).
		\item Significance level: $\alpha=0.05$.
	\end{itemize}
	
	\textbf{Question.} Explain what a statistical hypothesis is in this context, and set up a hypothesis test to check the supplier's claim.
	
	\textbf{Solution.}
	
	A statistical hypothesis is a statement about a population parameter (here, the true mean capacity $\mu$). We use sample data to decide whether the claim is reasonable.
	
	\textbf{Hypotheses.}
	\[
	\begin{aligned}
	H_0&: \mu = 750 \text{ mL} \\
	H_a&: \mu \neq 750 \text{ mL}
	\end{aligned}
	\]
	
	\textbf{Type of test.} Two-tailed, because the concern is any difference from $750$ mL.
	
	\textbf{Test statistic.} If the population standard deviation is known or a large sample is used, use
	\[
	z = \frac{\bar{x}-\mu_0}{\sigma/\sqrt{n}}.
	\]
	
	\textbf{Reference distribution.} Standard normal under $H_0$.
	
	\textbf{Decision rule.} Compare the P-value to $\alpha=0.05$ (or use the two-tailed critical region).
	
	\textbf{Conclusion in context.} Decide whether the sample provides evidence that the average capacity differs from $750$ mL.

	\textbf{Problem.} A school bulletin states that students study an average of $2.5$ hours per day in the library. A survey of $n=40$ students is conducted to check the statement.
	\begin{itemize}
		\item Population parameter: mean study time $\mu$ (hours).
		\item Significance level: $\alpha=0.10$.
	\end{itemize}
	
	\textbf{Question.} Describe the statistical hypothesis in this setting and state the hypotheses to be tested.
	
	\textbf{Solution.}
	
	The statistical hypothesis concerns the population mean study time $\mu$. The claim $\mu=2.5$ hours is tested using sample data.
	
	\textbf{Hypotheses.}
	\[
	\begin{aligned}
	H_0&: \mu = 2.5 \text{ hours} \\
	H_a&: \mu \neq 2.5 \text{ hours}
	\end{aligned}
	\]
	
	\textbf{Type of test.} Two-tailed because the claim is an equality and deviations in either direction matter.
	
	\textbf{Test statistic and reference distribution.} Use
	\[
	z = \frac{\bar{x}-\mu_0}{\sigma/\sqrt{n}},
	\]
	with a standard normal reference distribution (or a $t$ distribution if $\sigma$ is unknown).
	
	\textbf{Decision rule.} Compare the P-value to $\alpha=0.10$.
	
	\textbf{Conclusion in context.} Decide if the data support or contradict the bulletin's mean study time.

	\hypertarget{c2-ex1}{}
	\subsection*{Problem 1 — Bakery Cookie Weight}
	
	\textbf{Problem.} A bakery advertises that its cookies weigh $35$ g on average. A sample of $n=25$ cookies has mean $\bar{x}=33.8$ g, and the population standard deviation is known to be $\sigma=3.0$ g.
	\begin{itemize}
		\item Population parameter: mean cookie weight $\mu$ (g).
		\item Significance level: $\alpha=0.05$.
	\end{itemize}
	
	\textbf{Question.} Identify the null and alternative hypotheses and perform the test.
	
	\textbf{Solution.}
	
	\textbf{Hypotheses.}
	\[
	\begin{aligned}
	H_0&: \mu = 35 \text{ g} \\
	H_a&: \mu < 35 \text{ g}
	\end{aligned}
	\]
	
	\textbf{Type of test.} One-tailed (left) because the concern is underfilling.
	
	\textbf{Test statistic.}
	\[
	z = \frac{\bar{x}-\mu_0}{\sigma/\sqrt{n}} = \frac{33.8-35}{3.0/\sqrt{25}} = \frac{-1.2}{0.6} = -2.00.
	\]
	
	\textbf{Reference distribution.} $Z \sim N(0,1)$ under $H_0$.
	
	\textbf{P-value.} $P(Z\le -2.00)=0.0228$.
	
	\textbf{Decision.} Since $0.0228<0.05$, reject $H_0$.
	
	\textbf{Conclusion.} There is evidence that the mean cookie weight is less than $35$ g.

	\subsection*{Problem 2 — Campus Wi-Fi Reliability}
	
	\textbf{Problem.} The IT office claims that campus Wi-Fi is available at least $95\%$ of the time. In a random sample of $n=200$ checks, the network is available in $186$ checks.
	\begin{itemize}
		\item Population parameter: availability proportion $p$.
		\item Significance level: $\alpha=0.05$.
	\end{itemize}
	
	\textbf{Question.} Identify the null and alternative hypotheses and test the claim.
	
	\textbf{Solution.}
	
	\textbf{Hypotheses.}
	\[
	\begin{aligned}
	H_0&: p = 0.95 \\
	H_a&: p < 0.95
	\end{aligned}
	\]
	
	\textbf{Type of test.} One-tailed (left) because the claim is “at least $95\%$,” and we test for a decrease.
	
	\textbf{Test statistic.}
	\[
	z = \frac{\hat{p}-p_0}{\sqrt{p_0(1-p_0)/n}} = \frac{0.93-0.95}{\sqrt{0.95(0.05)/200}} \approx \frac{-0.02}{0.0154} \approx -1.30.
	\]
	
	\textbf{Reference distribution.} Standard normal under $H_0$.
	
	\textbf{P-value.} $P(Z\le -1.30)\approx 0.0968$.
	
	\textbf{Decision.} Since $0.0968>0.05$, fail to reject $H_0$.
	
	\textbf{Conclusion.} The sample does not provide enough evidence that availability is below $95\%$.

		\item Significance level: $\alpha=0.05$.

	\textbf{Hypotheses.}

	\begin{aligned}
	H_0&: \mu = 12 \\
	H_a&: \mu > 12

	\textbf{Reference distribution.} $Z \sim N(0,1)$.
	
	\textbf{P-value.} $P(Z\ge 2.26)\approx 0.0119$.
	
	\textbf{Decision.} Since $0.0119<0.05$, reject $H_0$.
	
	\textbf{Conclusion.} There is evidence that average bus arrival times exceed $12$ minutes.

	\[
	\begin{aligned}
	H_0&: p = 0.80 \\
	H_a&: p < 0.80
	\end{aligned}
	\]
	
	\textbf{Step 2: Type of test.} One-tailed (left), because we test for a drop below $80\%$.

	\[
	z = \frac{\hat{p}-p_0}{\sqrt{p_0(1-p_0)/n}} = \frac{0.76-0.80}{\sqrt{0.80(0.20)/150}} \approx \frac{-0.04}{0.0326} \approx -1.23.

	\textbf{Step 4: Reference distribution.} Standard normal under $H_0$.
	
	\textbf{Step 5: P-value.} $P(Z\le -1.23)\approx 0.1093$.
	
	\textbf{Step 6: Decision.} Since $0.1093>0.05$, fail to reject $H_0$.
	
	\textbf{Conclusion.} The data do not show that completion is below $80\%$.

	% -------------------------
	% Problem 1 — Classroom Noise Level
	% -------------------------
	\hypertarget{c8-ex1}{}
	\subsection*{Problem 1 — Classroom Noise Level}
	
	\textbf{Problem.} The school claims that the mean noise level during group work is $65$ dB. A random sample of $n=36$ readings gives $\bar{x}=66.5$ dB with $\sigma=6$ dB.
	\begin{itemize}
		\item Population parameter: mean noise level $\mu$ (dB).
		\item Significance level: $\alpha=0.05$.
	\end{itemize}
	
	\textbf{Question.} Use a hypothesis test and a confidence interval to evaluate the claim and explain the link between them.
	
	\textbf{Solution.}
	
	\textbf{Hypotheses.}
	\[
	\begin{aligned}
	H_0&: \mu = 65 \\
	H_a&: \mu \neq 65
	\end{aligned}
	\]
	
	\textbf{Type of test.} Two-tailed because any difference from $65$ dB is important.
	
	\textbf{Test statistic.}
	\[
	z = \frac{\bar{x}-\mu_0}{\sigma/\sqrt{n}} = \frac{66.5-65}{6/\sqrt{36}} = \frac{1.5}{1} = 1.50.
	\]
	
	\textbf{P-value.} $2P(Z\ge 1.50)\approx 2(0.0668)=0.1336$.
	
	\textbf{Decision.} Since $0.1336>0.05$, fail to reject $H_0$.
	
	\textbf{Confidence interval link.} A $95\%$ confidence interval is
	\[
	\bar{x} \pm z_{0.025}\frac{\sigma}{\sqrt{n}} = 66.5 \pm 1.96(1) = (64.54, 68.46).
	\]
	Because $65$ dB lies in the interval, the two-tailed test at $\alpha=0.05$ does not reject $H_0$.
	
	\textbf{Conclusion.} Both methods suggest the data do not show a significant difference from $65$ dB.

	\textbf{Problem.} The school cafeteria reports that $55\%$ of students are satisfied with lunch quality. A survey of $n=200$ students finds $120$ are satisfied.
	\begin{itemize}
		\item Population parameter: satisfaction proportion $p$.
		\item Significance level: $\alpha=0.10$.
	\end{itemize}
	
	\textbf{Question.} Use a hypothesis test and a confidence interval to analyze the claim and explain their connection.
	
	\textbf{Solution.}
	
	\textbf{Hypotheses.}
	\[
	\begin{aligned}
	H_0&: p = 0.55 \\
	H_a&: p \neq 0.55
	\end{aligned}
	\]
	
	\textbf{Type of test.} Two-tailed because deviations in either direction matter.
	
	\textbf{Test statistic.}
	\[
	z = \frac{\hat{p}-p_0}{\sqrt{p_0(1-p_0)/n}} = \frac{0.60-0.55}{\sqrt{0.55(0.45)/200}} \approx \frac{0.05}{0.0352} \approx 1.42.
	\]
	
	\textbf{P-value.} $2P(Z\ge 1.42)\approx 2(0.0778)=0.1556$.
	
	\textbf{Decision.} Since $0.1556>0.10$, fail to reject $H_0$.
	
	\textbf{Confidence interval link.} A $90\%$ confidence interval for $p$ is
	\[
	\hat{p} \pm z_{0.05}\sqrt{\frac{\hat{p}(1-\hat{p})}{n}} = 0.60 \pm 1.645\sqrt{\frac{0.60(0.40)}{200}}\approx 0.60 \pm 0.057.
	\]
	So the interval is approximately $(0.543, 0.657)$. Because $0.55$ is inside this interval, the two-tailed test at $\alpha=0.10$ does not reject $H_0$.
	
	\textbf{Conclusion.} The confidence interval and test agree that the data do not show a significant difference from $55\%$.

	\textbf{Problem.} A community bank expects that at most $4\%$ of microloans default. This affects interest pricing and reserves. In the last year, $n=500$ loans were issued and $28$ defaulted.
	\begin{itemize}
		\item Population parameter: default proportion $p$.
		\item Significance level: $\alpha=0.05$.
	\end{itemize}
	
	\textbf{Question.} Test whether the default rate exceeds $4\%$ and explain why a hypothesis test is needed.
	
	\textbf{Solution.}
	
	A hypothesis test is needed because the bank must decide if the observed defaults suggest a real increase in the population default rate, which would require higher reserves.
	
	\textbf{Hypotheses.}
	\[
	\begin{aligned}
	H_0&: p = 0.04 \\
	H_a&: p > 0.04
	\end{aligned}
	\]
	
	\textbf{Type of test.} One-tailed (right), because we test for a higher default rate.
	
	\textbf{Test statistic.}
	\[
	z = \frac{\hat{p}-p_0}{\sqrt{p_0(1-p_0)/n}} = \frac{0.056-0.04}{\sqrt{0.04(0.96)/500}} \approx \frac{0.016}{0.0088} \approx 1.82.
	\]
	
	\textbf{Reference distribution.} $Z \sim N(0,1)$.
	
	\textbf{P-value.} $P(Z\ge 1.82)\approx 0.0344$.
	
	\textbf{Decision.} Since $0.0344<0.05$, reject $H_0$.
	
	\textbf{Conclusion.} The data suggest the default rate is above $4\%$, so the bank should consider higher reserves.

	\subsection*{Problem 2 — Market Stall Daily Revenue}
	
	\textbf{Problem.} A cooperative of market vendors estimates that the mean daily revenue per stall is $\$120. To decide whether a new layout improved sales, they collect $n=25$ days of data with mean $\bar{x}=126$ and known $\sigma=15$.
	\begin{itemize}
		\item Population parameter: mean revenue $\mu$ (USD).
		\item Significance level: $\alpha=0.10$.
	\end{itemize}
	
	\textbf{Question.} Test whether the mean revenue has increased, and explain the economic decision involved.
	
	\textbf{Solution.}
	
	A hypothesis test is required because the cooperative must decide if the observed increase is large enough to justify keeping the new layout.
	
	\textbf{Hypotheses.}
	\[
	\begin{aligned}
	H_0&: \mu = 120 \\
	H_a&: \mu > 120
	\end{aligned}
	\]
	
	\textbf{Type of test.} One-tailed (right), because the decision depends on improvement.
	
	\textbf{Test statistic.}
	\[
	z = \frac{\bar{x}-\mu_0}{\sigma/\sqrt{n}} = \frac{126-120}{15/\sqrt{25}} = \frac{6}{3} = 2.00.
	\]
	
	\textbf{Reference distribution.} Standard normal under $H_0$.
	
	\textbf{P-value.} $P(Z\ge 2.00)=0.0228$.
	
	\textbf{Decision.} Since $0.0228<0.10$, reject $H_0$.
	
	\textbf{Conclusion.} The data provide evidence that mean daily revenue increased, supporting the new layout.

	\subsection*{Problem 1 — Savings Program Impact}
	
	\textbf{Problem.} A school introduces a savings club and wants to know if the average monthly savings of participants exceeds $\$15. In a sample of $n=30$ students, the mean savings is $\bar{x}=16.8$ with $\sigma=4.2$.
	\begin{itemize}
		\item Population parameter: mean monthly savings $\mu$ (USD).
		\item Significance level: $\alpha=0.05$.
	\end{itemize}
	
	\textbf{Question.} Model the research question with a statistical hypothesis test and provide the decision.
	
	\textbf{Solution.}
	
	The research question “Does the savings club increase mean monthly savings above $\$15?” is modeled by testing whether $\mu$ exceeds $15$.
	
	\textbf{Hypotheses.}
	\[
	\begin{aligned}
	H_0&: \mu = 15 \\
	H_a&: \mu > 15
	\end{aligned}
	\]
	
	\textbf{Type of test.} One-tailed (right), because we test for an increase.
	
	\textbf{Test statistic.}
	\[
	z = \frac{\bar{x}-\mu_0}{\sigma/\sqrt{n}} = \frac{16.8-15}{4.2/\sqrt{30}} \approx \frac{1.8}{0.767} \approx 2.35.
	\]
	
	\textbf{Reference distribution.} $Z \sim N(0,1)$.
	
	\textbf{P-value.} $P(Z\ge 2.35)\approx 0.0094$.
	
	\textbf{Decision.} Since $0.0094<0.05$, reject $H_0$.
	
	\textbf{Conclusion.} The data support that the savings club increases average monthly savings beyond $\$15.

	\textbf{Problem.} A student entrepreneurship group wants to test whether the proportion of customers who return within a week is at least $50\%$ after introducing a loyalty stamp card. In $n=120$ customers, $54$ return within a week.
	\begin{itemize}
		\item Population parameter: return proportion $p$.
		\item Significance level: $\alpha=0.05$.
	\end{itemize}
	
	\textbf{Question.} Model the research question with a hypothesis test and state the conclusion.
	
	\textbf{Solution.}
	
	The research question is whether the return rate reaches the target of $50\%$, so we test if $p$ is below $0.50$.
	
	\textbf{Hypotheses.}
	\[
	\begin{aligned}
	H_0&: p = 0.50 \\
	H_a&: p < 0.50
	\end{aligned}
	\]
	
	\textbf{Type of test.} One-tailed (left), because the concern is not meeting the target rate.
	
	\textbf{Test statistic.}
	\[
	z = \frac{\hat{p}-p_0}{\sqrt{p_0(1-p_0)/n}} = \frac{0.45-0.50}{\sqrt{0.50(0.50)/120}} \approx \frac{-0.05}{0.0456} \approx -1.10.
	\]
	
	\textbf{Reference distribution.} Standard normal under $H_0$.
	
	\textbf{P-value.} $P(Z\le -1.10)\approx 0.1357$.
	
	\textbf{Decision.} Since $0.1357>0.05$, fail to reject $H_0$.
	
	\textbf{Conclusion.} The data do not show the return rate is below $50\%$, so the target cannot be ruled out.
