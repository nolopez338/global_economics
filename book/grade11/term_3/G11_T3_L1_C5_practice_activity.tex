\ExamSection{C5 Calculates hypothesis tests on the mean of a population.}

\ExamSection{Evaluated criteria}
\begin{ExamCriteria}
    \item C5: Perform a hypothesis test for a population mean with known standard deviation using the z statistic.
\end{ExamCriteria}

\ExamSection{Problems}
\begin{ExamProblems}

% -------------------------
% Problem 1 — Production Time
% -------------------------
\hypertarget{c5-ex1}{}
\item
\subsection*{Problem 1 — Production Time}

\textbf{Problem.} A factory claims that the mean production time for a component is \(45\) minutes. A quality team collects a random sample of \(n=64\) components and finds \(\bar{x}=45.7\) minutes. The population standard deviation is known as \(\sigma=3.2\) minutes.
\begin{itemize}
    \item Population parameter: mean production time \(\mu\).
    \item Significance level: \(\alpha=0.05\).
\end{itemize}

\textbf{Question.} Test whether the true mean production time is greater than \(45\) minutes.

\textbf{Solution.}

\textbf{Hypotheses.}
\[
\begin{aligned}
H_0&: \mu = 45 \\
H_a&: \mu > 45
\end{aligned}
\]

\textbf{Type of test.} This is a right-tailed test because the claim states that the mean is greater than \(45\).

\textbf{Test statistic.}
\[
z = \frac{\bar{x}-\mu_0}{\sigma/\sqrt{n}} = \frac{45.7-45}{3.2/\sqrt{64}} = \frac{0.7}{3.2/8} = \frac{0.7}{0.4} = 1.75
\]

\textbf{Reference distribution.} Standard normal under \(H_0\).

\textbf{P-value.}
\[
p\text{-value}=P(Z\ge 1.75)=0.0401
\]

\textbf{Decision.} Since \(0.0401<0.05\), reject \(H_0\).

\textbf{Conclusion.} There is sufficient evidence that the mean production time is greater than \(45\) minutes.

\subsection*{Problem 2 — Battery Life}

\textbf{Problem.} A company states that the mean battery life of a device is \(12\) hours. A random sample with \(n=100\) devices gives \(\bar{x}=11.68\) hours. The population standard deviation is known as \(\sigma=2.4\) hours.
\begin{itemize}
    \item Population parameter: mean battery life \(\mu\).
    \item Significance level: \(\alpha=0.05\).
\end{itemize}

\textbf{Question.} Test whether the true mean battery life is less than \(12\) hours.

\textbf{Solution.}

\textbf{Hypotheses.}
\[
\begin{aligned}
H_0&: \mu = 12 \\
H_a&: \mu < 12
\end{aligned}
\]

\textbf{Type of test.} This is a left-tailed test because the claim states that the mean is less than \(12\).

\textbf{Test statistic.}
\[
z = \frac{\bar{x}-\mu_0}{\sigma/\sqrt{n}} = \frac{11.68-12}{2.4/\sqrt{100}} = \frac{-0.32}{2.4/10} = \frac{-0.32}{0.24} = -1.33
\]

\textbf{Reference distribution.} Standard normal under \(H_0\).

\textbf{P-value.}
\[
p\text{-value}=P(Z\le -1.33)=0.0918
\]

\textbf{Decision.} Since \(0.0918>0.05\), fail to reject \(H_0\).

\textbf{Conclusion.} There is not sufficient evidence that the mean battery life is less than \(12\) hours.

\item
\subsection*{Problem 3 — Exam Score Average}

\textbf{Problem.} A school reports that the mean score on a standardized exam is \(78\) points. A random sample of \(n=81\) students gives \(\bar{x}=80.1\) points. The population standard deviation is known as \(\sigma=9\) points.
\begin{itemize}
    \item Population parameter: mean exam score \(\mu\).
    \item Significance level: \(\alpha=0.05\).
\end{itemize}

\textbf{Question.} Test whether the true mean exam score differs from \(78\) points.

\textbf{Solution.}

\textbf{Hypotheses.}
\[
\begin{aligned}
H_0&: \mu = 78 \\
H_a&: \mu \ne 78
\end{aligned}
\]

\textbf{Type of test.} This is a two-tailed test because the claim asks whether the mean differs from \(78\).

\textbf{Test statistic.}
\[
z = \frac{\bar{x}-\mu_0}{\sigma/\sqrt{n}} = \frac{80.1-78}{9/\sqrt{81}} = \frac{2.1}{9/9} = \frac{2.1}{1} = 2.10
\]

\textbf{Reference distribution.} Standard normal under \(H_0\).

\textbf{P-value.}
\[
p\text{-value}=2P(Z\ge |2.10|)=2P(Z\ge 2.10)=0.0358
\]

\textbf{Decision.} Since \(0.0358<0.05\), reject \(H_0\).

\textbf{Conclusion.} There is sufficient evidence that the mean exam score is different from \(78\) points.

\subsection*{Problem 6 — Packaging Weight}

\textbf{Problem.} A packing line is calibrated to a mean weight of \(500\) g. A random sample with \(n=100\) packages gives \(\bar{x}=498.45\) g. The population standard deviation is known as \(\sigma=12\) g.
\begin{itemize}
    \item Population parameter: mean package weight \(\mu\).
    \item Significance level: \(\alpha=0.05\).
\end{itemize}

\textbf{Question.} Test whether the true mean package weight differs from \(500\) g.

\textbf{Solution.}

\textbf{Hypotheses.}
\[
\begin{aligned}
H_0&: \mu = 500 \\
H_a&: \mu \ne 500
\end{aligned}
\]

\textbf{Type of test.} This is a two-tailed test because the claim asks whether the mean differs from \(500\).

\textbf{Test statistic.}
\[
z = \frac{\bar{x}-\mu_0}{\sigma/\sqrt{n}} = \frac{498.45-500}{12/\sqrt{100}} = \frac{-1.55}{12/10} = \frac{-1.55}{1.2} = -1.29
\]

\textbf{Reference distribution.} Standard normal under \(H_0\).

\textbf{P-value.}

p\text{-value}=2P(Z\ge |{-1.29}|)=2P(Z\ge 1.29)=0.1974
\]

\textbf{Decision.} Since \(0.1974>0.05\), fail to reject \(H_0\).

\textbf{Conclusion.} There is not sufficient evidence that the mean package weight is different from \(500\) g.
