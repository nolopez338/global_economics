\section{General Problem (One-Sample Mean Test Template in Counting Contexts)}
A benchmark claim concerns a population mean count. A random sample of size \(n\) has \(x\) successes, so \(\bar x=x/n\). At significance level \(\alpha\), test whether the true mean count is lower than, higher than, or different from the benchmark \(\mu_0\).

\textbf{Reference table for common significance levels}

\begin{center}
	\renewcommand{\arraystretch}{1.3}
	\begin{tabular}{|c|c|c|}
		\hline
		\textbf{Significance level} & \textbf{Upper-tail p-value} & \textbf{\(z_\alpha\) such that \(P(Z>z_\alpha)=\alpha\)} \\
		\hline
		\(10\%\) & \(0.10\) & \(0.842\) \\
		\hline
		\(5\%\) & \(0.05\) & \(1.645\) \\
		\hline
		\(2\%\) & \(0.02\) & \(2.054\) \\
		\hline
		\(1\%\) & \(0.01\) & \(2.326\) \\
		\hline
	\end{tabular}
\end{center}

\textbf{Questions/Tasks.}
\begin{enumerate}
	\item[(a)] Identify the population parameter and write hypotheses for lower-tail, higher-tail, and two-tailed alternatives.
	\item[(b)] State the condition checks for a one-sample mean count \(z\)-test.
	\item[(c)] Write the one-sample mean count test statistic.
	\item[(d)] State how the p-value is computed for lower-tail, higher-tail, and two-tailed tests.
	\item[(e)] State the decision rule and write a contextual conclusion template.
\end{enumerate}

\subsection*{C5}
\begin{itemize}
	\item \textbf{Parameter:} \(\mu\) = true population mean count for the context.
	\item \textbf{Hypotheses:}
	\[
	H_0\colon \mu=\mu_0,\qquad
	H_a\colon \mu<\mu_0\ \text{or}\ \mu>\mu_0\ \text{or}\ \mu\ne \mu_0.
	\]
	\item \textbf{Conditions:} Check \(n\mu_0\ge 10\) and \(n(1-\mu_0)\ge 10\).
	\item \textbf{Test statistic:}
	\[
	z=\frac{\bar x-\mu_0}{\sqrt{\frac{\mu_0(1-\mu_0)}{n}}}.
	\]
	\item \textbf{p-value by tail type:}
	\begin{itemize}
		\item Lower-tail: \(P(Z\le z_{obs})\)
		\item Higher-tail: \(P(Z\ge z_{obs})\)
		\item Two-tailed: \(2P(Z\ge |z_{obs}|)\)
	\end{itemize}
	\item \textbf{Decision rule:} Reject \(H_0\) if p-value \(\le \alpha\); otherwise fail to reject \(H_0\).
	\item \textbf{Conclusion template:} ``There is sufficient evidence'' (reject \(H_0\)) or ``There is not enough evidence'' (fail to reject \(H_0\)) for the contextual claim in \(H_a\).
\end{itemize}

\section{On-time assignment completion mean count (introductory higher-tail)}
		A school reports that students submit an average of 0.72 on-time assignments per assignment opportunity (counting variable: 1 for on time, 0 for not on time). Consider two random samples:
		\begin{itemize}
		\item Sample A has 250 assignments, of which 190 are submitted on time.
		\item Sample B has 500 assignments, of which 380 are submitted on time.
		\end{itemize}
		Notice that both samples have the same sample mean count. At \(\alpha=5\%\), test whether the true on-time mean count is higher than 0.72 for each sample.
		
	\subsection*{C5}
	\begin{itemize}
	\item \textbf{Parameter:} \(\mu\) = true mean number of on-time submissions per assignment opportunity (0 or 1).
		\item \textbf{Hypotheses:} \(H_0\colon \mu=0.72\), \(H_a\colon \mu>0.72\).
	\end{itemize}
	
	\textbf{Sample A}
	
	\begin{itemize}
	\item \textbf{Conditions:} \(n\mu_0=250(0.72)=180\ge10\), \(n(1-\mu_0)=250(0.28)=70\ge10\).
	\item \textbf{Sample mean count:} \(\bar x=190/250=0.76\).
		\item \textbf{Test statistic:}
		\[
		z=\frac{0.76-0.72}{\sigma/\sqrt{250}}\approx 1.41,\quad \sigma=\sqrt{0.72(0.28)}\approx0.4489.
		\]
		\item \textbf{p-value:} \(P(Z\ge 1.41)\approx 0.0795\).
		\item \textbf{Decision:} Fail to reject \(H_0\) since \(0.0795>0.05\).
		\item \textbf{Conclusion:} There is not enough evidence that the true on-time submission mean count is above 0.72.
		\end{itemize}
		
	\begin{center}
	\OneTailGraph{1}{1.41}{1.645}{Sample A higher-tail test: \(H_a:\mu>0.72\)}
	\end{center}
		
	\PropDecisionLine{0.72}{0.76}{1}{red}{fail to reject \(H_0\)}{1.41}{0.72}{5}

	
	\textbf{Sample B}
	
	\begin{itemize}
	\item \textbf{Conditions:} \(n\mu_0=500(0.72)=360\ge10\), \(n(1-\mu_0)=500(0.28)=140\ge10\).
	\item \textbf{Sample mean count:} \(\bar x=380/500=0.76\).
		\item \textbf{Test statistic:}
		\[
	z=\frac{0.76-0.72}{\sigma/\sqrt{500}}\approx 1.99,\quad \sigma=\sqrt{0.72(0.28)}\approx0.4489.
	\]
	\item \textbf{p-value:} \(P(Z\ge 1.99)\approx 0.0232\).
	\item \textbf{Decision:} Reject \(H_0\) since \(0.0232<0.05\).
	\item \textbf{Conclusion:} There is enough evidence that the true on-time submission mean count is above 0.72.
		\end{itemize}
		
	\begin{center}
	\OneTailGraph{1}{1.99}{1.645}{Sample B higher-tail test: \(H_a:\mu>0.72\)}
	\end{center}
	
	\PropDecisionLine{0.72}{0.76}{1}{green!60!black}{reject \(H_0\)}{1.99}{0.72}{5}

		
	Although both samples have the same sample mean count \(\bar x=0.76\), the larger sample gives a more positive test statistic and a smaller p-value because the standard error is smaller.

\section{Platform adoption rate after training (higher-tail test compared across three significance levels)}
A district dashboard states that teachers average 0.40 regular uses of a new digital platform per observation period (counting variable: 1 if used regularly, 0 otherwise). After training, a random sample of 300 teachers shows 138 regular users. At significance levels \(\alpha=1\%\), \(\alpha=2\%\), and \(\alpha=5\%\), test whether the true adoption mean count is higher than 0.40.

\subsection*{C5}
\begin{itemize}
	\item \textbf{Parameter:} \(\mu\) = true mean number of regular platform uses per observation period (0 or 1).
	\item \textbf{Hypotheses at \(\alpha=1\%\), \(\alpha=2\%\), and \(\alpha=5\%\):} \(H_0\colon \mu=0.40\), \(H_a\colon \mu>0.40\).
	\item \textbf{Conditions:} \(n\mu_0=120\ge10\), \(n(1-\mu_0)=180\ge10\).
	\item \textbf{Sample mean count:} \(\bar x=138/300=0.46\).
	\item \textbf{Test statistic:}
	\[
	z=\frac{0.46-0.40}{\sqrt{\frac{\sigma^2}{300}}}\approx 2.12.
	\]
	\item \textbf{p-value:} \(P(Z\ge 2.12)\approx 0.0170\).
	\item \textbf{Decision at \(\alpha=5\%\):} Reject \(H_0\) since \(0.0170<0.05\).
	\item \textbf{Decision at \(\alpha=2\%\):} Reject \(H_0\) since \(0.0170<0.02\).
	\item \textbf{Decision at \(\alpha=1\%\):} Fail to reject \(H_0\) since \(0.0170>0.01\).
	\item \textbf{Conclusion:} The null hypothesis is rejected at the 5\% and 2\% significance levels, but it is not rejected at the 1\% significance level. Therefore, only one of the three significance levels fails to reject \(H_0\).
\end{itemize}

\begin{center}
