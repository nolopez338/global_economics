\section{General Problem (One-Sample Proportion Test Template)}
A benchmark claim concerns a population proportion. A random sample of size \(n\) has \(x\) successes, so \(\hat p=x/n\). At significance level \(\alpha\), test whether the true proportion is lower than, higher than, or different from the benchmark \(p_0\).

\textbf{Reference table for common significance levels}

\begin{center}
	\renewcommand{\arraystretch}{1.3}
	\begin{tabular}{|c|c|c|}
		\hline
		\textbf{Significance level} & \textbf{Upper-tail p-value} & \textbf{\(z_\alpha\) such that \(P(Z>z_\alpha)=\alpha\)} \\
		\hline
		\(10\%\) & \(0.10\) & \(0.842\) \\
		\hline
		\(5\%\) & \(0.05\) & \(1.645\) \\
		\hline
		\(2\%\) & \(0.02\) & \(2.054\) \\
		\hline
		\(1\%\) & \(0.01\) & \(2.326\) \\
		\hline
	\end{tabular}
\end{center}

\textbf{Questions/Tasks.}
\begin{enumerate}
	\item[(a)] Identify the population parameter and write hypotheses for lower-tail, higher-tail, and two-tailed alternatives.
	\item[(b)] State the condition checks for a one-sample proportion \(z\)-test.
	\item[(c)] Write the one-sample proportion test statistic.
	\item[(d)] State how the p-value is computed for lower-tail, higher-tail, and two-tailed tests.
	\item[(e)] State the decision rule and write a contextual conclusion template.
\end{enumerate}

\subsection*{C4}
\begin{itemize}
	\item \textbf{Parameter:} \(p\) = true population proportion for the context.
	\item \textbf{Hypotheses:}
	\[
	H_0\colon p=p_0,\qquad
	H_a\colon p<p_0\ \text{or}\ p>p_0\ \text{or}\ p\ne p_0.
	\]
	\item \textbf{Conditions:} Check \(np_0\ge 10\) and \(n(1-p_0)\ge 10\).
	\item \textbf{Test statistic:}
	\[
	z=\frac{\hat p-p_0}{\sqrt{\frac{p_0(1-p_0)}{n}}}.
	\]
	\item \textbf{p-value by tail type:}
	\begin{itemize}
		\item Lower-tail: \(P(Z\le z_{obs})\)
		\item Higher-tail: \(P(Z\ge z_{obs})\)
		\item Two-tailed: \(2P(Z\ge |z_{obs}|)\)
	\end{itemize}
	\item \textbf{Decision rule:} Reject \(H_0\) if p-value \(\le \alpha\); otherwise fail to reject \(H_0\).
	\item \textbf{Conclusion template:} ``There is sufficient evidence'' (reject \(H_0\)) or ``There is not enough evidence'' (fail to reject \(H_0\)) for the contextual claim in \(H_a\).
\end{itemize}

\section{On-time assignment completion rate (introductory higher-tail)}
		A school reports that 72\% of assignments are submitted on time. Consider two random samples:
		\begin{itemize}
		\item Sample A has 250 assignments, of which 190 are submitted on time.
		\item Sample B has 500 assignments, of which 380 are submitted on time.
		\end{itemize}
		Notice that both samples have the same sample proportion. At \(\alpha=5\%\), test whether the true on-time proportion is higher than 72\% for each sample.
		
	\subsection*{C4}
	\begin{itemize}
	\item \textbf{Parameter:} \(p\) = true on-time submission proportion.
		\item \textbf{Hypotheses:} \(H_0\colon p=0.72\), \(H_a\colon p>0.72\).
	\end{itemize}
	
	\textbf{Sample A}
	
	\begin{itemize}
	\item \textbf{Conditions:} \(np_0=250(0.72)=180\ge10\), \(n(1-p_0)=250(0.28)=70\ge10\).
	\item \textbf{Sample proportion:} \(\hat p=190/250=0.76\).
		\item \textbf{Test statistic:}
		\[
		z=\frac{0.76-0.72}{\sqrt{\frac{0.72(0.28)}{250}}}\approx 1.41.
		\]
		\item \textbf{p-value:} \(P(Z\ge 1.41)\approx 0.0795\).
		\item \textbf{Decision:} Fail to reject \(H_0\) since \(0.0795>0.05\).
		\item \textbf{Conclusion:} There is not enough evidence that the true on-time submission rate is above 72\%.
		\end{itemize}
		
	\begin{center}
	\OneTailGraph{1}{1.41}{1.645}{Sample A higher-tail test: \(H_a:p>0.72\)}
	\end{center}
		
	\PropDecisionLine{0.72}{0.76}{1}{red}{fail to reject \(H_0\)}{1.41}{0.72}{5}

	
	\textbf{Sample B}
	
	\begin{itemize}
	\item \textbf{Conditions:} \(np_0=500(0.72)=360\ge10\), \(n(1-p_0)=500(0.28)=140\ge10\).
	\item \textbf{Sample proportion:} \(\hat p=380/500=0.76\).
		\item \textbf{Test statistic:}
		\[
	z=\frac{0.76-0.72}{\sqrt{\frac{0.72(0.28)}{500}}}\approx 1.99.
	\]
	\item \textbf{p-value:} \(P(Z\ge 1.99)\approx 0.0232\).
	\item \textbf{Decision:} Reject \(H_0\) since \(0.0232<0.05\).
	\item \textbf{Conclusion:} There is enough evidence that the true on-time submission rate is above 72\%.
		\end{itemize}
		
	\begin{center}
	\OneTailGraph{1}{1.99}{1.645}{Sample B higher-tail test: \(H_a:p>0.72\)}
	\end{center}
	
	\PropDecisionLine{0.72}{0.76}{1}{green!60!black}{reject \(H_0\)}{1.99}{0.72}{5}

		
	Although both samples have the same sample proportion \(\hat p=0.76\), the larger sample gives a more positive test statistic and a smaller p-value because the standard error is smaller.

\section{Platform adoption rate after training (higher-tail test compared across three significance levels)}
A district dashboard states that 40\% of teachers regularly use a new digital platform. After training, a random sample of 300 teachers shows 138 regular users. At significance levels \(\alpha=1\%\), \(\alpha=2\%\), and \(\alpha=5\%\), test whether the true adoption proportion is higher than 40\%.

\subsection*{C4}
\begin{itemize}
	\item \textbf{Parameter:} \(p\) = true regular-use proportion.
	\item \textbf{Hypotheses at \(\alpha=1\%\), \(\alpha=2\%\), and \(\alpha=5\%\):} \(H_0\colon p=0.40\), \(H_a\colon p>0.40\).
	\item \textbf{Conditions:} \(np_0=120\ge10\), \(n(1-p_0)=180\ge10\).
	\item \textbf{Sample proportion:} \(\hat p=138/300=0.46\).
	\item \textbf{Test statistic:}
	\[
	z=\frac{0.46-0.40}{\sqrt{\frac{0.40(0.60)}{300}}}\approx 2.12.
	\]
	\item \textbf{p-value:} \(P(Z\ge 2.12)\approx 0.0170\).
	\item \textbf{Decision at \(\alpha=5\%\):} Reject \(H_0\) since \(0.0170<0.05\).
	\item \textbf{Decision at \(\alpha=2\%\):} Reject \(H_0\) since \(0.0170<0.02\).
	\item \textbf{Decision at \(\alpha=1\%\):} Fail to reject \(H_0\) since \(0.0170>0.01\).
	\item \textbf{Conclusion:} The null hypothesis is rejected at the 5\% and 2\% significance levels, but it is not rejected at the 1\% significance level. Therefore, only one of the three significance levels fails to reject \(H_0\).
\end{itemize}

\section{On-time assignment completion rate (higher-tail test with fixed sample proportion and variable sample size)}
A school reports that 72\% of assignments are submitted on time. Suppose that for samples of different sizes, the observed sample proportion is always
\[
\hat p=0.76.
\]
At significance levels \(\alpha=10\%\), \(\alpha=5\%\), and \(\alpha=1\%\), test whether the true on-time proportion is higher than 72\%, and describe how the value of the test statistic changes as the sample size \(n\) increases.

\subsection*{C4}
\begin{itemize}
	\item \textbf{Parameter:} \(p\) = true on-time submission proportion.
	\item \textbf{Hypotheses at \(\alpha=10\%\), \(\alpha=5\%\), and \(\alpha=1\%\):} \(H_0\colon p=0.72\), \(H_a\colon p>0.72\).
	\item \textbf{Condition:} For a one-sample proportion test, check
	\[
	np_0=n(0.72)\ge10
	\qquad\text{and}\qquad
	n(1-p_0)=n(0.28)\ge10.
	\]
	Since
	\[
	n\ge \frac{10}{0.28}\approx 35.72,
	\]
	the conditions are satisfied for \(n\ge36\).
	\item \textbf{Sample proportion:} \(\hat p=0.76\).
	\item \textbf{Test statistic as a function of \(n\):}
	\[
	z(n)=\frac{0.76-0.72}{\sqrt{\frac{0.72(0.28)}{n}}}.
	\]
	Simplifying,
	\[
	z(n)=\frac{0.04}{\sqrt{\frac{0.2016}{n}}}
	=\frac{0.04\sqrt{n}}{\sqrt{0.2016}}
	\approx 0.0891\sqrt{n}.
	\]
	\item \textbf{p-value as a function of \(n\):}
	\[
	p\text{-value}(n)=P(Z\ge z(n)).
	\]
	Since this is a higher-tail test,
	\[
	p\text{-value}(n)=1-\Phi(z(n)),
	\]
	where \(\Phi\) is the standard normal cumulative distribution function.
	\item \textbf{Interpretation of the formulas:} Since \(z(n)\approx 0.0891\sqrt{n}\), the test statistic increases as \(n\) increases when the sample proportion stays fixed at \(0.76\). Therefore, the p-value decreases as \(n\) increases.
	\item \textbf{Critical values:} For a higher-tail test,
	\[

\section{Workshop attendance rate (higher-tail test with fixed sample size and variable sample proportion)}
A school reports that 72\% of students attend the weekly academic support workshop. Suppose that the sample size is fixed at
\[
n=400,
\]
and that the observed sample proportion \(\hat p\) varies from one sample to another. At significance levels \(\alpha=10\%\), \(\alpha=5\%\), and \(\alpha=1\%\), test whether the true attendance proportion is higher than 72\%, and describe how the value of the test statistic and the p-value change as the observed sample proportion changes.

\subsection*{C4}
\begin{itemize}
	\item \textbf{Parameter:} \(p\) = true workshop-attendance proportion.
	
	\item \textbf{Hypotheses at \(\alpha=10\%\), \(\alpha=5\%\), and \(\alpha=1\%\):}
	\[
	H_0\colon p=0.72,
	\qquad
	H_a\colon p>0.72.
	\]
	
	\item \textbf{Conditions:} For a one-sample proportion test, check
	\[
	np_0=400(0.72)=288\ge10
	\qquad\text{and}\qquad
	n(1-p_0)=400(0.28)=112\ge10.
	\]
	So the normal conditions are satisfied.
	
	\item \textbf{Test statistic as a function of \(\hat p\):}
	\[
	z(\hat p)=\frac{\hat p-0.72}{\sqrt{\frac{0.72(0.28)}{400}}}.
	\]
	Since
	\[
	\sqrt{\frac{0.72(0.28)}{400}}=\sqrt{0.000504}\approx 0.02245,
	\]
	we obtain
	\[
	z(\hat p)\approx \frac{\hat p-0.72}{0.02245}
	\approx 44.54(\hat p-0.72).
	\]

	\[
	p\text{-value}(\hat p)=P(Z\ge z(\hat p))=1-\Phi(z(\hat p)).
