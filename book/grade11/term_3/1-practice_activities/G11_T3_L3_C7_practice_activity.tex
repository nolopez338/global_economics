\section*{Evaluated criterion}
\begin{itemize}
\item C7: Uses one-sample mean \(z\)-tests with known \(\sigma\) to find the minimum sample size required to reject \(H_0\), then interprets the result in context.
\end{itemize}

\section{Reference rule for all problems}
\[z(n)=\frac{\bar x-\mu_0}{\sigma/\sqrt{n}}\]
Reject \(H_0\) when: higher-tail \(z(n)\ge z_\alpha\), lower-tail \(z(n)\le -z_\alpha\), two-tailed \(|z(n)|\ge z_{\alpha/2}\).

\section{Problem 1: Higher-tail, one significance level}
\subsection*{C7}
\begin{itemize}
\item \textbf{Context:} Export price records (historical benchmark and new customs sample).
\item \textbf{Parameter:} \(\mu\) average export price.
\item \textbf{Hypotheses:} \(H_0:\mu=120\), \(H_a:\mu>120\).
\item \textbf{Known standard deviation:} \(\sigma=20\), \textbf{Observed mean:} \(\bar x=124\), \textbf{Significance:} \(\alpha=5\%\).
\item \textbf{Test statistic:} \(z(n)=\frac{124-120}{20/\sqrt n}=0.2\sqrt n\).
\item \textbf{Critical value:} \(z_{5\%}=1.645\).
\item \textbf{Inequality:} \(0.2\sqrt n\ge1.645\Rightarrow n\ge67.65\).
\item \textbf{Minimum sample size:} \(n_{\min}=68\).
\item \textbf{Interpretation:} Need at least 68 records to reject \(H_0\).
\end{itemize}
\MinNGraph{0.2}{1.645}{100}{4}{68}

\section{Problem 2: Lower-tail, one significance level}
\subsection*{C7}
\begin{itemize}
\item \textbf{Context:} Trade logistics study on delivery-time records.
\item \textbf{Parameter:} \(\mu\) average delivery days.
\item \textbf{Hypotheses:} \(H_0:\mu=18\), \(H_a:\mu<18\).
\item \textbf{Known standard deviation:} \(\sigma=4\), \textbf{Observed mean:} \(\bar x=17\), \textbf{Significance:} \(\alpha=10\%\).
\item \textbf{Test statistic:} \(z(n)=-0.25\sqrt n\).
\item \textbf{Critical value:} \(-z_{10\%}=-1.282\).
\item \textbf{Inequality:} \(-0.25\sqrt n\le-1.282\Rightarrow n\ge26.30\).
\item \textbf{Minimum sample size:} \(n_{\min}=27\).
\item \textbf{Interpretation:} Need at least 27 observations to reject \(H_0\).
\end{itemize}
\MinNGraph{0.25}{1.282}{80}{4}{27}

\section{Problem 3: Two-tailed, one significance level}
\subsection*{C7}
\begin{itemize}
\item \textbf{Context:} Household survey on imported food spending.
\item \textbf{Parameter:} \(\mu\) average weekly spending.
\item \textbf{Hypotheses:} \(H_0:\mu=75\), \(H_a:\mu\ne75\).
\item \textbf{Known standard deviation:} \(\sigma=12\), \textbf{Observed mean:} \(\bar x=78\), \textbf{Significance:} \(\alpha=5\%\).
\item \textbf{Test statistic:} \(|z(n)|=0.25\sqrt n\).
\item \textbf{Critical value:} \(z_{5\%/2}=1.96\).
\item \textbf{Inequality:} \(0.25\sqrt n\ge1.96\Rightarrow n\ge61.47\).
\item \textbf{Minimum sample size:} \(n_{\min}=62\).
\item \textbf{Interpretation:} Need at least 62 surveyed households to reject \(H_0\).
\end{itemize}
\MinNGraph{0.25}{1.96}{100}{4}{62}

\section{Problem 4: Higher-tail, multiple significance levels}
\subsection*{C7}
\begin{itemize}

\item \(\alpha=10\%\Rightarrow n_{\min}=27\), \(\alpha=5\%\Rightarrow n_{\min}=44\), \(\alpha=1\%\Rightarrow n_{\min}=87\).
\item Stricter \(\alpha\) needs larger \(n\).
\end{itemize}
