\section*{Evaluated criterion}
\begin{itemize}
\item C7: Uses one-sample mean \(z\)-tests with known \(\sigma\) to find the minimum sample size required to reject \(H_0\), then interprets the result in context.
\end{itemize}

\section{Reference rule for all problems}

For the sample mean case,
\[
z=\frac{\bar x-\mu_0}{\sigma/\sqrt{n}}.
\]

Solving for \(n\),
\[
z=\frac{(\bar x-\mu_0)\sqrt{n}}{\sigma}
\qquad \longrightarrow \qquad
z\sigma=(\bar x-\mu_0)\sqrt{n}
\qquad \longrightarrow \qquad
\sqrt{n}=\frac{z\sigma}{\bar x-\mu_0}
\qquad \longrightarrow \qquad
n=\left(\frac{z\sigma}{\bar x-\mu_0}\right)^2.
\]

Using a required critical value \(z^*\), the full expression is
\[
n=\left(\frac{z^*\sigma}{\bar x-\mu_0}\right)^2.
\]

For the proportion case, the notation changes from \(\bar x\) to \(\hat p\), and from \(\mu_0\) to \(p_0\). The standard deviation can be calculated directly using \(p_0\):
\[
\sigma_{\hat p}=\sqrt{p_0(1-p_0)}.
\]

Therefore, for the proportion case,
\[
z=\frac{\hat p-p_0}{\frac{\sqrt{p_0(1-p_0)}}{\sqrt{n}}} = \frac{\hat p-p_0}{\sqrt{\frac{p_0(1-p_0)}{n}}}..
\]

Solving for \(n\),
\[
z=\frac{(\hat p-p_0)\sqrt{n}}{\sqrt{p_0(1-p_0)}}
\qquad \longrightarrow \qquad
\sqrt{n}=\frac{z\sqrt{p_0(1-p_0)}}{\hat p-p_0}
\qquad \longrightarrow \qquad
n=\left(\frac{z\sqrt{p_0(1-p_0)}}{\hat p-p_0}\right)^2.
\]

Using a required critical value \(z^*\), the full expression is
\[
n=\frac{(z^*)^2p_0(1-p_0)}{(\hat p-p_0)^2}.
\]

Reject \(H_0\) when: higher-tail \(z\ge z_\alpha\), lower-tail \(z\le -z_\alpha\), two-tailed \(|z|\ge z_{\alpha/2}\).

\section{Problem 1: Right-tail, one significance level}
\subsection*{C7}
\begin{itemize}
	\item \textbf{Context:} A previous study found that the average export price is \(120\). However, newly obtained data showed an observed export average of \(124\). We want to determine whether there is enough evidence to update the previous study's claim.
	\item \textbf{Parameter:} \(\mu\) average export price.
	\item \textbf{Hypotheses:} \(H_0:\mu=120\), \(H_a:\mu>120\).
	\item \textbf{Known standard deviation:} \(\sigma=20\), \textbf{Observed mean:} \(\bar x=124\), \textbf{Significance:} \(\alpha=5\%\).
	\item \textbf{Test statistic:} \(z(n)=\frac{124-120}{20/\sqrt n}=0.2\sqrt n\).
	\item \textbf{Critical value:} \(z_{5\%}=1.645\).
	\item \textbf{Inequality:} \(0.2\sqrt n\ge1.645\Rightarrow n\ge67.65\).
	\item \textbf{Minimum sample size:} \(n_{\min}=68\).
	\item \textbf{Interpretation:} Need at least 68 records to reject \(H_0\) and conclude that there is enough evidence to update the previous study's claim.
\end{itemize}

\section{Problem 2: Left-tail, one significance level}

\section{Problem 3: Comparing a left-tail test and a two-tailed test}
\subsection*{C7}
\begin{itemize}
	\item \textbf{Context:} A previous study found that the average income is \(900\). However, newly obtained survey data showed an observed average income of \(885\). Since the observed average is lower than \(900\), a left-tail test can be used if the goal is only to test whether income decreased. A two-tailed test can also be used if the goal is to determine whether the previous study's claim should be updated in either direction.
	\item \textbf{Test setup:} \(\mu_0=900\), \(\bar x=885\), \(\sigma=60\), so
	\[
	z=\frac{885-900}{60/\sqrt n}=-0.25\sqrt n.
	\]
	\item \textbf{Left-tail test:} \(H_0:\mu=900\), \(H_a:\mu<900\). Reject \(H_0\) when \(z\le -z_{\alpha}\).
	\[
	\alpha=10\%\Rightarrow n_{\min}=27,
	\qquad
	\alpha=5\%\Rightarrow n_{\min}=44.
	\]
	\item \textbf{Two-tailed test:} \(H_0:\mu=900\), \(H_a:\mu\ne900\). Reject \(H_0\) when \(|z|\ge z_{\alpha/2}\).
	\[
	\alpha=10\%\Rightarrow n_{\min}=44,
	\qquad
	\alpha=5\%\Rightarrow n_{\min}=62.
	\]
	\item \textbf{Comparison:} For the same significance level, the two-tailed test requires a larger sample size because it uses a larger critical value. In this context, the left-tail test only checks for a decrease, while the two-tailed test checks for any meaningful change from the previous study's value.
\end{itemize}

\section{Problem 7: Left-tail test with multiple significance levels}
\subsection*{C7}
\begin{itemize}
	\item \textbf{Context:} A previous study found that the average production index is \(150\). However, newly obtained data showed an observed average production index of \(144\). Since the observed average is lower than \(150\), a left-tail test is used to determine whether there is enough evidence to update the previous study's claim and conclude that the average production index has decreased.
	\item \textbf{Test setup:} \(\mu_0=150\), \(\bar x=144\), \(\sigma=30\), so
	\[
	z=\frac{144-150}{30/\sqrt n}=-0.2\sqrt n.
	\]
	\item \textbf{Left-tail test:} \(H_0:\mu=150\), \(H_a:\mu<150\). Reject \(H_0\) when \(z\le -z_{\alpha}\).
	\[
	\alpha=10\%\Rightarrow -z_{\alpha}=-1.282
	\qquad \longrightarrow \qquad
	-0.2\sqrt n\le-1.282
	\qquad \longrightarrow \qquad
	n_{\min}=42.
	\]
	\[
	\alpha=5\%\Rightarrow -z_{\alpha}=-1.645
	\qquad \longrightarrow \qquad
	-0.2\sqrt n\le-1.645
	\qquad \longrightarrow \qquad
	n_{\min}=68.
	\]
	\[
	\alpha=1\%\Rightarrow -z_{\alpha}=-2.326
	\qquad \longrightarrow \qquad
	-0.2\sqrt n\le-2.326
	\qquad \longrightarrow \qquad
	n_{\min}=136.
	\]
	\item \textbf{Comparison:} As \(\alpha\) becomes smaller, the critical value moves farther into the left tail. Therefore, a stricter significance level requires a larger minimum sample size.
\end{itemize}
