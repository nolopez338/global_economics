	\ExamTitleBlock{11th grade}{Term 3 — Lesson 1 Practice Activity (C1–C3) with Solutions}{}

	\ExamSection{Evaluated criteria}
	\begin{ExamCriteria}
		\item C1: Describe the concept of a statistical hypothesis in context.
		\item C2: Identify the null and alternative hypotheses using correct parameter notation.
		\item C3: Determine whether a test is one tailed or two tailed based on the claim.
	\end{ExamCriteria}

	\ExamSection{Problems}

	\begin{ExamProblems}

		\item
		\subsection*{Case 1 — Daily app check-ins}
		A school learning app reports that the daily check-in proportion is \(p_0=0.62\). A class project samples \(n=200\) active users and records \(x=112\) daily check-ins. At \(\alpha=0.05\), test whether the true proportion is lower than \(0.62\).

		\subsection*{C1}
		The population parameter is the proportion \(p\) of all active users who check in daily. The hypothesis is about whether this population proportion is below the reported value.

		\subsection*{C2}
		\(H_0\colon p=0.62\), \(H_a\colon p<0.62\), with \(\alpha=0.05\).

		\subsection*{C3}
		This is a left-tailed test because the wording says \emph{lower than} 0.62.
		% --------------------------------------------------

		\item
		\subsection*{Case 2 — Study time per evening}
		A school report claims that students study an average of \(\mu_0=75\) minutes per evening. A sample of \(n=36\) students has \(\bar{x}=79\) minutes, and assume the population standard deviation is known as \(\sigma=12\) minutes. At \(\alpha=0.01\), test whether the true mean is greater than 75 minutes.

		\subsection*{C1}
		The population parameter is the mean \(\mu\) study time per evening for all students in the school. The hypothesis is about whether this mean exceeds the reported benchmark.

		\subsection*{C2}
		\(H_0\colon \mu=75\), \(H_a\colon \mu>75\), with \(\alpha=0.01\).

		\subsection*{C3}
		This is a right-tailed test because the wording says \emph{greater than} 75.
		% --------------------------------------------------

		\item
		\subsection*{Case 3 — Defect rate in chargers}
		A factory process benchmark is a defect proportion of \(p_0=0.06\). In a quality sample of \(n=400\) chargers, \(x=34\) are defective. At \(\alpha=0.05\), test whether the true defect proportion is different from 0.06.

		\subsection*{C1}
		The population parameter is the proportion \(p\) of defective chargers from the process. The hypothesis is about whether this proportion differs from the benchmark value.

		\subsection*{C2}
		\(H_0\colon p=0.06\), \(H_a\colon p\neq 0.06\), with \(\alpha=0.05\).

		\subsection*{C3}
		This is a two-tailed test because the wording says \emph{different from} 0.06, so both directions matter.
		% --------------------------------------------------

		\item

		% --------------------------------------------------

		\item

		\item

		\item
		\subsection*{Case 7 — Cafeteria card usage}
