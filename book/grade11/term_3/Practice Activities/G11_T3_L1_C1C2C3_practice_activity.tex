	\ExamTitleBlock{11th grade}{Term 3 — Lesson 1 Practice Activity (C1–C3) with Solutions}{}

	\ExamSection{Evaluated criteria}
	\begin{ExamCriteria}
		\item C1: Describe the concept of a statistical hypothesis in context.
		\item C2: Identify the null and alternative hypotheses using correct parameter notation.
		\item C3: Determine whether a test is one tailed or two tailed based on the claim.
	\end{ExamCriteria}

	\ExamSection{Problems}

	\begin{ExamProblems}

		\item
		\subsection*{Case 1 — Daily app check-ins}
		A school learning app reports that the daily check-in proportion is 0.62. A class project samples \(n=200\) active users and records \(x=112\) daily check-ins (so \(\hat p=0.56\)).

		\subsection*{C1}
		A statistical hypothesis here is a testable claim about the population proportion \(p\) of all active users who open the app every day.

		\subsection*{C2}
		\(H_0\colon p=0.62\), \(H_a\colon p<0.62\), with \(\alpha=0.05\).

		\subsection*{C3}
		This is a left-tailed test because the suspicion is that the true proportion is below 0.62.

		\subsection*{Probability tasks}
		Using the standardized value \(z=-1.75\):
		\begin{itemize}
			\item Compute the p-value as a probability: \(P(Z\le -1.75)\approx 0.0401\).
		\end{itemize}
		% --------------------------------------------------

		\item
		\subsection*{Case 2 — On-time bus arrivals}
		A transportation office states that the on-time arrival proportion is 0.78. A student team samples \(n=250\) arrivals and observes \(x=209\) on-time (so \(\hat p=0.836\)).

		\subsection*{C1}
		A statistical hypothesis in this context is a claim about the population proportion \(p\) of bus arrivals that are on time.

		\subsection*{C2}
		\(H_0\colon p=0.78\), \(H_a\colon p>0.78\), with \(\alpha=0.01\).

		\subsection*{C3}
		This is a right-tailed test because the suspicion is that the true proportion is greater than 0.78.

		\subsection*{Probability tasks}
		For a right-tailed \(z\)-test at \(\alpha=0.01\):
		\begin{itemize}
			\item Find the critical value \(z^*\approx 2.326\).
			\item State the rejection region: reject \(H_0\) if \(z>2.326\).
		\end{itemize}
		% --------------------------------------------------

		\subsection*{C3}

		\subsection*{Probability tasks}
		Using the standardized value \(z=2.10\):

			\item Compute the two-tailed p-value: \(2P(Z\ge 2.10)\approx 0.0358\).
		\end{itemize}

		\subsection*{Case 8 — Delivery within one day}
		A local courier advertises that 0.80 of orders are delivered within one day. A check of \(n=220\) orders finds \(x=187\) one-day deliveries (so \(\hat p=0.85\)).

		\subsection*{C1}
		A statistical hypothesis in this setting is a claim about the population proportion \(p\) of orders delivered within one day.

		\subsection*{C2}
		\(H_0\colon p=0.80\), \(H_a\colon p>0.80\), with \(\alpha=0.05\).

		\subsection*{C3}
		This is a right-tailed test because the suspicion is that the true one-day delivery proportion exceeds 0.80.

		\subsection*{Probability tasks}
		Given rejection region \(z>1.645\) for \(\alpha=0.05\), suppose the true proportion is \(p_1=0.84\).
		\begin{itemize}
			\item Approximate \(\beta=P(\text{fail to reject }H_0\mid p=0.84)\).
			\item Using \(\hat p_c=0.80+1.645\sqrt{\frac{0.80(0.20)}{220}}\approx 0.8444\), compute
			\[
			\beta\approx P\!\left(\hat p\le 0.8444\mid p=0.84\right)
			= P\!\left(Z\le \frac{0.8444-0.84}{\sqrt{\frac{0.84(0.16)}{220}}}\right)
			\approx P(Z\le 0.18)\approx 0.5714.
			\]
		\end{itemize}
