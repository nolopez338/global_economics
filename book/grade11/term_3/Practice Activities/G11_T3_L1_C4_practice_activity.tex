\begin{ExamProblems}
\item
\textbf{Statement.} Applies the general procedure for hypothesis testing with fixed probability in contextual situations.

\textbf{Problem.} A company states that the proportion of products that pass quality control is \(p_0=0.85\). In a random sample of \(n=200\) products, \(x=160\) products pass quality control. At \(\alpha=0.05\), test whether the true pass proportion is lower than 0.85.

\textbf{Solution.}
\begin{itemize}
    \item \textbf{Hypotheses:}
    \[
    H_0\colon p=0.85,\qquad H_1\colon p<0.85
    \]

    \item \textbf{Conditions:}
    \[
    np_0=200(0.85)=170\ge 10,\qquad n(1-p_0)=200(0.15)=30\ge 10
    \]
    So the normal approximation is valid.

    \item \textbf{Sample proportion:}
    \[
    \hat p=\frac{x}{n}=\frac{160}{200}=0.80
    \]

    \item \textbf{Test statistic:}
    \[
    z=\frac{\hat p-p_0}{\sqrt{\frac{p_0(1-p_0)}{n}}}
    =\frac{0.80-0.85}{\sqrt{\frac{0.85(0.15)}{200}}}
    \approx -1.98
    \]

    \item \textbf{\(p\)-value (left-tailed):}
    \[
    p\text{-value}=P(Z\le -1.98)\approx 0.0239
    \]

    \item \textbf{Decision:} Since \(0.0239<0.05\), reject \(H_0\).

    \item \textbf{Conclusion:} At the 5\% significance level, there is sufficient evidence that the true proportion of products that pass quality control is lower than 0.85.
\end{itemize}
\end{ExamProblems}
