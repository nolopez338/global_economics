\begin{ExamProblems}
\item
\textbf{Statement.} Calculates hypothesis tests on the mean of a population.

\textbf{Problem.} A manufacturer claims that the mean fill mass is \(\mu_0=500\) g. A random sample of size \(n=49\) has mean \(\bar{x}=496\) g. Assume the population standard deviation is known: \(\sigma=14\) g. Test at \(\alpha=0.05\) whether the population mean differs from 500 g.

\textbf{Solution.}
\begin{itemize}
    \item \textbf{Hypotheses:}
    \[
    H_0\colon \mu=500,\qquad H_1\colon \mu\ne 500
    \]

    \item \textbf{Test statistic:}
    \[
    z=\frac{\bar{x}-\mu_0}{\sigma/\sqrt{n}}
    =\frac{496-500}{14/\sqrt{49}}
    =\frac{-4}{2}=-2.00
    \]

    \item \textbf{Two-tailed \(p\)-value:}
    \[
    p\text{-value}=2P(Z\le -2.00)\approx 2(0.0228)=0.0456
    \]

    \item \textbf{Decision:} Since \(0.0456<0.05\), reject \(H_0\).

    \item \textbf{Conclusion:} At the 5\% significance level, there is sufficient evidence that the population mean fill mass is different from 500 g.
\end{itemize}
\end{ExamProblems}
