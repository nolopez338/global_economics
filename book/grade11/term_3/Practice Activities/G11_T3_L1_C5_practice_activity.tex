\ExamSection{Problems}
\begin{ExamProblems}

\item
\section{Average export clearance time}
A customs agency claims the mean clearance time is \(\mu_0=18\) hours. A random sample of \(n=64\) shipments has \(\bar x=19.1\) hours. Assume \(\sigma=4\) hours. At \(\alpha=0.05\), test whether the mean is greater than 18 hours.

\subsection*{C5}
\begin{itemize}
    \item \textbf{Hypotheses:} \(H_0\colon \mu=18\), \(H_1\colon \mu>18\).
    \item \textbf{Test statistic:}
    \[
    z=\frac{19.1-18}{4/\sqrt{64}}=\frac{1.1}{0.5}=2.20.
    \]
    \item \textbf{\(p\)-value:} \(P(Z\ge 2.20)\approx 0.0139\).
    \item \textbf{Decision:} Reject \(H_0\) since \(0.0139<0.05\).
    \item \textbf{Conclusion:} There is sufficient evidence that the mean clearance time exceeds 18 hours.
\end{itemize}

\section{Mean monthly inflation expectation}
Economists test whether the mean household inflation expectation is 6.0\%. A sample of \(n=100\) households gives \(\bar x=5.7\%\). Known \(\sigma=1.5\%\). At \(\alpha=0.10\), test if the mean is lower than 6.0\%.

\subsection*{C5}
\begin{itemize}
    \item \textbf{Hypotheses:} \(H_0\colon \mu=6.0\), \(H_1\colon \mu<6.0\).
    \item \textbf{Test statistic:}
    \[
    z=\frac{5.7-6.0}{1.5/\sqrt{100}}=\frac{-0.3}{0.15}=-2.00.
    \]
    \item \textbf{\(p\)-value:} \(P(Z\le -2.00)\approx 0.0228\).
    \item \textbf{Decision:} Reject \(H_0\) since \(0.0228<0.10\).
    \item \textbf{Conclusion:} Evidence suggests mean inflation expectation is below 6.0\%.
\end{itemize}

\section{Average household electricity bill}
A regulator monitors whether the mean monthly electricity bill differs from \(\mu_0=92\) dollars. Sample: \(n=81\), \(\bar x=95.2\), \(\sigma=13.5\). Use \(\alpha=0.05\).

\subsection*{C5}
\begin{itemize}
    \item \textbf{Hypotheses:} \(H_0\colon \mu=92\), \(H_1\colon \mu\ne 92\).
    \item \textbf{Test statistic:}
    \[
    z=\frac{95.2-92}{13.5/\sqrt{81}}=\frac{3.2}{1.5}=2.13.
    \]
    \item \textbf{\(p\)-value:} \(2P(Z\ge 2.13)\approx 0.0332\).
    \item \textbf{Decision:} Reject \(H_0\) since \(0.0332<0.05\).
    \item \textbf{Conclusion:} The mean bill is significantly different from 92 dollars.
\end{itemize}

\section{Mean agricultural loan approval time}
A development bank states average approval time is \(\mu_0=12\) days. For \(n=49\) loans, \(\bar x=11.4\) days and known \(\sigma=2.8\) days. At \(\alpha=0.01\), test whether the mean is lower than 12 days.

\subsection*{C5}
\begin{itemize}
    \item \textbf{Hypotheses:} \(H_0\colon \mu=12\), \(H_1\colon \mu<12\).
    \item \textbf{Test statistic:}
    \[
    z=\frac{11.4-12}{2.8/\sqrt{49}}=\frac{-0.6}{0.4}=-1.50.
    \]
    \item \textbf{\(p\)-value:} \(P(Z\le -1.50)\approx 0.0668\).
    \item \textbf{Decision:} Fail to reject \(H_0\) since \(0.0668>0.01\).
    \item \textbf{Conclusion:} At 1\%, there is insufficient evidence that mean approval time is below 12 days.
\end{itemize}
