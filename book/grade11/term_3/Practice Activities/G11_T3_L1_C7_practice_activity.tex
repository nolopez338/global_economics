\begin{ExamProblems}
\item
\textbf{Statement.} Contrasts p-values in hypothesis tests with the reference probability.

\textbf{Problem.} In a hypothesis test, let the significance level be \(\alpha=0.05\).
\begin{itemize}
    \item Suppose the test gives \(p\)-value \(=0.018\). Answer:
    \begin{itemize}
        \item[(a)] Compare the \(p\)-value with \(\alpha\).
        \item[(b)] State the decision.
        \item[(c)] State the conclusion in context (evidence for the alternative hypothesis or not).
    \end{itemize}
    \item Now suppose instead that the test gives \(p\)-value \(=0.11\). Answer:
    \begin{itemize}
        \item[(d)] Compare the \(p\)-value with \(\alpha\).
        \item[(e)] State the decision.
        \item[(f)] State the conclusion in context.
    \end{itemize}
\end{itemize}

\textbf{Solution.}
\begin{itemize}
    \item[(a)] \(0.018<0.05\), so the \(p\)-value is smaller than \(\alpha\).
    \item[(b)] Reject \(H_0\).
    \item[(c)] There is sufficient evidence in favor of the alternative hypothesis.

    \item[(d)] \(0.11>0.05\), so the \(p\)-value is greater than \(\alpha\).
    \item[(e)] Fail to reject \(H_0\).
    \item[(f)] There is not sufficient evidence in favor of the alternative hypothesis.
\end{itemize}
\end{ExamProblems}
