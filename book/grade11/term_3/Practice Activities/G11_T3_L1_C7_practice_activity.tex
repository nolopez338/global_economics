\ExamSection{Evaluated criteria}
\begin{ExamCriteria}
    \item C7: Contrasts p-values in hypothesis tests with the reference probability.
\end{ExamCriteria}

\ExamSection{Problems}
\begin{ExamProblems}

\item
\section{Inflation survey evidence}
A test on a mean inflation expectation uses \(\alpha=0.05\) and reports \(p\)-value \(=0.018\). Compare \(p\)-value and \(\alpha\), state the decision, and interpret in context.

\subsection*{C7}
\begin{itemize}
    \item \(0.018<0.05\).
    \item Decision: reject \(H_0\).
    \item Interpretation: the data provide sufficient evidence for the alternative claim about inflation expectation.
\end{itemize}

\item
\section{Export delay claim under weak evidence}
For a right-tailed test on average export delay, \(\alpha=0.10\) and \(p\)-value \(=0.27\). Compare and conclude.

\subsection*{C7}
\begin{itemize}
    \item \(0.27>0.10\).
    \item Decision: fail to reject \(H_0\).
    \item Interpretation: evidence is insufficient to support that delays are higher than the benchmark.
\end{itemize}

\item
\section{Two p-values for unemployment duration}
A ministry compares two regional studies for the claim \(\mu<11\) weeks at \(\alpha=0.05\): Study A has \(p=0.041\), Study B has \(p=0.22\). Decide for each study.

\subsection*{C7}
\begin{itemize}
    \item Study A: \(0.041<0.05\) \(\Rightarrow\) reject \(H_0\); sufficient evidence that mean duration is below 11 weeks.
    \item Study B: \(0.22>0.05\) \(\Rightarrow\) fail to reject \(H_0\); insufficient evidence for that claim.
\end{itemize}

\item
\section{Recycling participation with strict significance}
A city uses \(\alpha=0.01\) to test whether recycling participation changed. The reported \(p\)-value is 0.014. Compare and interpret.

\subsection*{C7}
\begin{itemize}
    \item \(0.014>0.01\).
    \item Decision: fail to reject \(H_0\).
    \item Interpretation: with this strict 1\% threshold, evidence is not strong enough to conclude a change.
\end{itemize}

\item
\section{Comparing two audits of tax filing rates}
A hypothesis test uses \(\alpha=0.05\). Audit 1 gives \(p=0.003\), and Audit 2 gives \(p=0.049\). Compare both with \(\alpha\), then conclude.

\subsection*{C7}
\begin{itemize}
    \item Audit 1: \(0.003<0.05\) \(\Rightarrow\) reject \(H_0\), very strong evidence for the alternative.
    \item Audit 2: \(0.049<0.05\) \(\Rightarrow\) reject \(H_0\), but evidence is weaker and close to the cutoff.
\end{itemize}

\item
\section{Hospital wait-time policy review}
For a two-tailed test on mean registration wait time, \(\alpha=0.10\), \(p\)-value \(=0.083\). Make the comparison and decision.

\subsection*{C7}
\begin{itemize}
    \item \(0.083<0.10\).
    \item Decision: reject \(H_0\).
    \item Interpretation: at 10\% significance, there is enough evidence that mean wait time differs from the target.
\end{itemize}

\item
\section{Two studies on fuel subsidy awareness}
A campaign tests whether awareness exceeds 65\% at \(\alpha=0.05\). Study X reports \(p=0.058\), Study Y reports \(p=0.006\). Compare and conclude each.

\subsection*{C7}
\begin{itemize}
    \item Study X: \(0.058>0.05\) \(\Rightarrow\) fail to reject \(H_0\); insufficient evidence that awareness exceeds 65\%.
    \item Study Y: \(0.006<0.05\) \(\Rightarrow\) reject \(H_0\); sufficient evidence that awareness exceeds 65\%.
\end{itemize}

\item
\section{Microcredit repayment benchmark}
A lender tests whether the on-time repayment proportion is below 88\% with \(\alpha=0.05\). The test returns \(p=0.51\). Compare and conclude.

\subsection*{C7}
\begin{itemize}
    \item \(0.51>0.05\).
    \item Decision: fail to reject \(H_0\).
    \item Interpretation: evidence does not support that repayment is below 88\%.
\end{itemize}

\item
\section{Two p-values for broadband access programs}
At \(\alpha=0.01\), Program A yields \(p=0.009\) and Program B yields \(p=0.031\) when testing whether broadband access improved. Compare and decide.

\subsection*{C7}
\begin{itemize}
    \item Program A: \(0.009<0.01\) \(\Rightarrow\) reject \(H_0\); sufficient evidence of improvement at 1\%.
    \item Program B: \(0.031>0.01\) \(\Rightarrow\) fail to reject \(H_0\); not enough evidence at 1\%.
\end{itemize}

\item
\section{Customs compliance checks under different evidence strengths}
A compliance test uses \(\alpha=0.05\). Month 1 has \(p=0.12\), and Month 2 has \(p=0.0008\). Compare each \(p\)-value to \(\alpha\) and conclude.

\subsection*{C7}
\begin{itemize}
    \item Month 1: \(0.12>0.05\) \(\Rightarrow\) fail to reject \(H_0\); insufficient evidence of a change.
    \item Month 2: \(0.0008<0.05\) \(\Rightarrow\) reject \(H_0\); very strong evidence of a change in compliance.
\end{itemize}

\end{ExamProblems}
